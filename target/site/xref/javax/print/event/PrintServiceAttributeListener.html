26  package javax.print.event;
27  
28  /**
29    * Implementations of this listener interface are attached to a
30    * {@link javax.print.PrintService PrintService} to monitor
31    * the status of the print service.
32    * <p>
33    * To monitor a particular job see {@link PrintJobListener} and
34    * {@link PrintJobAttributeListener}.
35    */
36  
37  public interface PrintServiceAttributeListener {
38  
39      /**
40       * Called to notify a listener of an event in the print service.
41       * The service will call this method on an event notification thread.
42       * The client should not perform lengthy processing in this callback
43       * or subsequent event notifications may be blocked.
44       * @param psae the event being notified
45       */
46      public void attributeUpdate(PrintServiceAttributeEvent psae) ;
47  
48  }
