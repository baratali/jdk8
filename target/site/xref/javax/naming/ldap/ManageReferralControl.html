26  package javax.naming.ldap;
27  
28  /**
29   * Requests that referral and other special LDAP objects be manipulated
30   * as normal LDAP objects. It enables the requestor to interrogate or
31   * update such objects.
32   *<p>
33   * This class implements the LDAPv3 Request Control for ManageDsaIT
34   * as defined in
35   * <a href="http://www.ietf.org/rfc/rfc3296.txt">RFC 3296</a>.
36   *
37   * The control has no control value.
38   *
39   * @since 1.5
40   * @author Vincent Ryan
41   */
42  final public class ManageReferralControl extends BasicControl {
43  
44      /**
45       * The ManageReferral control's assigned object identifier
46       * is 2.16.840.1.113730.3.4.2.
47       */
48      public static final String OID = "2.16.840.1.113730.3.4.2";
49  
50      private static final long serialVersionUID = 3017756160149982566L;
51  
52      /**
53       * Constructs a critical ManageReferral control.
54       */
55      public ManageReferralControl() {
56          super(OID, true, null);
57      }
58  
59      /**
60       * Constructs a ManageReferral control.
61       *
62       * @param   criticality The control's criticality setting.
63       */
64      public ManageReferralControl(boolean criticality) {
65          super(OID, criticality, null);
66      }
67  }
