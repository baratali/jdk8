52  /**
53   * <p>An RMI object representing a connector server.  Remote clients
54   * can make connections using the {@link #newClient(Object)} method.  This
55   * method returns an RMI object representing the connection.</p>
56   *
57   * <p>User code does not usually reference this class directly.
58   * RMI connection servers are usually created with the class {@link
59   * RMIConnectorServer}.  Remote clients usually create connections
60   * either with {@link javax.management.remote.JMXConnectorFactory}
61   * or by instantiating {@link RMIConnector}.</p>
62   *
63   * <p>This is an abstract class.  Concrete subclasses define the
64   * details of the client connection objects, such as whether they use
65   * JRMP or IIOP.</p>
66   *
67   * @since 1.5
68   */

75       * to an empty Map.
76       */

78          this.env = (env == null) ? Collections.<String,Object>emptyMap() : env;

168     /**
169      * <p>Creates a new client connection.  This method calls {@link
170      * #makeClient makeClient} and adds the returned client connection
171      * object to an internal list.  When this
172      * <code>RMIServerImpl</code> is shut down via its {@link
173      * #close()} method, the {@link RMIConnection#close() close()}
174      * method of each object remaining in the list is called.</p>
175      *
176      * <p>The fact that a client connection object is in this internal
177      * list does not prevent it from being garbage collected.</p>
178      *
179      * @param credentials this object specifies the user-defined
180      * credentials to be passed in to the server in order to
181      * authenticate the caller before creating the
182      * <code>RMIConnection</code>.  Can be null.
183      *
184      * @return the newly-created <code>RMIConnection</code>.  This is
185      * usually the object created by <code>makeClient</code>, though
186      * an implementation may choose to wrap that object in another
187      * object implementing <code>RMIConnection</code>.
188      *
189      * @exception IOException if the new client object cannot be
190      * created or exported.
191      *
192      * @exception SecurityException if the given credentials do not allow
193      * the server to authenticate the user successfully.
194      *
195      * @exception IllegalStateException if {@link #getMBeanServer()}
196      * is null.
197      */
198     public RMIConnection newClient(Object credentials) throws IOException {
199         return doNewClient(credentials);
200     }

202     /**
203      * This method could be overridden by subclasses defined in this package
204      * to perform additional operations specific to the underlying transport
205      * before creating the new client connection.
206      */
207     RMIConnection doNewClient(Object credentials) throws IOException {
208         final boolean tracing = logger.traceOn();
209 
210         if (tracing) logger.trace("newClient","making new client");
211 
212         if (getMBeanServer() == null)
213             throw new IllegalStateException("Not attached to an MBean server");
214 
215         Subject subject = null;
216         JMXAuthenticator authenticator =
217             (JMXAuthenticator) env.get(JMXConnectorServer.AUTHENTICATOR);
218         if (authenticator == null) {
219             /*
220              * Create the JAAS-based authenticator only if authentication
221              * has been enabled
222              */
223             if (env.get("jmx.remote.x.password.file") != null ||
224                 env.get("jmx.remote.x.login.config") != null) {
225                 authenticator = new JMXPluggableAuthenticator(env);
226             }
227         }
228         if (authenticator != null) {
229             if (tracing) logger.trace("newClient","got authenticator: " +
230                                authenticator.getClass().getName());
231             try {
232                 subject = authenticator.authenticate(credentials);
233             } catch (SecurityException e) {
234                 logger.trace("newClient", "Authentication failed: " + e);
235                 throw e;
236             }
237         }
238 
239         if (tracing) {
240             if (subject != null)
241                 logger.trace("newClient","subject is not null");
242             else logger.trace("newClient","no subject");
243         }
244 
245         final String connectionId = makeConnectionId(getProtocol(), subject);
246 
247         if (tracing)
248             logger.trace("newClient","making new connection: " + connectionId);
249 
250         RMIConnection client = makeClient(connectionId, subject);
251 
252         dropDeadReferences();
253         WeakReference<RMIConnection> wr = new WeakReference<RMIConnection>(client);
254         synchronized (clientList) {
255             clientList.add(wr);
256         }
257 
258         connServer.connectionOpened(connectionId, "Connection opened", null);
259 
260         synchronized (clientList) {
261             if (!clientList.contains(wr)) {
262                 // can be removed only by a JMXConnectionNotification listener
263                 throw new IOException("The connection is refused.");
264             }
265         }
266 
267         if (tracing)
268             logger.trace("newClient","new connection done: " + connectionId );
269 
270         return client;
271     }

316     /**
317      * <p>Method called when a client connection created by {@link
318      * #makeClient makeClient} is closed.  A subclass that defines
319      * <code>makeClient</code> must arrange for this method to be
320      * called when the resultant object's {@link RMIConnection#close()
321      * close} method is called.  This enables it to be removed from
322      * the <code>RMIServerImpl</code>'s list of connections.  It is
323      * not an error for <code>client</code> not to be in that
324      * list.</p>
325      *
326      * <p>After removing <code>client</code> from the list of
327      * connections, this method calls {@link #closeClient
328      * closeClient(client)}.</p>
329      *
330      * @param client the client connection that has been closed.
331      *
332      * @exception IOException if {@link #closeClient} throws this
333      * exception.
334      *
335      * @exception NullPointerException if <code>client</code> is null.
336      */
337     protected void clientClosed(RMIConnection client) throws IOException {
338         final boolean debug = logger.debugOn();
339 
340         if (debug) logger.trace("clientClosed","client="+client);
341 
342         if (client == null)
343             throw new NullPointerException("Null client");
344 
345         synchronized (clientList) {
346             dropDeadReferences();
347             for (Iterator<WeakReference<RMIConnection>> it = clientList.iterator();
348                  it.hasNext(); ) {
349                 WeakReference<RMIConnection> wr = it.next();
350                 if (wr.get() == client) {
351                     it.remove();
352                     break;
353                 }
354             }
355             /* It is not a bug for this loop not to find the client.  In
356                our close() method, we remove a client from the list before
357                calling its close() method.  */
358         }
359 
360         if (debug) logger.trace("clientClosed", "closing client.");
361         closeClient(client);
362 
363         if (debug) logger.trace("clientClosed", "sending notif");
364         connServer.connectionClosed(client.getConnectionId(),
365                                     "Client connection closed", null);
366 
367         if (debug) logger.trace("clientClosed","done");
368     }

370     /**
371      * <p>Closes this connection server.  This method first calls the
372      * {@link #closeServer()} method so that no new client connections
373      * will be accepted.  Then, for each remaining {@link
374      * RMIConnection} object returned by {@link #makeClient
375      * makeClient}, its {@link RMIConnection#close() close} method is
376      * called.</p>
377      *
378      * <p>The behavior when this method is called more than once is
379      * unspecified.</p>
380      *
381      * <p>If {@link #closeServer()} throws an
382      * <code>IOException</code>, the individual connections are
383      * nevertheless closed, and then the <code>IOException</code> is
384      * thrown from this method.</p>
385      *
386      * <p>If {@link #closeServer()} returns normally but one or more
387      * of the individual connections throws an
388      * <code>IOException</code>, then, after closing all the
389      * connections, one of those <code>IOException</code>s is thrown
390      * from this method.  If more than one connection throws an
391      * <code>IOException</code>, it is unspecified which one is thrown
392      * from this method.</p>
393      *
394      * @exception IOException if {@link #closeServer()} or one of the
395      * {@link RMIConnection#close()} calls threw
396      * <code>IOException</code>.
397      */
398     public synchronized void close() throws IOException {
399         final boolean tracing = logger.traceOn();
400         final boolean debug   = logger.debugOn();
401 
402         if (tracing) logger.trace("close","closing");
403 
404         IOException ioException = null;
405         try {
406             if (debug)   logger.debug("close","closing Server");
407             closeServer();
408         } catch (IOException e) {
409             if (tracing) logger.trace("close","Failed to close server: " + e);
410             if (debug)   logger.debug("close",e);
411             ioException = e;
412         }
413 
414         if (debug)   logger.debug("close","closing Clients");
415         // Loop to close all clients
416         while (true) {
417             synchronized (clientList) {
418                 if (debug) logger.debug("close","droping dead references");
419                 dropDeadReferences();
420 
421                 if (debug) logger.debug("close","client count: "+clientList.size());
422                 if (clientList.size() == 0)
423                     break;
424                 /* Loop until we find a non-null client.  Because we called
425                    dropDeadReferences(), this will usually be the first
426                    element of the list, but a garbage collection could have
427                    happened in between.  */
428                 for (Iterator<WeakReference<RMIConnection>> it = clientList.iterator();
429                      it.hasNext(); ) {
430                     WeakReference<RMIConnection> wr = it.next();
431                     RMIConnection client = wr.get();
432                     it.remove();
433                     if (client != null) {
434                         try {
435                             client.close();
436                         } catch (IOException e) {
437                             if (tracing)
438                                 logger.trace("close","Failed to close client: " + e);
439                             if (debug) logger.debug("close",e);
440                             if (ioException == null)
441                                 ioException = e;
442                         }
443                         break;
444                     }
445                 }
446             }
447         }
448 
449         if(notifBuffer != null)
450             notifBuffer.dispose();
451 
452         if (ioException != null) {
453             if (tracing) logger.trace("close","close failed.");
454             throw ioException;
455         }
456 
457         if (tracing) logger.trace("close","closed.");
458     }

470     private static synchronized String makeConnectionId(String protocol,
471                                                         Subject subject) {
472         connectionIdNumber++;
473 
474         String clientHost = "";
475         try {
476             clientHost = RemoteServer.getClientHost();
477             /*
478              * According to the rules specified in the javax.management.remote
479              * package description, a numeric IPv6 address (detected by the
480              * presence of otherwise forbidden ":" character) forming a part
481              * of the connection id must be enclosed in square brackets.
482              */
483             if (clientHost.contains(":")) {
484                 clientHost = "[" + clientHost + "]";
485             }
486         } catch (ServerNotActiveException e) {
487             logger.trace("makeConnectionId", "getClientHost", e);
488         }
489 
490         final StringBuilder buf = new StringBuilder();
491         buf.append(protocol).append(":");
492         if (clientHost.length() > 0)
493             buf.append("//").append(clientHost);
494         buf.append(" ");
495         if (subject != null) {
496             Set<Principal> principals = subject.getPrincipals();
497             String sep = "";
498             for (Iterator<Principal> it = principals.iterator(); it.hasNext(); ) {
499                 Principal p = it.next();
500                 String name = p.getName().replace(' ', '_').replace(';', ':');
501                 buf.append(sep).append(name);
502                 sep = ";";
503             }
504         }
505         buf.append(" ").append(connectionIdNumber);
506         if (logger.traceOn())
507             logger.trace("newConnectionId","connectionId="+buf);
508         return buf.toString();
509     }

511     private void dropDeadReferences() {
512         synchronized (clientList) {
513             for (Iterator<WeakReference<RMIConnection>> it = clientList.iterator();
514                  it.hasNext(); ) {
515                 WeakReference<RMIConnection> wr = it.next();
516                 if (wr.get() == null)
517                     it.remove();
518             }
519         }
520     }

524         if(notifBuffer == null)

527                                                               env);
528         return notifBuffer;
