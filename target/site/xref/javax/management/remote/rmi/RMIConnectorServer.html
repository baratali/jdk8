167     /**
168      * <p>Makes an <code>RMIConnectorServer</code> for the given MBean
169      * server.</p>
170      *
171      * @param url the URL defining how to create the connector server.
172      * Cannot be null.
173      *
174      * @param environment attributes governing the creation and
175      * storing of the RMI object.  Can be null, which is equivalent to
176      * an empty Map.
177      *
178      * @param rmiServerImpl An implementation of the RMIServer interface,
179      *  consistent with the protocol type specified in <var>url</var>.
180      *  If this parameter is non null, the protocol type specified by
181      *  <var>url</var> is not constrained, and is assumed to be valid.
182      *  Otherwise, only "rmi" and "iiop" will be recognized.
183      *
184      * @param mbeanServer the MBean server to which the new connector
185      * server is attached, or null if it will be attached by being
186      * registered as an MBean in the MBean server.
187      *
188      * @exception IllegalArgumentException if <code>url</code> is null.
189      *
190      * @exception MalformedURLException if <code>url</code> does not
191      * conform to the syntax for an RMI connector, or if its protocol
192      * is not recognized by this implementation. Only "rmi" and "iiop"
193      * are recognized when <var>rmiServerImpl</var> is null.
194      *
195      * @exception IOException if the connector server cannot be created
196      * for some reason or if it is inevitable that its {@link #start()
197      * start} method will fail.
198      *
199      * @see #start
200      */
201     public RMIConnectorServer(JMXServiceURL url, Map<String,?> environment,
202                               RMIServerImpl rmiServerImpl,
203                               MBeanServer mbeanServer)
204             throws IOException {
205         super(mbeanServer);
206 
207         if (url == null) throw new
208             IllegalArgumentException("Null JMXServiceURL");
209         if (rmiServerImpl == null) {
210             final String prt = url.getProtocol();
211             if (prt == null || !(prt.equals("rmi") || prt.equals("iiop"))) {
212                 final String msg = "Invalid protocol type: " + prt;
213                 throw new MalformedURLException(msg);
214             }
215             final String urlPath = url.getURLPath();
216             if (!urlPath.equals("")
217                 && !urlPath.equals("/")
218                 && !urlPath.startsWith("/jndi/")) {
219                 final String msg = "URL path must be empty or start with " +
220                     "/jndi/";
221                 throw new MalformedURLException(msg);
222             }
223         }
224 
225         if (environment == null)
226             this.attributes = Collections.emptyMap();
227         else {
228             EnvHelp.checkAttributes(environment);
229             this.attributes = Collections.unmodifiableMap(environment);
230         }
231 
232         this.address = url;
233         this.rmiServerImpl = rmiServerImpl;
234     }

236     /**
237      * <p>Returns a client stub for this connector server.  A client
238      * stub is a serializable object whose {@link
239      * JMXConnector#connect(Map) connect} method can be used to make
240      * one new connection to this connector server.</p>
241      *
242      * @param env client connection parameters of the same sort that
243      * could be provided to {@link JMXConnector#connect(Map)
244      * JMXConnector.connect(Map)}.  Can be null, which is equivalent
245      * to an empty map.
246      *
247      * @return a client stub that can be used to make a new connection
248      * to this connector server.
249      *
250      * @exception UnsupportedOperationException if this connector
251      * server does not support the generation of client stubs.
252      *
253      * @exception IllegalStateException if the JMXConnectorServer is
254      * not started (see {@link #isActive()}).
255      *
256      * @exception IOException if a communications problem means that a
257      * stub cannot be created.
258      **/
259     public JMXConnector toJMXConnector(Map<String,?> env) throws IOException {
260         // The serialized for of rmiServerImpl is automatically
261         // a RMI server stub.
262         if (!isActive()) throw new
263             IllegalStateException("Connector is not active");
264 
265         // Merge maps
266         Map<String, Object> usemap = new HashMap<String, Object>(
267                 (this.attributes==null)?Collections.<String, Object>emptyMap():
268                     this.attributes);
269 
270         if (env != null) {
271             EnvHelp.checkAttributes(env);
272             usemap.putAll(env);
273         }
274 
275         usemap = EnvHelp.filterAttributes(usemap);
276 
277         final RMIServer stub=(RMIServer)rmiServerImpl.toStub();
278 
279         return new RMIConnector(stub, usemap);
280     }

282     /**
283      * <p>Activates the connector server, that is starts listening for
284      * client connections.  Calling this method when the connector
285      * server is already active has no effect.  Calling this method
286      * when the connector server has been stopped will generate an
287      * <code>IOException</code>.</p>
288      *
289      * <p>The behavior of this method when called for the first time
290      * depends on the parameters that were supplied at construction,
291      * as described below.</p>
292      *
293      * <p>First, an object of a subclass of {@link RMIServerImpl} is
294      * required, to export the connector server through RMI:</p>
295      *
296      * <ul>
297      *
298      * <li>If an <code>RMIServerImpl</code> was supplied to the
299      * constructor, it is used.
300      *
301      * <li>Otherwise, if the protocol part of the
302      * <code>JMXServiceURL</code> supplied to the constructor was
303      * <code>iiop</code>, an object of type {@link RMIIIOPServerImpl}
304      * is created.
305      *
306      * <li>Otherwise, if the <code>JMXServiceURL</code>
307      * was null, or its protocol part was <code>rmi</code>, an object
308      * of type {@link RMIJRMPServerImpl} is created.
309      *
310      * <li>Otherwise, the implementation can create an
311      * implementation-specific {@link RMIServerImpl} or it can throw
312      * {@link MalformedURLException}.
313      *
314      * </ul>
315      *
316      * <p>If the given address includes a JNDI directory URL as
317      * specified in the package documentation for {@link
318      * javax.management.remote.rmi}, then this
319      * <code>RMIConnectorServer</code> will bootstrap by binding the
320      * <code>RMIServerImpl</code> to the given address.</p>
321      *
322      * <p>If the URL path part of the <code>JMXServiceURL</code> was
323      * empty or a single slash (<code>/</code>), then the RMI object
324      * will not be bound to a directory.  Instead, a reference to it
325      * will be encoded in the URL path of the RMIConnectorServer
326      * address (returned by {@link #getAddress()}).  The encodings for
327      * <code>rmi</code> and <code>iiop</code> are described in the
328      * package documentation for {@link
329      * javax.management.remote.rmi}.</p>
330      *
331      * <p>The behavior when the URL path is neither empty nor a JNDI
332      * directory URL, or when the protocol is neither <code>rmi</code>
333      * nor <code>iiop</code>, is implementation defined, and may
334      * include throwing {@link MalformedURLException} when the
335      * connector server is created or when it is started.</p>
336      *
337      * @exception IllegalStateException if the connector server has
338      * not been attached to an MBean server.
339      * @exception IOException if the connector server cannot be
340      * started, or in the case of the {@code iiop} protocol, that
341      * RMI/IIOP is not supported.
342      */
343     public synchronized void start() throws IOException {
344         final boolean tracing = logger.traceOn();
345 
346         if (state == STARTED) {
347             if (tracing) logger.trace("start", "already started");
348             return;
349         } else if (state == STOPPED) {
350             if (tracing) logger.trace("start", "already stopped");
351             throw new IOException("The server has been stopped.");
352         }
353 
354         if (getMBeanServer() == null)
355             throw new IllegalStateException("This connector server is not " +
356                                             "attached to an MBean server");
357 
358         // Check the internal access file property to see
359         // if an MBeanServerForwarder is to be provided
360         //
361         if (attributes != null) {
362             // Check if access file property is specified
363             //
364             String accessFile =
365                 (String) attributes.get("jmx.remote.x.access.file");
366             if (accessFile != null) {
367                 // Access file property specified, create an instance
368                 // of the MBeanServerFileAccessController class
369                 //
370                 MBeanServerForwarder mbsf;
371                 try {
372                     mbsf = new MBeanServerFileAccessController(accessFile);
373                 } catch (IOException e) {
374                     throw EnvHelp.initCause(
375                         new IllegalArgumentException(e.getMessage()), e);
376                 }
377                 // Set the MBeanServerForwarder
378                 //
379                 setMBeanServerForwarder(mbsf);
380             }
381         }
382 
383         try {
384             if (tracing) logger.trace("start", "setting default class loader");
385             defaultClassLoader = EnvHelp.resolveServerClassLoader(
386                     attributes, getMBeanServer());
387         } catch (InstanceNotFoundException infc) {
388             IllegalArgumentException x = new
389                 IllegalArgumentException("ClassLoader not found: "+infc);
390             throw EnvHelp.initCause(x,infc);
391         }
392 
393         if (tracing) logger.trace("start", "setting RMIServer object");
394         final RMIServerImpl rmiServer;
395 
396         if (rmiServerImpl != null)
397             rmiServer = rmiServerImpl;
398         else
399             rmiServer = newServer();
400 
401         rmiServer.setMBeanServer(getMBeanServer());
402         rmiServer.setDefaultClassLoader(defaultClassLoader);
403         rmiServer.setRMIConnectorServer(this);
404         rmiServer.export();
405 
406         try {
407             if (tracing) logger.trace("start", "getting RMIServer object to export");
408             final RMIServer objref = objectToBind(rmiServer, attributes);
409 
410             if (address != null && address.getURLPath().startsWith("/jndi/")) {
411                 final String jndiUrl = address.getURLPath().substring(6);
412 
413                 if (tracing)
414                     logger.trace("start", "Using external directory: " + jndiUrl);
415 
416                 String stringBoolean = (String) attributes.get(JNDI_REBIND_ATTRIBUTE);
417                 final boolean rebind = EnvHelp.computeBooleanFromString( stringBoolean );
418 
419                 if (tracing)
420                     logger.trace("start", JNDI_REBIND_ATTRIBUTE + "=" + rebind);
421 
422                 try {
423                     if (tracing) logger.trace("start", "binding to " + jndiUrl);
424 
425                     final Hashtable<?, ?> usemap = EnvHelp.mapToHashtable(attributes);
426 
427                     bind(jndiUrl, usemap, objref, rebind);
428 
429                     boundJndiUrl = jndiUrl;
430                 } catch (NamingException e) {
431                     // fit e in the nested exception if we are on 1.4
432                     throw newIOException("Cannot bind to URL ["+jndiUrl+"]: "
433                                          + e, e);
434                 }
435             } else {
436                 // if jndiURL is null, we must encode the stub into the URL.
437                 if (tracing) logger.trace("start", "Encoding URL");
438 
439                 encodeStubInAddress(objref, attributes);
440 
441                 if (tracing) logger.trace("start", "Encoded URL: " + this.address);
442             }
443         } catch (Exception e) {
444             try {
445                 rmiServer.close();
446             } catch (Exception x) {
447                 // OK: we are already throwing another exception
448             }
449             if (e instanceof RuntimeException)
450                 throw (RuntimeException) e;
451             else if (e instanceof IOException)
452                 throw (IOException) e;
453             else
454                 throw newIOException("Got unexpected exception while " +
455                                      "starting the connector server: "
456                                      + e, e);
457         }
458 
459         rmiServerImpl = rmiServer;
460 
461         synchronized(openedServers) {
462             openedServers.add(this);
463         }
464 
465         state = STARTED;
466 
467         if (tracing) {
468             logger.trace("start", "Connector Server Address = " + address);
469             logger.trace("start", "started.");
470         }
471     }

473     /**
474      * <p>Deactivates the connector server, that is, stops listening for
475      * client connections.  Calling this method will also close all
476      * client connections that were made by this server.  After this
477      * method returns, whether normally or with an exception, the
478      * connector server will not create any new client
479      * connections.</p>
480      *
481      * <p>Once a connector server has been stopped, it cannot be started
482      * again.</p>
483      *
484      * <p>Calling this method when the connector server has already
485      * been stopped has no effect.  Calling this method when the
486      * connector server has not yet been started will disable the
487      * connector server object permanently.</p>
488      *
489      * <p>If closing a client connection produces an exception, that
490      * exception is not thrown from this method.  A {@link
491      * JMXConnectionNotification} is emitted from this MBean with the
492      * connection ID of the connection that could not be closed.</p>
493      *
494      * <p>Closing a connector server is a potentially slow operation.
495      * For example, if a client machine with an open connection has
496      * crashed, the close operation might have to wait for a network
497      * protocol timeout.  Callers that do not want to block in a close
498      * operation should do it in a separate thread.</p>
499      *
500      * <p>This method calls the method {@link RMIServerImpl#close()
501      * close} on the connector server's <code>RMIServerImpl</code>
502      * object.</p>
503      *
504      * <p>If the <code>RMIServerImpl</code> was bound to a JNDI
505      * directory by the {@link #start() start} method, it is unbound
506      * from the directory by this method.</p>
507      *
508      * @exception IOException if the server cannot be closed cleanly,
509      * or if the <code>RMIServerImpl</code> cannot be unbound from the
510      * directory.  When this exception is thrown, the server has
511      * already attempted to close all client connections, if
512      * appropriate; to call {@link RMIServerImpl#close()}; and to
513      * unbind the <code>RMIServerImpl</code> from its directory, if
514      * appropriate.  All client connections are closed except possibly
515      * those that generated exceptions when the server attempted to
516      * close them.
517      */
518     public void stop() throws IOException {
519         final boolean tracing = logger.traceOn();
520 
521         synchronized (this) {
522             if (state == STOPPED) {
523                 if (tracing) logger.trace("stop","already stopped.");
524                 return;
525             } else if (state == CREATED) {
526                 if (tracing) logger.trace("stop","not started yet.");
527             }
528 
529             if (tracing) logger.trace("stop", "stopping.");
530             state = STOPPED;
531         }
532 
533         synchronized(openedServers) {
534             openedServers.remove(this);
535         }
536 
537         IOException exception = null;
538 
539         // rmiServerImpl can be null if stop() called without start()
540         if (rmiServerImpl != null) {
541             try {
542                 if (tracing) logger.trace("stop", "closing RMI server.");
543                 rmiServerImpl.close();
544             } catch (IOException e) {
545                 if (tracing) logger.trace("stop", "failed to close RMI server: " + e);
546                 if (logger.debugOn()) logger.debug("stop",e);
547                 exception = e;
548             }
549         }
550 
551         if (boundJndiUrl != null) {
552             try {
553                 if (tracing)
554                     logger.trace("stop",
555                           "unbind from external directory: " + boundJndiUrl);
556 
557                 final Hashtable<?, ?> usemap = EnvHelp.mapToHashtable(attributes);
558 
559                 InitialContext ctx =
560                     new InitialContext(usemap);
561 
562                 ctx.unbind(boundJndiUrl);
563 
564                 ctx.close();
565             } catch (NamingException e) {
566                 if (tracing) logger.trace("stop", "failed to unbind RMI server: "+e);
567                 if (logger.debugOn()) logger.debug("stop",e);
568                 // fit e in as the nested exception if we are on 1.4
569                 if (exception == null)
570                     exception = newIOException("Cannot bind to URL: " + e, e);
571             }
572         }
573 
574         if (exception != null) throw exception;
575 
576         if (tracing) logger.trace("stop", "stopped");
577     }

624     /**
625      * Bind a stub to a registry.
626      * @param jndiUrl URL of the stub in the registry, extracted
627      *        from the <code>JMXServiceURL</code>.
628      * @param attributes A Hashtable containing environment parameters,
629      *        built from the Map specified at this object creation.
630      * @param rmiServer The object to bind in the registry
631      * @param rebind true if the object must be rebound.
632      **/
633     void bind(String jndiUrl, Hashtable<?, ?> attributes,
634               RMIServer rmiServer, boolean rebind)
635         throws NamingException, MalformedURLException {
636         // if jndiURL is not null, we nust bind the stub to a
637         // directory.
638         InitialContext ctx =
639             new InitialContext(attributes);
640 
641         if (rebind)
642             ctx.rebind(jndiUrl, rmiServer);
643         else
644             ctx.bind(jndiUrl, rmiServer);
645         ctx.close();
646     }

648     /**
649      * Creates a new RMIServerImpl.
650      **/
651     RMIServerImpl newServer() throws IOException {
652         final boolean iiop = isIiopURL(address,true);
653         final int port;
654         if (address == null)
655             port = 0;
656         else
657             port = address.getPort();
658         if (iiop)
659             return newIIOPServer(attributes);
660         else
661             return newJRMPServer(attributes, port);
662     }

664     /**
665      * Encode a stub into the JMXServiceURL.
666      * @param rmiServer The stub object to encode in the URL
667      * @param attributes A Map containing environment parameters,
668      *        built from the Map specified at this object creation.
669      **/
670     private void encodeStubInAddress(
671             RMIServer rmiServer, Map<String, ?> attributes)
672             throws IOException {
673 
674         final String protocol, host;
675         final int port;
676 
677         if (address == null) {
678             if (IIOPHelper.isStub(rmiServer))
679                 protocol = "iiop";
680             else
681                 protocol = "rmi";
682             host = null; // will default to local host name
683             port = 0;
684         } else {
685             protocol = address.getProtocol();
686             host = (address.getHost().equals("")) ? null : address.getHost();
687             port = address.getPort();
688         }
689 
690         final String urlPath = encodeStub(rmiServer, attributes);
691 
692         address = new JMXServiceURL(protocol, host, port, urlPath);
693     }

695     static boolean isIiopURL(JMXServiceURL directoryURL, boolean strict)
696         throws MalformedURLException {
697         String protocol = directoryURL.getProtocol();
698         if (protocol.equals("rmi"))
699             return false;
700         else if (protocol.equals("iiop"))
701             return true;
702         else if (strict) {
703 
704             throw new MalformedURLException("URL must have protocol " +
705                                             "\"rmi\" or \"iiop\": \"" +
706                                             protocol + "\"");
707         }
708         return false;
709     }

769     private static String byteArrayToBase64(byte[] a) {
770         int aLen = a.length;
771         int numFullGroups = aLen/3;
772         int numBytesInPartialGroup = aLen - 3*numFullGroups;
773         int resultLen = 4*((aLen + 2)/3);
774         final StringBuilder result = new StringBuilder(resultLen);
775 
776         // Translate all full groups from byte array elements to Base64
777         int inCursor = 0;
778         for (int i=0; i<numFullGroups; i++) {
779             int byte0 = a[inCursor++] & 0xff;
780             int byte1 = a[inCursor++] & 0xff;
781             int byte2 = a[inCursor++] & 0xff;
782             result.append(intToAlpha[byte0 >> 2]);
783             result.append(intToAlpha[(byte0 << 4)&0x3f | (byte1 >> 4)]);
784             result.append(intToAlpha[(byte1 << 2)&0x3f | (byte2 >> 6)]);
785             result.append(intToAlpha[byte2 & 0x3f]);
786         }
787 
788         // Translate partial group if present
789         if (numBytesInPartialGroup != 0) {
790             int byte0 = a[inCursor++] & 0xff;
791             result.append(intToAlpha[byte0 >> 2]);
792             if (numBytesInPartialGroup == 1) {
793                 result.append(intToAlpha[(byte0 << 4) & 0x3f]);
794                 result.append("==");
795             } else {
796                 // assert numBytesInPartialGroup == 2;
797                 int byte1 = a[inCursor++] & 0xff;
798                 result.append(intToAlpha[(byte0 << 4)&0x3f | (byte1 >> 4)]);
799                 result.append(intToAlpha[(byte1 << 2)&0x3f]);
800                 result.append('=');
801             }
802         }
803         // assert inCursor == a.length;
804         // assert result.length() == resultLen;
805         return result.toString();
806     }
