28  package javax.xml.crypto;
29  
30  import java.util.Iterator;
31  
32  /**
33   * An abstract representation of a <code>Data</code> type containing a
34   * node-set. The type (class) and ordering of the nodes contained in the set
35   * are not defined by this class; instead that behavior should be
36   * defined by <code>NodeSetData</code> subclasses.
37   *
38   * @author Sean Mullan
39   * @author JSR 105 Expert Group
40   * @since 1.6
41   */
42  public interface NodeSetData extends Data {
43  
44      /**
45       * Returns a read-only iterator over the nodes contained in this
46       * <code>NodeSetData</code> in
47       * <a href="http://www.w3.org/TR/1999/REC-xpath-19991116#dt-document-order">
48       * document order</a>. Attempts to modify the returned iterator
49       * via the <code>remove</code> method throw
50       * <code>UnsupportedOperationException</code>.
51       *
52       * @return an <code>Iterator</code> over the nodes in this
53       *    <code>NodeSetData</code> in document order
54       */
55      @SuppressWarnings("rawtypes")
56      Iterator iterator();
57  }
