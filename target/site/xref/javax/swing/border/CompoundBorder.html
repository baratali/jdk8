25  package javax.swing.border;
26  
27  import java.awt.Graphics;
28  import java.awt.Insets;
29  import java.awt.Component;
30  import java.beans.ConstructorProperties;
31  
32  /**
33   * A composite Border class used to compose two Border objects
34   * into a single border by nesting an inside Border object within
35   * the insets of an outside Border object.
36   *
37   * For example, this class may be used to add blank margin space
38   * to a component with an existing decorative border:
39   *
40   * <pre>
41   *    Border border = comp.getBorder();
42   *    Border margin = new EmptyBorder(10,10,10,10);
43   *    comp.setBorder(new CompoundBorder(border, margin));
44   * </pre>
45   * <p>
46   * <strong>Warning:</strong>
47   * Serialized objects of this class will not be compatible with
48   * future Swing releases. The current serialization support is
49   * appropriate for short term storage or RMI between applications running
50   * the same version of Swing.  As of 1.4, support for long term storage
51   * of all JavaBeans&trade;
52   * has been added to the <code>java.beans</code> package.
53   * Please see {@link java.beans.XMLEncoder}.
54   *
55   * @author David Kloba
56   */
57  @SuppressWarnings("serial")
58  public class CompoundBorder extends AbstractBorder {
59      protected Border outsideBorder;
60      protected Border insideBorder;
61  
62      /**
63       * Creates a compound border with null outside and inside borders.
64       */
65      public CompoundBorder() {
66          this.outsideBorder = null;
67          this.insideBorder = null;
68      }
69  
70      /**
71       * Creates a compound border with the specified outside and
72       * inside borders.  Either border may be null.
73       * @param outsideBorder the outside border
74       * @param insideBorder the inside border to be nested
75       */
76      @ConstructorProperties({"outsideBorder", "insideBorder"})
77      public CompoundBorder(Border outsideBorder, Border insideBorder) {
78          this.outsideBorder = outsideBorder;
79          this.insideBorder = insideBorder;
80      }
81  
82      /**
83       * Returns whether or not the compound border is opaque.
84       *
85       * @return {@code true} if the inside and outside borders
86       *         are each either {@code null} or opaque;
87       *         or {@code false} otherwise
88       */
89      @Override
90      public boolean isBorderOpaque() {
91          return (outsideBorder == null || outsideBorder.isBorderOpaque()) &&
92                 (insideBorder == null || insideBorder.isBorderOpaque());
93      }
94  
95      /**
96       * Paints the compound border by painting the outside border
97       * with the specified position and size and then painting the
98       * inside border at the specified position and size offset by
99       * the insets of the outside border.
100      * @param c the component for which this border is being painted
101      * @param g the paint graphics
102      * @param x the x position of the painted border
103      * @param y the y position of the painted border
104      * @param width the width of the painted border
105      * @param height the height of the painted border
106      */
107     public void paintBorder(Component c, Graphics g, int x, int y, int width, int height) {
108         Insets  nextInsets;
109         int px, py, pw, ph;
110 
111         px = x;
112         py = y;
113         pw = width;
114         ph = height;
115 
116         if(outsideBorder != null) {
117             outsideBorder.paintBorder(c, g, px, py, pw, ph);
118 
119             nextInsets = outsideBorder.getBorderInsets(c);
120             px += nextInsets.left;
121             py += nextInsets.top;
122             pw = pw - nextInsets.right - nextInsets.left;
123             ph = ph - nextInsets.bottom - nextInsets.top;
124         }
125         if(insideBorder != null)
126             insideBorder.paintBorder(c, g, px, py, pw, ph);
127 
128     }
129 
130     /**
131      * Reinitialize the insets parameter with this Border's current Insets.
132      * @param c the component for which this border insets value applies
133      * @param insets the object to be reinitialized
134      */
135     public Insets getBorderInsets(Component c, Insets insets) {
136         Insets  nextInsets;
137 
138         insets.top = insets.left = insets.right = insets.bottom = 0;
139         if(outsideBorder != null) {
140             nextInsets = outsideBorder.getBorderInsets(c);
141             insets.top += nextInsets.top;
142             insets.left += nextInsets.left;
143             insets.right += nextInsets.right;
144             insets.bottom += nextInsets.bottom;
145         }
146         if(insideBorder != null) {
147             nextInsets = insideBorder.getBorderInsets(c);
148             insets.top += nextInsets.top;
149             insets.left += nextInsets.left;
150             insets.right += nextInsets.right;
151             insets.bottom += nextInsets.bottom;
152         }
153         return insets;
154     }
155 
156     /**
157      * Returns the outside border object.
158      */
159     public Border getOutsideBorder() {
160         return outsideBorder;
161     }
162 
163     /**
164      * Returns the inside border object.
165      */
166     public Border getInsideBorder() {
167         return insideBorder;
168     }
169 }
