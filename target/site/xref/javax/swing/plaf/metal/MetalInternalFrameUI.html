49    private static final Border handyEmptyBorder = new EmptyBorder(0,0,0,0);

51    protected static String IS_PALETTE   = "JInternalFrame.isPalette";
52    private static String IS_PALETTE_KEY = "JInternalFrame.isPalette";
53    private static String FRAME_TYPE     = "JInternalFrame.frameType";
54    private static String NORMAL_FRAME   = "normal";
55    private static String PALETTE_FRAME  = "palette";
56    private static String OPTION_DIALOG  = "optionDialog";

58    public MetalInternalFrameUI(JInternalFrame b)   {
59      super(b);
60    }
61  
62    public static ComponentUI createUI(JComponent c)    {
63        return new MetalInternalFrameUI( (JInternalFrame) c);
64    }
65  
66    public void installUI(JComponent c) {
67      super.installUI(c);
68  
69      Object paletteProp = c.getClientProperty(IS_PALETTE_KEY);
70      if ( paletteProp != null ) {
71          setPalette( ((Boolean)paletteProp).booleanValue() );
72      }
73  
74      Container content = frame.getContentPane();
75      stripContentBorder(content);
76      //c.setOpaque(false);
77    }
78  
79    public void uninstallUI(JComponent c) {
80        frame = (JInternalFrame)c;
81  
82        Container cont = ((JInternalFrame)(c)).getContentPane();
83        if (cont instanceof JComponent) {
84          JComponent content = (JComponent)cont;
85          if ( content.getBorder() == handyEmptyBorder) {
86            content.setBorder(null);
87          }
88        }
89        super.uninstallUI(c);
90    }
91  
92      protected void installListeners() {
93          super.installListeners();
94          frame.addPropertyChangeListener(metalPropertyChangeListener);
95      }
96  
97      protected void uninstallListeners() {
98          frame.removePropertyChangeListener(metalPropertyChangeListener);
99          super.uninstallListeners();
100     }

102   protected void installKeyboardActions(){
103       super.installKeyboardActions();
104       ActionMap map = SwingUtilities.getUIActionMap(frame);
105       if (map != null) {
106           // BasicInternalFrameUI creates an action with the same name, we override
107           // it as Metal frames do not have system menus.
108           map.remove("showSystemMenu");
109       }
110   }
111 
112   protected void uninstallKeyboardActions(){
113       super.uninstallKeyboardActions();
114   }
115 
116     protected void uninstallComponents() {
117         titlePane = null;
118         super.uninstallComponents();
119     }

121   private void stripContentBorder(Object c) {
122         if ( c instanceof JComponent ) {
123             JComponent contentComp = (JComponent)c;
124             Border contentBorder = contentComp.getBorder();
125             if (contentBorder == null || contentBorder instanceof UIResource) {
126                 contentComp.setBorder( handyEmptyBorder );
127             }
128         }
129   }

133       return new MetalInternalFrameTitlePane(w);
134   }

137   private void setFrameType( String frameType )
138   {
139       if ( frameType.equals( OPTION_DIALOG ) )
140       {
141           LookAndFeel.installBorder(frame, "InternalFrame.optionDialogBorder");
142           ((MetalInternalFrameTitlePane)titlePane).setPalette( false );
143       }
144       else if ( frameType.equals( PALETTE_FRAME ) )
145       {
146           LookAndFeel.installBorder(frame, "InternalFrame.paletteBorder");
147           ((MetalInternalFrameTitlePane)titlePane).setPalette( true );
148       }
149       else
150       {
151           LookAndFeel.installBorder(frame, "InternalFrame.border");
152           ((MetalInternalFrameTitlePane)titlePane).setPalette( false );
153       }
154   }
155

158     if (isPalette) {
159         LookAndFeel.installBorder(frame, "InternalFrame.paletteBorder");
160     } else {
161         LookAndFeel.installBorder(frame, "InternalFrame.border");
162     }
163     ((MetalInternalFrameTitlePane)titlePane).setPalette(isPalette);

167   private static class MetalPropertyChangeHandler implements
168         PropertyChangeListener
169   {
170       public void propertyChange(PropertyChangeEvent e)
171       {
172           String name = e.getPropertyName();
173           JInternalFrame jif = (JInternalFrame)e.getSource();
174 
175           if (!(jif.getUI() instanceof MetalInternalFrameUI)) {
176               return;
177           }
178 
179           MetalInternalFrameUI ui = (MetalInternalFrameUI)jif.getUI();
180 
181           if ( name.equals( FRAME_TYPE ) )
182           {
183               if ( e.getNewValue() instanceof String )
184               {
185                   ui.setFrameType( (String) e.getNewValue() );
186               }
187           }
188           else if ( name.equals(IS_PALETTE_KEY) )
189           {
190               if ( e.getNewValue() != null )
191               {
192                   ui.setPalette( ((Boolean)e.getNewValue()).booleanValue() );
193               }
194               else
195               {
196                   ui.setPalette( false );
197               }
198           } else if ( name.equals( JInternalFrame.CONTENT_PANE_PROPERTY ) ) {
199               ui.stripContentBorder(e.getNewValue());
200           }
201       }
202   } // end class MetalPropertyChangeHandler

205     private class BorderListener1 extends BorderListener implements SwingConstants
206     {
207 
208         Rectangle getIconBounds() {
209             boolean leftToRight = MetalUtils.isLeftToRight(frame);
210             int xOffset = leftToRight ? 5 : titlePane.getWidth() - 5;
211             Rectangle rect = null;
212 
213             Icon icon = frame.getFrameIcon();
214             if ( icon != null ) {
215                 if ( !leftToRight ) {
216                     xOffset -= icon.getIconWidth();
217                 }
218                 int iconY = ((titlePane.getHeight() / 2) - (icon.getIconHeight() /2));
219                 rect = new Rectangle(xOffset, iconY,
220                     icon.getIconWidth(), icon.getIconHeight());
221             }
222             return rect;
223         }
224 
225         public void mouseClicked(MouseEvent e) {
226             if (e.getClickCount() == 2 && e.getSource() == getNorthPane() &&
227                 frame.isClosable() && !frame.isIcon()) {
228                 Rectangle rect = getIconBounds();
229                 if ((rect != null) && rect.contains(e.getX(), e.getY())) {
230                     frame.doDefaultCloseAction();
231                 }
232                 else {
233                     super.mouseClicked(e);
234                 }
235             }
236             else {
237                 super.mouseClicked(e);
238             }
239         }
240     };    /// End BorderListener Class
