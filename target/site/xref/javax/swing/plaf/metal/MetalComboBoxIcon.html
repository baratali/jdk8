26  package javax.swing.plaf.metal;
27  
28  import java.awt.*;
29  import java.awt.event.*;
30  import javax.swing.*;
31  import javax.swing.plaf.*;
32  import javax.swing.border.*;
33  import java.io.Serializable;
34  import javax.swing.plaf.basic.BasicComboBoxUI;
35  
36  
37  /**
38   * This utility class draws the horizontal bars which indicate a MetalComboBox
39   *
40   * @see MetalComboBoxUI
41   * @author Tom Santos
42   */
43  public class MetalComboBoxIcon implements Icon, Serializable {
44  
45      /**
46       * Paints the horizontal bars for the
47       */
48      public void paintIcon(Component c, Graphics g, int x, int y){
49          JComponent component = (JComponent)c;
50          int iconWidth = getIconWidth();
51  
52          g.translate( x, y );
53  
54          g.setColor( component.isEnabled() ? MetalLookAndFeel.getControlInfo() : MetalLookAndFeel.getControlShadow() );
55          g.drawLine( 0, 0, iconWidth - 1, 0 );
56          g.drawLine( 1, 1, 1 + (iconWidth - 3), 1 );
57          g.drawLine( 2, 2, 2 + (iconWidth - 5), 2 );
58          g.drawLine( 3, 3, 3 + (iconWidth - 7), 3 );
59          g.drawLine( 4, 4, 4 + (iconWidth - 9), 4 );
60  
61          g.translate( -x, -y );
62      }
63  
64      /**
65       * Created a stub to satisfy the interface.
66       */
67      public int getIconWidth() { return 10; }
68  
69      /**
70       * Created a stub to satisfy the interface.
71       */
72      public int getIconHeight()  { return 5; }
73  
74  }
