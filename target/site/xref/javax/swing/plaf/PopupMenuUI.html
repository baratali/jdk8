26  package javax.swing.plaf;
27  
28  import java.awt.event.MouseEvent;
29  import javax.swing.Popup;
30  import javax.swing.PopupFactory;
31  import javax.swing.JPopupMenu;
32  
33  /**
34   * Pluggable look and feel interface for JPopupMenu.
35   *
36   * @author Georges Saab
37   * @author David Karlton
38   */
39  
40  public abstract class PopupMenuUI extends ComponentUI {
41      /**
42       * @since 1.3
43       */
44      public boolean isPopupTrigger(MouseEvent e) {
45          return e.isPopupTrigger();
46      }
47  
48      /**
49       * Returns the <code>Popup</code> that will be responsible for
50       * displaying the <code>JPopupMenu</code>.
51       *
52       * @param popup JPopupMenu requesting Popup
53       * @param x     Screen x location Popup is to be shown at
54       * @param y     Screen y location Popup is to be shown at.
55       * @return Popup that will show the JPopupMenu
56       * @since 1.4
57       */
58      public Popup getPopup(JPopupMenu popup, int x, int y) {
59          PopupFactory popupFactory = PopupFactory.getSharedInstance();
60  
61          return popupFactory.getPopup(popup.getInvoker(), popup, x, y);
62      }
63  }
