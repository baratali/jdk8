65  /**
66   * This class contains all the implementation details related to
67   * Nimbus. It contains all the code for initializing the UIDefaults table,
68   * as well as for selecting
69   * a SynthStyle based on a JComponent/Region pair.

71   * @author Richard Bair

73  final class NimbusDefaults {
74      /**
75       * The map of SynthStyles. This map is keyed by Region. Each Region maps
76       * to a List of LazyStyles. Each LazyStyle has a reference to the prefix
77       * that was registered with it. This reference can then be inspected to see
78       * if it is the proper lazy style.
79       * <p/>
80       * There can be more than one LazyStyle for a single Region if there is more
81       * than one prefix defined for a given region. For example, both Button and
82       * "MyButton" might be prefixes assigned to the Region.Button region.
83       */
84      private Map<Region, List<LazyStyle>> m;
85      /**
86       * A map of regions which have been registered.
87       * This mapping is maintained so that the Region can be found based on
88       * prefix in a very fast manner. This is used in the "matches" method of
89       * LazyStyle.
90       */

92              new HashMap<String, Region>();

95              new WeakHashMap<JComponent, Map<Region, SynthStyle>>();

114     void initialize() {

122         // remove listener for derived colors

1076     /**
1077      * <p>Registers the given region and prefix. The prefix, if it contains
1078      * quoted sections, refers to certain named components. If there are not
1079      * quoted sections, then the prefix refers to a generic component type.</p>
1080      *
1081      * <p>If the given region/prefix combo has already been registered, then
1082      * it will not be registered twice. The second registration attempt will
1083      * fail silently.</p>
1084      *
1085      * @param region The Synth Region that is being registered. Such as Button,
1086      *        or ScrollBarThumb.
1087      * @param prefix The UIDefault prefix. For example, could be ComboBox, or if
1088      *        a named components, "MyComboBox", or even something like
1089      *        ToolBar:"MyComboBox":"ComboBox.arrowButton"
1090      */
1091     void register(Region region, String prefix) {
1092         //validate the method arguments
1093         if (region == null || prefix == null) {
1094             throw new IllegalArgumentException(
1095                     "Neither Region nor Prefix may be null");
1096         }
1097 
1098         //Add a LazyStyle for this region/prefix to m.
1099         List<LazyStyle> styles = m.get(region);
1100         if (styles == null) {
1101             styles = new LinkedList<LazyStyle>();
1102             styles.add(new LazyStyle(prefix));
1103             m.put(region, styles);
1104         } else {
1105             //iterate over all the current styles and see if this prefix has
1106             //already been registered. If not, then register it.
1107             for (LazyStyle s : styles) {
1108                 if (prefix.equals(s.prefix)) {
1109                     return;
1110                 }
1111             }
1112             styles.add(new LazyStyle(prefix));
1113         }

1119     /**
1120      * <p>Locate the style associated with the given region, and component.
1121      * This is called from NimbusLookAndFeel in the SynthStyleFactory
1122      * implementation.</p>
1123      *
1124      * <p>Lookup occurs as follows:<br/>
1125      * Check the map of styles <code>m</code>. If the map contains no styles at
1126      * all, then simply return the defaultStyle. If the map contains styles,
1127      * then iterate over all of the styles for the Region <code>r</code> looking
1128      * for the best match, based on prefix. If a match was made, then return
1129      * that SynthStyle. Otherwise, return the defaultStyle.</p>
1130      *
1131      * @param comp The component associated with this region. For example, if
1132      *        the Region is Region.Button then the component will be a JButton.
1133      *        If the Region is a subregion, such as ScrollBarThumb, then the
1134      *        associated component will be the component that subregion belongs
1135      *        to, such as JScrollBar. The JComponent may be named. It may not be
1136      *        null.
1137      * @param r The region we are looking for a style for. May not be null.
1138      */
1139     SynthStyle getStyle(JComponent comp, Region r) {
1140         //validate method arguments
1141         if (comp == null || r == null) {
1142             throw new IllegalArgumentException(
1143                     "Neither comp nor r may be null");
1144         }
1145 
1146         //if there are no lazy styles registered for the region r, then return
1147         //the default style
1148         List<LazyStyle> styles = m.get(r);
1149         if (styles == null || styles.size() == 0) {
1150             return defaultStyle;
1151         }
1152 
1153         //Look for the best SynthStyle for this component/region pair.
1154         LazyStyle foundStyle = null;
1155         for (LazyStyle s : styles) {
1156             if (s.matches(comp)) {
1157                 //replace the foundStyle if foundStyle is null, or
1158                 //if the new style "s" is more specific (ie, its path was
1159                 //longer), or if the foundStyle was "simple" and the new style
1160                 //was not (ie: the foundStyle was for something like Button and
1161                 //the new style was for something like "MyButton", hence, being
1162                 //more specific.) In all cases, favor the most specific style
1163                 //found.
1164                 if (foundStyle == null ||
1165                    (foundStyle.parts.length < s.parts.length) ||
1166                    (foundStyle.parts.length == s.parts.length 
1167                     && foundStyle.simple && !s.simple)) {
1168                     foundStyle = s;
1169                 }
1170             }
1171         }
1172 
1173         //return the style, if found, or the default style if not found
1174         return foundStyle == null ? defaultStyle : foundStyle.getStyle(comp, r);
1175     }

1347      * A class which creates the NimbusStyle associated with it lazily, but also
1348      * manages a lot more information about the style. It is less of a LazyValue
1349      * type of class, and more of an Entry or Item type of class, as it

1351      *

1358      *   <li>Creating and vending a NimbusStyle lazily.</li>
1359      * </ul>
1360      */
1361     private final class LazyStyle {
1362         /**
1363          * The prefix this LazyStyle was registered with. Something like
1364          * Button or ComboBox:"ComboBox.arrowButton"
1365          */
1366         private String prefix;
1367         /**
1368          * Whether or not this LazyStyle represents an unnamed component
1369          */
1370         private boolean simple = true;
1371         /**
1372          * The various parts, or sections, of the prefix. For example,
1373          * the prefix:
1374          *     ComboBox:"ComboBox.arrowButton"
1375          *
1376          * will be broken into two parts,
1377          *     ComboBox and "ComboBox.arrowButton"
1378          */

1381          * Cached shared style.
1382          */

1385         /**
1386          * Create a new LazyStyle.
1387          *
1388          * @param prefix The prefix associated with this style. Cannot be null.
1389          */
1390         private LazyStyle(String prefix) {
1391             if (prefix == null) {
1392                 throw new IllegalArgumentException(
1393                         "The prefix must not be null");
1394             }
1395 
1396             this.prefix = prefix;
1397 
1398             //there is one odd case that needs to be supported here: cell
1399             //renderers. A cell renderer is defined as a named internal
1400             //component, so for example:
1401             // List."List.cellRenderer"
1402             //The problem is that the component named List.cellRenderer is not a
1403             //child of a JList. Rather, it is treated more as a direct component
1404             //Thus, if the prefix ends with "cellRenderer", then remove all the
1405             //previous dotted parts of the prefix name so that it becomes, for
1406             //example: "List.cellRenderer"
1407             //Likewise, we have a hacked work around for cellRenderer, renderer,
1408             //and listRenderer.
1409             String temp = prefix;
1410             if (temp.endsWith("cellRenderer\"")
1411                     || temp.endsWith("renderer\"")
1412                     || temp.endsWith("listRenderer\"")) {
1413                 temp = temp.substring(temp.lastIndexOf(":\"") + 1);
1414             }
1415 
1416             //otherwise, normal code path
1417             List<String> sparts = split(temp);
1418             parts = new Part[sparts.size()];
1419             for (int i = 0; i < parts.length; i++) {
1420                 parts[i] = new Part(sparts.get(i));
1421                 if (parts[i].named) {
1422                     simple = false;
1423                 }
1424             }

1458         /**
1459          * This LazyStyle is a match for the given component if, and only if,
1460          * for each part of the prefix the component hierarchy matches exactly.
1461          * That is, if given "a":something:"b", then:
1462          * c.getName() must equals "b"
1463          * c.getParent() can be anything
1464          * c.getParent().getParent().getName() must equal "a".
1465          */
1466         boolean matches(JComponent c) {
1467             return matches(c, parts.length - 1);
1468         }

1470         private boolean matches(Component c, int partIndex) {
1471             if (partIndex < 0) return true;
1472             if (c == null) return false;
1473             //only get here if partIndex > 0 and c == null
1474 
1475             String name = c.getName();
1476             if (parts[partIndex].named && parts[partIndex].s.equals(name)) {
1477                 //so far so good, recurse
1478                 return matches(c.getParent(), partIndex - 1);
1479             } else if (!parts[partIndex].named) {
1480                 //if c is not named, and parts[partIndex] has an expected class
1481                 //type registered, then check to make sure c is of the
1482                 //right type;
1483                 Class clazz = parts[partIndex].c;
1484                 if (clazz != null && clazz.isAssignableFrom(c.getClass())) {
1485                     //so far so good, recurse
1486                     return matches(c.getParent(), partIndex - 1);
1487                 } else if (clazz == null &&
1488                            registeredRegions.containsKey(parts[partIndex].s)) {
1489                     Region r = registeredRegions.get(parts[partIndex].s);
1490                     Component parent = r.isSubregion() ? c : c.getParent();
1491                     //special case the JInternalFrameTitlePane, because it
1492                     //doesn't fit the mold. very, very funky.
1493                     if (r == Region.INTERNAL_FRAME_TITLE_PANE && parent != null
1494                         && parent instanceof JInternalFrame.JDesktopIcon) {
1495                         JInternalFrame.JDesktopIcon icon =
1496                                 (JInternalFrame.JDesktopIcon) parent;
1497                         parent = icon.getInternalFrame();
1498                     }
1499                     //it was the name of a region. So far, so good. Recurse.
1500                     return matches(parent, partIndex - 1);
1501                 }
1502             }
1503 
1504             return false;
1505         }

1507         /**
1508          * Given some dot separated prefix, split on the colons that are
1509          * not within quotes, and not within brackets.
1510          *
1511          * @param prefix
1512          * @return
1513          */
1514         private List<String> split(String prefix) {
1515             List<String> parts = new ArrayList<String>();
1516             int bracketCount = 0;
1517             boolean inquotes = false;
1518             int lastIndex = 0;
1519             for (int i = 0; i < prefix.length(); i++) {
1520                 char c = prefix.charAt(i);
1521 
1522                 if (c == '[') {
1523                     bracketCount++;
1524                     continue;
1525                 } else if (c == '"') {
1526                     inquotes = !inquotes;
1527                     continue;
1528                 } else if (c == ']') {
1529                     bracketCount--;
1530                     if (bracketCount < 0) {
1531                         throw new RuntimeException(
1532                                 "Malformed prefix: " + prefix);
1533                     }
1534                     continue;
1535                 }
1536 
1537                 if (c == ':' && !inquotes && bracketCount == 0) {
1538                     //found a character to split on.
1539                     parts.add(prefix.substring(lastIndex, i));
1540                     lastIndex = i + 1;
1541                 }
1542             }
1543             if (lastIndex < prefix.length() - 1 && !inquotes
1544                     && bracketCount == 0) {
1545                 parts.add(prefix.substring(lastIndex));
1546             }
1547             return parts;
1548 
1549         }
