25  package javax.swing.plaf.nimbus;
26  
27  import java.awt.Color;
28  
29  /**
30   * ShadowEffect - base class with all the standard properties for shadow effects
31   *
32   * @author Created by Jasper Potts (Jun 18, 2007)
33   */
34  abstract class ShadowEffect extends Effect {
35      protected Color color = Color.BLACK;
36      /** Opacity a float 0-1 for percentage */
37      protected float opacity = 0.75f;
38      /** Angle in degrees between 0-360 */
39      protected int angle = 135;
40      /** Distance in pixels */
41      protected int distance = 5;
42      /** The shadow spread between 0-100 % */
43      protected int spread = 0;
44      /** Size in pixels */
45      protected int size = 5;
46  
47      // =================================================================================================================
48      // Bean methods
49  
50      Color getColor() {
51          return color;
52      }
53  
54      void setColor(Color color) {
55          Color old = getColor();
56          this.color = color;
57      }
58  
59      float getOpacity() {
60          return opacity;
61      }
62  
63      void setOpacity(float opacity) {
64          float old = getOpacity();
65          this.opacity = opacity;
66      }
67  
68      int getAngle() {
69          return angle;
70      }
71  
72      void setAngle(int angle) {
73          int old = getAngle();
74          this.angle = angle;
75      }
76  
77      int getDistance() {
78          return distance;
79      }
80  
81      void setDistance(int distance) {
82          int old = getDistance();
83          this.distance = distance;
84      }
85  
86      int getSpread() {
87          return spread;
88      }
89  
90      void setSpread(int spread) {
91          int old = getSpread();
92          this.spread = spread;
93      }
94  
95      int getSize() {
96          return size;
97      }
98  
99      void setSize(int size) {
100         int old = getSize();
101         this.size = size;
102     }
103 }
