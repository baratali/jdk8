112     @Override
113     protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
114         //populate componentColors array with colors calculated in getExtendedCacheKeys call
115         componentColors = extendedCacheKeys;
116         //generate this entire method. Each state/bg/fg/border combo that has
117         //been painted gets its own KEY and paint method.
118         switch(state) {
119             case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
120             case BACKGROUND_ENABLED_MOUSEOVER: paintBackgroundEnabledAndMouseOver(g); break;
121             case BACKGROUND_ENABLED_PRESSED: paintBackgroundEnabledAndPressed(g); break;
122             case BACKGROUND_DISABLED: paintBackgroundDisabled(g); break;
123             case BACKGROUND_SELECTED_DISABLED: paintBackgroundSelectedAndDisabled(g); break;
124             case BACKGROUND_SELECTED: paintBackgroundSelected(g); break;
125             case BACKGROUND_SELECTED_MOUSEOVER: paintBackgroundSelectedAndMouseOver(g); break;
126             case BACKGROUND_SELECTED_PRESSED: paintBackgroundSelectedAndPressed(g); break;
127             case BACKGROUND_SELECTED_FOCUSED: paintBackgroundSelectedAndFocused(g); break;
128             case BACKGROUND_SELECTED_MOUSEOVER_FOCUSED: paintBackgroundSelectedAndMouseOverAndFocused(g); break;
129             case BACKGROUND_SELECTED_PRESSED_FOCUSED: paintBackgroundSelectedAndPressedAndFocused(g); break;
130 
131         }
132     }
