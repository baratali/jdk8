103     public ComboBoxArrowButtonPainter(PaintContext ctx, int state) {
104         super();
105         this.state = state;
106         this.ctx = ctx;
107     }

109     @Override
110     protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
111         //populate componentColors array with colors calculated in getExtendedCacheKeys call
112         componentColors = extendedCacheKeys;
113         //generate this entire method. Each state/bg/fg/border combo that has
114         //been painted gets its own KEY and paint method.
115         switch(state) {
116             case BACKGROUND_DISABLED_EDITABLE: paintBackgroundDisabledAndEditable(g); break;
117             case BACKGROUND_ENABLED_EDITABLE: paintBackgroundEnabledAndEditable(g); break;
118             case BACKGROUND_MOUSEOVER_EDITABLE: paintBackgroundMouseOverAndEditable(g); break;
119             case BACKGROUND_PRESSED_EDITABLE: paintBackgroundPressedAndEditable(g); break;
120             case BACKGROUND_SELECTED_EDITABLE: paintBackgroundSelectedAndEditable(g); break;
121             case FOREGROUND_ENABLED: paintForegroundEnabled(g); break;
122             case FOREGROUND_MOUSEOVER: paintForegroundMouseOver(g); break;
123             case FOREGROUND_DISABLED: paintForegroundDisabled(g); break;
124             case FOREGROUND_PRESSED: paintForegroundPressed(g); break;
125             case FOREGROUND_SELECTED: paintForegroundSelected(g); break;
126 
127         }
128     }

342     private Paint decodeGradient1(Shape s) {
343         Rectangle2D bounds = s.getBounds2D();
344         float x = (float)bounds.getX();
345         float y = (float)bounds.getY();
346         float w = (float)bounds.getWidth();
347         float h = (float)bounds.getHeight();
348         return decodeGradient((0.5f * w) + x, (0.0f * h) + y, (0.5f * w) + x, (1.0f * h) + y,
349                 new float[] { 0.0f,0.5f,1.0f },
350                 new Color[] { color2,
351                             decodeColor(color2,color3,0.5f),
352                             color3});
353     }
