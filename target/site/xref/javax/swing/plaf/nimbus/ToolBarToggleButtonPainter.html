105     @Override
106     protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
107         //populate componentColors array with colors calculated in getExtendedCacheKeys call
108         componentColors = extendedCacheKeys;
109         //generate this entire method. Each state/bg/fg/border combo that has
110         //been painted gets its own KEY and paint method.
111         switch(state) {
112             case BACKGROUND_FOCUSED: paintBackgroundFocused(g); break;
113             case BACKGROUND_MOUSEOVER: paintBackgroundMouseOver(g); break;
114             case BACKGROUND_MOUSEOVER_FOCUSED: paintBackgroundMouseOverAndFocused(g); break;
115             case BACKGROUND_PRESSED: paintBackgroundPressed(g); break;
116             case BACKGROUND_PRESSED_FOCUSED: paintBackgroundPressedAndFocused(g); break;
117             case BACKGROUND_SELECTED: paintBackgroundSelected(g); break;
118             case BACKGROUND_SELECTED_FOCUSED: paintBackgroundSelectedAndFocused(g); break;
119             case BACKGROUND_PRESSED_SELECTED: paintBackgroundPressedAndSelected(g); break;
120             case BACKGROUND_PRESSED_SELECTED_FOCUSED: paintBackgroundPressedAndSelectedAndFocused(g); break;
121             case BACKGROUND_MOUSEOVER_SELECTED: paintBackgroundMouseOverAndSelected(g); break;
122             case BACKGROUND_MOUSEOVER_SELECTED_FOCUSED: paintBackgroundMouseOverAndSelectedAndFocused(g); break;
123             case BACKGROUND_DISABLED_SELECTED: paintBackgroundDisabledAndSelected(g); break;
124 
125         }
126     }
