73      //Array of current component colors, updated in each paint call
74      private Object[] componentColors;
75  
76      public ScrollBarTrackPainter(PaintContext ctx, int state) {
77          super();
78          this.state = state;
79          this.ctx = ctx;
80      }
81  
82      @Override
83      protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
84          //populate componentColors array with colors calculated in getExtendedCacheKeys call
85          componentColors = extendedCacheKeys;
86          //generate this entire method. Each state/bg/fg/border combo that has
87          //been painted gets its own KEY and paint method.
88          switch(state) {
89              case BACKGROUND_DISABLED: paintBackgroundDisabled(g); break;
90              case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
91  
92          }
93      }
94          
95  
96  
97      @Override
98      protected final PaintContext getPaintContext() {
99          return ctx;
100     }

102     private void paintBackgroundDisabled(Graphics2D g) {
103         rect = decodeRect1();
104         g.setPaint(decodeGradient1(rect));
105         g.fill(rect);
106 
107     }
108 
109     private void paintBackgroundEnabled(Graphics2D g) {
110         rect = decodeRect1();
111         g.setPaint(decodeGradient2(rect));
112         g.fill(rect);
113         path = decodePath1();
114         g.setPaint(decodeGradient3(path));
115         g.fill(path);
116         path = decodePath2();
117         g.setPaint(decodeGradient4(path));
118         g.fill(path);
119         path = decodePath3();
120         g.setPaint(decodeGradient5(path));
121         g.fill(path);
122         path = decodePath4();
123         g.setPaint(decodeGradient6(path));
124         g.fill(path);
125 
126     }
