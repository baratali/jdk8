102     public InternalFrameTitlePaneIconifyButtonPainter(PaintContext ctx, int state) {
103         super();
104         this.state = state;
105         this.ctx = ctx;

108     @Override
109     protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
110         //populate componentColors array with colors calculated in getExtendedCacheKeys call
111         componentColors = extendedCacheKeys;
112         //generate this entire method. Each state/bg/fg/border combo that has
113         //been painted gets its own KEY and paint method.
114         switch(state) {
115             case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
116             case BACKGROUND_DISABLED: paintBackgroundDisabled(g); break;
117             case BACKGROUND_MOUSEOVER: paintBackgroundMouseOver(g); break;
118             case BACKGROUND_PRESSED: paintBackgroundPressed(g); break;
119             case BACKGROUND_ENABLED_WINDOWNOTFOCUSED: paintBackgroundEnabledAndWindowNotFocused(g); break;
120             case BACKGROUND_MOUSEOVER_WINDOWNOTFOCUSED: paintBackgroundMouseOverAndWindowNotFocused(g); break;
121             case BACKGROUND_PRESSED_WINDOWNOTFOCUSED: paintBackgroundPressedAndWindowNotFocused(g); break;
122 
123         }
124     }
