76      public TextFieldPainter(PaintContext ctx, int state) {
77          super();
78          this.state = state;
79          this.ctx = ctx;
80      }

82      @Override
83      protected void doPaint(Graphics2D g, JComponent c, int width, int height, Object[] extendedCacheKeys) {
84          //populate componentColors array with colors calculated in getExtendedCacheKeys call
85          componentColors = extendedCacheKeys;
86          //generate this entire method. Each state/bg/fg/border combo that has
87          //been painted gets its own KEY and paint method.
88          switch(state) {
89              case BACKGROUND_DISABLED: paintBackgroundDisabled(g); break;
90              case BACKGROUND_ENABLED: paintBackgroundEnabled(g); break;
91              case BACKGROUND_SELECTED: paintBackgroundSelected(g); break;
92              case BORDER_DISABLED: paintBorderDisabled(g); break;
93              case BORDER_FOCUSED: paintBorderFocused(g); break;
94              case BORDER_ENABLED: paintBorderEnabled(g); break;
95  
96          }
97      }
98          
99      protected Object[] getExtendedCacheKeys(JComponent c) {
100         Object[] extendedCacheKeys = null;
101         switch(state) {
102             case BACKGROUND_ENABLED:
103                 extendedCacheKeys = new Object[] {
104                      getComponentColor(c, "background", color2, 0.0f, 0.0f, 0)};
105                 break;
106             case BORDER_FOCUSED:
107                 extendedCacheKeys = new Object[] {
108                      getComponentColor(c, "background", color9, 0.004901961f, -0.19999999f, 0),
109                      getComponentColor(c, "background", color2, 0.0f, 0.0f, 0)};
110                 break;
111             case BORDER_ENABLED:
112                 extendedCacheKeys = new Object[] {
113                      getComponentColor(c, "background", color9, 0.004901961f, -0.19999999f, 0),
114                      getComponentColor(c, "background", color2, 0.0f, 0.0f, 0)};
115                 break;
116         }
117         return extendedCacheKeys;
118     }

296     private Path2D decodePath1() {
297         path.reset();
298         path.moveTo(decodeX(0.4f), decodeY(0.4f));
299         path.lineTo(decodeX(0.4f), decodeY(2.6f));
300         path.lineTo(decodeX(2.6f), decodeY(2.6f));
301         path.lineTo(decodeX(2.6f), decodeY(0.4f));
302         path.curveTo(decodeAnchorX(2.5999999046325684f, 0.0f), decodeAnchorY(0.4000000059604645f, 0.0f), decodeAnchorX(2.880000352859497f, 0.09999999999999432f), decodeAnchorY(0.4000000059604645f, 0.0f), decodeX(2.8800004f), decodeY(0.4f));
303         path.curveTo(decodeAnchorX(2.880000352859497f, 0.09999999999999432f), decodeAnchorY(0.4000000059604645f, 0.0f), decodeAnchorX(2.880000352859497f, 0.0f), decodeAnchorY(2.879999876022339f, 0.0f), decodeX(2.8800004f), decodeY(2.8799999f));
304         path.lineTo(decodeX(0.120000005f), decodeY(2.8799999f));
305         path.lineTo(decodeX(0.120000005f), decodeY(0.120000005f));
306         path.lineTo(decodeX(2.8800004f), decodeY(0.120000005f));
307         path.lineTo(decodeX(2.8800004f), decodeY(0.4f));
308         path.lineTo(decodeX(0.4f), decodeY(0.4f));
309         path.closePath();
310         return path;
311     }

315     private Paint decodeGradient1(Shape s) {
316         Rectangle2D bounds = s.getBounds2D();
317         float x = (float)bounds.getX();
318         float y = (float)bounds.getY();
319         float w = (float)bounds.getWidth();
320         float h = (float)bounds.getHeight();
321         return decodeGradient((0.5f * w) + x, (0.0f * h) + y, (0.5f * w) + x, (1.0f * h) + y,
322                 new float[] { 0.0f,0.5f,1.0f },
323                 new Color[] { color3,
324                             decodeColor(color3,color4,0.5f),
325                             color4});
326     }
