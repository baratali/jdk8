26  package javax.swing.text.html.parser;
27  
28  import java.io.InputStream;
29  
30  /**
31   * Simple class to load resources using the 1.2
32   * security model.  Since the html support is loaded
33   * lazily, it's resources are potentially fetched with
34   * applet code in the call stack.  By providing this
35   * functionality in a class that is only built on 1.2,
36   * reflection can be used from the code that is also
37   * built on 1.1 to call this functionality (and avoid
38   * the evils of preprocessing).  This functionality
39   * is called from ParserDelegator.getResourceAsStream.
40   *
41   * @author  Timothy Prinzing
42   */
43  class ResourceLoader implements java.security.PrivilegedAction {
44  
45      ResourceLoader(String name) {
46          this.name = name;
47      }
48  
49      public Object run() {
50          Object o = ParserDelegator.class.getResourceAsStream(name);
51          return o;
52      }
53  
54      public static InputStream getResourceAsStream(String name) {
55          java.security.PrivilegedAction a = new ResourceLoader(name);
56          return (InputStream) java.security.AccessController.doPrivileged(a);
57      }
58  
59      private String name;
60  }
