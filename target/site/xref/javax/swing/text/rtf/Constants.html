25  package javax.swing.text.rtf;
26  
27  /**
28     Class to hold dictionary keys used by the RTF reader/writer.
29     These should be moved into StyleConstants.
30  */
31  class Constants
32  {
33      /** An array of TabStops */
34      static final String Tabs = "tabs";
35  
36      /** The name of the character set the original RTF file was in */
37      static final String RTFCharacterSet = "rtfCharacterSet";
38  
39      /** Indicates the domain of a Style */
40      static final String StyleType = "style:type";
41  
42      /** Value for StyleType indicating a section style */
43      static final String STSection = "section";
44      /** Value for StyleType indicating a paragraph style */
45      static final String STParagraph = "paragraph";
46      /** Value for StyleType indicating a character style */
47      static final String STCharacter = "character";
48  
49      /** The style of the text following this style */
50      static final String StyleNext = "style:nextStyle";
51  
52      /** Whether the style is additive */
53      static final String StyleAdditive = "style:additive";
54  
55      /** Whether the style is hidden from the user */
56      static final String StyleHidden = "style:hidden";
57  
58      /* Miscellaneous character attributes */
59      static final String Caps          = "caps";
60      static final String Deleted       = "deleted";
61      static final String Outline       = "outl";
62      static final String SmallCaps     = "scaps";
63      static final String Shadow        = "shad";
64      static final String Strikethrough = "strike";
65      static final String Hidden        = "v";
66  
67      /* Miscellaneous document attributes */
68      static final String PaperWidth    = "paperw";
69      static final String PaperHeight   = "paperh";
70      static final String MarginLeft    = "margl";
71      static final String MarginRight   = "margr";
72      static final String MarginTop     = "margt";
73      static final String MarginBottom  = "margb";
74      static final String GutterWidth   = "gutter";
75  
76      /* This is both a document and a paragraph attribute */
77      static final String WidowControl  = "widowctrl";
78  }
