25  package javax.swing.text.rtf;
26  
27  import javax.swing.text.AttributeSet;
28  import javax.swing.text.MutableAttributeSet;
29  import java.io.IOException;
30  
31  /**
32   * This interface describes a class which defines a 1-1 mapping between
33   * an RTF keyword and a SwingText attribute.
34   */
35  interface RTFAttribute
36  {
37      static final int D_CHARACTER = 0;
38      static final int D_PARAGRAPH = 1;
39      static final int D_SECTION = 2;
40      static final int D_DOCUMENT = 3;
41      static final int D_META = 4;
42  
43      /* These next three should really be public variables,
44         but you can't declare public variables in an interface... */
45      /* int domain; */
46      public int domain();
47      /* String swingName; */
48      public Object swingName();
49      /* String rtfName; */
50      public String rtfName();
51  
52      public boolean set(MutableAttributeSet target);
53      public boolean set(MutableAttributeSet target, int parameter);
54  
55      public boolean setDefault(MutableAttributeSet target);
56  
57      /* TODO: This method is poorly thought out */
58      public boolean write(AttributeSet source,
59                           RTFGenerator target,
60                           boolean force)
61          throws IOException;
62  
63      public boolean writeValue(Object value,
64                                RTFGenerator target,
65                                boolean force)
66          throws IOException;
67  }
