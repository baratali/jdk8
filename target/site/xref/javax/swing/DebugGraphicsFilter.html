26  package javax.swing;
27  
28  import java.awt.*;
29  import java.awt.image.*;
30  
31  /** Color filter for DebugGraphics, used for images only.
32    *
33    * @author Dave Karlton
34    */
35  class DebugGraphicsFilter extends RGBImageFilter {
36      Color color;
37  
38      DebugGraphicsFilter(Color c) {
39          canFilterIndexColorModel = true;
40          color = c;
41      }
42  
43      public int filterRGB(int x, int y, int rgb) {
44          return color.getRGB() | (rgb & 0xFF000000);
45      }
46  }
