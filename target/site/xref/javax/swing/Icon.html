25  package javax.swing;
26  
27  import java.awt.Graphics;
28  import java.awt.Component;
29  
30  
31  /**
32   * A small fixed size picture, typically used to decorate components.
33   *
34   * @see ImageIcon
35   */
36  
37  public interface Icon
38  {
39      /**
40       * Draw the icon at the specified location.  Icon implementations
41       * may use the Component argument to get properties useful for
42       * painting, e.g. the foreground or background color.
43       */
44      void paintIcon(Component c, Graphics g, int x, int y);
45  
46      /**
47       * Returns the icon's width.
48       *
49       * @return an int specifying the fixed width of the icon.
50       */
51      int getIconWidth();
52  
53      /**
54       * Returns the icon's height.
55       *
56       * @return an int specifying the fixed height of the icon.
57       */
58      int getIconHeight();
59  }
