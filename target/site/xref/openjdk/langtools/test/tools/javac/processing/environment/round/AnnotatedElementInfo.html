24  import java.lang.annotation.*;
25  import static java.lang.annotation.RetentionPolicy.*;
26  
27  /**
28   * Hold information about the set of elements expected to be annotated
29   * with a given annotation.
30   */
31  @Retention(RUNTIME)
32  public @interface AnnotatedElementInfo {
33      String annotationName();
34      int expectedSize();
35      String[] names();
36  }
