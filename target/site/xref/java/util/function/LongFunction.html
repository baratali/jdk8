25  package java.util.function;
26  
27  /**
28   * Represents a function that accepts a long-valued argument and produces a
29   * result.  This is the {@code long}-consuming primitive specialization for
30   * {@link Function}.
31   *
32   * <p>This is a <a href="package-summary.html">functional interface</a>
33   * whose functional method is {@link #apply(long)}.
34   *
35   * @param <R> the type of the result of the function
36   *
37   * @see Function
38   * @since 1.8
39   */
40  @FunctionalInterface
41  public interface LongFunction<R> {
42  
43      /**
44       * Applies this function to the given argument.
45       *
46       * @param value the function argument
47       * @return the function result
48       */
49      R apply(long value);
50  }
