49  /**
50   * An optionally-bounded {@linkplain BlockingQueue blocking queue} based on
51   * linked nodes.
52   * This queue orders elements FIFO (first-in-first-out).
53   * The <em>head</em> of the queue is that element that has been on the
54   * queue the longest time.
55   * The <em>tail</em> of the queue is that element that has been on the
56   * queue the shortest time. New elements
57   * are inserted at the tail of the queue, and the queue retrieval
58   * operations obtain elements at the head of the queue.
59   * Linked queues typically have higher throughput than array-based queues but
60   * less predictable performance in most concurrent applications.
61   *
62   * <p>The optional capacity bound constructor argument serves as a
63   * way to prevent excessive queue expansion. The capacity, if unspecified,
64   * is equal to {@link Integer#MAX_VALUE}.  Linked nodes are
65   * dynamically created upon each insertion unless this would bring the
66   * queue above capacity.
67   *
68   * <p>This class and its iterator implement all of the
69   * <em>optional</em> methods of the {@link Collection} and {@link
70   * Iterator} interfaces.
71   *
72   * <p>This class is a member of the
73   * <a href="{@docRoot}/../technotes/guides/collections/index.html">
74   * Java Collections Framework</a>.
75   *
76   * @since 1.5
77   * @author Doug Lea
78   * @param <E> the type of elements held in this collection
79   */
80  public class LinkedBlockingQueue<E> extends AbstractQueue<E>
81          implements BlockingQueue<E>, java.io.Serializable {
82      private static final long serialVersionUID = -6903933977591709194L;

84      /*
85       * A variant of the "two lock queue" algorithm.  The putLock gates
86       * entry to put (and offer), and has an associated condition for
87       * waiting puts.  Similarly for the takeLock.  The "count" field
88       * that they both rely on is maintained as an atomic to avoid
89       * needing to get both locks in most cases. Also, to minimize need
90       * for puts to get takeLock and vice-versa, cascading notifies are
91       * used. When a put notices that it has enabled at least one take,
92       * it signals taker. That taker in turn signals others if more
93       * items have been entered since the signal. And symmetrically for
94       * takes signalling puts. Operations such as remove(Object) and
95       * iterators acquire both locks.
96       *
97       * Visibility between writers and readers is provided as follows:
98       *
99       * Whenever an element is enqueued, the putLock is acquired and
100      * count updated.  A subsequent reader guarantees visibility to the
101      * enqueued Node by either acquiring the putLock (via fullyLock)
102      * or by acquiring the takeLock, and then reading n = count.get();
103      * this gives visibility to the first n items.
104      *
105      * To implement weakly consistent iterators, it appears we need to
106      * keep all Nodes GC-reachable from a predecessor dequeued Node.
107      * That would cause two problems:
108      * - allow a rogue Iterator to cause unbounded memory retention
109      * - cause cross-generational linking of old Nodes to new Nodes if
110      *   a Node was tenured while live, which generational GCs have a
111      *   hard time dealing with, causing repeated major collections.
112      * However, only non-deleted Nodes need to be reachable from
113      * dequeued Nodes, and reachability does not necessarily have to
114      * be of the kind understood by the GC.  We use the trick of
115      * linking a Node that has just been dequeued to itself.  Such a
116      * self-link implicitly means to advance to head.next.
117      */
118

125         /**
126          * One of:
127          * - the real successor Node
128          * - this Node, meaning the successor is head.next
129          * - null, meaning there is no successor (this is the last node)

131         Node<E> next;
132 
133         Node(E x) { item = x; }
134     }

166     /**
167      * Signals a waiting take. Called only from put/offer (which do not
168      * otherwise ordinarily lock takeLock.)
169      */
170     private void signalNotEmpty() {
171         final ReentrantLock takeLock = this.takeLock;
172         takeLock.lock();
173         try {
174             notEmpty.signal();
175         } finally {
176             takeLock.unlock();
177         }
178     }
179 
180     /**
181      * Signals a waiting put. Called only from take/poll.
182      */
183     private void signalNotFull() {
184         final ReentrantLock putLock = this.putLock;
185         putLock.lock();
186         try {
187             notFull.signal();
188         } finally {
189             putLock.unlock();
190         }
191     }

193     /**
194      * Links node at end of queue.
195      *
196      * @param node the node
197      */
198     private void enqueue(Node<E> node) {
199         // assert putLock.isHeldByCurrentThread();
200         // assert last.next == null;
201         last = last.next = node;
202     }

204     /**
205      * Removes a node from head of queue.
206      *
207      * @return the node
208      */
209     private E dequeue() {
210         // assert takeLock.isHeldByCurrentThread();
211         // assert head.item == null;
212         Node<E> h = head;
213         Node<E> first = h.next;
214         h.next = h; // help GC
215         head = first;
216         E x = first.item;
217         first.item = null;
218         return x;
219     }

223      */
224     void fullyLock() {
225         putLock.lock();
226         takeLock.lock();
227     }
228 
229     /**
230      * Unlocks to allow both puts and takes.
231      */
232     void fullyUnlock() {
233         takeLock.unlock();
234         putLock.unlock();
235     }

266     /**
267      * Creates a {@code LinkedBlockingQueue} with a capacity of
268      * {@link Integer#MAX_VALUE}, initially containing the elements of the
269      * given collection,
270      * added in traversal order of the collection's iterator.
271      *
272      * @param c the collection of elements to initially contain
273      * @throws NullPointerException if the specified collection or any
274      *         of its elements are null
275      */
276     public LinkedBlockingQueue(Collection<? extends E> c) {
277         this(Integer.MAX_VALUE);
278         final ReentrantLock putLock = this.putLock;
279         putLock.lock(); // Never contended, but necessary for visibility
280         try {
281             int n = 0;
282             for (E e : c) {
283                 if (e == null)
284                     throw new NullPointerException();
285                 if (n == capacity)
286                     throw new IllegalStateException("Queue full");
287                 enqueue(new Node<E>(e));
288                 ++n;
289             }
290             count.set(n);
291         } finally {
292             putLock.unlock();
293         }
294     }

309     /**
310      * Returns the number of additional elements that this queue can ideally
311      * (in the absence of memory or resource constraints) accept without
312      * blocking. This is always equal to the initial capacity of this queue
313      * less the current {@code size} of this queue.
314      *
315      * <p>Note that you <em>cannot</em> always tell if an attempt to insert
316      * an element will succeed by inspecting {@code remainingCapacity}
317      * because it may be the case that another thread is about to
318      * insert or remove an element.
319      */
320     public int remainingCapacity() {
321         return capacity - count.get();

324     /**
325      * Inserts the specified element at the tail of this queue, waiting if
326      * necessary for space to become available.
327      *
328      * @throws InterruptedException {@inheritDoc}
329      * @throws NullPointerException {@inheritDoc}
330      */
331     public void put(E e) throws InterruptedException {
332         if (e == null) throw new NullPointerException();
333         // Note: convention in all put/take/etc is to preset local var
334         // holding count negative to indicate failure unless set.
335         int c = -1;
336         Node<E> node = new Node<E>(e);
337         final ReentrantLock putLock = this.putLock;
338         final AtomicInteger count = this.count;
339         putLock.lockInterruptibly();
340         try {
341             /*
342              * Note that count is used in wait guard even though it is
343              * not protected by lock. This works because count can
344              * only decrease at this point (all other puts are shut
345              * out by lock), and we (or some other waiting put) are
346              * signalled if it ever changes from capacity. Similarly
347              * for all other uses of count in other wait guards.
348              */
349             while (count.get() == capacity) {
350                 notFull.await();
351             }
352             enqueue(node);
353             c = count.getAndIncrement();
354             if (c + 1 < capacity)
355                 notFull.signal();
356         } finally {
357             putLock.unlock();
358         }
359         if (c == 0)
360             signalNotEmpty();
361     }

363     /**
364      * Inserts the specified element at the tail of this queue, waiting if
365      * necessary up to the specified wait time for space to become available.
366      *
367      * @return {@code true} if successful, or {@code false} if
368      *         the specified waiting time elapses before space is available
369      * @throws InterruptedException {@inheritDoc}
370      * @throws NullPointerException {@inheritDoc}
371      */
372     public boolean offer(E e, long timeout, TimeUnit unit)
373         throws InterruptedException {
374 
375         if (e == null) throw new NullPointerException();
376         long nanos = unit.toNanos(timeout);
377         int c = -1;
378         final ReentrantLock putLock = this.putLock;
379         final AtomicInteger count = this.count;
380         putLock.lockInterruptibly();
381         try {
382             while (count.get() == capacity) {
383                 if (nanos <= 0)
384                     return false;
385                 nanos = notFull.awaitNanos(nanos);
386             }
387             enqueue(new Node<E>(e));
388             c = count.getAndIncrement();
389             if (c + 1 < capacity)
390                 notFull.signal();
391         } finally {
392             putLock.unlock();
393         }
394         if (c == 0)
395             signalNotEmpty();
396         return true;
397     }

399     /**
400      * Inserts the specified element at the tail of this queue if it is
401      * possible to do so immediately without exceeding the queue's capacity,
402      * returning {@code true} upon success and {@code false} if this queue
403      * is full.
404      * When using a capacity-restricted queue, this method is generally
405      * preferable to method {@link BlockingQueue#add add}, which can fail to
406      * insert an element only by throwing an exception.
407      *
408      * @throws NullPointerException if the specified element is null
409      */
410     public boolean offer(E e) {
411         if (e == null) throw new NullPointerException();
412         final AtomicInteger count = this.count;
413         if (count.get() == capacity)
414             return false;
415         int c = -1;
416         Node<E> node = new Node<E>(e);
417         final ReentrantLock putLock = this.putLock;
418         putLock.lock();
419         try {
420             if (count.get() < capacity) {
421                 enqueue(node);
422                 c = count.getAndIncrement();
423                 if (c + 1 < capacity)
424                     notFull.signal();
425             }
426         } finally {
427             putLock.unlock();
428         }
429         if (c == 0)
430             signalNotEmpty();
431         return c >= 0;
432     }

434     public E take() throws InterruptedException {
435         E x;
436         int c = -1;
437         final AtomicInteger count = this.count;
438         final ReentrantLock takeLock = this.takeLock;
439         takeLock.lockInterruptibly();
440         try {
441             while (count.get() == 0) {
442                 notEmpty.await();
443             }
444             x = dequeue();
445             c = count.getAndDecrement();
446             if (c > 1)
447                 notEmpty.signal();
448         } finally {
449             takeLock.unlock();
450         }
451         if (c == capacity)
452             signalNotFull();
453         return x;
454     }

456     public E poll(long timeout, TimeUnit unit) throws InterruptedException {
457         E x = null;
458         int c = -1;
459         long nanos = unit.toNanos(timeout);
460         final AtomicInteger count = this.count;
461         final ReentrantLock takeLock = this.takeLock;
462         takeLock.lockInterruptibly();
463         try {
464             while (count.get() == 0) {
465                 if (nanos <= 0)
466                     return null;
467                 nanos = notEmpty.awaitNanos(nanos);
468             }
469             x = dequeue();
470             c = count.getAndDecrement();
471             if (c > 1)
472                 notEmpty.signal();
473         } finally {
474             takeLock.unlock();
475         }
476         if (c == capacity)
477             signalNotFull();
478         return x;
479     }

481     public E poll() {
482         final AtomicInteger count = this.count;
483         if (count.get() == 0)
484             return null;
485         E x = null;
486         int c = -1;
487         final ReentrantLock takeLock = this.takeLock;
488         takeLock.lock();
489         try {
490             if (count.get() > 0) {
491                 x = dequeue();
492                 c = count.getAndDecrement();
493                 if (c > 1)
494                     notEmpty.signal();
495             }
496         } finally {
497             takeLock.unlock();
498         }
499         if (c == capacity)
500             signalNotFull();
501         return x;
502     }

504     public E peek() {
505         if (count.get() == 0)
506             return null;
507         final ReentrantLock takeLock = this.takeLock;
508         takeLock.lock();
509         try {
510             Node<E> first = head.next;
511             if (first == null)
512                 return null;
513             else
514                 return first.item;
515         } finally {
516             takeLock.unlock();
517         }
518     }

520     /**
521      * Unlinks interior Node p with predecessor trail.
522      */
523     void unlink(Node<E> p, Node<E> trail) {
524         // assert isFullyLocked();
525         // p.next is not changed, to allow iterators that are
526         // traversing p to maintain their weak-consistency guarantee.
527         p.item = null;
528         trail.next = p.next;
529         if (last == p)
530             last = trail;
531         if (count.getAndDecrement() == capacity)
532             notFull.signal();
533     }

535     /**
536      * Removes a single instance of the specified element from this queue,
537      * if it is present.  More formally, removes an element {@code e} such
538      * that {@code o.equals(e)}, if this queue contains one or more such
539      * elements.
540      * Returns {@code true} if this queue contained the specified element
541      * (or equivalently, if this queue changed as a result of the call).
542      *
543      * @param o element to be removed from this queue, if present
544      * @return {@code true} if this queue changed as a result of the call
545      */
546     public boolean remove(Object o) {
547         if (o == null) return false;
548         fullyLock();
549         try {
550             for (Node<E> trail = head, p = trail.next;
551                  p != null;
552                  trail = p, p = p.next) {
553                 if (o.equals(p.item)) {
554                     unlink(p, trail);
555                     return true;
556                 }
557             }
558             return false;
559         } finally {
560             fullyUnlock();
561         }
562     }

564     /**
565      * Returns {@code true} if this queue contains the specified element.
566      * More formally, returns {@code true} if and only if this queue contains
567      * at least one element {@code e} such that {@code o.equals(e)}.
568      *
569      * @param o object to be checked for containment in this queue
570      * @return {@code true} if this queue contains the specified element
571      */
572     public boolean contains(Object o) {
573         if (o == null) return false;
574         fullyLock();
575         try {
576             for (Node<E> p = head.next; p != null; p = p.next)
577                 if (o.equals(p.item))
578                     return true;
579             return false;
580         } finally {
581             fullyUnlock();
582         }
583     }

585     /**
586      * Returns an array containing all of the elements in this queue, in
587      * proper sequence.
588      *
589      * <p>The returned array will be "safe" in that no references to it are
590      * maintained by this queue.  (In other words, this method must allocate
591      * a new array).  The caller is thus free to modify the returned array.
592      *
593      * <p>This method acts as bridge between array-based and collection-based
594      * APIs.
595      *
596      * @return an array containing all of the elements in this queue
597      */
598     public Object[] toArray() {
599         fullyLock();
600         try {
601             int size = count.get();
602             Object[] a = new Object[size];
603             int k = 0;
604             for (Node<E> p = head.next; p != null; p = p.next)
605                 a[k++] = p.item;
606             return a;
607         } finally {
608             fullyUnlock();
609         }
610     }

612     /**
613      * Returns an array containing all of the elements in this queue, in
614      * proper sequence; the runtime type of the returned array is that of
615      * the specified array.  If the queue fits in the specified array, it
616      * is returned therein.  Otherwise, a new array is allocated with the
617      * runtime type of the specified array and the size of this queue.
618      *
619      * <p>If this queue fits in the specified array with room to spare
620      * (i.e., the array has more elements than this queue), the element in
621      * the array immediately following the end of the queue is set to
622      * {@code null}.
623      *
624      * <p>Like the {@link #toArray()} method, this method acts as bridge between
625      * array-based and collection-based APIs.  Further, this method allows
626      * precise control over the runtime type of the output array, and may,
627      * under certain circumstances, be used to save allocation costs.
628      *
629      * <p>Suppose {@code x} is a queue known to contain only strings.
630      * The following code can be used to dump the queue into a newly
631      * allocated array of {@code String}:
632      *
633      *  <pre> {@code String[] y = x.toArray(new String[0]);}</pre>
634      *
635      * Note that {@code toArray(new Object[0])} is identical in function to
636      * {@code toArray()}.
637      *
638      * @param a the array into which the elements of the queue are to
639      *          be stored, if it is big enough; otherwise, a new array of the
640      *          same runtime type is allocated for this purpose
641      * @return an array containing all of the elements in this queue
642      * @throws ArrayStoreException if the runtime type of the specified array
643      *         is not a supertype of the runtime type of every element in
644      *         this queue
645      * @throws NullPointerException if the specified array is null
646      */
647     @SuppressWarnings("unchecked")
648     public <T> T[] toArray(T[] a) {
649         fullyLock();
650         try {
651             int size = count.get();
652             if (a.length < size)
653                 a = (T[])java.lang.reflect.Array.newInstance
654                     (a.getClass().getComponentType(), size);
655 
656             int k = 0;
657             for (Node<E> p = head.next; p != null; p = p.next)
658                 a[k++] = (T)p.item;
659             if (a.length > k)
660                 a[k] = null;
661             return a;
662         } finally {
663             fullyUnlock();
664         }
665     }

667     public String toString() {
668         fullyLock();
669         try {
670             Node<E> p = head.next;
671             if (p == null)
672                 return "[]";
673 
674             StringBuilder sb = new StringBuilder();
675             sb.append('[');
676             for (;;) {
677                 E e = p.item;
678                 sb.append(e == this ? "(this Collection)" : e);
679                 p = p.next;
680                 if (p == null)
681                     return sb.append(']').toString();
682                 sb.append(',').append(' ');
683             }
684         } finally {
685             fullyUnlock();
686         }
687     }

689     /**
690      * Atomically removes all of the elements from this queue.
691      * The queue will be empty after this call returns.
692      */
693     public void clear() {
694         fullyLock();
695         try {
696             for (Node<E> p, h = head; (p = h.next) != null; h = p) {
697                 h.next = h;
698                 p.item = null;
699             }
700             head = last;
701             // assert head.item == null && head.next == null;
702             if (count.getAndSet(0) == capacity)
703                 notFull.signal();
704         } finally {
705             fullyUnlock();
706         }
707     }

719     /**
720      * @throws UnsupportedOperationException {@inheritDoc}
721      * @throws ClassCastException            {@inheritDoc}
722      * @throws NullPointerException          {@inheritDoc}
723      * @throws IllegalArgumentException      {@inheritDoc}
724      */
725     public int drainTo(Collection<? super E> c, int maxElements) {
726         if (c == null)
727             throw new NullPointerException();
728         if (c == this)
729             throw new IllegalArgumentException();
730         if (maxElements <= 0)
731             return 0;
732         boolean signalNotFull = false;
733         final ReentrantLock takeLock = this.takeLock;
734         takeLock.lock();
735         try {
736             int n = Math.min(maxElements, count.get());
737             // count.get provides visibility to first n Nodes
738             Node<E> h = head;
739             int i = 0;
740             try {
741                 while (i < n) {
742                     Node<E> p = h.next;
743                     c.add(p.item);
744                     p.item = null;
745                     h.next = h;
746                     h = p;
747                     ++i;
748                 }
749                 return n;
750             } finally {
751                 // Restore invariants even if c.add() threw
752                 if (i > 0) {
753                     // assert h.item == null;
754                     head = h;
755                     signalNotFull = (count.getAndAdd(-i) == capacity);
756                 }
757             }
758         } finally {
759             takeLock.unlock();
760             if (signalNotFull)
761                 signalNotFull();
762         }
763     }

778     private class Itr implements Iterator<E> {
779         /*
780          * Basic weakly-consistent iterator.  At all times hold the next
781          * item to hand out so that if hasNext() reports true, we will
782          * still have it to return even if lost race with a take etc.
783          */
784 
785         private Node<E> current;
786         private Node<E> lastRet;
787         private E currentElement;
788 
789         Itr() {
790             fullyLock();
791             try {
792                 current = head.next;
793                 if (current != null)
794                     currentElement = current.item;
795             } finally {
796                 fullyUnlock();
797             }
798         }
799 
800         public boolean hasNext() {
801             return current != null;
802         }
803 
804         /**
805          * Returns the next live successor of p, or null if no such.
806          *
807          * Unlike other traversal methods, iterators need to handle both:
808          * - dequeued nodes (p.next == p)
809          * - (possibly multiple) interior removed nodes (p.item == null)
810          */
811         private Node<E> nextNode(Node<E> p) {
812             for (;;) {
813                 Node<E> s = p.next;
814                 if (s == p)
815                     return head.next;
816                 if (s == null || s.item != null)
817                     return s;
818                 p = s;
819             }
820         }
821 
822         public E next() {
823             fullyLock();
824             try {
825                 if (current == null)
826                     throw new NoSuchElementException();
827                 E x = currentElement;
828                 lastRet = current;
829                 current = nextNode(current);
830                 currentElement = (current == null) ? null : current.item;
831                 return x;
832             } finally {
833                 fullyUnlock();
834             }
835         }
836 
837         public void remove() {
838             if (lastRet == null)
839                 throw new IllegalStateException();
840             fullyLock();
841             try {
842                 Node<E> node = lastRet;
843                 lastRet = null;
844                 for (Node<E> trail = head, p = trail.next;
845                      p != null;
846                      trail = p, p = p.next) {
847                     if (p == node) {
848                         unlink(p, trail);
849                         break;
850                     }
851                 }
852             } finally {
853                 fullyUnlock();
854             }
855         }
856     }

858     /** A customized variant of Spliterators.IteratorSpliterator */
859     static final class LBQSpliterator<E> implements Spliterator<E> {
860         static final int MAX_BATCH = 1 << 25;  // max batch array size;
861         final LinkedBlockingQueue<E> queue;
862         Node<E> current;    // current node; null until initialized
863         int batch;          // batch size for splits
864         boolean exhausted;  // true when no more nodes
865         long est;           // size estimate
866         LBQSpliterator(LinkedBlockingQueue<E> queue) {
867             this.queue = queue;
868             this.est = queue.size();
869         }
870 
871         public long estimateSize() { return est; }
872 
873         public Spliterator<E> trySplit() {
874             Node<E> h;
875             final LinkedBlockingQueue<E> q = this.queue;
876             int b = batch;
877             int n = (b <= 0) ? 1 : (b >= MAX_BATCH) ? MAX_BATCH : b + 1;
878             if (!exhausted &&
879                 ((h = current) != null || (h = q.head.next) != null) &&
880                 h.next != null) {
881                 Object[] a = new Object[n];
882                 int i = 0;
883                 Node<E> p = current;
884                 q.fullyLock();
885                 try {
886                     if (p != null || (p = q.head.next) != null) {
887                         do {
888                             if ((a[i] = p.item) != null)
889                                 ++i;
890                         } while ((p = p.next) != null && i < n);
891                     }
892                 } finally {
893                     q.fullyUnlock();
894                 }
895                 if ((current = p) == null) {
896                     est = 0L;
897                     exhausted = true;
898                 }
899                 else if ((est -= i) < 0L)
900                     est = 0L;
901                 if (i > 0) {
902                     batch = i;
903                     return Spliterators.spliterator
904                         (a, 0, i, Spliterator.ORDERED | Spliterator.NONNULL |
905                          Spliterator.CONCURRENT);
906                 }
907             }
908             return null;
909         }
910 
911         public void forEachRemaining(Consumer<? super E> action) {
912             if (action == null) throw new NullPointerException();
913             final LinkedBlockingQueue<E> q = this.queue;
914             if (!exhausted) {
915                 exhausted = true;
916                 Node<E> p = current;
917                 do {
918                     E e = null;
919                     q.fullyLock();
920                     try {
921                         if (p == null)
922                             p = q.head.next;
923                         while (p != null) {
924                             e = p.item;
925                             p = p.next;
926                             if (e != null)
927                                 break;
928                         }
929                     } finally {
930                         q.fullyUnlock();
931                     }
932                     if (e != null)
933                         action.accept(e);
934                 } while (p != null);
935             }
936         }
937 
938         public boolean tryAdvance(Consumer<? super E> action) {
939             if (action == null) throw new NullPointerException();
940             final LinkedBlockingQueue<E> q = this.queue;
941             if (!exhausted) {
942                 E e = null;
943                 q.fullyLock();
944                 try {
945                     if (current == null)
946                         current = q.head.next;
947                     while (current != null) {
948                         e = current.item;
949                         current = current.next;
950                         if (e != null)
951                             break;
952                     }
953                 } finally {
954                     q.fullyUnlock();
955                 }
956                 if (current == null)
957                     exhausted = true;
958                 if (e != null) {
959                     action.accept(e);
960                     return true;
961                 }
962             }
963             return false;
964         }
965 
966         public int characteristics() {
967             return Spliterator.ORDERED | Spliterator.NONNULL |
968                 Spliterator.CONCURRENT;
969         }
970     }

993      * Saves this queue to a stream (that is, serializes it).
994      *
995      * @param s the stream
996      * @throws java.io.IOException if an I/O error occurs
997      * @serialData The capacity is emitted (int), followed by all of
998      * its elements (each an {@code Object}) in the proper order,
999      * followed by a null
1000      */
1001     private void writeObject(java.io.ObjectOutputStream s)
1002         throws java.io.IOException {
1003 
1004         fullyLock();
1005         try {
1006             // Write out any hidden stuff, plus capacity
1007             s.defaultWriteObject();
1008 
1009             // Write out all elements in the proper order.
1010             for (Node<E> p = head.next; p != null; p = p.next)
1011                 s.writeObject(p.item);
1012 
1013             // Use trailing null as sentinel
1014             s.writeObject(null);
1015         } finally {
1016             fullyUnlock();
1017         }
1018     }

1020     /**
1021      * Reconstitutes this queue from a stream (that is, deserializes it).
1022      * @param s the stream
1023      * @throws ClassNotFoundException if the class of a serialized object
1024      *         could not be found
1025      * @throws java.io.IOException if an I/O error occurs
1026      */
1027     private void readObject(java.io.ObjectInputStream s)
1028         throws java.io.IOException, ClassNotFoundException {
1029         // Read in capacity, and any hidden stuff
1030         s.defaultReadObject();
1031 
1032         count.set(0);
1033         last = head = new Node<E>(null);
1034 
1035         // Read in all elements and place in queue
1036         for (;;) {
1037             @SuppressWarnings("unchecked")
1038             E item = (E)s.readObject();
1039             if (item == null)
1040                 break;
1041             add(item);
