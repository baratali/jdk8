96      public JarVerifier(byte rawBytes[]) {
97          manifestRawBytes = rawBytes;
98          sigFileSigners = new Hashtable<>();
99          verifiedSigners = new Hashtable<>();

106     /**
107      * This method scans to see which entry we're parsing and
108      * keeps various state information depending on what type of
109      * file is being parsed.
110      */
111     public void beginEntry(JarEntry je, ManifestEntryVerifier mev)
112         throws IOException
113     {
114         if (je == null)
115             return;
116 
117         if (debug != null) {
118             debug.println("beginEntry "+je.getName());
119         }
120 
121         String name = je.getName();
122 
123         /*
124          * Assumptions:
125          * 1. The manifest should be the first entry in the META-INF directory.
126          * 2. The .SF/.DSA/.EC files follow the manifest, before any normal entries
127          * 3. Any of the following will throw a SecurityException:
128          *    a. digest mismatch between a manifest section and
129          *       the SF section.
130          *    b. digest mismatch between the actual jar entry and the manifest
131          */
132 
133         if (parsingMeta) {
134             String uname = name.toUpperCase(Locale.ENGLISH);
135             if ((uname.startsWith("META-INF/") ||
136                  uname.startsWith("/META-INF/"))) {
137 
138                 if (je.isDirectory()) {
139                     mev.setEntry(null, je);
140                     return;
141                 }
142 
143                 if (uname.equals(JarFile.MANIFEST_NAME) ||
144                         uname.equals(JarIndex.INDEX_NAME)) {
145                     return;
146                 }
147 
148                 if (SignatureFileVerifier.isBlockOrSF(uname)) {
149                     /* We parse only DSA, RSA or EC PKCS7 blocks. */
150                     parsingBlockOrSF = true;
151                     baos.reset();
152                     mev.setEntry(null, je);
153                     return;
154                 }
155 
156                 // If a META-INF entry is not MF or block or SF, they should
157                 // be normal entries. According to 2 above, no more block or
158                 // SF will appear. Let's doneWithMeta.
159             }
160         }
161 
162         if (parsingMeta) {
163             doneWithMeta();
164         }
165 
166         if (je.isDirectory()) {
167             mev.setEntry(null, je);
168             return;
169         }
170 
171         // be liberal in what you accept. If the name starts with ./, remove
172         // it as we internally canonicalize it with out the ./.
173         if (name.startsWith("./"))
174             name = name.substring(2);
175 
176         // be liberal in what you accept. If the name starts with /, remove
177         // it as we internally canonicalize it with out the /.
178         if (name.startsWith("/"))
179             name = name.substring(1);
180 
181         // only set the jev object for entries that have a signature
182         // (either verified or not)
183         if (sigFileSigners.get(name) != null ||
184                 verifiedSigners.get(name) != null) {
185             mev.setEntry(name, je);
186             return;
187         }
188 
189         // don't compute the digest for this entry
190         mev.setEntry(null, je);
191 
192         return;
193     }

195     /**
196      * update a single byte.
197      */
198 
199     public void update(int b, ManifestEntryVerifier mev)
200         throws IOException
201     {
202         if (b != -1) {
203             if (parsingBlockOrSF) {
204                 baos.write(b);
205             } else {
206                 mev.update((byte)b);
207             }
208         } else {
209             processEntry(mev);
210         }
211     }

213     /**
214      * update an array of bytes.
215      */
216 
217     public void update(int n, byte[] b, int off, int len,
218                        ManifestEntryVerifier mev)
219         throws IOException
220     {
221         if (n != -1) {
222             if (parsingBlockOrSF) {
223                 baos.write(b, off, n);
224             } else {
225                 mev.update(b, off, n);
226             }
227         } else {
228             processEntry(mev);
229         }
230     }

232     /**
233      * called when we reach the end of entry in one of the read() methods.
234      */
235     private void processEntry(ManifestEntryVerifier mev)
236         throws IOException
237     {
238         if (!parsingBlockOrSF) {
239             JarEntry je = mev.getEntry();
240             if ((je != null) && (je.signers == null)) {
241                 je.signers = mev.verify(verifiedSigners, sigFileSigners);
242                 je.certs = mapSignersToCertArray(je.signers);
243             }
244         } else {
245 
246             try {
247                 parsingBlockOrSF = false;
248 
249                 if (debug != null) {
250                     debug.println("processEntry: processing block");
251                 }
252 
253                 String uname = mev.getEntry().getName()
254                                              .toUpperCase(Locale.ENGLISH);
255 
256                 if (uname.endsWith(".SF")) {
257                     String key = uname.substring(0, uname.length()-3);
258                     byte bytes[] = baos.toByteArray();
259                     // add to sigFileData in case future blocks need it
260                     sigFileData.put(key, bytes);
261                     // check pending blocks, we can now process
262                     // anyone waiting for this .SF file
263                     Iterator<SignatureFileVerifier> it = pendingBlocks.iterator();
264                     while (it.hasNext()) {
265                         SignatureFileVerifier sfv = it.next();
266                         if (sfv.needSignatureFile(key)) {
267                             if (debug != null) {
268                                 debug.println(
269                                  "processEntry: processing pending block");
270                             }
271 
272                             sfv.setSignatureFile(bytes);
273                             sfv.process(sigFileSigners, manifestDigests);
274                         }
275                     }
276                     return;
277                 }
278 
279                 // now we are parsing a signature block file
280 
281                 String key = uname.substring(0, uname.lastIndexOf("."));
282 
283                 if (signerCache == null)
284                     signerCache = new ArrayList<>();
285 
286                 if (manDig == null) {
287                     synchronized(manifestRawBytes) {
288                         if (manDig == null) {
289                             manDig = new ManifestDigester(manifestRawBytes);
290                             manifestRawBytes = null;
291                         }
292                     }
293                 }
294 
295                 SignatureFileVerifier sfv =
296                   new SignatureFileVerifier(signerCache,
297                                             manDig, uname, baos.toByteArray());
298 
299                 if (sfv.needSignatureFileBytes()) {
300                     // see if we have already parsed an external .SF file
301                     byte[] bytes = sigFileData.get(key);
302 
303                     if (bytes == null) {
304                         // put this block on queue for later processing
305                         // since we don't have the .SF bytes yet
306                         // (uname, block);
307                         if (debug != null) {
308                             debug.println("adding pending block");
309                         }
310                         pendingBlocks.add(sfv);
311                         return;
312                     } else {
313                         sfv.setSignatureFile(bytes);
314                     }
315                 }
316                 sfv.process(sigFileSigners, manifestDigests);
317 
318             } catch (IOException ioe) {
319                 // e.g. sun.security.pkcs.ParsingException
320                 if (debug != null) debug.println("processEntry caught: "+ioe);
321                 // ignore and treat as unsigned
322             } catch (SignatureException se) {
323                 if (debug != null) debug.println("processEntry caught: "+se);
324                 // ignore and treat as unsigned
325             } catch (NoSuchAlgorithmException nsae) {
326                 if (debug != null) debug.println("processEntry caught: "+nsae);
327                 // ignore and treat as unsigned
328             } catch (CertificateException ce) {
329                 if (debug != null) debug.println("processEntry caught: "+ce);
330                 // ignore and treat as unsigned
331             }
332         }
333     }

361     public CodeSigner[] getCodeSigners(JarFile jar, JarEntry entry)
362     {
363         String name = entry.getName();
364         if (eagerValidation && sigFileSigners.get(name) != null) {
365             /*
366              * Force a read of the entry data to generate the
367              * verification hash.
368              */
369             try {
370                 InputStream s = jar.getInputStream(entry);
371                 byte[] buffer = new byte[1024];
372                 int n = buffer.length;
373                 while (n != -1) {
374                     n = s.read(buffer, 0, buffer.length);
375                 }
376                 s.close();
377             } catch (IOException e) {
378             }
379         }
380         return getCodeSigners(name);
381     }

383     /*
384      * Convert an array of signers into an array of concatenated certificate
385      * arrays.
386      */
387     private static java.security.cert.Certificate[] mapSignersToCertArray(
388         CodeSigner[] signers) {
389 
390         if (signers != null) {
391             ArrayList<java.security.cert.Certificate> certChains = new ArrayList<>();
392             for (int i = 0; i < signers.length; i++) {
393                 certChains.addAll(
394                     signers[i].getSignerCertPath().getCertificates());
395             }
396 
397             // Convert into a Certificate[]
398             return certChains.toArray(
399                     new java.security.cert.Certificate[certChains.size()]);
400         }
401         return null;
402     }

404     /**
405      * returns true if there no files to verify.
406      * should only be called after all the META-INF entries
407      * have been processed.
408      */
409     boolean nothingToVerify()
410     {
411         return (anyToVerify == false);
412     }
413 
414     /**
415      * called to let us know we have processed all the
416      * META-INF entries, and if we re-read one of them, don't
417      * re-process it. Also gets rid of any data structures
418      * we needed when parsing META-INF entries.
419      */
420     void doneWithMeta()
421     {
422         parsingMeta = false;
423         anyToVerify = !sigFileSigners.isEmpty();
424         baos = null;
425         sigFileData = null;
426         pendingBlocks = null;
427         signerCache = null;
428         manDig = null;
429         // MANIFEST.MF is always treated as signed and verified,
430         // move its signers from sigFileSigners to verifiedSigners.
431         if (sigFileSigners.containsKey(JarFile.MANIFEST_NAME)) {
432             CodeSigner[] codeSigners = sigFileSigners.remove(JarFile.MANIFEST_NAME);
433             verifiedSigners.put(JarFile.MANIFEST_NAME, codeSigners);
434         }
435     }

437     static class VerifierStream extends java.io.InputStream {
438 
439         private InputStream is;
440         private JarVerifier jv;
441         private ManifestEntryVerifier mev;
442         private long numLeft;
443 
444         VerifierStream(Manifest man,
445                        JarEntry je,
446                        InputStream is,
447                        JarVerifier jv) throws IOException
448         {
449             this.is = is;
450             this.jv = jv;
451             this.mev = new ManifestEntryVerifier(man);
452             this.jv.beginEntry(je, mev);
453             this.numLeft = je.getSize();
454             if (this.numLeft == 0)
455                 this.jv.update(-1, this.mev);
456         }
457 
458         public int read() throws IOException
459         {
460             if (numLeft > 0) {
461                 int b = is.read();
462                 jv.update(b, mev);
463                 numLeft--;
464                 if (numLeft == 0)
465                     jv.update(-1, mev);
466                 return b;
467             } else {
468                 return -1;
469             }
470         }
471 
472         public int read(byte b[], int off, int len) throws IOException {
473             if ((numLeft > 0) && (numLeft < len)) {
474                 len = (int)numLeft;
475             }
476 
477             if (numLeft > 0) {
478                 int n = is.read(b, off, len);
479                 jv.update(n, b, off, len, mev);
480                 numLeft -= n;
481                 if (numLeft == 0)
482                     jv.update(-1, b, off, len, mev);
483                 return n;
484             } else {
485                 return -1;
486             }
487         }
488 
489         public void close()
490             throws IOException
491         {
492             if (is != null)
493                 is.close();
494             is = null;
495             mev = null;
496             jv = null;
497         }
498 
499         public int available() throws IOException {
500             return is.available();
501         }

507     private Map<URL, Map<CodeSigner[], CodeSource>> urlToCodeSourceMap = new HashMap<>();
508     private Map<CodeSigner[], CodeSource> signerToCodeSource = new HashMap<>();
509     private URL lastURL;
510     private Map<CodeSigner[], CodeSource> lastURLMap;
511 
512     /*
513      * Create a unique mapping from codeSigner cache entries to CodeSource.
514      * In theory, multiple URLs origins could map to a single locally cached
515      * and shared JAR file although in practice there will be a single URL in use.
516      */
517     private synchronized CodeSource mapSignersToCodeSource(URL url, CodeSigner[] signers) {
518         Map<CodeSigner[], CodeSource> map;
519         if (url == lastURL) {
520             map = lastURLMap;
521         } else {
522             map = urlToCodeSourceMap.get(url);
523             if (map == null) {
524                 map = new HashMap<>();
525                 urlToCodeSourceMap.put(url, map);
526             }
527             lastURLMap = map;
528             lastURL = url;
529         }
530         CodeSource cs = map.get(signers);
531         if (cs == null) {
532             cs = new VerifierCodeSource(csdomain, url, signers);
533             signerToCodeSource.put(signers, cs);
534         }
535         return cs;
536     }
537 
538     private CodeSource[] mapSignersToCodeSources(URL url, List<CodeSigner[]> signers, boolean unsigned) {
539         List<CodeSource> sources = new ArrayList<>();
540 
541         for (int i = 0; i < signers.size(); i++) {
542             sources.add(mapSignersToCodeSource(url, signers.get(i)));
543         }
544         if (unsigned) {
545             sources.add(mapSignersToCodeSource(url, null));
546         }
547         return sources.toArray(new CodeSource[sources.size()]);
548     }
549     private CodeSigner[] emptySigner = new CodeSigner[0];
550 
551     /*
552      * Match CodeSource to a CodeSigner[] in the signer cache.
553      */
554     private CodeSigner[] findMatchingSigners(CodeSource cs) {
555         if (cs instanceof VerifierCodeSource) {
556             VerifierCodeSource vcs = (VerifierCodeSource) cs;
557             if (vcs.isSameDomain(csdomain)) {
558                 return ((VerifierCodeSource) cs).getPrivateSigners();
559             }
560         }
561 
562         /*
563          * In practice signers should always be optimized above
564          * but this handles a CodeSource of any type, just in case.
565          */
566         CodeSource[] sources = mapSignersToCodeSources(cs.getLocation(), getJarCodeSigners(), true);
567         List<CodeSource> sourceList = new ArrayList<>();
568         for (int i = 0; i < sources.length; i++) {
569             sourceList.add(sources[i]);
570         }
571         int j = sourceList.indexOf(cs);
572         if (j != -1) {
573             CodeSigner[] match;
574             match = ((VerifierCodeSource) sourceList.get(j)).getPrivateSigners();
575             if (match == null) {
576                 match = emptySigner;
577             }
578             return match;
579         }
580         return null;
581     }

583     /*
584      * Instances of this class hold uncopied references to internal
585      * signing data that can be compared by object reference identity.
586      */
587     private static class VerifierCodeSource extends CodeSource {
588         private static final long serialVersionUID = -9047366145967768825L;
589 
590         URL vlocation;
591         CodeSigner[] vsigners;
592         java.security.cert.Certificate[] vcerts;
593         Object csdomain;
594 
595         VerifierCodeSource(Object csdomain, URL location, CodeSigner[] signers) {
596             super(location, signers);
597             this.csdomain = csdomain;
598             vlocation = location;
599             vsigners = signers; // from signerCache
600         }
601 
602         VerifierCodeSource(Object csdomain, URL location, java.security.cert.Certificate[] certs) {
603             super(location, certs);
604             this.csdomain = csdomain;
605             vlocation = location;
606             vcerts = certs; // from signerCache
607         }
608 
609         /*
610          * All VerifierCodeSource instances are constructed based on
611          * singleton signerCache or signerCacheCert entries for each unique signer.
612          * No CodeSigner<->Certificate[] conversion is required.
613          * We use these assumptions to optimize equality comparisons.
614          */
615         public boolean equals(Object obj) {
616             if (obj == this) {
617                 return true;
618             }
619             if (obj instanceof VerifierCodeSource) {
620                 VerifierCodeSource that = (VerifierCodeSource) obj;
621 
622                 /*
623                  * Only compare against other per-signer singletons constructed
624                  * on behalf of the same JarFile instance. Otherwise, compare
625                  * things the slower way.
626                  */
627                 if (isSameDomain(that.csdomain)) {
628                     if (that.vsigners != this.vsigners
629                             || that.vcerts != this.vcerts) {
630                         return false;
631                     }
632                     if (that.vlocation != null) {
633                         return that.vlocation.equals(this.vlocation);
634                     } else if (this.vlocation != null) {
635                         return this.vlocation.equals(that.vlocation);
636                     } else { // both null
637                         return true;
638                     }
639                 }
640             }
641             return super.equals(obj);
642         }
643 
644         boolean isSameDomain(Object csdomain) {
645             return this.csdomain == csdomain;
646         }

656     private Map<String, CodeSigner[]> signerMap;

658     private synchronized Map<String, CodeSigner[]> signerMap() {
659         if (signerMap == null) {
660             /*
661              * Snapshot signer state so it doesn't change on us. We care
662              * only about the asserted signatures. Verification of
663              * signature validity happens via the JarEntry apis.
664              */
665             signerMap = new HashMap<>(verifiedSigners.size() + sigFileSigners.size());
666             signerMap.putAll(verifiedSigners);
667             signerMap.putAll(sigFileSigners);
668         }
669         return signerMap;
670     }

672     public synchronized Enumeration<String> entryNames(JarFile jar, final CodeSource[] cs) {
673         final Map<String, CodeSigner[]> map = signerMap();
674         final Iterator<Map.Entry<String, CodeSigner[]>> itor = map.entrySet().iterator();
675         boolean matchUnsigned = false;
676 
677         /*
678          * Grab a single copy of the CodeSigner arrays. Check
679          * to see if we can optimize CodeSigner equality test.
680          */
681         List<CodeSigner[]> req = new ArrayList<>(cs.length);
682         for (int i = 0; i < cs.length; i++) {
683             CodeSigner[] match = findMatchingSigners(cs[i]);
684             if (match != null) {
685                 if (match.length > 0) {
686                     req.add(match);
687                 } else {
688                     matchUnsigned = true;
689                 }
690             } else {
691                 matchUnsigned = true;
692             }
693         }
694 
695         final List<CodeSigner[]> signersReq = req;
696         final Enumeration<String> enum2 = (matchUnsigned) ? unsignedEntryNames(jar) : emptyEnumeration;
697 
698         return new Enumeration<String>() {
699 
700             String name;
701 
702             public boolean hasMoreElements() {
703                 if (name != null) {
704                     return true;
705                 }
706 
707                 while (itor.hasNext()) {
708                     Map.Entry<String, CodeSigner[]> e = itor.next();
709                     if (signersReq.contains(e.getValue())) {
710                         name = e.getKey();
711                         return true;
712                     }
713                 }
714                 while (enum2.hasMoreElements()) {
715                     name = enum2.nextElement();
716                     return true;
717                 }
718                 return false;
719             }
720 
721             public String nextElement() {
722                 if (hasMoreElements()) {
723                     String value = name;
724                     name = null;
725                     return value;
726                 }
727                 throw new NoSuchElementException();
728             }
729         };
730     }

732     /*
733      * Like entries() but screens out internal JAR mechanism entries
734      * and includes signed entries with no ZIP data.
735      */
736     public Enumeration<JarEntry> entries2(final JarFile jar, Enumeration<? extends ZipEntry> e) {
737         final Map<String, CodeSigner[]> map = new HashMap<>();
738         map.putAll(signerMap());
739         final Enumeration<? extends ZipEntry> enum_ = e;
740         return new Enumeration<JarEntry>() {
741 
742             Enumeration<String> signers = null;
743             JarEntry entry;
744 
745             public boolean hasMoreElements() {
746                 if (entry != null) {
747                     return true;
748                 }
749                 while (enum_.hasMoreElements()) {
750                     ZipEntry ze = enum_.nextElement();
751                     if (JarVerifier.isSigningRelated(ze.getName())) {
752                         continue;
753                     }
754                     entry = jar.newEntry(ze);
755                     return true;
756                 }
757                 if (signers == null) {
758                     signers = Collections.enumeration(map.keySet());
759                 }
760                 while (signers.hasMoreElements()) {
761                     String name = signers.nextElement();
762                     entry = jar.newEntry(new ZipEntry(name));
763                     return true;
764                 }
765 
766                 // Any map entries left?
767                 return false;
768             }
769 
770             public JarEntry nextElement() {
771                 if (hasMoreElements()) {
772                     JarEntry je = entry;
773                     map.remove(je.getName());
774                     entry = null;
775                     return je;
776                 }
777                 throw new NoSuchElementException();
778             }
779         };

797     private Enumeration<String> unsignedEntryNames(JarFile jar) {
798         final Map<String, CodeSigner[]> map = signerMap();
799         final Enumeration<JarEntry> entries = jar.entries();
800         return new Enumeration<String>() {
801 
802             String name;
803 
804             /*
805              * Grab entries from ZIP directory but screen out
806              * metadata.
807              */
808             public boolean hasMoreElements() {
809                 if (name != null) {
810                     return true;
811                 }
812                 while (entries.hasMoreElements()) {
813                     String value;
814                     ZipEntry e = entries.nextElement();
815                     value = e.getName();
816                     if (e.isDirectory() || isSigningRelated(value)) {
817                         continue;
818                     }
819                     if (map.get(value) == null) {
820                         name = value;
821                         return true;
822                     }
823                 }
824                 return false;
825             }
826 
827             public String nextElement() {
828                 if (hasMoreElements()) {
829                     String value = name;
830                     name = null;
831                     return value;
832                 }
833                 throw new NoSuchElementException();
834             }
835         };
836     }
