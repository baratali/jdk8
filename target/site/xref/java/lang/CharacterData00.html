355

357             // There is a titlecase equivalent.  Perform further checks:
358             if ((val & 0x00010000) == 0) {
359                 // The character does not have an uppercase equivalent, so it must

362             }
363             else if ((val & 0x00020000) == 0) {
364                 // The character does not have a lowercase equivalent, so it must
365                 // already be lowercase; so subtract 1 to get the titlecase form.
366                 mapChar = ch - 1;
367             }
368             // else {
369             // The character has both an uppercase equivalent and a lowercase

373         }

376             // uppercase equivalent, so use that (subtract the signed case offset).
377             mapChar = toUpperCase(ch);
378         }

382     int digit(int ch, int radix) {
383         int value = -1;
384         if (radix >= Character.MIN_RADIX && radix <= Character.MAX_RADIX) {
385             int val = getProperties(ch);
386             int kind = val & 0x1F;
387             if (kind == Character.DECIMAL_DIGIT_NUMBER) {
388                 value = ch + ((val & 0x3E0) >> 5) & 0x1F;
389             }
390             else if ((val & 0xC00) == 0x00000C00) {
391                 // Java supradecimal digit
392                 value = (ch + ((val & 0x3E0) >> 5) & 0x1F) + 10;
393             }
394         }
395         return (value < radix) ? value : -1;
