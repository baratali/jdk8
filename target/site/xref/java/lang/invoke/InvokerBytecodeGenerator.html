86      private InvokerBytecodeGenerator(LambdaForm lambdaForm, int localsMapSize,
87                                       String className, String invokerName, MethodType invokerType) {
88          if (invokerName.contains(".")) {
89              int p = invokerName.indexOf(".");
90              className = invokerName.substring(0, p);
91              invokerName = invokerName.substring(p+1);
92          }
93          if (DUMP_CLASS_FILES) {
94              className = makeDumpableClassName(className);
95          }
96          this.className  = superName + "$" + className;
97          this.sourceFile = "LambdaForm$" + className;
98          this.lambdaForm = lambdaForm;
99          this.invokerName = invokerName;
100         this.invokerType = invokerType;
101         this.localsMap = new int[localsMapSize];
102     }

104     private InvokerBytecodeGenerator(String className, String invokerName, MethodType invokerType) {
105         this(null, invokerType.parameterCount(),
106              className, invokerName, invokerType);
107         // Create an array to map name indexes to locals indexes.
108         for (int i = 0; i < localsMap.length; i++) {
109             localsMap[i] = invokerType.parameterSlotCount() - invokerType.parameterSlotDepth(i);
110         }
111     }
112 
113     private InvokerBytecodeGenerator(String className, LambdaForm form, MethodType invokerType) {
114         this(form, form.names.length,
115              className, form.debugName, invokerType);
116         // Create an array to map name indexes to locals indexes.
117         Name[] names = form.names;
118         for (int i = 0, index = 0; i < localsMap.length; i++) {
119             localsMap[i] = index;
120             index += Wrapper.forBasicType(names[i].type).stackSlots();
121         }
122     }

173     private static String makeDumpableClassName(String className) {
174         Integer ctr;
175         synchronized (DUMP_CLASS_FILES_COUNTERS) {
176             ctr = DUMP_CLASS_FILES_COUNTERS.get(className);
177             if (ctr == null)  ctr = 0;
178             DUMP_CLASS_FILES_COUNTERS.put(className, ctr+1);
179         }
180         String sfx = ctr.toString();
181         while (sfx.length() < 3)
182             sfx = "0"+sfx;
183         className += sfx;
184         return className;
185     }

187     class CpPatch {
188         final int index;
189         final String placeholder;
190         final Object value;
191         CpPatch(int index, String placeholder, Object value) {
192             this.index = index;
193             this.placeholder = placeholder;
194             this.value = value;
195         }
196         public String toString() {
197             return "CpPatch/index="+index+",placeholder="+placeholder+",value="+value;
198         }
199     }

201     Map<Object, CpPatch> cpPatches = new HashMap<>();

203     int cph = 0;  // for counting constant placeholders

205     String constantPlaceholder(Object arg) {
206         String cpPlaceholder = "CONSTANT_PLACEHOLDER_" + cph++;
207         if (DUMP_CLASS_FILES) cpPlaceholder += " <<" + arg.toString() + ">>";  // debugging aid
208         if (cpPatches.containsKey(cpPlaceholder)) {
209             throw new InternalError("observed CP placeholder twice: " + cpPlaceholder);
210         }
211         // insert placeholder in CP and remember the patch
212         int index = cw.newConst((Object) cpPlaceholder);  // TODO check if aready in the constant pool
213         cpPatches.put(cpPlaceholder, new CpPatch(index, cpPlaceholder, arg));
214         return cpPlaceholder;
215     }

217     Object[] cpPatches(byte[] classFile) {
218         int size = getConstantPoolSize(classFile);
219         Object[] res = new Object[size];
220         for (CpPatch p : cpPatches.values()) {
221             if (p.index >= size)
222                 throw new InternalError("in cpool["+size+"]: "+p+"\n"+Arrays.toString(Arrays.copyOf(classFile, 20)));
223             res[p.index] = p.value;
224         }
225         return res;
226     }

228     /**
229      * Extract the number of constant pool entries from a given class file.
230      *
231      * @param classFile the bytes of the class file in question.
232      * @return the number of entries in the constant pool.
233      */
234     private static int getConstantPoolSize(byte[] classFile) {
235         // The first few bytes:
236         // u4 magic;
237         // u2 minor_version;
238         // u2 major_version;
239         // u2 constant_pool_count;
240         return ((classFile[8] & 0xFF) << 8) | (classFile[9] & 0xFF);
241     }

260     private static MemberName resolveInvokerMember(Class<?> invokerClass, String name, MethodType type) {
261         MemberName member = new MemberName(invokerClass, name, type, REF_invokeStatic);
262         //System.out.println("resolveInvokerMember => "+member);
263         //for (Method m : invokerClass.getDeclaredMethods())  System.out.println("  "+m);
264         try {
265             member = MEMBERNAME_FACTORY.resolveOrFail(REF_invokeStatic, member, HOST_CLASS, ReflectiveOperationException.class);
266         } catch (ReflectiveOperationException e) {
267             throw newInternalError(e);
268         }
269         //System.out.println("resolveInvokerMember => "+member);
270         return member;
271     }

273     /**
274      * Set up class file generation.
275      */
276     private void classFilePrologue() {
277         final int NOT_ACC_PUBLIC = 0;  // not ACC_PUBLIC
278         cw = new ClassWriter(ClassWriter.COMPUTE_MAXS + ClassWriter.COMPUTE_FRAMES);
279         cw.visit(Opcodes.V1_8, NOT_ACC_PUBLIC + Opcodes.ACC_FINAL + Opcodes.ACC_SUPER, className, null, superName, null);
280         cw.visitSource(sourceFile, null);
281 
282         String invokerDesc = invokerType.toMethodDescriptorString();
283         mv = cw.visitMethod(Opcodes.ACC_STATIC, invokerName, invokerDesc, null, null);
284     }

294     /*
295      * Low-level emit helpers.
296      */
297     private void emitConst(Object con) {
298         if (con == null) {
299             mv.visitInsn(Opcodes.ACONST_NULL);
300             return;
301         }
302         if (con instanceof Integer) {
303             emitIconstInsn((int) con);
304             return;
305         }
306         if (con instanceof Long) {
307             long x = (long) con;
308             if (x == (short) x) {
309                 emitIconstInsn((int) x);
310                 mv.visitInsn(Opcodes.I2L);
311                 return;
312             }
313         }
314         if (con instanceof Float) {
315             float x = (float) con;
316             if (x == (short) x) {
317                 emitIconstInsn((int) x);
318                 mv.visitInsn(Opcodes.I2F);
319                 return;
320             }
321         }
322         if (con instanceof Double) {
323             double x = (double) con;
324             if (x == (short) x) {
325                 emitIconstInsn((int) x);
326                 mv.visitInsn(Opcodes.I2D);
327                 return;
328             }
329         }
330         if (con instanceof Boolean) {
331             emitIconstInsn((boolean) con ? 1 : 0);
332             return;
333         }
334         // fall through:
335         mv.visitLdcInsn(con);
336     }

338     private void emitIconstInsn(int i) {
339         int opcode;
340         switch (i) {
341         case 0:  opcode = Opcodes.ICONST_0;  break;
342         case 1:  opcode = Opcodes.ICONST_1;  break;
343         case 2:  opcode = Opcodes.ICONST_2;  break;
344         case 3:  opcode = Opcodes.ICONST_3;  break;
345         case 4:  opcode = Opcodes.ICONST_4;  break;
346         case 5:  opcode = Opcodes.ICONST_5;  break;
347         default:
348             if (i == (byte) i) {
349                 mv.visitIntInsn(Opcodes.BIPUSH, i & 0xFF);
350             } else if (i == (short) i) {
351                 mv.visitIntInsn(Opcodes.SIPUSH, (char) i);
352             } else {
353                 mv.visitLdcInsn(i);
354             }
355             return;
356         }
357         mv.visitInsn(opcode);
358     }

360     /*
361      * NOTE: These load/store methods use the localsMap to find the correct index!
362      */
363     private void emitLoadInsn(char type, int index) {
364         int opcode;
365         switch (type) {
366         case 'I':  opcode = Opcodes.ILOAD;  break;
367         case 'J':  opcode = Opcodes.LLOAD;  break;
368         case 'F':  opcode = Opcodes.FLOAD;  break;
369         case 'D':  opcode = Opcodes.DLOAD;  break;
370         case 'L':  opcode = Opcodes.ALOAD;  break;
371         default:
372             throw new InternalError("unknown type: " + type);
373         }
374         mv.visitVarInsn(opcode, localsMap[index]);
375     }
376     private void emitAloadInsn(int index) {
377         emitLoadInsn('L', index);
378     }

380     private void emitStoreInsn(char type, int index) {
381         int opcode;
382         switch (type) {
383         case 'I':  opcode = Opcodes.ISTORE;  break;
384         case 'J':  opcode = Opcodes.LSTORE;  break;
385         case 'F':  opcode = Opcodes.FSTORE;  break;
386         case 'D':  opcode = Opcodes.DSTORE;  break;
387         case 'L':  opcode = Opcodes.ASTORE;  break;
388         default:
389             throw new InternalError("unknown type: " + type);
390         }
391         mv.visitVarInsn(opcode, localsMap[index]);
392     }
393     private void emitAstoreInsn(int index) {
394         emitStoreInsn('L', index);

418         String name  = wrapper.primitiveSimpleName() + "Value";

420         mv.visitTypeInsn(Opcodes.CHECKCAST, owner);
421         mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, owner, name, desc);
422     }

424     /**
425      * Emit an implicit conversion.
426      *
427      * @param ptype type of value present on stack
428      * @param pclass type of value required on stack
429      */
430     private void emitImplicitConversion(char ptype, Class<?> pclass) {
431         switch (ptype) {
432         case 'L':
433             if (VerifyType.isNullConversion(Object.class, pclass))
434                 return;
435             if (isStaticallyNameable(pclass)) {
436                 mv.visitTypeInsn(Opcodes.CHECKCAST, getInternalName(pclass));
437             } else {
438                 mv.visitLdcInsn(constantPlaceholder(pclass));
439                 mv.visitTypeInsn(Opcodes.CHECKCAST, CLS);
440                 mv.visitInsn(Opcodes.SWAP);
441                 mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, CLS, "cast", LL_SIG);
442                 if (pclass.isArray())
443                     mv.visitTypeInsn(Opcodes.CHECKCAST, OBJARY);
444             }
445             return;
446         case 'I':
447             if (!VerifyType.isNullConversion(int.class, pclass))
448                 emitPrimCast(ptype, Wrapper.basicTypeChar(pclass));
449             return;
450         case 'J':
451             assert(pclass == long.class);
452             return;
453         case 'F':
454             assert(pclass == float.class);
455             return;
456         case 'D':
457             assert(pclass == double.class);
458             return;
459         }
460         throw new InternalError("bad implicit conversion: tc="+ptype+": "+pclass);
461     }

463     /**
464      * Emits an actual return instruction conforming to the given return type.
465      */
466     private void emitReturnInsn(Class<?> type) {
467         int opcode;
468         switch (Wrapper.basicTypeChar(type)) {
469         case 'I':  opcode = Opcodes.IRETURN;  break;
470         case 'J':  opcode = Opcodes.LRETURN;  break;
471         case 'F':  opcode = Opcodes.FRETURN;  break;
472         case 'D':  opcode = Opcodes.DRETURN;  break;
473         case 'L':  opcode = Opcodes.ARETURN;  break;
474         case 'V':  opcode = Opcodes.RETURN;   break;
475         default:
476             throw new InternalError("unknown return type: " + type);
477         }
478         mv.visitInsn(opcode);
479     }

494     /**
495      * Generate an invoker method for the passed {@link LambdaForm}.
496      */
497     private byte[] generateCustomizedCodeBytes() {
498         classFilePrologue();
499 
500         // Suppress this method in backtraces displayed to the user.
501         mv.visitAnnotation("Ljava/lang/invoke/LambdaForm$Hidden;", true);
502 
503         // Mark this method as a compiled LambdaForm
504         mv.visitAnnotation("Ljava/lang/invoke/LambdaForm$Compiled;", true);
505 
506         // Force inlining of this invoker method.
507         mv.visitAnnotation("Ljava/lang/invoke/ForceInline;", true);
508 
509         // iterate over the form's names, generating bytecode instructions for each
510         // start iterating at the first name following the arguments
511         for (int i = lambdaForm.arity; i < lambdaForm.names.length; i++) {
512             Name name = lambdaForm.names[i];
513             MemberName member = name.function.member();
514 
515             if (isSelectAlternative(member)) {
516                 // selectAlternative idiom
517                 // FIXME: make sure this idiom is really present!
518                 emitSelectAlternative(name, lambdaForm.names[i + 1]);
519                 i++;  // skip MH.invokeBasic of the selectAlternative result
520             } else if (isStaticallyInvocable(member)) {
521                 emitStaticInvoke(member, name);
522             } else {
523                 emitInvoke(name);
524             }
525 
526             // store the result from evaluating to the target name in a local if required
527             // (if this is the last value, i.e., the one that is going to be returned,
528             // avoid store/load/return and just return)
529             if (i == lambdaForm.names.length - 1 && i == lambdaForm.result) {
530                 // return value - do nothing
531             } else if (name.type != 'V') {
532                 // non-void: actually assign
533                 emitStoreInsn(name.type, name.index());
534             }
535         }
536 
537         // return statement
538         emitReturn();
539 
540         classFileEpilogue();
541         bogusMethod(lambdaForm);
542 
543         final byte[] classFile = cw.toByteArray();
544         maybeDump(className, classFile);
545         return classFile;
546     }

548     /**
549      * Emit an invoke for the given name.
550      */
551     void emitInvoke(Name name) {
552         if (true) {
553             // push receiver
554             MethodHandle target = name.function.resolvedHandle;
555             assert(target != null) : name.exprString();
556             mv.visitLdcInsn(constantPlaceholder(target));
557             mv.visitTypeInsn(Opcodes.CHECKCAST, MH);
558         } else {
559             // load receiver
560             emitAloadInsn(0);
561             mv.visitTypeInsn(Opcodes.CHECKCAST, MH);
562             mv.visitFieldInsn(Opcodes.GETFIELD, MH, "form", LF_SIG);
563             mv.visitFieldInsn(Opcodes.GETFIELD, LF, "names", LFN_SIG);
564             // TODO more to come
565         }
566 
567         // push arguments
568         for (int i = 0; i < name.arguments.length; i++) {
569             emitPushArgument(name, i);
570         }
571 
572         // invocation
573         MethodType type = name.function.methodType();
574         mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, MH, "invokeBasic", type.basicType().toMethodDescriptorString());
575     }

578         // Sample classes from each package we are willing to bind to statically:

585     static boolean isStaticallyInvocable(MemberName member) {
586         if (member == null)  return false;
587         if (member.isConstructor())  return false;
588         Class<?> cls = member.getDeclaringClass();
589         if (cls.isArray() || cls.isPrimitive())
590             return false;  // FIXME
591         if (cls.isAnonymousClass() || cls.isLocalClass())
592             return false;  // inner class of some sort
593         if (cls.getClassLoader() != MethodHandle.class.getClassLoader())
594             return false;  // not on BCP
595         MethodType mtype = member.getMethodOrFieldType();
596         if (!isStaticallyNameable(mtype.returnType()))
597             return false;
598         for (Class<?> ptype : mtype.parameterArray())
599             if (!isStaticallyNameable(ptype))
600                 return false;
601         if (!member.isPrivate() && VerifyAccess.isSamePackage(MethodHandle.class, cls))
602             return true;   // in java.lang.invoke package
603         if (member.isPublic() && isStaticallyNameable(cls))
604             return true;
605         return false;
606     }
607 
608     static boolean isStaticallyNameable(Class<?> cls) {
609         while (cls.isArray())
610             cls = cls.getComponentType();
611         if (cls.isPrimitive())
612             return true;  // int[].class, for example
613         // could use VerifyAccess.isClassAccessible but the following is a safe approximation
614         if (cls.getClassLoader() != Object.class.getClassLoader())
615             return false;
616         if (VerifyAccess.isSamePackage(MethodHandle.class, cls))
617             return true;
618         if (!Modifier.isPublic(cls.getModifiers()))
619             return false;
620         for (Class<?> pkgcls : STATICALLY_INVOCABLE_PACKAGES) {
621             if (VerifyAccess.isSamePackage(pkgcls, cls))
622                 return true;
623         }
624         return false;
625     }

627     /**
628      * Emit an invoke for the given name, using the MemberName directly.
629      */
630     void emitStaticInvoke(MemberName member, Name name) {
631         assert(member.equals(name.function.member()));
632         String cname = getInternalName(member.getDeclaringClass());
633         String mname = member.getName();
634         String mtype;
635         byte refKind = member.getReferenceKind();
636         if (refKind == REF_invokeSpecial) {
637             // in order to pass the verifier, we need to convert this to invokevirtual in all cases
638             assert(member.canBeStaticallyBound()) : member;
639             refKind = REF_invokeVirtual;
640         }
641 
642         if (member.getDeclaringClass().isInterface() && refKind == REF_invokeVirtual) {
643             // Methods from Object declared in an interface can be resolved by JVM to invokevirtual kind.
644             // Need to convert it back to invokeinterface to pass verification and make the invocation works as expected.
645             refKind = REF_invokeInterface;
646         }
647 
648         // push arguments
649         for (int i = 0; i < name.arguments.length; i++) {
650             emitPushArgument(name, i);
651         }
652 
653         // invocation
654         if (member.isMethod()) {
655             mtype = member.getMethodType().toMethodDescriptorString();
656             mv.visitMethodInsn(refKindOpcode(refKind), cname, mname, mtype,
657                                member.getDeclaringClass().isInterface());
658         } else {
659             mtype = MethodType.toFieldDescriptorString(member.getFieldType());
660             mv.visitFieldInsn(refKindOpcode(refKind), cname, mname, mtype);
661         }
662     }
663     int refKindOpcode(byte refKind) {
664         switch (refKind) {
665         case REF_invokeVirtual:      return Opcodes.INVOKEVIRTUAL;
666         case REF_invokeStatic:       return Opcodes.INVOKESTATIC;
667         case REF_invokeSpecial:      return Opcodes.INVOKESPECIAL;
668         case REF_invokeInterface:    return Opcodes.INVOKEINTERFACE;
669         case REF_getField:           return Opcodes.GETFIELD;
670         case REF_putField:           return Opcodes.PUTFIELD;
671         case REF_getStatic:          return Opcodes.GETSTATIC;
672         case REF_putStatic:          return Opcodes.PUTSTATIC;
673         }
674         throw new InternalError("refKind="+refKind);
675     }

686     /**
687      * Emit bytecode for the selectAlternative idiom.
688      *
689      * The pattern looks like (Cf. MethodHandleImpl.makeGuardWithTest):
690      * <blockquote><pre>{@code
691      *   Lambda(a0:L,a1:I)=>{
692      *     t2:I=foo.test(a1:I);
693      *     t3:L=MethodHandleImpl.selectAlternative(t2:I,(MethodHandle(int)int),(MethodHandle(int)int));
694      *     t4:I=MethodHandle.invokeBasic(t3:L,a1:I);t4:I}
695      * }</pre></blockquote>
696      */
697     private void emitSelectAlternative(Name selectAlternativeName, Name invokeBasicName) {
698         MethodType type = selectAlternativeName.function.methodType();
699 
700         Name receiver = (Name) invokeBasicName.arguments[0];
701 
702         Label L_fallback = new Label();
703         Label L_done     = new Label();
704 
705         // load test result
706         emitPushArgument(selectAlternativeName, 0);
707         mv.visitInsn(Opcodes.ICONST_1);
708 
709         // if_icmpne L_fallback
710         mv.visitJumpInsn(Opcodes.IF_ICMPNE, L_fallback);
711 
712         // invoke selectAlternativeName.arguments[1]
713         MethodHandle target = (MethodHandle) selectAlternativeName.arguments[1];
714         emitPushArgument(selectAlternativeName, 1);  // get 2nd argument of selectAlternative
715         emitAstoreInsn(receiver.index());  // store the MH in the receiver slot
716         emitInvoke(invokeBasicName);
717 
718         // goto L_done
719         mv.visitJumpInsn(Opcodes.GOTO, L_done);
720 
721         // L_fallback:
722         mv.visitLabel(L_fallback);
723 
724         // invoke selectAlternativeName.arguments[2]
725         MethodHandle fallback = (MethodHandle) selectAlternativeName.arguments[2];
726         emitPushArgument(selectAlternativeName, 2);  // get 3rd argument of selectAlternative
727         emitAstoreInsn(receiver.index());  // store the MH in the receiver slot
728         emitInvoke(invokeBasicName);
729 
730         // L_done:
731         mv.visitLabel(L_done);
732     }

734     private void emitPushArgument(Name name, int paramIndex) {
735         Object arg = name.arguments[paramIndex];
736         char ptype = name.function.parameterType(paramIndex);
737         MethodType mtype = name.function.methodType();
738         if (arg instanceof Name) {
739             Name n = (Name) arg;
740             emitLoadInsn(n.type, n.index());
741             emitImplicitConversion(n.type, mtype.parameterType(paramIndex));
742         } else if ((arg == null || arg instanceof String) && ptype == 'L') {
743             emitConst(arg);
744         } else {
745             if (Wrapper.isWrapperType(arg.getClass()) && ptype != 'L') {
746                 emitConst(arg);
747             } else {
748                 mv.visitLdcInsn(constantPlaceholder(arg));
749                 emitImplicitConversion('L', mtype.parameterType(paramIndex));
750             }
751         }
752     }

754     /**
755      * Emits a return statement from a LF invoker. If required, the result type is cast to the correct return type.
756      */
757     private void emitReturn() {
758         // return statement
759         if (lambdaForm.result == -1) {
760             // void
761             mv.visitInsn(Opcodes.RETURN);
762         } else {
763             LambdaForm.Name rn = lambdaForm.names[lambdaForm.result];
764             char rtype = Wrapper.basicTypeChar(invokerType.returnType());
765 
766             // put return value on the stack if it is not already there
767             if (lambdaForm.result != lambdaForm.names.length - 1) {
768                 emitLoadInsn(rn.type, lambdaForm.result);
769             }
770 
771             // potentially generate cast
772             // rtype is the return type of the invoker - generated code must conform to this
773             // rn.type is the type of the result Name in the LF
774             if (rtype != rn.type) {
775                 // need cast
776                 if (rtype == 'L') {
777                     // possibly cast the primitive to the correct type for boxing
778                     char boxedType = Wrapper.forWrapperType(invokerType.returnType()).basicTypeChar();
779                     if (boxedType != rn.type) {
780                         emitPrimCast(rn.type, boxedType);
781                     }
782                     // cast primitive to reference ("boxing")
783                     emitBoxing(invokerType.returnType());
784                 } else {
785                     // to-primitive cast
786                     if (rn.type != 'L') {
787                         // prim-to-prim cast
788                         emitPrimCast(rn.type, rtype);
789                     } else {
790                         // ref-to-prim cast ("unboxing")
791                         throw new InternalError("no ref-to-prim (unboxing) casts supported right now");
792                     }
793                 }
794             }
795 
796             // generate actual return statement
797             emitReturnInsn(invokerType.returnType());
798         }
799     }

801     /**
802      * Emit a type conversion bytecode casting from "from" to "to".
803      */
804     private void emitPrimCast(char from, char to) {
805         // Here's how.
806         // -   indicates forbidden
807         // <-> indicates implicit
808         //      to ----> boolean  byte     short    char     int      long     float    double
809         // from boolean    <->        -        -        -        -        -        -        -
810         //      byte        -       <->       i2s      i2c      <->      i2l      i2f      i2d
811         //      short       -       i2b       <->      i2c      <->      i2l      i2f      i2d
812         //      char        -       i2b       i2s      <->      <->      i2l      i2f      i2d
813         //      int         -       i2b       i2s      i2c      <->      i2l      i2f      i2d
814         //      long        -     l2i,i2b   l2i,i2s  l2i,i2c    l2i      <->      l2f      l2d
815         //      float       -     f2i,i2b   f2i,i2s  f2i,i2c    f2i      f2l      <->      f2d
816         //      double      -     d2i,i2b   d2i,i2s  d2i,i2c    d2i      d2l      d2f      <->
817         if (from == to) {
818             // no cast required, should be dead code anyway
819             return;
820         }
821         Wrapper wfrom = Wrapper.forBasicType(from);
822         Wrapper wto   = Wrapper.forBasicType(to);
823         if (wfrom.isSubwordOrInt()) {
824             // cast from {byte,short,char,int} to anything
825             emitI2X(to);
826         } else {
827             // cast from {long,float,double} to anything
828             if (wto.isSubwordOrInt()) {
829                 // cast to {byte,short,char,int}
830                 emitX2I(from);
831                 if (wto.bitWidth() < 32) {
832                     // targets other than int require another conversion
833                     emitI2X(to);
834                 }
835             } else {
836                 // cast to {long,float,double} - this is verbose
837                 boolean error = false;
838                 switch (from) {
839                 case 'J':
840                          if (to == 'F') { mv.visitInsn(Opcodes.L2F); }
841                     else if (to == 'D') { mv.visitInsn(Opcodes.L2D); }
842                     else error = true;
843                     break;
844                 case 'F':
845                          if (to == 'J') { mv.visitInsn(Opcodes.F2L); }
846                     else if (to == 'D') { mv.visitInsn(Opcodes.F2D); }
847                     else error = true;
848                     break;
849                 case 'D':
850                          if (to == 'J') { mv.visitInsn(Opcodes.D2L); }
851                     else if (to == 'F') { mv.visitInsn(Opcodes.D2F); }
852                     else error = true;
853                     break;
854                 default:
855                     error = true;
856                     break;
857                 }
858                 if (error) {
859                     throw new IllegalStateException("unhandled prim cast: " + from + "2" + to);
860                 }
861             }
862         }
863     }

865     private void emitI2X(char type) {
866         switch (type) {
867         case 'B':  mv.visitInsn(Opcodes.I2B);  break;
868         case 'S':  mv.visitInsn(Opcodes.I2S);  break;
869         case 'C':  mv.visitInsn(Opcodes.I2C);  break;
870         case 'I':  /* naught */                break;
871         case 'J':  mv.visitInsn(Opcodes.I2L);  break;
872         case 'F':  mv.visitInsn(Opcodes.I2F);  break;
873         case 'D':  mv.visitInsn(Opcodes.I2D);  break;
874         case 'Z':
875             // For compatibility with ValueConversions and explicitCastArguments:
876             mv.visitInsn(Opcodes.ICONST_1);
877             mv.visitInsn(Opcodes.IAND);
878             break;
879         default:   throw new InternalError("unknown type: " + type);
880         }
881     }

883     private void emitX2I(char type) {
884         switch (type) {
885         case 'J':  mv.visitInsn(Opcodes.L2I);  break;
886         case 'F':  mv.visitInsn(Opcodes.F2I);  break;
887         case 'D':  mv.visitInsn(Opcodes.D2I);  break;
888         default:   throw new InternalError("unknown type: " + type);
889         }
890     }

892     private static String basicTypeCharSignature(String prefix, MethodType type) {
893         StringBuilder buf = new StringBuilder(prefix);
894         for (Class<?> ptype : type.parameterList())
895             buf.append(Wrapper.forBasicType(ptype).basicTypeChar());
896         buf.append('_').append(Wrapper.forBasicType(type.returnType()).basicTypeChar());
897         return buf.toString();
898     }

900     /**
901      * Generate bytecode for a LambdaForm.vmentry which calls interpretWithArguments.
902      */
903     static MemberName generateLambdaFormInterpreterEntryPoint(String sig) {
904         assert(LambdaForm.isValidSignature(sig));
905         //System.out.println("generateExactInvoker "+sig);
906         // compute method type
907         // first parameter and return type
908         char tret = LambdaForm.signatureReturn(sig);
909         MethodType type = MethodType.methodType(LambdaForm.typeClass(tret), MethodHandle.class);
910         // other parameter types
911         int arity = LambdaForm.signatureArity(sig);
912         for (int i = 1; i < arity; i++) {
913             type = type.appendParameterTypes(LambdaForm.typeClass(sig.charAt(i)));
914         }
915         InvokerBytecodeGenerator g = new InvokerBytecodeGenerator("LFI", "interpret_"+tret, type);
916         return g.loadMethod(g.generateLambdaFormInterpreterEntryPointBytes());
917     }

919     private byte[] generateLambdaFormInterpreterEntryPointBytes() {
920         classFilePrologue();
921 
922         // Suppress this method in backtraces displayed to the user.
923         mv.visitAnnotation("Ljava/lang/invoke/LambdaForm$Hidden;", true);
924 
925         // Don't inline the interpreter entry.
926         mv.visitAnnotation("Ljava/lang/invoke/DontInline;", true);
927 
928         // create parameter array
929         emitIconstInsn(invokerType.parameterCount());
930         mv.visitTypeInsn(Opcodes.ANEWARRAY, "java/lang/Object");
931 
932         // fill parameter array
933         for (int i = 0; i < invokerType.parameterCount(); i++) {
934             Class<?> ptype = invokerType.parameterType(i);
935             mv.visitInsn(Opcodes.DUP);
936             emitIconstInsn(i);
937             emitLoadInsn(Wrapper.basicTypeChar(ptype), i);
938             // box if primitive type
939             if (ptype.isPrimitive()) {
940                 emitBoxing(ptype);
941             }
942             mv.visitInsn(Opcodes.AASTORE);
943         }
944         // invoke
945         emitAloadInsn(0);
946         mv.visitFieldInsn(Opcodes.GETFIELD, MH, "form", "Ljava/lang/invoke/LambdaForm;");
947         mv.visitInsn(Opcodes.SWAP);  // swap form and array; avoid local variable
948         mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, LF, "interpretWithArguments", "([Ljava/lang/Object;)Ljava/lang/Object;");
949 
950         // maybe unbox
951         Class<?> rtype = invokerType.returnType();
952         if (rtype.isPrimitive() && rtype != void.class) {
953             emitUnboxing(Wrapper.asWrapperType(rtype));
954         }
955 
956         // return statement
957         emitReturnInsn(rtype);
958 
959         classFileEpilogue();
960         bogusMethod(invokerType);
961 
962         final byte[] classFile = cw.toByteArray();
963         maybeDump(className, classFile);
964         return classFile;
965     }

979     private byte[] generateNamedFunctionInvokerImpl(MethodTypeForm typeForm) {
980         MethodType dstType = typeForm.erasedType();
981         classFilePrologue();
982 
983         // Suppress this method in backtraces displayed to the user.
984         mv.visitAnnotation("Ljava/lang/invoke/LambdaForm$Hidden;", true);
985 
986         // Force inlining of this invoker method.
987         mv.visitAnnotation("Ljava/lang/invoke/ForceInline;", true);
988 
989         // Load receiver
990         emitAloadInsn(0);
991 
992         // Load arguments from array
993         for (int i = 0; i < dstType.parameterCount(); i++) {
994             emitAloadInsn(1);
995             emitIconstInsn(i);
996             mv.visitInsn(Opcodes.AALOAD);
997 
998             // Maybe unbox
999             Class<?> dptype = dstType.parameterType(i);
1000             if (dptype.isPrimitive()) {
1001                 Class<?> sptype = dstType.basicType().wrap().parameterType(i);
1002                 Wrapper dstWrapper = Wrapper.forBasicType(dptype);
1003                 Wrapper srcWrapper = dstWrapper.isSubwordOrInt() ? Wrapper.INT : dstWrapper;  // narrow subword from int
1004                 emitUnboxing(srcWrapper.wrapperType());
1005                 emitPrimCast(srcWrapper.basicTypeChar(), dstWrapper.basicTypeChar());
1006             }
1007         }
1008 
1009         // Invoke
1010         String targetDesc = dstType.basicType().toMethodDescriptorString();
1011         mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, MH, "invokeBasic", targetDesc);
1012 
1013         // Box primitive types
1014         Class<?> rtype = dstType.returnType();
1015         if (rtype != void.class && rtype.isPrimitive()) {
1016             Wrapper srcWrapper = Wrapper.forBasicType(rtype);
1017             Wrapper dstWrapper = srcWrapper.isSubwordOrInt() ? Wrapper.INT : srcWrapper;  // widen subword to int
1018             // boolean casts not allowed
1019             emitPrimCast(srcWrapper.basicTypeChar(), dstWrapper.basicTypeChar());
1020             emitBoxing(dstWrapper.primitiveType());
1021         }
1022 
1023         // If the return type is void we return a null reference.
1024         if (rtype == void.class) {
1025             mv.visitInsn(Opcodes.ACONST_NULL);
1026         }
1027         emitReturnInsn(Object.class);  // NOTE: NamedFunction invokers always return a reference value.
1028 
1029         classFileEpilogue();
1030         bogusMethod(dstType);
1031 
1032         final byte[] classFile = cw.toByteArray();
1033         maybeDump(className, classFile);
1034         return classFile;
1035     }
