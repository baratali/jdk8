26  package java.beans;
27  
28  /**
29   * A "PropertyChange" event gets fired whenever a bean changes a "bound"
30   * property.  You can register a PropertyChangeListener with a source
31   * bean so as to be notified of any bound property updates.
32   */
33  
34  public interface PropertyChangeListener extends java.util.EventListener {
35  
36      /**
37       * This method gets called when a bound property is changed.
38       * @param evt A PropertyChangeEvent object describing the event source
39       *          and the property that has changed.
40       */
41  
42      void propertyChange(PropertyChangeEvent evt);
43  
44  }
