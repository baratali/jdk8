26  package java.security.cert;
27  
28  /**
29   * Certificate is not yet valid exception. This is thrown whenever
30   * the current {@code Date} or the specified {@code Date}
31   * is before the {@code notBefore} date/time in the Certificate
32   * validity period.
33   *
34   * @author Hemma Prafullchandra
35   */
36  public class CertificateNotYetValidException extends CertificateException {
37  
38      static final long serialVersionUID = 4355919900041064702L;
39  
40      /**
41       * Constructs a CertificateNotYetValidException with no detail message. A
42       * detail message is a String that describes this particular
43       * exception.
44       */
45      public CertificateNotYetValidException() {
46          super();
47      }
48  
49      /**
50       * Constructs a CertificateNotYetValidException with the specified detail
51       * message. A detail message is a String that describes this
52       * particular exception.
53       *
54       * @param message the detail message.
55       */
56      public CertificateNotYetValidException(String message) {
57          super(message);
58      }
59  }
