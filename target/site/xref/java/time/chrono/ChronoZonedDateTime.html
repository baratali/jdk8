91  /**
92   * A date-time with a time-zone in an arbitrary chronology,
93   * intended for advanced globalization use cases.
94   * <p>
95   * <b>Most applications should declare method signatures, fields and variables
96   * as {@link ZonedDateTime}, not this interface.</b>
97   * <p>
98   * A {@code ChronoZonedDateTime} is the abstract representation of an offset date-time
99   * where the {@code Chronology chronology}, or calendar system, is pluggable.
100  * The date-time is defined in terms of fields expressed by {@link TemporalField},
101  * where most common implementations are defined in {@link ChronoField}.
102  * The chronology defines how the calendar system operates and the meaning of
103  * the standard fields.
104  *
105  * <h3>When to use this interface</h3>
106  * The design of the API encourages the use of {@code ZonedDateTime} rather than this
107  * interface, even in the case where the application needs to deal with multiple
108  * calendar systems. The rationale for this is explored in detail in {@link ChronoLocalDate}.
109  * <p>
110  * Ensure that the discussion in {@code ChronoLocalDate} has been read and understood
111  * before using this interface.
112  *
113  * @implSpec
114  * This interface must be implemented with care to ensure other classes operate correctly.
115  * All implementations that can be instantiated must be final, immutable and thread-safe.
116  * Subclasses should be Serializable wherever possible.
117  *
118  * @param <D> the concrete type for the date of this date-time
119  * @since 1.8
120  */
121 public interface ChronoZonedDateTime<D extends ChronoLocalDate>
122         extends Temporal, Comparable<ChronoZonedDateTime<?>> {
123 
124     /**
125      * Gets a comparator that compares {@code ChronoZonedDateTime} in
126      * time-line order ignoring the chronology.
127      * <p>
128      * This comparator differs from the comparison in {@link #compareTo} in that it
129      * only compares the underlying instant and not the chronology.
130      * This allows dates in different calendar systems to be compared based
131      * on the position of the date-time on the instant time-line.
132      * The underlying comparison is equivalent to comparing the epoch-second and nano-of-second.
133      *
134      * @return a comparator that compares in time-line order ignoring the chronology
135      *
136      * @see #isAfter
137      * @see #isBefore
138      * @see #isEqual
139      */
140     static Comparator<ChronoZonedDateTime<?>> timeLineOrder() {
141         return AbstractChronology.INSTANT_ORDER;
142     }

144     //-----------------------------------------------------------------------
145     /**
146      * Obtains an instance of {@code ChronoZonedDateTime} from a temporal object.
147      * <p>
148      * This creates a zoned date-time based on the specified temporal.
149      * A {@code TemporalAccessor} represents an arbitrary set of date and time information,
150      * which this factory converts to an instance of {@code ChronoZonedDateTime}.
151      * <p>
152      * The conversion extracts and combines the chronology, date, time and zone
153      * from the temporal object. The behavior is equivalent to using
154      * {@link Chronology#zonedDateTime(TemporalAccessor)} with the extracted chronology.
155      * Implementations are permitted to perform optimizations such as accessing
156      * those fields that are equivalent to the relevant objects.
157      * <p>
158      * This method matches the signature of the functional interface {@link TemporalQuery}
159      * allowing it to be used as a query via method reference, {@code ChronoZonedDateTime::from}.
160      *
161      * @param temporal  the temporal object to convert, not null
162      * @return the date-time, not null
163      * @throws DateTimeException if unable to convert to a {@code ChronoZonedDateTime}
164      * @see Chronology#zonedDateTime(TemporalAccessor)
165      */
166     static ChronoZonedDateTime<?> from(TemporalAccessor temporal) {
167         if (temporal instanceof ChronoZonedDateTime) {
168             return (ChronoZonedDateTime<?>) temporal;
169         }
170         Objects.requireNonNull(temporal, "temporal");
171         Chronology chrono = temporal.query(TemporalQueries.chronology());
172         if (chrono == null) {
173             throw new DateTimeException("Unable to obtain ChronoZonedDateTime from TemporalAccessor: " + temporal.getClass());
174         }
175         return chrono.zonedDateTime(temporal);
176     }

178     //-----------------------------------------------------------------------
179     @Override
180     default ValueRange range(TemporalField field) {
181         if (field instanceof ChronoField) {
182             if (field == INSTANT_SECONDS || field == OFFSET_SECONDS) {
183                 return field.range();
184             }
185             return toLocalDateTime().range(field);
186         }
187         return field.rangeRefinedBy(this);
188     }
189 
190     @Override
191     default int get(TemporalField field) {
192         if (field instanceof ChronoField) {
193             switch ((ChronoField) field) {
194                 case INSTANT_SECONDS:
195                     throw new UnsupportedTemporalTypeException("Invalid field 'InstantSeconds' for get() method, use getLong() instead");
196                 case OFFSET_SECONDS:
197                     return getOffset().getTotalSeconds();
198             }
199             return toLocalDateTime().get(field);
200         }
201         return Temporal.super.get(field);
202     }
203 
204     @Override
205     default long getLong(TemporalField field) {
206         if (field instanceof ChronoField) {
207             switch ((ChronoField) field) {
208                 case INSTANT_SECONDS: return toEpochSecond();
209                 case OFFSET_SECONDS: return getOffset().getTotalSeconds();
210             }
211             return toLocalDateTime().getLong(field);
212         }
213         return field.getFrom(this);
214     }

382      * This checks if the specified unit can be added to or subtracted from this date-time.
383      * If false, then calling the {@link #plus(long, TemporalUnit)} and
384      * {@link #minus(long, TemporalUnit) minus} methods will throw an exception.

395      * @return true if the unit can be added/subtracted, false if not
396      */
397     @Override
398     default boolean isSupported(TemporalUnit unit) {
399         if (unit instanceof ChronoUnit) {
400             return unit != FOREVER;
401         }
402         return unit != null && unit.isSupportedBy(this);
403     }

464     /**
465      * Queries this date-time using the specified query.
466      * <p>
467      * This queries this date-time using the specified query strategy object.
468      * The {@code TemporalQuery} object defines the logic to be used to
469      * obtain the result. Read the documentation of the query to understand
470      * what the result of this method will be.
471      * <p>
472      * The result of this method is obtained by invoking the
473      * {@link java.time.temporal.TemporalQuery#queryFrom(TemporalAccessor)} method on the
474      * specified query passing {@code this} as the argument.
475      *
476      * @param <R> the type of the result
477      * @param query  the query to invoke, not null
478      * @return the query result, null may be returned (defined by the query)
479      * @throws DateTimeException if unable to query (defined by the query)
480      * @throws ArithmeticException if numeric overflow occurs (defined by the query)
481      */
482     @SuppressWarnings("unchecked")
483     @Override
484     default <R> R query(TemporalQuery<R> query) {
485         if (query == TemporalQueries.zone() || query == TemporalQueries.zoneId()) {
486             return (R) getZone();
487         } else if (query == TemporalQueries.offset()) {
488             return (R) getOffset();
489         } else if (query == TemporalQueries.localTime()) {
490             return (R) toLocalTime();
491         } else if (query == TemporalQueries.chronology()) {
492             return (R) getChronology();
493         } else if (query == TemporalQueries.precision()) {
494             return (R) NANOS;
495         }
496         // inline TemporalAccessor.super.query(query) as an optimization
497         // non-JDK classes are not permitted to make this optimization
498         return query.queryFrom(this);
499     }

521     /**
522      * Converts this date-time to an {@code Instant}.
523      * <p>
524      * This returns an {@code Instant} representing the same point on the
525      * time-line as this date-time. The calculation combines the
526      * {@linkplain #toLocalDateTime() local date-time} and
527      * {@linkplain #getOffset() offset}.
528      *
529      * @return an {@code Instant} representing the same instant, not null
530      */
531     default Instant toInstant() {
532         return Instant.ofEpochSecond(toEpochSecond(), toLocalTime().getNano());
533     }

535     /**
536      * Converts this date-time to the number of seconds from the epoch
537      * of 1970-01-01T00:00:00Z.
538      * <p>
539      * This uses the {@linkplain #toLocalDateTime() local date-time} and
540      * {@linkplain #getOffset() offset} to calculate the epoch-second value,
541      * which is the number of elapsed seconds from 1970-01-01T00:00:00Z.
542      * Instants on the time-line after the epoch are positive, earlier are negative.
543      *
544      * @return the number of seconds from the epoch of 1970-01-01T00:00:00Z
545      */
546     default long toEpochSecond() {
547         long epochDay = toLocalDate().toEpochDay();
548         long secs = epochDay * 86400 + toLocalTime().toSecondOfDay();
549         secs -= getOffset().getTotalSeconds();
550         return secs;
551     }

553     //-----------------------------------------------------------------------
554     /**
555      * Compares this date-time to another date-time, including the chronology.
556      * <p>
557      * The comparison is based first on the instant, then on the local date-time,
558      * then on the zone ID, then on the chronology.
559      * It is "consistent with equals", as defined by {@link Comparable}.
560      * <p>
561      * If all the date-time objects being compared are in the same chronology, then the
562      * additional chronology stage is not required.
563      * <p>
564      * This default implementation performs the comparison defined above.
565      *
566      * @param other  the other date-time to compare to, not null
567      * @return the comparator value, negative if less, positive if greater
568      */
569     @Override
570     default int compareTo(ChronoZonedDateTime<?> other) {
571         int cmp = Long.compare(toEpochSecond(), other.toEpochSecond());
572         if (cmp == 0) {
573             cmp = toLocalTime().getNano() - other.toLocalTime().getNano();
574             if (cmp == 0) {
575                 cmp = toLocalDateTime().compareTo(other.toLocalDateTime());
576                 if (cmp == 0) {
577                     cmp = getZone().getId().compareTo(other.getZone().getId());
578                     if (cmp == 0) {
579                         cmp = getChronology().compareTo(other.getChronology());
580                     }
581                 }
582             }
583         }
584         return cmp;
585     }

587     /**
588      * Checks if the instant of this date-time is before that of the specified date-time.
589      * <p>
590      * This method differs from the comparison in {@link #compareTo} in that it
591      * only compares the instant of the date-time. This is equivalent to using
592      * {@code dateTime1.toInstant().isBefore(dateTime2.toInstant());}.
593      * <p>
594      * This default implementation performs the comparison based on the epoch-second
595      * and nano-of-second.
596      *
597      * @param other  the other date-time to compare to, not null
598      * @return true if this point is before the specified date-time
599      */
600     default boolean isBefore(ChronoZonedDateTime<?> other) {
601         long thisEpochSec = toEpochSecond();
602         long otherEpochSec = other.toEpochSecond();
603         return thisEpochSec < otherEpochSec ||
604             (thisEpochSec == otherEpochSec && toLocalTime().getNano() < other.toLocalTime().getNano());
605     }

607     /**
608      * Checks if the instant of this date-time is after that of the specified date-time.
609      * <p>
610      * This method differs from the comparison in {@link #compareTo} in that it
611      * only compares the instant of the date-time. This is equivalent to using
612      * {@code dateTime1.toInstant().isAfter(dateTime2.toInstant());}.
613      * <p>
614      * This default implementation performs the comparison based on the epoch-second
615      * and nano-of-second.
616      *
617      * @param other  the other date-time to compare to, not null
618      * @return true if this is after the specified date-time
619      */
620     default boolean isAfter(ChronoZonedDateTime<?> other) {
621         long thisEpochSec = toEpochSecond();
622         long otherEpochSec = other.toEpochSecond();
623         return thisEpochSec > otherEpochSec ||
624             (thisEpochSec == otherEpochSec && toLocalTime().getNano() > other.toLocalTime().getNano());
625     }

627     /**
628      * Checks if the instant of this date-time is equal to that of the specified date-time.
629      * <p>
630      * This method differs from the comparison in {@link #compareTo} and {@link #equals}
631      * in that it only compares the instant of the date-time. This is equivalent to using
632      * {@code dateTime1.toInstant().equals(dateTime2.toInstant());}.
633      * <p>
634      * This default implementation performs the comparison based on the epoch-second
635      * and nano-of-second.
636      *
637      * @param other  the other date-time to compare to, not null
638      * @return true if the instant equals the instant of the specified date-time
639      */
640     default boolean isEqual(ChronoZonedDateTime<?> other) {
641         return toEpochSecond() == other.toEpochSecond() &&
642                 toLocalTime().getNano() == other.toLocalTime().getNano();
643     }
