53   * correspond to platform-specific data formats, and "flavors" (DataFlavors),

55   * by the data transfer subsystem to transfer data between Java and native
56   * applications, and between Java applications in separate VMs.

59   * @since 1.2

195     /**
196      * Initializes a SystemFlavorMap by reading flavormap.properties and
197      * AWT.DnD.flavorMapFileURL.
198      * For thread-safety must be called under lock on this.
199      */
200     private void initSystemFlavorMap() {
201         if (isMapInitialized) {
202             return;
203         }
204 
205         isMapInitialized = true;
206         BufferedReader flavormapDotProperties =
207             java.security.AccessController.doPrivileged(
208                 new java.security.PrivilegedAction<BufferedReader>() {
209                     public BufferedReader run() {
210                         String fileName =
211                             System.getProperty("java.home") +
212                             File.separator +
213                             "lib" +
214                             File.separator +
215                             "flavormap.properties";
216                         try {
217                             return new BufferedReader
218                                 (new InputStreamReader
219                                     (new File(fileName).toURI().toURL().openStream(), "ISO-8859-1"));
220                         } catch (MalformedURLException e) {
221                             System.err.println("MalformedURLException:" + e + " while loading default flavormap.properties file:" + fileName);
222                         } catch (IOException e) {
223                             System.err.println("IOException:" + e + " while loading default flavormap.properties file:" + fileName);
224                         }
225                         return null;
226                     }
227                 });
228 
229         String url =
230             java.security.AccessController.doPrivileged(
231                 new java.security.PrivilegedAction<String>() {
232                     public String run() {
233                         return Toolkit.getProperty("AWT.DnD.flavorMapFileURL", null);
234                     }
235                 });
236 
237         if (flavormapDotProperties != null) {
238             try {
239                 parseAndStoreReader(flavormapDotProperties);
240             } catch (IOException e) {
241                 System.err.println("IOException:" + e + " while parsing default flavormap.properties file");
242             }
243         }
244 
245         BufferedReader flavormapURL = null;
246         if (url != null) {
247             try {
248                 flavormapURL = new BufferedReader(new InputStreamReader(new URL(url).openStream(), "ISO-8859-1"));
249             } catch (MalformedURLException e) {
250                 System.err.println("MalformedURLException:" + e + " while reading AWT.DnD.flavorMapFileURL:" + url);
251             } catch (IOException e) {
252                 System.err.println("IOException:" + e + " while reading AWT.DnD.flavorMapFileURL:" + url);
253             } catch (SecurityException e) {
254                 // ignored
255             }
256         }
257 
258         if (flavormapURL != null) {
259             try {
260                 parseAndStoreReader(flavormapURL);
261             } catch (IOException e) {
262                 System.err.println("IOException:" + e + " while parsing AWT.DnD.flavorMapFileURL");
263             }
264         }
265     }
266     /**
267      * Copied code from java.util.Properties. Parsing the data ourselves is the
268      * only way to handle duplicate keys and values.
269      */
270     private void parseAndStoreReader(BufferedReader in) throws IOException {
271         while (true) {
272             // Get next line
273             String line = in.readLine();
274             if (line == null) {
275                 return;
276             }
277 
278             if (line.length() > 0) {
279                 // Continue lines that end in slashes if they are not comments
280                 char firstChar = line.charAt(0);
281                 if (firstChar != '#' && firstChar != '!') {
282                     while (continueLine(line)) {
283                         String nextLine = in.readLine();
284                         if (nextLine == null) {
285                             nextLine = "";
286                         }
287                         String loppedLine =
288                             line.substring(0, line.length() - 1);
289                         // Advance beyond whitespace on new line
290                         int startIndex = 0;
291                         for(; startIndex < nextLine.length(); startIndex++) {
292                             if (whiteSpaceChars.
293                                     indexOf(nextLine.charAt(startIndex)) == -1)
294                             {
295                                 break;
296                             }
297                         }
298                         nextLine = nextLine.substring(startIndex,
299                                                       nextLine.length());
300                         line = loppedLine+nextLine;
301                     }
302 
303                     // Find start of key
304                     int len = line.length();
305                     int keyStart = 0;
306                     for(; keyStart < len; keyStart++) {
307                         if(whiteSpaceChars.
308                                indexOf(line.charAt(keyStart)) == -1) {
309                             break;
310                         }
311                     }
312 
313                     // Blank lines are ignored
314                     if (keyStart == len) {
315                         continue;
316                     }
317 
318                     // Find separation between key and value
319                     int separatorIndex = keyStart;
320                     for(; separatorIndex < len; separatorIndex++) {
321                         char currentChar = line.charAt(separatorIndex);
322                         if (currentChar == '\\') {
323                             separatorIndex++;
324                         } else if (keyValueSeparators.
325                                        indexOf(currentChar) != -1) {
326                             break;
327                         }
328                     }
329 
330                     // Skip over whitespace after key if any
331                     int valueIndex = separatorIndex;
332                     for (; valueIndex < len; valueIndex++) {
333                         if (whiteSpaceChars.
334                                 indexOf(line.charAt(valueIndex)) == -1) {
335                             break;
336                         }
337                     }
338 
339                     // Skip over one non whitespace key value separators if any
340                     if (valueIndex < len) {
341                         if (strictKeyValueSeparators.
342                                 indexOf(line.charAt(valueIndex)) != -1) {
343                             valueIndex++;
344                         }
345                     }
346 
347                     // Skip over white space after other separators if any
348                     while (valueIndex < len) {
349                         if (whiteSpaceChars.
350                                 indexOf(line.charAt(valueIndex)) == -1) {
351                             break;
352                         }
353                         valueIndex++;
354                     }
355 
356                     String key = line.substring(keyStart, separatorIndex);
357                     String value = (separatorIndex < len)
358                         ? line.substring(valueIndex, len)
359                         : "";
360 
361                     // Convert then store key and value
362                     key = loadConvert(key);
363                     value = loadConvert(value);
364 
365                     try {
366                         MimeType mime = new MimeType(value);
367                         if ("text".equals(mime.getPrimaryType())) {
368                             String charset = mime.getParameter("charset");
369                             if (DataTransferer.doesSubtypeSupportCharset
370                                     (mime.getSubType(), charset))
371                             {
372                                 // We need to store the charset and eoln
373                                 // parameters, if any, so that the
374                                 // DataTransferer will have this information
375                                 // for conversion into the native format.
376                                 DataTransferer transferer =
377                                     DataTransferer.getInstance();
378                                 if (transferer != null) {
379                                     transferer.registerTextFlavorProperties
380                                         (key, charset,
381                                          mime.getParameter("eoln"),
382                                          mime.getParameter("terminators"));
383                                 }
384                             }
385 
386                             // But don't store any of these parameters in the
387                             // DataFlavor itself for any text natives (even
388                             // non-charset ones). The SystemFlavorMap will
389                             // synthesize the appropriate mappings later.
390                             mime.removeParameter("charset");
391                             mime.removeParameter("class");
392                             mime.removeParameter("eoln");
393                             mime.removeParameter("terminators");
394                             value = mime.toString();
395                         }
396                     } catch (MimeTypeParseException e) {
397                         e.printStackTrace();
398                         continue;
399                     }
400 
401                     DataFlavor flavor;
402                     try {
403                         flavor = new DataFlavor(value);
404                     } catch (Exception e) {
405                         try {
406                             flavor = new DataFlavor(value, (String)null);
407                         } catch (Exception ee) {
408                             ee.printStackTrace();
409                             continue;
410                         }
411                     }
412 
413                     final LinkedHashSet<DataFlavor> dfs = new LinkedHashSet<>();
414 
415                     dfs.add(flavor);
416 
417                     if ("text".equals(flavor.getPrimaryType())) {
418                         dfs.addAll(convertMimeTypeToDataFlavors(value));
419                     }
420 
421                     for (DataFlavor df : dfs) {
422                         store(df, key, getFlavorToNative());
423                         store(key, df, getNativeToFlavor());
424                     }
425                 }
426             }
427         }
428     }

430     /**
431      * Copied from java.util.Properties.
432      */
433     private boolean continueLine (String line) {
434         int slashCount = 0;
435         int index = line.length() - 1;
436         while((index >= 0) && (line.charAt(index--) == '\\')) {
437             slashCount++;
438         }
439         return (slashCount % 2 == 1);
440     }

442     /**
443      * Copied from java.util.Properties.
444      */
445     private String loadConvert(String theString) {
446         char aChar;
447         int len = theString.length();
448         StringBuilder outBuffer = new StringBuilder(len);
449 
450         for (int x = 0; x < len; ) {
451             aChar = theString.charAt(x++);
452             if (aChar == '\\') {
453                 aChar = theString.charAt(x++);
454                 if (aChar == 'u') {
455                     // Read the xxxx
456                     int value = 0;
457                     for (int i = 0; i < 4; i++) {
458                         aChar = theString.charAt(x++);
459                         switch (aChar) {
460                           case '0': case '1': case '2': case '3': case '4':
461                           case '5': case '6': case '7': case '8': case '9': {
462                              value = (value << 4) + aChar - '0';
463                              break;
464                           }
465                           case 'a': case 'b': case 'c':
466                           case 'd': case 'e': case 'f': {
467                              value = (value << 4) + 10 + aChar - 'a';
468                              break;
469                           }
470                           case 'A': case 'B': case 'C':
471                           case 'D': case 'E': case 'F': {
472                              value = (value << 4) + 10 + aChar - 'A';
473                              break;
474                           }
475                           default: {
476                               throw new IllegalArgumentException(
477                                            "Malformed \\uxxxx encoding.");
478                           }
479                         }
480                     }
481                     outBuffer.append((char)value);
482                 } else {
483                     if (aChar == 't') {
484                         aChar = '\t';
485                     } else if (aChar == 'r') {
486                         aChar = '\r';
487                     } else if (aChar == 'n') {
488                         aChar = '\n';
489                     } else if (aChar == 'f') {
490                         aChar = '\f';
491                     }
492                     outBuffer.append(aChar);
493                 }
494             } else {
495                 outBuffer.append(aChar);
496             }
497         }
498         return outBuffer.toString();
499     }

501     /**
502      * Stores the listed object under the specified hash key in map. Unlike a
503      * standard map, the listed object will not replace any object already at
504      * the appropriate Map location, but rather will be appended to a List
505      * stored in that location.
506      */
507     private <H, L> void store(H hashed, L listed, Map<H, List<L>> map) {
508         List<L> list = map.get(hashed);
509         if (list == null) {
510             list = new ArrayList<>(1);
511             map.put(hashed, list);
512         }
513         if (!list.contains(listed)) {
514             list.add(listed);
515         }
516     }

518     /**
519      * Semantically equivalent to 'nativeToFlavor.get(nat)'. This method
520      * handles the case where 'nat' is not found in 'nativeToFlavor'. In that
521      * case, a new DataFlavor is synthesized, stored, and returned, if and
522      * only if the specified native is encoded as a Java MIME type.
523      */
524     private List<DataFlavor> nativeToFlavorLookup(String nat) {
525         List<DataFlavor> flavors = getNativeToFlavor().get(nat);
526 
527         if (nat != null && !disabledMappingGenerationKeys.contains(nat)) {
528             DataTransferer transferer = DataTransferer.getInstance();
529             if (transferer != null) {
530                 List<DataFlavor> platformFlavors =
531                     transferer.getPlatformMappingsForNative(nat);
532                 if (!platformFlavors.isEmpty()) {
533                     if (flavors != null) {
534                         platformFlavors.removeAll(new HashSet<>(flavors));
535                         // Prepending the platform-specific mappings ensures
536                         // that the flavors added with
537                         // addFlavorForUnencodedNative() are at the end of
538                         // list.
539                         platformFlavors.addAll(flavors);
540                     }
541                     flavors = platformFlavors;
542                 }
543             }
544         }
545 
546         if (flavors == null && isJavaMIMEType(nat)) {
547             String decoded = decodeJavaMIMEType(nat);
548             DataFlavor flavor = null;
549 
550             try {
551                 flavor = new DataFlavor(decoded);
552             } catch (Exception e) {
553                 System.err.println("Exception \"" + e.getClass().getName() +
554                                    ": " + e.getMessage()  +
555                                    "\"while constructing DataFlavor for: " +
556                                    decoded);
557             }
558 
559             if (flavor != null) {
560                 flavors = new ArrayList<>(1);
561                 getNativeToFlavor().put(nat, flavors);
562                 flavors.add(flavor);
563                 getFlavorsForNativeCache.remove(nat);
564                 getFlavorsForNativeCache.remove(null);
565 
566                 List<String> natives = getFlavorToNative().get(flavor);
567                 if (natives == null) {
568                     natives = new ArrayList<>(1);
569                     getFlavorToNative().put(flavor, natives);
570                 }
571                 natives.add(nat);
572                 getNativesForFlavorCache.remove(flavor);
573                 getNativesForFlavorCache.remove(null);
574             }
575         }
576 
577         return (flavors != null) ? flavors : new ArrayList<>(0);
578     }

580     /**
581      * Semantically equivalent to 'flavorToNative.get(flav)'. This method
582      * handles the case where 'flav' is not found in 'flavorToNative' depending
583      * on the value of passes 'synthesize' parameter. If 'synthesize' is
584      * SYNTHESIZE_IF_NOT_FOUND a native is synthesized, stored, and returned by
585      * encoding the DataFlavor's MIME type. Otherwise an empty List is returned
586      * and 'flavorToNative' remains unaffected.
587      */
588     private List<String> flavorToNativeLookup(final DataFlavor flav,
589                                               final boolean synthesize) {
590         List<String> natives = getFlavorToNative().get(flav);
591 
592         if (flav != null && !disabledMappingGenerationKeys.contains(flav)) {
593             DataTransferer transferer = DataTransferer.getInstance();
594             if (transferer != null) {
595                 List<String> platformNatives =
596                     transferer.getPlatformMappingsForFlavor(flav);
597                 if (!platformNatives.isEmpty()) {
598                     if (natives != null) {
599                         platformNatives.removeAll(new HashSet<>(natives));
600                         // Prepend the platform-specific mappings to ensure
601                         // that the natives added with
602                         // addUnencodedNativeForFlavor() are at the end of
603                         // list.
604                         platformNatives.addAll(natives);
605                     }
606                     natives = platformNatives;
607                 }
608             }
609         }
610 
611         if (natives == null) {
612             if (synthesize) {
613                 String encoded = encodeDataFlavor(flav);
614                 natives = new ArrayList<>(1);
615                 getFlavorToNative().put(flav, natives);
616                 natives.add(encoded);
617                 getNativesForFlavorCache.remove(flav);
618                 getNativesForFlavorCache.remove(null);
619 
620                 List<DataFlavor> flavors = getNativeToFlavor().get(encoded);
621                 if (flavors == null) {
622                     flavors = new ArrayList<>(1);
623                     getNativeToFlavor().put(encoded, flavors);
624                 }
625                 flavors.add(flav);
626                 getFlavorsForNativeCache.remove(encoded);
627                 getFlavorsForNativeCache.remove(null);
628             } else {
629                 natives = new ArrayList<>(0);
630             }
631         }
632 
633         return natives;
634     }

636     /**
637      * Returns a <code>List</code> of <code>String</code> natives to which the
638      * specified <code>DataFlavor</code> can be translated by the data transfer
639      * subsystem. The <code>List</code> will be sorted from best native to
640      * worst. That is, the first native will best reflect data in the specified
641      * flavor to the underlying native platform.
642      * <p>
643      * If the specified <code>DataFlavor</code> is previously unknown to the
644      * data transfer subsystem and the data transfer subsystem is unable to
645      * translate this <code>DataFlavor</code> to any existing native, then
646      * invoking this method will establish a
647      * mapping in both directions between the specified <code>DataFlavor</code>
648      * and an encoded version of its MIME type as its native.
649      *
650      * @param flav the <code>DataFlavor</code> whose corresponding natives
651      *        should be returned. If <code>null</code> is specified, all
652      *        natives currently known to the data transfer subsystem are
653      *        returned in a non-deterministic order.
654      * @return a <code>java.util.List</code> of <code>java.lang.String</code>
655      *         objects which are platform-specific representations of platform-
656      *         specific data formats
657      *
658      * @see #encodeDataFlavor
659      * @since 1.4
660      */
661     public synchronized List<String> getNativesForFlavor(DataFlavor flav) {
662         List<String> retval = null;
663 
664         // Check cache, even for null flav
665         SoftReference<List<String>> ref = getNativesForFlavorCache.get(flav);
666         if (ref != null) {
667             retval = ref.get();
668             if (retval != null) {
669                 // Create a copy, because client code can modify the returned
670                 // list.
671                 return new ArrayList<>(retval);
672             }
673         }
674 
675         if (flav == null) {
676             retval = new ArrayList<>(getNativeToFlavor().keySet());
677         } else if (disabledMappingGenerationKeys.contains(flav)) {
678             // In this case we shouldn't synthesize a native for this flavor,
679             // since its mappings were explicitly specified.
680             retval = flavorToNativeLookup(flav, !SYNTHESIZE_IF_NOT_FOUND);
681         } else if (DataTransferer.isFlavorCharsetTextType(flav)) {
682 
683             // For text/* flavors, flavor-to-native mappings specified in
684             // flavormap.properties are stored per flavor's base type.
685             if ("text".equals(flav.getPrimaryType())) {
686                 retval = getAllNativesForType(flav.mimeType.getBaseType());
687                 if (retval != null) {
688                     // To prevent the List stored in the map from modification.
689                     retval = new ArrayList(retval);
690                 }
691             }
692 
693             // Also include text/plain natives, but don't duplicate Strings
694             List<String> textPlainList = getAllNativesForType(TEXT_PLAIN_BASE_TYPE);
695 
696             if (textPlainList != null && !textPlainList.isEmpty()) {
697                 // To prevent the List stored in the map from modification.
698                 // This also guarantees that removeAll() is supported.
699                 textPlainList = new ArrayList<>(textPlainList);
700                 if (retval != null && !retval.isEmpty()) {
701                     // Use HashSet to get constant-time performance for search.
702                     textPlainList.removeAll(new HashSet<>(retval));
703                     retval.addAll(textPlainList);
704                 } else {
705                     retval = textPlainList;
706                 }
707             }
708 
709             if (retval == null || retval.isEmpty()) {
710                 retval = flavorToNativeLookup(flav, SYNTHESIZE_IF_NOT_FOUND);
711             } else {
712                 // In this branch it is guaranteed that natives explicitly
713                 // listed for flav's MIME type were added with
714                 // addUnencodedNativeForFlavor(), so they have lower priority.
715                 List<String> explicitList =
716                     flavorToNativeLookup(flav, !SYNTHESIZE_IF_NOT_FOUND);
717 
718                 // flavorToNativeLookup() never returns null.
719                 // It can return an empty List, however.
720                 if (!explicitList.isEmpty()) {
721                     // To prevent the List stored in the map from modification.
722                     // This also guarantees that removeAll() is supported.
723                     explicitList = new ArrayList<>(explicitList);
724                     // Use HashSet to get constant-time performance for search.
725                     explicitList.removeAll(new HashSet<>(retval));
726                     retval.addAll(explicitList);
727                 }
728             }
729         } else if (DataTransferer.isFlavorNoncharsetTextType(flav)) {
730             retval = getAllNativesForType(flav.mimeType.getBaseType());
731 
732             if (retval == null || retval.isEmpty()) {
733                 retval = flavorToNativeLookup(flav, SYNTHESIZE_IF_NOT_FOUND);
734             } else {
735                 // In this branch it is guaranteed that natives explicitly
736                 // listed for flav's MIME type were added with
737                 // addUnencodedNativeForFlavor(), so they have lower priority.
738                 List<String> explicitList =
739                     flavorToNativeLookup(flav, !SYNTHESIZE_IF_NOT_FOUND);
740 
741                 // flavorToNativeLookup() never returns null.
742                 // It can return an empty List, however.
743                 if (!explicitList.isEmpty()) {
744                     // To prevent the List stored in the map from modification.
745                     // This also guarantees that add/removeAll() are supported.
746                     retval = new ArrayList<>(retval);
747                     explicitList = new ArrayList<>(explicitList);
748                     // Use HashSet to get constant-time performance for search.
749                     explicitList.removeAll(new HashSet<>(retval));
750                     retval.addAll(explicitList);
751                 }
752             }
753         } else {
754             retval = flavorToNativeLookup(flav, SYNTHESIZE_IF_NOT_FOUND);
755         }
756 
757         getNativesForFlavorCache.put(flav, new SoftReference<>(retval));
758         // Create a copy, because client code can modify the returned list.
759         return new ArrayList<>(retval);
760     }

762     /**
763      * Returns a <code>List</code> of <code>DataFlavor</code>s to which the
764      * specified <code>String</code> native can be translated by the data
765      * transfer subsystem. The <code>List</code> will be sorted from best
766      * <code>DataFlavor</code> to worst. That is, the first
767      * <code>DataFlavor</code> will best reflect data in the specified
768      * native to a Java application.
769      * <p>
770      * If the specified native is previously unknown to the data transfer
771      * subsystem, and that native has been properly encoded, then invoking this
772      * method will establish a mapping in both directions between the specified
773      * native and a <code>DataFlavor</code> whose MIME type is a decoded
774      * version of the native.
775      * <p>
776      * If the specified native is not a properly encoded native and the
777      * mappings for this native have not been altered with
778      * <code>setFlavorsForNative</code>, then the contents of the
779      * <code>List</code> is platform dependent, but <code>null</code>
780      * cannot be returned.
781      *
782      * @param nat the native whose corresponding <code>DataFlavor</code>s
783      *        should be returned. If <code>null</code> is specified, all
784      *        <code>DataFlavor</code>s currently known to the data transfer
785      *        subsystem are returned in a non-deterministic order.
786      * @return a <code>java.util.List</code> of <code>DataFlavor</code>
787      *         objects into which platform-specific data in the specified,
788      *         platform-specific native can be translated
789      *
790      * @see #encodeJavaMIMEType
791      * @since 1.4
792      */
793     public synchronized List<DataFlavor> getFlavorsForNative(String nat) {
794 
795         // Check cache, even for null nat
796         SoftReference<List<DataFlavor>> ref = getFlavorsForNativeCache.get(nat);
797         if (ref != null) {
798             List<DataFlavor> retval = ref.get();
799             if (retval != null) {
800                 return new ArrayList<>(retval);
801             }
802         }
803 
804         final LinkedHashSet <DataFlavor> returnValue =
805             new LinkedHashSet<>();
806 
807         if (nat == null) {
808             final List<String> natives = getNativesForFlavor(null);
809 
810             for (String n : natives)
811             {
812                 final List<DataFlavor> flavors = getFlavorsForNative(n);
813 
814                 for (DataFlavor df : flavors)
815                 {
816                     returnValue.add(df);
817                 }
818             }
819         } else {
820 
821             final List<DataFlavor> flavors = nativeToFlavorLookup(nat);
822 
823             if (disabledMappingGenerationKeys.contains(nat)) {
824                 return flavors;
825             }
826 
827             final List<DataFlavor> flavorsAndBaseTypes =
828                 nativeToFlavorLookup(nat);
829 
830             for (DataFlavor df : flavorsAndBaseTypes) {
831                 returnValue.add(df);
832                 if ("text".equals(df.getPrimaryType())) {
833                     try {
834                         returnValue.addAll(
835                                 convertMimeTypeToDataFlavors(
836                                         new MimeType(df.getMimeType()
837                                         ).getBaseType()));
838                     } catch (MimeTypeParseException e) {
839                         e.printStackTrace();
840                     }
841                 }
842             }
843 
844         }
845 
846         final List<DataFlavor> arrayList = new ArrayList<>(returnValue);
847         getFlavorsForNativeCache.put(nat, new SoftReference<>(arrayList));
848         return new ArrayList<>(arrayList);
849     }

851     private static Set<DataFlavor> convertMimeTypeToDataFlavors(
852         final String baseType) {
853 
854         final Set<DataFlavor> returnValue = new LinkedHashSet<>();
855 
856         String subType = null;
857 
858         try {
859             final MimeType mimeType = new MimeType(baseType);
860             subType = mimeType.getSubType();
861         } catch (MimeTypeParseException mtpe) {
862             // Cannot happen, since we checked all mappings
863             // on load from flavormap.properties.
864             assert(false);
865         }
866 
867         if (DataTransferer.doesSubtypeSupportCharset(subType, null)) {
868             if (TEXT_PLAIN_BASE_TYPE.equals(baseType))
869             {
870                 returnValue.add(DataFlavor.stringFlavor);
871             }
872 
873             for (String unicodeClassName : UNICODE_TEXT_CLASSES) {
874                 final String mimeType = baseType + ";charset=Unicode;class=" +
875                                             unicodeClassName;
876 
877                 final LinkedHashSet<String> mimeTypes =
878                     handleHtmlMimeTypes(baseType, mimeType);
879                 for (String mt : mimeTypes) {
880                     DataFlavor toAdd = null;
881                     try {
882                         toAdd = new DataFlavor(mt);
883                     } catch (ClassNotFoundException cannotHappen) {
884                     }
885                     returnValue.add(toAdd);
886                 }
887             }
888 
889             for (String charset : DataTransferer.standardEncodings()) {
890 
891                 for (String encodedTextClass : ENCODED_TEXT_CLASSES) {
892                     final String mimeType =
893                             baseType + ";charset=" + charset +
894                             ";class=" + encodedTextClass;
895 
896                     final LinkedHashSet<String> mimeTypes =
897                         handleHtmlMimeTypes(baseType, mimeType);
898 
899                     for (String mt : mimeTypes) {
900 
901                         DataFlavor df = null;
902 
903                         try {
904                             df = new DataFlavor(mt);
905                             // Check for equality to plainTextFlavor so
906                             // that we can ensure that the exact charset of
907                             // plainTextFlavor, not the canonical charset
908                             // or another equivalent charset with a
909                             // different name, is used.
910                             if (df.equals(DataFlavor.plainTextFlavor)) {
911                                 df = DataFlavor.plainTextFlavor;
912                             }
913                         } catch (ClassNotFoundException cannotHappen) {
914                         }
915 
916                         returnValue.add(df);
917                     }
918                 }
919             }
920 
921             if (TEXT_PLAIN_BASE_TYPE.equals(baseType))
922             {
923                 returnValue.add(DataFlavor.plainTextFlavor);
924             }
925         } else {
926             // Non-charset text natives should be treated as
927             // opaque, 8-bit data in any of its various
928             // representations.
929             for (String encodedTextClassName : ENCODED_TEXT_CLASSES) {
930                 DataFlavor toAdd = null;
931                 try {
932                     toAdd = new DataFlavor(baseType +
933                          ";class=" + encodedTextClassName);
934                 } catch (ClassNotFoundException cannotHappen) {
935                 }
936                 returnValue.add(toAdd);
937             }
938         }
939         return returnValue;
940     }

1295      * @return the decoded <code>DataFlavor</code>, or <code>null</code> if
1296      *         nat is not an encoded <code>String</code> native
1297      */
1298     public static DataFlavor decodeDataFlavor(String nat)
1299         throws ClassNotFoundException
1300     {
1301         String retval_str = SystemFlavorMap.decodeJavaMIMEType(nat);
1302         return (retval_str != null)

1308         Set<String> retval = null;

1317         }
