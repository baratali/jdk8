26  package java.awt.print;
27  
28  /**
29   * The <code>PrinterException</code> class and its subclasses are used
30   * to indicate that an exceptional condition has occurred in the print
31   * system.
32   */
33  
34  public class PrinterException extends Exception {
35  
36      /**
37       * Constructs a new <code>PrinterException</code> object
38       * without a detail message.
39       */
40      public PrinterException() {
41  
42      }
43  
44      /**
45       * Constructs a new <code>PrinterException</code> object
46       * with the specified detail message.
47       * @param msg the message to generate when a
48       * <code>PrinterException</code> is thrown
49       */
50      public PrinterException(String msg) {
51          super(msg);
52      }
53  }
