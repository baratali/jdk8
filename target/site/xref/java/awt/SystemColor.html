31  /**
32   * A class to encapsulate symbolic colors representing the color of
33   * native GUI objects on a system.  For systems which support the dynamic
34   * update of the system colors (when the user changes the colors)
35   * the actual RGB values of these symbolic colors will also change
36   * dynamically.  In order to compare the "current" RGB value of a
37   * <code>SystemColor</code> object with a non-symbolic Color object,
38   * <code>getRGB</code> should be used rather than <code>equals</code>.
39   * <p>
40   * Note that the way in which these system colors are applied to GUI objects
41   * may vary slightly from platform to platform since GUI objects may be
42   * rendered differently on each platform.
43   * <p>
44   * System color values may also be available through the <code>getDesktopProperty</code>
45   * method on <code>java.awt.Toolkit</code>.
46   *
47   * @see Toolkit#getDesktopProperty
48   *
49   * @author      Carl Quinn
50   * @author      Amy Fowler
51   */
52  public final class SystemColor extends Color implements java.io.Serializable {

465     /**
466      * Called from {@code <init>} and toolkit to update the above systemColors cache.
467      */
468     private static void updateSystemColors() {
469         if (!GraphicsEnvironment.isHeadless()) {
470             Toolkit.getDefaultToolkit().loadSystemColors(systemColors);
471         }
472         for (int i = 0; i < systemColors.length; i++) {
473             systemColorObjects[i].value = systemColors[i];
474         }
475     }

477     /**
478      * Creates a symbolic color that represents an indexed entry into system
479      * color cache. Used by above static system colors.
480      */
481     private SystemColor(byte index) {
482         super(systemColors[index]);
483         this.index = index;
484     }

486     /**
487      * Returns a string representation of this <code>Color</code>'s values.
488      * This method is intended to be used only for debugging purposes,
489      * and the content and format of the returned string may vary between
490      * implementations.
491      * The returned string may be empty but may not be <code>null</code>.
492      *
493      * @return  a string representation of this <code>Color</code>
494      */
495     public String toString() {
496         return getClass().getName() + "[i=" + (index) + "]";
497     }

499     /**
500      * The design of the {@code SystemColor} class assumes that
501      * the {@code SystemColor} object instances stored in the
502      * static final fields above are the only instances that can
503      * be used by developers.
504      * This method helps maintain those limits on instantiation
505      * by using the index stored in the value field of the
506      * serialized form of the object to replace the serialized
507      * object with the equivalent static object constant field
508      * of {@code SystemColor}.
509      * See the {@link #writeReplace} method for more information
510      * on the serialized form of these objects.
511      * @return one of the {@code SystemColor} static object
512      *         fields that refers to the same system color.
513      */
514     private Object readResolve() {
515         // The instances of SystemColor are tightly controlled and
516         // only the canonical instances appearing above as static
517         // constants are allowed.  The serial form of SystemColor
518         // objects stores the color index as the value.  Here we
519         // map that index back into the canonical instance.
520         return systemColorObjects[value];
521     }
522 
523     /**
524      * Returns a specialized version of the {@code SystemColor}
525      * object for writing to the serialized stream.
526      * @serialData
527      * The value field of a serialized {@code SystemColor} object
528      * contains the array index of the system color instead of the
529      * rgb data for the system color.
530      * This index is used by the {@link #readResolve} method to
531      * resolve the deserialized objects back to the original
532      * static constant versions to ensure unique instances of
533      * each {@code SystemColor} object.
534      * @return a proxy {@code SystemColor} object with its value
535      *         replaced by the corresponding system color index.
536      */
537     private Object writeReplace() throws ObjectStreamException
538     {
539         // we put an array index in the SystemColor.value while serialize
540         // to keep compatibility.
541         SystemColor color = new SystemColor((byte)index);
542         color.value = index;
543         return color;
544     }
