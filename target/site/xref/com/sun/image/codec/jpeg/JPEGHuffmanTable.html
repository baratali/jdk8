153     /**
154      * Creates a Huffman Table and initializes it.
155      * @param lengths lengths[k] = # of symbols with codes of length k
156      * bits; lengths[0] is ignored
157      * @param symbols symbols in order of increasing code length
158      * @exception IllegalArgumentException if the length of
159      * <code>lengths</code> is greater than 17
160      * @exception IllegalArgumentException if the length of
161      * <code>symbols</code> is greater than 256
162      * @exception IllegalArgumentException if any of the values in
163      * <code>lengths</code> or <code>symbols</code> is less than zero
164      */
165     public JPEGHuffmanTable( short lengths[], short symbols[] ) {
166                 if ( lengths.length > HUFF_MAX_LEN )
167                         throw new IllegalArgumentException( "lengths array is too long" );
168                 for (int i=1; i < lengths.length; i++)
169                         if (lengths[i] < 0)
170                                 throw new IllegalArgumentException
171                                         ( "Values in lengths array must be non-negative." );
172 
173 
174                 if ( symbols.length > HUFF_MAX_SYM )
175                         throw new IllegalArgumentException( "symbols array is too long" );
176                 for (int i=0; i < symbols.length; i++)
177                         if (symbols[i] < 0)
178                                 throw new IllegalArgumentException
179                                         ( "Values in symbols array must be non-negative." );
180 
181                 this.lengths = new short[lengths.length];
182                 this.symbols = new short[symbols.length];
183 
184                 System.arraycopy( lengths, 0, this.lengths, 0, lengths.length );
185                 System.arraycopy( symbols, 0, this.symbols, 0, symbols.length );
186 
187                 checkTable();
188     }
189 
190         /**
191          * This checks that the table they gave us isn't 'illegal' It
192          * checks that the symbol length counts are possible, and that
193          * they gave us at least enough symbols for the symbol length
194          * counts. Eventually this might check that there aren't duplicate
195          * symbols.
196          */
197         private void checkTable() {
198                 int numVals=2;
199                 int sum=0;
200                 for (int i=1; i<lengths.length; i++) {
201                         sum     += lengths[i];
202                         numVals -= lengths[i];
203                         numVals *= 2;
204                 }
205 
206                 //System.out.println("NumVals: " + numVals);
207                 //System.out.println("Sum: " + sum + " Symbols: " + symbols.length);
208 
209                 if (numVals < 0)
210                         throw new IllegalArgumentException
211                                 ("Invalid Huffman Table provided, lengths are incorrect.");
212 
213                 // I'll let them go if they gave us 'extra' symbols...
214                 if (sum > symbols.length)
215                         throw new IllegalArgumentException
216                                 ("Invalid Huffman Table provided, not enough symbols.");
217         }

219         /**
220          * Return a copy of the array containing the number of symbols
221          * for each length in the Huffman table.
222          * @return A short array where array[k] = # of symbols in the
223          *         table of length k. array[0] is unused
224          */
225     public short[] getLengths() {
226                 short[] ret = new short[ lengths.length];
227                 System.arraycopy( lengths, 0, ret, 0, lengths.length);
228                 return ret;
229     }
230 
231     /**
232          * Return an array containing the Huffman symbols arranged by
233          * increasing length.  To make use of this array you must refer
234          * the the lengths array.
235          * @return A short array of Huffman symbols
236      */
237     public short[] getSymbols() {
238                 short[] ret = new short[symbols.length];
239                 System.arraycopy( symbols, 0, ret, 0, symbols.length);
240                 return ret;
241     }
