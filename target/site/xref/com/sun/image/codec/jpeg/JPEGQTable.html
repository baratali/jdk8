21  /** Class to encapsulate the JPEG quantization tables.
22   * <p>
23   * Note that the classes in the com.sun.image.codec.jpeg package are not
24   * part of the core Java APIs.  They are a part of Sun's JDK and JRE
25   * distributions.  Although other licensees may choose to distribute these
26   * classes, developers cannot depend on their availability in non-Sun
27   * implementations.  We expect that equivalent functionality will eventually
28   * be available in a core API or standard extension.
29   * <p>
30   */
31  public class JPEGQTable {
32  
33          /** Quantization step for each coefficient in zig-zag order */
34          private int quantval[];
35  
36          /** The number of coefficients in a DCT block */
37          private static final byte QTABLESIZE = 64;
38  
39          /**
40           * This is the sample luminance quantization table given in the
41           * JPEG spec section K.1, expressed in zigzag order. The spec says
42           * that the values given produce "good" quality, and when divided
43           * by 2, "very good" quality.
44           */
45          public static final JPEGQTable StdLuminance = new JPEGQTable();
46          static {
47                  int [] lumVals = {
48                          16,   11,  12,  14,  12,  10,  16,  14,
49                          13,   14,  18,  17,  16,  19,  24,  40,
50                          26,   24,  22,  22,  24,  49,  35,  37,
51                          29,   40,  58,  51,  61,  60,  57,  51,
52                          56,   55,  64,  72,  92,  78,  64,  68,
53                          87,   69,  55,  56,  80, 109,  81,  87,
54                          95,   98, 103, 104, 103,  62,  77, 113,
55                          121, 112, 100, 120,  92, 101, 103,  99
56                  };
57  
58                  StdLuminance.quantval = lumVals;
59          }
60  
61          /**
62           * This is the sample luminance quantization table given in the
63           * JPEG spec section K.1, expressed in zigzag order. The spec says
64           * that the values given produce "good" quality, and when divided
65           * by 2, "very good" quality.
66           */
67          public static final JPEGQTable StdChrominance = new JPEGQTable();
68          static {
69                  int [] chromVals = {
70                          17,  18,  18,  24,  21,  24,  47,  26,
71                          26,  47,  99,  66,  56,  66,  99,  99,
72                          99,  99,  99,  99,  99,  99,  99,  99,
73                          99,  99,  99,  99,  99,  99,  99,  99,
74                          99,  99,  99,  99,  99,  99,  99,  99,
75                          99,  99,  99,  99,  99,  99,  99,  99,
76                          99,  99,  99,  99,  99,  99,  99,  99,
77                          99,  99,  99,  99,  99,  99,  99,  99
78                  };
79                  StdChrominance.quantval = chromVals;
80          }
81  
82  
83          /**
84           * Constructs an empty quantization table. This is used to create
85           * the Std Q-Tables.
86           */
87          private JPEGQTable() {
88                  quantval = new int[QTABLESIZE];
89          }
90  
91          /**
92           * Constructs an quantization table from the array that was
93           * passed.  The coefficents must be in zig-zag order. The array
94           * must be of length 64.
95           *  @param table the quantization table (this is copied).
96           */
97          public JPEGQTable( int table[] ) {
98                  if ( table.length != QTABLESIZE ) {
99                          throw new IllegalArgumentException
100                                 ("Quantization table is the wrong size.");
101                 } else {
102                         quantval = new int[QTABLESIZE];
103                         System.arraycopy( table, 0, quantval, 0, QTABLESIZE );
104                 }
105         }
106 
107 
108         /**
109          * Returns the current quantization table as an array of ints in
110          * zig zag order.
111          *  @return A copy of the contained quantization table.
112          */
113         public int[] getTable() {
114                 int[] table = new int[QTABLESIZE];
115                 System.arraycopy( quantval, 0, table, 0, QTABLESIZE );
116                 return table;
117         }
118 
119         /**
120          * Returns a new Quantization table where the values are
121          * multiplied by scaleFactor and then clamped to the range
122          * 1..32767 (or to 1..255 if forceBaseline is 'true'). <P>
123 
124          * Values less than one tend to improve the quality level of the
125          * table, and values greater than one degrade the quality level of
126          * the table.
127 
128          * @param scaleFactor the multiplication factor for the table
129          * @param forceBaseline if true the values will be clamped
130          * to the range  [1 .. 255]
131          * @return A new Q-Table that is a linear multiple of this Q-Table
132          */
133         public JPEGQTable getScaledInstance(float scaleFactor,
134                                                                                 boolean forceBaseline ) {
135                 long  max    = (forceBaseline)?255L:32767L;
136                 int []ret    = new int[QTABLESIZE];
137 
138                 for (int i=0; i<QTABLESIZE; i++ ) {
139                         long holder = (long)((quantval[i] * scaleFactor) + 0.5);
140 
141                         // limit to valid range
142                         if (holder <= 0L) holder = 1L;
143 
144                         // Max quantizer for 12 bits
145                         if (holder > max ) holder = max;
146 
147                         ret[i] = (int)holder;
148                 }
149                 return new JPEGQTable(ret);
150         }
151 }
