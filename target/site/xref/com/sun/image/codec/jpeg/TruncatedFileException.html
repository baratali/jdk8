17  package com.sun.image.codec.jpeg;
18  
19  import java.awt.image.Raster;
20  import java.awt.image.BufferedImage;
21  /**
22   * Signals that a truncated file was detected.  This object contains
23   * the Raster/BufferedImage that has the partially decoded image data
24   * in it.  There is no indication of the portion of the Raster that
25   * may or may not be good.
26   * <p>
27   * Note that the classes in the com.sun.image.codec.jpeg package are not
28   * part of the core Java APIs.  They are a part of Sun's JDK and JRE
29   * distributions.  Although other licensees may choose to distribute these
30   * classes, developers cannot depend on their availability in non-Sun
31   * implementations.  We expect that equivalent functionality will eventually
32   * be available in a core API or standard extension.
33   * <p>
34   *
35   * @author  Thomas DeWeese
36   * @see     JPEGImageDecoder
37   * @since   1.2
38   */
39  public
40  class TruncatedFileException extends RuntimeException {
41          private Raster                  ras = null;
42          private BufferedImage   bi  = null;
43  
44  
45      /**
46       * Constructs a <code>TruncatedFileException</code> with the
47       * partially decoded BufferedImage.
48       *
49       * @param   bi the partially decoded BufferedImage (may be null).
50       * @since   1.2
51       */
52      public TruncatedFileException(BufferedImage bi) {
53                  super("Premature end of input file");
54                  this.bi  = bi;
55                  this.ras = bi.getData();
56      }
57  
58      /**
59       * Constructs an <code>TruncatedFileException</code> with the
60       * partially decoded Raster
61       *
62       * @param   ras the partially decoded Raster (may be null).
63       * @since   1.2
64       */
65      public TruncatedFileException(Raster ras) {
66                  super("Premature end of input file");
67                  this.ras = ras;
68      }
69  
70          /** Allows access to the raster that was in the progress of being
71           * decoded may be null, it is likely to be only partially filled
72           * with image data.
73       * @since   1.2
74           */
75          public Raster getRaster() { return ras; }
76  
77          /** Allows access to the BufferedImage that was in the progress of
78           * being decoded, this may be null, it is likely to be only
79           * partially filled with image data.
80       * @since   1.2
81           */
82          public BufferedImage getBufferedImage() { return bi; }
83  }
