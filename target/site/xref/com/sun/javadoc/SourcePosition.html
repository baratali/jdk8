26  package com.sun.javadoc;
27  
28  import java.io.File;
29  
30  /**
31   * This interface describes a source position: filename, line number,
32   * and column number.
33   *
34   * @since 1.4
35   * @author Neal M Gafter
36   */
37  public interface SourcePosition {
38      /** The source file. Returns null if no file information is
39       *  available. */
40      File file();
41  
42      /** The line in the source file. The first line is numbered 1;
43       *  0 means no line number information is available. */
44      int line();
45  
46      /** The column in the source file. The first column is
47       *  numbered 1; 0 means no column information is available.
48       *  Columns count characters in the input stream; a tab
49       *  advances the column number to the next 8-column tab stop.
50       */
51      int column();
52  
53      /** Convert the source position to the form "Filename:line". */
54      String toString();
55  }
