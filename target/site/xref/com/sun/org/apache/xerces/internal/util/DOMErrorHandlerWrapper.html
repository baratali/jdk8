112     /**
113      * Reports a warning. Warnings are non-fatal and can be safely ignored
114      * by most applications.
115      *
116      * @param domain    The domain of the warning. The domain can be any
117      *                  string but is suggested to be a valid URI. The
118      *                  domain can be used to conveniently specify a web
119      *                  site location of the relevent specification or
120      *                  document pertaining to this warning.
121      * @param key       The warning key. This key can be any string and
122      *                  is implementation dependent.
123      * @param exception Exception.
124      *
125      * @throws XNIException Thrown to signal that the parser should stop
126      *                      parsing the document.
127      */
128 
129     public void warning(String domain, String key,
130                         XMLParseException exception) throws XNIException {
131         fDOMError.fSeverity = DOMError.SEVERITY_WARNING;
132         fDOMError.fException = exception;
133         // REVISIT: May need to lookup from DOMErrorTypeMap in the future.
134         fDOMError.fType = key;
135         fDOMError.fRelatedData = fDOMError.fMessage = exception.getMessage();
136         DOMLocatorImpl locator = fDOMError.fLocator;
137         if (locator != null) {
138             locator.fColumnNumber = exception.getColumnNumber();
139             locator.fLineNumber = exception.getLineNumber();
140             locator.fUtf16Offset = exception.getCharacterOffset();
141             locator.fUri = exception.getExpandedSystemId();
142             locator.fRelatedNode = fCurrentNode;
143         }
144         if (fDomErrorHandler != null) {
145             fDomErrorHandler.handleError(fDOMError);
146         }
147     } // warning(String,String,XMLParseException)

149     /**

185     /**
186      * Report a fatal error. Fatal errors usually occur when the document
187      * is not well-formed and signifies that the parser cannot continue
188      * normal operation.
189      * <p>
190      * <strong>Note:</strong> The error handler should <em>always</em>
191      * throw an <code>XNIException</code> from this method. This exception
192      * can either be the same exception that is passed as a parameter to
193      * the method or a new XNI exception object. If the registered error
194      * handler fails to throw an exception, the continuing operation of
195      * the parser is undetermined.
196      *
197      * @param domain    The domain of the fatal error. The domain can be
198      *                  any string but is suggested to be a valid URI. The
199      *                  domain can be used to conveniently specify a web
200      *                  site location of the relevent specification or
201      *                  document pertaining to this fatal error.
202      * @param key       The fatal error key. This key can be any string
203      *                  and is implementation dependent.
204      * @param exception Exception.
205      *
206      * @throws XNIException Thrown to signal that the parser should stop
207      *                      parsing the document.
208      */
209     public void fatalError(String domain, String key,
210                            XMLParseException exception) throws XNIException {
211         fDOMError.fSeverity = DOMError.SEVERITY_FATAL_ERROR;
212         fDOMError.fException = exception;
213         fErrorCode.setValues(domain, key);
214         String domErrorType = DOMErrorTypeMap.getDOMErrorType(fErrorCode);
215         fDOMError.fType = (domErrorType != null) ? domErrorType : key;
216         fDOMError.fRelatedData = fDOMError.fMessage = exception.getMessage();
217         DOMLocatorImpl locator = fDOMError.fLocator;
218         if (locator != null) {
219             locator.fColumnNumber = exception.getColumnNumber();
220             locator.fLineNumber = exception.getLineNumber();
221             locator.fUtf16Offset = exception.getCharacterOffset();
222             locator.fUri = exception.getExpandedSystemId();
223             locator.fRelatedNode = fCurrentNode;
224         }
225         if (fDomErrorHandler != null) {
226             fDomErrorHandler.handleError(fDOMError);
227         }
228     } // fatalError(String,String,XMLParseException)

231     public boolean handleError(DOMError error) {
232         printError(error);
233         return eStatus;
234     }

236     /** Prints the error message. */
237 
238     private void printError(DOMError error) {
239         int severity = error.getSeverity();
240         fOut.print("[");
241         if ( severity == DOMError.SEVERITY_WARNING) {
242             fOut.print("Warning");
243         } else if ( severity == DOMError.SEVERITY_ERROR) {
244             fOut.print("Error");
245         } else {
246             fOut.print("FatalError");
247             eStatus = false ; //REVISIT: Abort processing if fatal error, do we need to??
248         }
249         fOut.print("] ");
250         DOMLocator locator = error.getLocation();
251         if (locator != null) {
252             fOut.print(locator.getLineNumber());
253             fOut.print(":");
254             fOut.print(locator.getColumnNumber());
255             fOut.print(":");
256             fOut.print(locator.getByteOffset());
257             fOut.print(",");
258             fOut.print(locator.getUtf16Offset());
259             Node node = locator.getRelatedNode();
260             if (node != null) {
261                 fOut.print("[");
262                 fOut.print(node.getNodeName());
263                 fOut.print("]");
264             }
265             String systemId = locator.getUri();
266             if (systemId != null) {
267                 int index = systemId.lastIndexOf('/');
268                 if (index != -1)
269                     systemId = systemId.substring(index + 1);
270                 fOut.print(": ");
271                 fOut.print(systemId);
272             }
273 
274         }
275 
276         fOut.print(":");
277         fOut.print(error.getMessage());
278         fOut.println();
279         fOut.flush();
280 
281     } // printError(DOMError)
