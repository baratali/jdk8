25  /**
26   * This class defines the basic XML character properties. The data
27   * in this class can be used to verify that a character is a valid
28   * XML character or if the character is a space, name start, or name
29   * character.
30   * <p>

36   * <code>MASK_VALID</code>), yields the same results as calling the

1034     /**
1035      * Trims space characters as defined by production [3] in
1036      * the XML 1.0 specification from both ends of the given string.
1037      *
1038      * @param value the string to be trimmed
1039      * @return the given string with the space characters trimmed
1040      * from both ends
1041      */
1042     public static String trim(String value) {
1043         int start;
1044         int end;
1045         final int lengthMinusOne = value.length() - 1;
1046         for (start = 0; start <= lengthMinusOne; ++start) {
1047             if (!isSpace(value.charAt(start))) {
1048                 break;
1049             }
1050         }
1051         for (end = lengthMinusOne; end >= start; --end) {
1052             if (!isSpace(value.charAt(end))) {
1053                 break;
1054             }
1055         }
1056         if (start == 0 && end == lengthMinusOne) {
1057             return value;
1058         }
1059         if (start > lengthMinusOne) {
1060             return "";
1061         }
1062         return value.substring(start, end + 1);
1063     } // trim(String):String
