137     /**
138      * Scans public ID literal.
139      *
140      * [12] PubidLiteral ::= '"' PubidChar* '"' | "'" (PubidChar - "'")* "'"
141      * [13] PubidChar::= #x20 | #xD | #xA | [a-zA-Z0-9] | [-'()+,./:=?;!*#@$_%]
142      *
143      * The returned string is normalized according to the following rule,
144      * from http://www.w3.org/TR/REC-xml#dt-pubid:
145      *
146      * Before a match is attempted, all strings of white space in the public
147      * identifier must be normalized to single space characters (#x20), and
148      * leading and trailing white space must be removed.
149      *
150      * @param literal The string to fill in with the public ID literal.
151      * @return True on success.
152      *
153      * <strong>Note:</strong> This method uses fStringBuffer, anything in it at
154      * the time of calling is lost.
155      */
156     protected boolean scanPubidLiteral(XMLString literal)
157         throws IOException, XNIException
158     {
159         int quote = fEntityScanner.scanChar();
160         if (quote != '\'' && quote != '"') {
161             reportFatalError("QuoteRequiredInPublicID", null);
162             return false;
163         }
164 
165         fStringBuffer.clear();
166         // skip leading whitespace
167         boolean skipSpace = true;
168         boolean dataok = true;
169         while (true) {
170             int c = fEntityScanner.scanChar();
171             // REVISIT:  it could really only be \n or 0x20; all else is normalized, no?  - neilg
172             if (c == ' ' || c == '\n' || c == '\r' || c == 0x85 || c == 0x2028) {
173                 if (!skipSpace) {
174                     // take the first whitespace as a space and skip the others
175                     fStringBuffer.append(' ');
176                     skipSpace = true;
177                 }
178             }
179             else if (c == quote) {
180                 if (skipSpace) {
181                     // if we finished on a space let's trim it
182                     fStringBuffer.length--;
183                 }
184                 literal.setValues(fStringBuffer);
185                 break;
186             }
187             else if (XMLChar.isPubid(c)) {
188                 fStringBuffer.append((char)c);
189                 skipSpace = false;
190             }
191             else if (c == -1) {
192                 reportFatalError("PublicIDUnterminated", null);
193                 return false;
194             }
195             else {
196                 dataok = false;
197                 reportFatalError("InvalidCharInPublicID",
198                                  new Object[]{Integer.toHexString(c)});
199             }
200         }
201         return dataok;
202    }

204     /**
205      * Normalize whitespace in an XMLString converting all whitespace
206      * characters to space characters.
207      */
208     protected void normalizeWhitespace(XMLString value) {
209         int end = value.offset + value.length;
210         for (int i = value.offset; i < end; ++i) {
211             int c = value.ch[i];
212             if (XMLChar.isSpace(c)) {
213                 value.ch[i] = ' ';
214             }
215         }
216     }
217 
218     /**
219      * Normalize whitespace in an XMLString converting all whitespace
220      * characters to space characters.
221      */
222     protected void normalizeWhitespace(XMLString value, int fromIndex) {
223         int end = value.offset + value.length;
224         for (int i = value.offset + fromIndex; i < end; ++i) {
225             int c = value.ch[i];
226             if (XMLChar.isSpace(c)) {
227                 value.ch[i] = ' ';
228             }
229         }
230     }

232     /**
233      * Checks whether this string would be unchanged by normalization.
234      *
235      * @return -1 if the value would be unchanged by normalization,
236      * otherwise the index of the first whitespace character which
237      * would be transformed.
238      */
239     protected int isUnchangedByNormalization(XMLString value) {
240         int end = value.offset + value.length;
241         for (int i = value.offset; i < end; ++i) {
242             int c = value.ch[i];
243             if (XMLChar.isSpace(c)) {
244                 return i - value.offset;
245             }
246         }
247         return -1;
248     }
