133     private XMLSchemaFactory(boolean useServicesMechanism) {
134         fUseServicesMechanism = useServicesMechanism;
135         fErrorHandlerWrapper = new ErrorHandlerWrapper(DraconianErrorHandler.getInstance());
136         fDOMEntityResolverWrapper = new DOMEntityResolverWrapper();
137         fXMLGrammarPoolWrapper = new XMLGrammarPoolWrapper();
138         fXMLSchemaLoader.setFeature(SCHEMA_FULL_CHECKING, true);
139         fXMLSchemaLoader.setProperty(XMLGRAMMAR_POOL, fXMLGrammarPoolWrapper);
140         fXMLSchemaLoader.setEntityResolver(fDOMEntityResolverWrapper);
141         fXMLSchemaLoader.setErrorHandler(fErrorHandlerWrapper);

144         fSecurityManager = new XMLSecurityManager(true);
145         fXMLSchemaLoader.setProperty(SECURITY_MANAGER, fSecurityManager);
146 
147         fSecurityPropertyMgr = new XMLSecurityPropertyManager();
148         fXMLSchemaLoader.setProperty(XML_SECURITY_PROPERTY_MANAGER,
149                 fSecurityPropertyMgr);

152     /**
153      * <p>Is specified schema supported by this <code>SchemaFactory</code>?</p>
154      *
155      * @param schemaLanguage Specifies the schema language which the returned <code>SchemaFactory</code> will understand.
156      *    <code>schemaLanguage</code> must specify a <a href="#schemaLanguage">valid</a> schema language.
157      *
158      * @return <code>true</code> if <code>SchemaFactory</code> supports <code>schemaLanguage</code>, else <code>false</code>.
159      *
160      * @throws NullPointerException If <code>schemaLanguage</code> is <code>null</code>.
161      * @throws IllegalArgumentException If <code>schemaLanguage.length() == 0</code>
162      *   or <code>schemaLanguage</code> does not specify a <a href="#schemaLanguage">valid</a> schema language.
163      */
164     public boolean isSchemaLanguageSupported(String schemaLanguage) {
165         if (schemaLanguage == null) {
166             throw new NullPointerException(JAXPValidationMessageFormatter.formatMessage(fXMLSchemaLoader.getLocale(),
167                     "SchemaLanguageNull", null));
168         }
169         if (schemaLanguage.length() == 0) {
170             throw new IllegalArgumentException(JAXPValidationMessageFormatter.formatMessage(fXMLSchemaLoader.getLocale(),
171                     "SchemaLanguageLengthZero", null));
172         }
173         // only W3C XML Schema 1.0 is supported
174         return schemaLanguage.equals(XMLConstants.W3C_XML_SCHEMA_NS_URI);
175     }

197     public Schema newSchema( Source[] schemas ) throws SAXException {
198 
199         // this will let the loader store parsed Grammars into the pool.
200         XMLGrammarPoolImplExtension pool = new XMLGrammarPoolImplExtension();
201         fXMLGrammarPoolWrapper.setGrammarPool(pool);
202 
203         XMLInputSource[] xmlInputSources = new XMLInputSource[schemas.length];
204         InputStream inputStream;
205         Reader reader;
206         for( int i=0; i<schemas.length; i++ ) {
207             Source source = schemas[i];
208             if (source instanceof StreamSource) {
209                 StreamSource streamSource = (StreamSource) source;
210                 String publicId = streamSource.getPublicId();
211                 String systemId = streamSource.getSystemId();
212                 inputStream = streamSource.getInputStream();
213                 reader = streamSource.getReader();
214                 xmlInputSources[i] = new XMLInputSource(publicId, systemId, null);
215                 xmlInputSources[i].setByteStream(inputStream);
216                 xmlInputSources[i].setCharacterStream(reader);
217             }
218             else if (source instanceof SAXSource) {
219                 SAXSource saxSource = (SAXSource) source;
220                 InputSource inputSource = saxSource.getInputSource();
221                 if (inputSource == null) {
222                     throw new SAXException(JAXPValidationMessageFormatter.formatMessage(fXMLSchemaLoader.getLocale(),
223                             "SAXSourceNullInputSource", null));
224                 }
225                 xmlInputSources[i] = new SAXInputSource(saxSource.getXMLReader(), inputSource);
226             }
227             else if (source instanceof DOMSource) {
228                 DOMSource domSource = (DOMSource) source;
229                 Node node = domSource.getNode();
230                 String systemID = domSource.getSystemId();
231                 xmlInputSources[i] = new DOMInputSource(node, systemID);
232             }
233              else if (source instanceof StAXSource) {
234                 StAXSource staxSource = (StAXSource) source;
235                 XMLEventReader eventReader = staxSource.getXMLEventReader();
236                 if (eventReader != null) {
237                     xmlInputSources[i] = new StAXInputSource(eventReader);
238                 }
239                 else {
240                     xmlInputSources[i] = new StAXInputSource(staxSource.getXMLStreamReader());
241                 }
242             }
243             else if (source == null) {
244                 throw new NullPointerException(JAXPValidationMessageFormatter.formatMessage(fXMLSchemaLoader.getLocale(),
245                         "SchemaSourceArrayMemberNull", null));
246             }
247             else {
248                 throw new IllegalArgumentException(JAXPValidationMessageFormatter.formatMessage(fXMLSchemaLoader.getLocale(),
249                         "SchemaFactorySourceUnrecognized",
250                         new Object [] {source.getClass().getName()}));
251             }
252         }
253 
254         try {
255             fXMLSchemaLoader.loadGrammar(xmlInputSources);
256         }
257         catch (XNIException e) {
258             // this should have been reported to users already.
259             throw Util.toSAXException(e);
260         }
261         catch (IOException e) {
262             // this hasn't been reported, so do so now.
263             SAXParseException se = new SAXParseException(e.getMessage(),null,e);
264             fErrorHandler.error(se);
265             throw se; // and we must throw it.
266         }
267 
268         // Clear reference to grammar pool.
269         fXMLGrammarPoolWrapper.setGrammarPool(null);
270 
271         // Select Schema implementation based on grammar count.
272         final int grammarCount = pool.getGrammarCount();
273         AbstractXMLSchema schema = null;
274         if (grammarCount > 1) {
275             schema = new XMLSchema(new ReadOnlyGrammarPool(pool));
276         }
277         else if (grammarCount == 1) {
278             Grammar[] grammars = pool.retrieveInitialGrammarSet(XMLGrammarDescription.XML_SCHEMA);
279             schema = new SimpleXMLSchema(grammars[0]);
280         }
281         else {
282             schema = new EmptyXMLSchema();
283         }
284         propagateFeatures(schema);
285         propagateProperties(schema);
286         return schema;
287     }

297     public boolean getFeature(String name)
298         throws SAXNotRecognizedException, SAXNotSupportedException {
299         if (name == null) {
300             throw new NullPointerException(JAXPValidationMessageFormatter.formatMessage(fXMLSchemaLoader.getLocale(),
301                     "FeatureNameNull", null));
302         }
303         if (name.equals(XMLConstants.FEATURE_SECURE_PROCESSING)) {
304             return (fSecurityManager != null && fSecurityManager.isSecureProcessing());
305         }
306         try {
307             return fXMLSchemaLoader.getFeature(name);
308         }
309         catch (XMLConfigurationException e) {
310             String identifier = e.getIdentifier();
311             if (e.getType() == Status.NOT_RECOGNIZED) {
312                 throw new SAXNotRecognizedException(
313                         SAXMessageFormatter.formatMessage(fXMLSchemaLoader.getLocale(),
314                         "feature-not-recognized", new Object [] {identifier}));
315             }
316             else {
317                 throw new SAXNotSupportedException(
318                         SAXMessageFormatter.formatMessage(fXMLSchemaLoader.getLocale(),
319                         "feature-not-supported", new Object [] {identifier}));
320             }
321         }
322     }

324     public Object getProperty(String name)
325         throws SAXNotRecognizedException, SAXNotSupportedException {
326         if (name == null) {
327             throw new NullPointerException(JAXPValidationMessageFormatter.formatMessage(fXMLSchemaLoader.getLocale(),
328                     "ProperyNameNull", null));
329         }
330         if (name.equals(SECURITY_MANAGER)) {
331             return fSecurityManager;
332         }
333         else if (name.equals(XMLGRAMMAR_POOL)) {
334             throw new SAXNotSupportedException(
335                     SAXMessageFormatter.formatMessage(fXMLSchemaLoader.getLocale(),
336                     "property-not-supported", new Object [] {name}));
337         }
338         try {
339             return fXMLSchemaLoader.getProperty(name);
340         }
341         catch (XMLConfigurationException e) {
342             String identifier = e.getIdentifier();
343             if (e.getType() == Status.NOT_RECOGNIZED) {
344                 throw new SAXNotRecognizedException(
345                         SAXMessageFormatter.formatMessage(fXMLSchemaLoader.getLocale(),
346                         "property-not-recognized", new Object [] {identifier}));
347             }
348             else {
349                 throw new SAXNotSupportedException(
350                         SAXMessageFormatter.formatMessage(fXMLSchemaLoader.getLocale(),
351                         "property-not-supported", new Object [] {identifier}));
352             }
353         }
354     }

356     public void setFeature(String name, boolean value)
357         throws SAXNotRecognizedException, SAXNotSupportedException {
358         if (name == null) {
359             throw new NullPointerException(JAXPValidationMessageFormatter.formatMessage(fXMLSchemaLoader.getLocale(),
360                     "FeatureNameNull", null));
361         }
362         if (name.equals(XMLConstants.FEATURE_SECURE_PROCESSING)) {
363             if (System.getSecurityManager() != null && (!value)) {
364                 throw new SAXNotSupportedException(
365                         SAXMessageFormatter.formatMessage(null,
366                         "jaxp-secureprocessing-feature", null));
367             }
368 
369             fSecurityManager.setSecureProcessing(value);
370             if (value) {
371                 if (Constants.IS_JDK8_OR_ABOVE) {
372                     fSecurityPropertyMgr.setValue(XMLSecurityPropertyManager.Property.ACCESS_EXTERNAL_DTD,
373                             XMLSecurityPropertyManager.State.FSP, Constants.EXTERNAL_ACCESS_DEFAULT_FSP);
374                     fSecurityPropertyMgr.setValue(XMLSecurityPropertyManager.Property.ACCESS_EXTERNAL_SCHEMA,
375                             XMLSecurityPropertyManager.State.FSP, Constants.EXTERNAL_ACCESS_DEFAULT_FSP);
376                 }
377             }
378 
379             fXMLSchemaLoader.setProperty(SECURITY_MANAGER, fSecurityManager);
380             return;
381         } else if (name.equals(Constants.ORACLE_FEATURE_SERVICE_MECHANISM)) {
382             //in secure mode, let _useServicesMechanism be determined by the constructor
383             if (System.getSecurityManager() != null)
384                 return;
385         }
386         try {
387             fXMLSchemaLoader.setFeature(name, value);
388         }
389         catch (XMLConfigurationException e) {
390             String identifier = e.getIdentifier();
391             if (e.getType() == Status.NOT_RECOGNIZED) {
392                 throw new SAXNotRecognizedException(
393                         SAXMessageFormatter.formatMessage(fXMLSchemaLoader.getLocale(),
394                         "feature-not-recognized", new Object [] {identifier}));
395             }
396             else {
397                 throw new SAXNotSupportedException(
398                         SAXMessageFormatter.formatMessage(fXMLSchemaLoader.getLocale(),
399                         "feature-not-supported", new Object [] {identifier}));
400             }
401         }
402     }

404     public void setProperty(String name, Object object)
405         throws SAXNotRecognizedException, SAXNotSupportedException {
406         if (name == null) {
407             throw new NullPointerException(JAXPValidationMessageFormatter.formatMessage(fXMLSchemaLoader.getLocale(),
408                     "ProperyNameNull", null));
409         }
410         if (name.equals(SECURITY_MANAGER)) {
411             fSecurityManager = XMLSecurityManager.convert(object, fSecurityManager);
412             fXMLSchemaLoader.setProperty(SECURITY_MANAGER, fSecurityManager);
413             return;
414         } else if (name.equals(Constants.XML_SECURITY_PROPERTY_MANAGER)) {
415             if (object == null) {
416                 fSecurityPropertyMgr = new XMLSecurityPropertyManager();
417             } else {
418                 fSecurityPropertyMgr = (XMLSecurityPropertyManager)object;
419             }
420             fXMLSchemaLoader.setProperty(Constants.XML_SECURITY_PROPERTY_MANAGER, fSecurityPropertyMgr);
421             return;
422         }
423         else if (name.equals(XMLGRAMMAR_POOL)) {
424             throw new SAXNotSupportedException(
425                     SAXMessageFormatter.formatMessage(fXMLSchemaLoader.getLocale(),
426                     "property-not-supported", new Object [] {name}));
427         }
428         try {
429             //check if the property is managed by security manager
430             if (fSecurityManager == null ||
431                     !fSecurityManager.setLimit(name, XMLSecurityManager.State.APIPROPERTY, object)) {
432                 //check if the property is managed by security property manager
433                 if (fSecurityPropertyMgr == null ||
434                         !fSecurityPropertyMgr.setValue(name, XMLSecurityPropertyManager.State.APIPROPERTY, object)) {
435                     //fall back to the existing property manager
436                     fXMLSchemaLoader.setProperty(name, object);
437                 }
438             }
439         }
440         catch (XMLConfigurationException e) {
441             String identifier = e.getIdentifier();
442             if (e.getType() == Status.NOT_RECOGNIZED) {
443                 throw new SAXNotRecognizedException(
444                         SAXMessageFormatter.formatMessage(fXMLSchemaLoader.getLocale(),
445                         "property-not-recognized", new Object [] {identifier}));
446             }
447             else {
448                 throw new SAXNotSupportedException(
449                         SAXMessageFormatter.formatMessage(fXMLSchemaLoader.getLocale(),
450                         "property-not-supported", new Object [] {identifier}));
451             }
452         }
453     }

455     private void propagateFeatures(AbstractXMLSchema schema) {
456         schema.setFeature(XMLConstants.FEATURE_SECURE_PROCESSING, fSecurityManager != null);
457         schema.setFeature(Constants.ORACLE_FEATURE_SERVICE_MECHANISM, fUseServicesMechanism);
458         String[] features = fXMLSchemaLoader.getRecognizedFeatures();
459         for (int i = 0; i < features.length; ++i) {
460             boolean state = fXMLSchemaLoader.getFeature(features[i]);
461             schema.setFeature(features[i], state);
462         }
463     }
464 
465     private void propagateProperties(AbstractXMLSchema schema) {
466         String[] properties = fXMLSchemaLoader.getRecognizedProperties();
467         for (int i = 0; i < properties.length; ++i) {
468             Object state = fXMLSchemaLoader.getProperty(properties[i]);
469             schema.setProperty(properties[i], state);
470         }
471     }
