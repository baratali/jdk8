30  /**
31   * Reader for UCS-2 and UCS-4 encodings.
32   * (i.e., encodings from ISO-10646-UCS-(2|4)).

34   * @xerces.internal

36   * @author Neil Graham, IBM
37   *

41      //
42      // Constants
43      //
44  
45      /** Default byte buffer size (8192, larger than that of ASCIIReader
46       * since it's reasonable to surmise that the average UCS-4-encoded
47       * file should be 4 times as large as the average ASCII-encoded file).
48       */
49      public static final int DEFAULT_BUFFER_SIZE = 8192;
50  
51      public static final short UCS2LE = 1;
52      public static final short UCS2BE = 2;
53      public static final short UCS4LE = 4;
54      public static final short UCS4BE = 8;

60      /** Input stream. */
61      protected InputStream fInputStream;
62  
63      /** Byte buffer. */
64      protected byte[] fBuffer;

66      // what kind of data we're dealing with
67      protected short fEncoding;

73      /**
74       * Constructs an ASCII reader from the specified input stream
75       * using the default buffer size.  The Endian-ness and whether this is
76       * UCS-2 or UCS-4 needs also to be known in advance.
77       *
78       * @param inputStream The input stream.
79       * @param encoding One of UCS2LE, UCS2BE, UCS4LE or UCS4BE.
80       */
81      public UCSReader(InputStream inputStream, short encoding) {
82          this(inputStream, DEFAULT_BUFFER_SIZE, encoding);
83      } // <init>(InputStream, short)

85      /**
86       * Constructs an ASCII reader from the specified input stream
87       * and buffer size.  The Endian-ness and whether this is
88       * UCS-2 or UCS-4 needs also to be known in advance.
89       *
90       * @param inputStream The input stream.
91       * @param size        The initial buffer size.
92       * @param encoding One of UCS2LE, UCS2BE, UCS4LE or UCS4BE.
93       */
94      public UCSReader(InputStream inputStream, int size, short encoding) {
95          fInputStream = inputStream;
96          BufferAllocator ba = ThreadLocalBufferAllocator.getBufferAllocator();
97          fBuffer = ba.getByteBuffer(size);
98          if (fBuffer == null) {
99              fBuffer = new byte[size];
100         }
101         fEncoding = encoding;
102     } // <init>(InputStream,int,short)

108     /**
109      * Read a single character.  This method will block until a character is
110      * available, an I/O error occurs, or the end of the stream is reached.
111      *
112      * <p> Subclasses that intend to support efficient single-character input
113      * should override this method.
114      *
115      * @return     The character read, as an integer in the range 0 to 127
116      *             (<tt>0x00-0x7f</tt>), or -1 if the end of the stream has
117      *             been reached
118      *
119      * @exception  IOException  If an I/O error occurs
120      */
121     public int read() throws IOException {
122         int b0 = fInputStream.read() & 0xff;
123         if (b0 == 0xff)
124             return -1;
125         int b1 = fInputStream.read() & 0xff;
126         if (b1 == 0xff)
127             return -1;
128         if(fEncoding >=4) {
129             int b2 = fInputStream.read() & 0xff;
130             if (b2 == 0xff)
131                 return -1;
132             int b3 = fInputStream.read() & 0xff;
133             if (b3 == 0xff)
134                 return -1;
135             System.err.println("b0 is " + (b0 & 0xff) + " b1 " + (b1 & 0xff) + " b2 " + (b2 & 0xff) + " b3 " + (b3 & 0xff));
136             if (fEncoding == UCS4BE)
137                 return (b0<<24)+(b1<<16)+(b2<<8)+b3;
138             else
139                 return (b3<<24)+(b2<<16)+(b1<<8)+b0;
140         } else { // UCS-2
141             if (fEncoding == UCS2BE)
142                 return (b0<<8)+b1;
143             else
144                 return (b1<<8)+b0;
145         }
146     } // read():int
147 
148     /**
149      * Read characters into a portion of an array.  This method will block
150      * until some input is available, an I/O error occurs, or the end of the
151      * stream is reached.
152      *
153      * @param      ch     Destination buffer
154      * @param      offset Offset at which to start storing characters
155      * @param      length Maximum number of characters to read
156      *
157      * @return     The number of characters read, or -1 if the end of the
158      *             stream has been reached
159      *
160      * @exception  IOException  If an I/O error occurs
161      */
162     public int read(char ch[], int offset, int length) throws IOException {
163         int byteLength = length << ((fEncoding >= 4)?2:1);
164         if (byteLength > fBuffer.length) {
165             byteLength = fBuffer.length;
166         }
167         int count = fInputStream.read(fBuffer, 0, byteLength);
168         if(count == -1) return -1;
169         // try and make count be a multiple of the number of bytes we're looking for
170         if(fEncoding >= 4) { // BigEndian
171             // this looks ugly, but it avoids an if at any rate...
172             int numToRead = (4 - (count & 3) & 3);
173             for(int i=0; i<numToRead; i++) {
174                 int charRead = fInputStream.read();
175                 if(charRead == -1) { // end of input; something likely went wrong!A  Pad buffer with nulls.
176                     for (int j = i;j<numToRead; j++)
177                         fBuffer[count+j] = 0;
178                     break;
179                 } else {
180                     fBuffer[count+i] = (byte)charRead;
181                 }
182             }
183             count += numToRead;
184         } else {
185             int numToRead = count & 1;
186             if(numToRead != 0) {
187                 count++;
188                 int charRead = fInputStream.read();
189                 if(charRead == -1) { // end of input; something likely went wrong!A  Pad buffer with nulls.
190                     fBuffer[count] = 0;
191                 } else {
192                     fBuffer[count] = (byte)charRead;
193                 }
194             }
195         }
196 
197         // now count is a multiple of the right number of bytes
198         int numChars = count >> ((fEncoding >= 4)?2:1);
199         int curPos = 0;
200         for (int i = 0; i < numChars; i++) {
201             int b0 = fBuffer[curPos++] & 0xff;
202             int b1 = fBuffer[curPos++] & 0xff;
203             if(fEncoding >=4) {
204                 int b2 = fBuffer[curPos++] & 0xff;
205                 int b3 = fBuffer[curPos++] & 0xff;
206                 if (fEncoding == UCS4BE)
207                     ch[offset+i] = (char)((b0<<24)+(b1<<16)+(b2<<8)+b3);
208                 else
209                     ch[offset+i] = (char)((b3<<24)+(b2<<16)+(b1<<8)+b0);
210             } else { // UCS-2
211                 if (fEncoding == UCS2BE)
212                     ch[offset+i] = (char)((b0<<8)+b1);
213                 else
214                     ch[offset+i] = (char)((b1<<8)+b0);
215             }
216         }
217         return numChars;
218     } // read(char[],int,int)

220     /**
221      * Skip characters.  This method will block until some characters are
222      * available, an I/O error occurs, or the end of the stream is reached.
223      *
224      * @param  n  The number of characters to skip
225      *
226      * @return    The number of characters actually skipped
227      *
228      * @exception  IOException  If an I/O error occurs
229      */
230     public long skip(long n) throws IOException {
231         // charWidth will represent the number of bits to move
232         // n leftward to get num of bytes to skip, and then move the result rightward
233         // to get num of chars effectively skipped.
234         // The trick with &'ing, as with elsewhere in this dcode, is
235         // intended to avoid an expensive use of / that might not be optimized
236         // away.
237         int charWidth = (fEncoding >=4)?2:1;
238         long bytesSkipped = fInputStream.skip(n<<charWidth);
239         if((bytesSkipped & (charWidth | 1)) == 0) return bytesSkipped >> charWidth;
240         return (bytesSkipped >> charWidth) + 1;
241     } // skip(long):long

243     /**
244      * Tell whether this stream is ready to be read.
245      *
246      * @return True if the next read() is guaranteed not to block for input,
247      * false otherwise.  Note that returning false does not guarantee that the
248      * next read will block.
249      *
250      * @exception  IOException  If an I/O error occurs
251      */
252     public boolean ready() throws IOException {
253             return false;
254     } // ready()

257      * Tell whether this stream supports the mark() operation.
258      */
259     public boolean markSupported() {
260             return fInputStream.markSupported();
261     } // markSupported()

263     /**
264      * Mark the present position in the stream.  Subsequent calls to reset()
265      * will attempt to reposition the stream to this point.  Not all
266      * character-input streams support the mark() operation.
267      *
268      * @param  readAheadLimit  Limit on the number of characters that may be
269      *                         read while still preserving the mark.  After
270      *                         reading this many characters, attempting to
271      *                         reset the stream may fail.
272      *
273      * @exception  IOException  If the stream does not support mark(),
274      *                          or if some other I/O error occurs
275      */
276     public void mark(int readAheadLimit) throws IOException {
277             fInputStream.mark(readAheadLimit);
278     } // mark(int)
279 
280     /**
281      * Reset the stream.  If the stream has been marked, then attempt to
282      * reposition it at the mark.  If the stream has not been marked, then
283      * attempt to reset it in some way appropriate to the particular stream,
284      * for example by repositioning it to its starting point.  Not all
285      * character-input streams support the reset() operation, and some support
286      * reset() without supporting mark().
287      *
288      * @exception  IOException  If the stream has not been marked,
289      *                          or if the mark has been invalidated,
290      *                          or if the stream does not support reset(),
291      *                          or if some other I/O error occurs
292      */
293     public void reset() throws IOException {
294         fInputStream.reset();
295     } // reset()
296 
297     /**
298      * Close the stream.  Once a stream has been closed, further read(),
299      * ready(), mark(), or reset() invocations will throw an IOException.
300      * Closing a previously-closed stream, however, has no effect.
301      *
302      * @exception  IOException  If an I/O error occurs
303      */
304      public void close() throws IOException {
305          BufferAllocator ba = ThreadLocalBufferAllocator.getBufferAllocator();
306          ba.returnByteBuffer(fBuffer);
307          fBuffer = null;
308          fInputStream.close();
309      } // close()
