86  /**
87   * This class implements xni.grammars.XMLGrammarLoader.
88   * It also serves as implementation of xs.XSLoader interface and DOMConfiguration interface.
89   *
90   * This class is designed to interact either with a proxy for a user application
91   * which wants to preparse schemas, or with our own Schema validator.
92   * It is hoped that none of these "external" classes will therefore need to communicate directly
93   * with XSDHandler in future.
94   * <p>This class only knows how to make XSDHandler do its thing.
95   * The caller must ensure that all its properties (schemaLocation, JAXPSchemaSource
96   * etc.) have been properly set.
97   *
98   * @xerces.internal
99   *
100  * @author Neil Graham, IBM
101  * @version $Id: XMLSchemaLoader.java,v 1.10 2010-11-01 04:39:55 joehw Exp $
102  */

314     XMLSchemaLoader(SymbolTable symbolTable,
315             XMLErrorReporter errorReporter,
316             XMLEntityManager entityResolver,
317             XSGrammarBucket grammarBucket,
318             SubstitutionGroupHandler sHandler,
319             CMBuilder builder) {
320 
321         // store properties and features in configuration
322         fLoaderConfig.addRecognizedFeatures(RECOGNIZED_FEATURES);
323         fLoaderConfig.addRecognizedProperties(RECOGNIZED_PROPERTIES);
324         if (symbolTable != null){
325             fLoaderConfig.setProperty(SYMBOL_TABLE, symbolTable);
326         }
327 
328         if(errorReporter == null) {
329             errorReporter = new XMLErrorReporter ();
330             errorReporter.setLocale(fLocale);
331             errorReporter.setProperty(ERROR_HANDLER, new DefaultErrorHandler());
332 
333         }
334         fErrorReporter = errorReporter;
335         // make sure error reporter knows about schemas...
336         if(fErrorReporter.getMessageFormatter(XSMessageFormatter.SCHEMA_DOMAIN) == null) {
337             fErrorReporter.putMessageFormatter(XSMessageFormatter.SCHEMA_DOMAIN, new XSMessageFormatter());
338         }
339         fLoaderConfig.setProperty(ERROR_REPORTER, fErrorReporter);
340         fEntityManager = entityResolver;
341         // entity manager is null if XMLSchemaValidator creates the loader
342         if (fEntityManager != null){
343             fLoaderConfig.setProperty(ENTITY_MANAGER, fEntityManager);
344         }
345 
346         // by default augment PSVI (i.e. don't use declaration pool)
347         fLoaderConfig.setFeature(AUGMENT_PSVI, true);
348 
349         if(grammarBucket == null ) {
350             grammarBucket = new XSGrammarBucket();
351         }
352         fGrammarBucket = grammarBucket;
353         if(sHandler == null) {
354             sHandler = new SubstitutionGroupHandler(fGrammarBucket);
355         }
356         fSubGroupHandler = sHandler;
357 
358         if(builder == null) {
359             builder = new CMBuilder(fNodeFactory);
360         }
361         fCMBuilder = builder;
362         fSchemaHandler = new XSDHandler(fGrammarBucket);
363         if (fDeclPool != null) {
364             fDeclPool.reset();
365         }
366         fJAXPCache = new HashMap();
367 
368         fSettingsChanged = true;
369     }

436      *

438      * @param state     The state of the property.
439      *
440      * @throws XMLConfigurationException Thrown when a property is not
441      *                  recognized or cannot be set.
442      */
443     public void setProperty(String propertyId,
444             Object state) throws XMLConfigurationException {
445         fSettingsChanged = true;
446         fLoaderConfig.setProperty(propertyId, state);
447         if (propertyId.equals(JAXP_SCHEMA_SOURCE)) {
448             fJAXPSource = state;
449             fJAXPProcessed = false;
450         }
451         else if (propertyId.equals(XMLGRAMMAR_POOL)) {
452             fGrammarPool = (XMLGrammarPool)state;
453         }
454         else if (propertyId.equals(SCHEMA_LOCATION)) {
455             fExternalSchemas = (String)state;
456         }
457         else if (propertyId.equals(SCHEMA_NONS_LOCATION)) {
458             fExternalNoNSSchema = (String) state;
459         }
460         else if (propertyId.equals(LOCALE)) {
461             setLocale((Locale) state);
462         }
463         else if (propertyId.equals(ENTITY_RESOLVER)) {
464             fEntityManager.setProperty(ENTITY_RESOLVER, state);
465         }
466         else if (propertyId.equals(ERROR_REPORTER)) {
467             fErrorReporter = (XMLErrorReporter)state;
468             if (fErrorReporter.getMessageFormatter(XSMessageFormatter.SCHEMA_DOMAIN) == null) {
469                 fErrorReporter.putMessageFormatter(XSMessageFormatter.SCHEMA_DOMAIN, new XSMessageFormatter());
470             }
471         }
472         else if (propertyId.equals(XML_SECURITY_PROPERTY_MANAGER)) {
473             XMLSecurityPropertyManager spm = (XMLSecurityPropertyManager)state;
474             faccessExternalSchema = spm.getValue(XMLSecurityPropertyManager.Property.ACCESS_EXTERNAL_SCHEMA);
475         }
476     } // setProperty(String, Object)

526     /**
527      * Returns a Grammar object by parsing the contents of the
528      * entities pointed to by sources.
529      *
530      * @param source[]  the locations of the entity which forms
531      *                      the staring point of the grammars to be constructed
532      * @throws IOException  when a problem is encounted reading the entity
533      * @throws XNIException when a condition arises (such as a FatalError) that requires parsing
534      *                          of the entity be terminated
535      */
536     public void loadGrammar(XMLInputSource source[])
537     throws IOException, XNIException {
538         int numSource = source.length;
539         for (int i = 0; i < numSource; ++i) {
540             loadGrammar(source[i]);
541         }
542     }
543 
544     /**
545      * Returns a Grammar object by parsing the contents of the
546      * entity pointed to by source.
547      *
548      * @param source        the location of the entity which forms
549      *                          the starting point of the grammar to be constructed.
550      * @throws IOException      When a problem is encountered reading the entity
551      *          XNIException    When a condition arises (such as a FatalError) that requires parsing
552      *                              of the entity be terminated.
553      */
554     public Grammar loadGrammar(XMLInputSource source)
555     throws IOException, XNIException {
556 
557         // REVISIT: this method should have a namespace parameter specified by
558         // user. In this case we can easily detect if a schema asked to be loaded
559         // is already in the local cache.
560 
561         reset(fLoaderConfig);
562         fSettingsChanged = false;
563         XSDDescription desc = new XSDDescription();
564         desc.fContextType = XSDDescription.CONTEXT_PREPARSE;
565         desc.setBaseSystemId(source.getBaseSystemId());
566         desc.setLiteralSystemId( source.getSystemId());
567         // none of the other fields make sense for preparsing
568         Map locationPairs = new HashMap();
569         // Process external schema location properties.
570         // We don't call tokenizeSchemaLocationStr here, because we also want
571         // to check whether the values are valid URI.
572         processExternalHints(fExternalSchemas, fExternalNoNSSchema,
573                 locationPairs, fErrorReporter);
574         SchemaGrammar grammar = loadSchema(desc, source, locationPairs);
575 
576         if(grammar != null && fGrammarPool != null) {
577             fGrammarPool.cacheGrammars(XMLGrammarDescription.XML_SCHEMA, fGrammarBucket.getGrammars());
578             // NOTE: we only need to verify full checking in case the schema was not provided via JAXP
579             // since full checking already verified for all JAXP schemas
580             if(fIsCheckedFully && fJAXPCache.get(grammar) != grammar) {
581                 XSConstraints.fullSchemaChecking(fGrammarBucket, fSubGroupHandler, fCMBuilder, fErrorReporter);
582             }
583         }
584         return grammar;
585     } // loadGrammar(XMLInputSource):  Grammar

621     /** This method tries to resolve location of the given schema.
622      * The loader stores the namespace/location pairs in a hashtable (use "" as the
623      * namespace of absent namespace). When resolving an entity, loader first tries
624      * to find in the hashtable whether there is a value for that namespace,
625      * if so, pass that location value to the user-defined entity resolver.
626      *
627      * @param desc
628      * @param locationPairs
629      * @param entityResolver
630      * @return
631      * @throws IOException
632      */
633     public static XMLInputSource resolveDocument(XSDDescription desc, Map locationPairs,
634             XMLEntityResolver entityResolver) throws IOException {
635         String loc = null;
636         // we consider the schema location properties for import
637         if (desc.getContextType() == XSDDescription.CONTEXT_IMPORT ||
638                 desc.fromInstance()) {
639             // use empty string as the key for absent namespace
640             String namespace = desc.getTargetNamespace();
641             String ns = namespace == null ? XMLSymbols.EMPTY_STRING : namespace;
642             // get the location hint for that namespace
643             LocationArray tempLA = (LocationArray)locationPairs.get(ns);
644             if(tempLA != null)
645                 loc = tempLA.getFirstLocation();
646         }
647 
648         // if it's not import, or if the target namespace is not set
649         // in the schema location properties, use location hint
650         if (loc == null) {
651             String[] hints = desc.getLocationHints();
652             if (hints != null && hints.length > 0)
653                 loc = hints[0];
654         }
655 
656         String expandedLoc = XMLEntityManager.expandSystemId(loc, desc.getBaseSystemId(), false);
657         desc.setLiteralSystemId(loc);
658         desc.setExpandedSystemId(expandedLoc);
659         return entityResolver.resolveEntity(desc);
660     }

710     // this method takes a SchemaLocation string.
711     // If an error is encountered, false is returned;
712     // otherwise, true is returned.  In either case, locations
713     // is augmented to include as many tokens as possible.
714     // @param schemaStr     The schemaLocation string to tokenize
715     // @param locations     HashMap mapping namespaces to LocationArray objects holding lists of locaitons
716     // @return true if no problems; false if string could not be tokenized
717     public static boolean tokenizeSchemaLocationStr(String schemaStr, Map locations) {
718         if (schemaStr!= null) {
719             StringTokenizer t = new StringTokenizer(schemaStr, " \n\t\r");
720             String namespace, location;
721             while (t.hasMoreTokens()) {
722                 namespace = t.nextToken ();
723                 if (!t.hasMoreTokens()) {
724                     return false; // error!
725                 }
726                 location = t.nextToken();
727                 LocationArray la = ((LocationArray)locations.get(namespace));
728                 if(la == null) {
729                     la = new LocationArray();
730                     locations.put(namespace, la);
731                 }
732                 la.addLocation(location);
733             }
734         }
735         return true;
736     } // tokenizeSchemaLocation(String, HashMap):  boolean

738     /**
739      * Translate the various JAXP SchemaSource property types to XNI
740      * XMLInputSource.  Valid types are: String, org.xml.sax.InputSource,
741      * InputStream, File, or Object[] of any of previous types.
742      * REVISIT:  the JAXP 1.2 spec is less than clear as to whether this property
743      * should be available to imported schemas.  I have assumed
744      * that it should.  - NG
745      * Note: all JAXP schema files will be checked for full-schema validity if the feature was set up
746      *
747      */
748     private void processJAXPSchemaSource(Map locationPairs) throws IOException {
749         fJAXPProcessed = true;
750         if (fJAXPSource == null) {
751             return;
752         }
753 
754         Class componentType = fJAXPSource.getClass().getComponentType();
755         XMLInputSource xis = null;
756         String sid = null;
757         if (componentType == null) {
758             // Not an array
759             if(fJAXPSource instanceof InputStream ||
760                     fJAXPSource instanceof InputSource) {
761                 SchemaGrammar g = (SchemaGrammar)fJAXPCache.get(fJAXPSource);
762                 if(g != null) {
763                     fGrammarBucket.putGrammar(g);
764                     return;
765                 }
766             }
767             fXSDDescription.reset();
768             xis = xsdToXMLInputSource(fJAXPSource);
769             sid = xis.getSystemId();
770             fXSDDescription.fContextType = XSDDescription.CONTEXT_PREPARSE;
771             if (sid != null) {
772                 fXSDDescription.setBaseSystemId(xis.getBaseSystemId());
773                 fXSDDescription.setLiteralSystemId(sid);
774                 fXSDDescription.setExpandedSystemId(sid);
775                 fXSDDescription.fLocationHints = new String[]{sid};
776             }
777             SchemaGrammar g = loadSchema(fXSDDescription, xis, locationPairs);
778             // it is possible that we won't be able to resolve JAXP schema-source location
779             if (g != null){
780                 if(fJAXPSource instanceof InputStream ||
781                         fJAXPSource instanceof InputSource) {
782                     fJAXPCache.put(fJAXPSource, g);
783                     if(fIsCheckedFully) {
784                         XSConstraints.fullSchemaChecking(fGrammarBucket, fSubGroupHandler, fCMBuilder, fErrorReporter);
785                     }
786                 }
787                 fGrammarBucket.putGrammar(g);
788             }
789             return ;
790         } else if ( (componentType != Object.class) &&
791                 (componentType != String.class) &&
792                 (componentType != File.class) &&
793                 (componentType != InputStream.class) &&
794                 (componentType != InputSource.class)
795         ) {
796             // Not an Object[], String[], File[], InputStream[], InputSource[]
797             throw new XMLConfigurationException(
798                     Status.NOT_SUPPORTED, "\""+JAXP_SCHEMA_SOURCE+
799                     "\" property cannot have an array of type {"+componentType.getName()+
800                     "}. Possible types of the array supported are Object, String, File, "+
801             "InputStream, InputSource.");
802         }
803 
804         // JAXP spec. allow []s of type String, File, InputStream,
805         // InputSource also, apart from [] of type Object.
806         Object[] objArr = (Object[]) fJAXPSource;
807         //make local vector for storing targetn namespaces of schemasources specified in object arrays.
808         Vector jaxpSchemaSourceNamespaces = new Vector() ;
809         for (int i = 0; i < objArr.length; i++) {
810             if(objArr[i] instanceof InputStream ||
811                     objArr[i] instanceof InputSource) {
812                 SchemaGrammar g = (SchemaGrammar)fJAXPCache.get(objArr[i]);
813                 if (g != null) {
814                     fGrammarBucket.putGrammar(g);
815                     continue;
816                 }
817             }
818             fXSDDescription.reset();
819             xis = xsdToXMLInputSource(objArr[i]);
820             sid = xis.getSystemId();
821             fXSDDescription.fContextType = XSDDescription.CONTEXT_PREPARSE;
822             if (sid != null) {
823                 fXSDDescription.setBaseSystemId(xis.getBaseSystemId());
824                 fXSDDescription.setLiteralSystemId(sid);
825                 fXSDDescription.setExpandedSystemId(sid);
826                 fXSDDescription.fLocationHints = new String[]{sid};
827             }
828             String targetNamespace = null ;
829             // load schema
830             SchemaGrammar grammar = fSchemaHandler.parseSchema(xis,fXSDDescription, locationPairs);
831 
832             if(fIsCheckedFully) {
833                 XSConstraints.fullSchemaChecking(fGrammarBucket, fSubGroupHandler, fCMBuilder, fErrorReporter);
834             }
835             if(grammar != null){
836                 targetNamespace = grammar.getTargetNamespace() ;
837                 if(jaxpSchemaSourceNamespaces.contains(targetNamespace)){
838                     //when an array of objects is passed it is illegal to have two schemas that share same namespace.
839                     throw new java.lang.IllegalArgumentException(
840                             " When using array of Objects as the value of SCHEMA_SOURCE property , " +
841                     "no two Schemas should share the same targetNamespace. " );
842                 }
843                 else{
844                     jaxpSchemaSourceNamespaces.add(targetNamespace) ;
845                 }
846                 if(objArr[i] instanceof InputStream ||
847                         objArr[i] instanceof InputSource) {
848                     fJAXPCache.put(objArr[i], grammar);
849                 }
850                 fGrammarBucket.putGrammar(grammar);
851             }
852             else{
853                 //REVISIT: What should be the acutal behavior if grammar can't be loaded as specified in schema source?
854             }
855         }
856     }//processJAXPSchemaSource

858     private XMLInputSource xsdToXMLInputSource(
859             Object val)
860     {
861         if (val instanceof String) {
862             // String value is treated as a URI that is passed through the
863             // EntityResolver
864             String loc = (String) val;
865             fXSDDescription.reset();
866             fXSDDescription.setValues(null, loc, null, null);
867             XMLInputSource xis = null;
868             try {
869                 xis = fEntityManager.resolveEntity(fXSDDescription);
870             } catch (IOException ex) {
871                 fErrorReporter.reportError(XSMessageFormatter.SCHEMA_DOMAIN,
872                         "schema_reference.4",
873                         new Object[] { loc }, XMLErrorReporter.SEVERITY_ERROR);
874             }
875             if (xis == null) {
876                 // REVISIT: can this happen?
877                 // Treat value as a URI and pass in as systemId
878                 return new XMLInputSource(null, loc, null);
879             }
880             return xis;
881         } else if (val instanceof InputSource) {
882             return saxToXMLInputSource((InputSource) val);
883         } else if (val instanceof InputStream) {
884             return new XMLInputSource(null, null, null,
885                     (InputStream) val, null);
886         } else if (val instanceof File) {
887             File file = (File) val;
888             InputStream is = null;
889             try {
890                 is = new BufferedInputStream(new FileInputStream(file));
891             } catch (FileNotFoundException ex) {
892                 fErrorReporter.reportError(XSMessageFormatter.SCHEMA_DOMAIN,
893                         "schema_reference.4", new Object[] { file.toString() },
894                         XMLErrorReporter.SEVERITY_ERROR);
895             }
896             return new XMLInputSource(null, null, null, is, null);
897         }
898         throw new XMLConfigurationException(
899                 Status.NOT_SUPPORTED, "\""+JAXP_SCHEMA_SOURCE+
900                 "\" property cannot have a value of type {"+val.getClass().getName()+
901                 "}. Possible types of the value supported are String, File, InputStream, "+
902         "InputSource OR an array of these types.");
903     }

909         String publicId = sis.getPublicId();

913         if (charStream != null) {
914             return new XMLInputSource(publicId, systemId, null, charStream,

922         }
923 
924         return new XMLInputSource(publicId, systemId, null);

1345         XMLInputSource dom2xmlInputSource(LSInput is) {
1346         // need to wrap the LSInput with an XMLInputSource
1347         XMLInputSource xis = null;
1348 
1349         /**
1350          * An LSParser looks at inputs specified in LSInput in
1351          * the following order: characterStream, byteStream,
1352          * stringData, systemId, publicId. For consistency
1353          * have the same behaviour for XSLoader.
1354          */
1355 
1356         // check whether there is a Reader
1357         // according to DOM, we need to treat such reader as "UTF-16".
1358         if (is.getCharacterStream() != null) {
1359             xis = new XMLInputSource(is.getPublicId(), is.getSystemId(),
1360                     is.getBaseURI(), is.getCharacterStream(),
1361             "UTF-16");
1362         }
1363         // check whether there is an InputStream
1364         else if (is.getByteStream() != null) {
1365             xis = new XMLInputSource(is.getPublicId(), is.getSystemId(),
1366                     is.getBaseURI(), is.getByteStream(),
1367                     is.getEncoding());
1368         }
1369         // if there is a string data, use a StringReader
1370         // according to DOM, we need to treat such data as "UTF-16".
1371         else if (is.getStringData() != null && is.getStringData().length() != 0) {
1372             xis = new XMLInputSource(is.getPublicId(), is.getSystemId(),
1373                     is.getBaseURI(), new StringReader(is.getStringData()),
1374             "UTF-16");
1375         }
1376         // otherwise, just use the public/system/base Ids
1377         else {
1378             xis = new XMLInputSource(is.getPublicId(), is.getSystemId(),
1379                     is.getBaseURI());
1380         }
1381 
1382         return xis;
1383     }
