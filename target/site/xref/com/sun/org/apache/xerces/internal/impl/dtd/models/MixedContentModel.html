62  package com.sun.org.apache.xerces.internal.impl.dtd.models;
63  
64  import com.sun.org.apache.xerces.internal.xni.QName;
65  
66  import com.sun.org.apache.xerces.internal.impl.dtd.XMLContentSpec;
67  
68  /**
69   * MixedContentModel is a derivative of the abstract content model base
70   * class that handles the special case of mixed model elements. If an element
71   * is mixed model, it has PCDATA as its first possible content, followed
72   * by an alternation of the possible children. The children cannot have any
73   * numeration or order, so it must look like this:
74   * <pre>
75   *   &lt;!ELEMENT Foo ((#PCDATA|a|b|c|)*)&gt;
76   * </pre>
77   * So, all we have to do is to keep an array of the possible children and
78   * validate by just looking up each child being validated by looking it up
79   * in the list.
80   *
81   * @xerces.internal
82   *
83   */
84  public class MixedContentModel
85      implements ContentModelValidator {
86  
87      //
88      // Data
89      //
90  
91      /** The count of possible children that we have to deal with. */
92      private int fCount;
93  
94      /** The list of possible children that we have to accept. */
95      private QName fChildren[];
96  
97      /** The type of the children to support ANY. */
98      private int fChildrenType[];
99  
100     /* this is the EquivClassComparator object */
101     //private EquivClassComparator comparator = null;
102 
103     /**
104      * True if mixed content model is ordered. DTD mixed content models
105      * are <em>always</em> unordered.
106      */
107     private boolean fOrdered;
108 
109     //
110     // Constructors
111     //
112 
113     /**
114      * Constructs a mixed content model.
115      *
116      * @param children The list of allowed children.
117      * @param type The list of the types of the children.
118      * @param offset The start offset position in the children.
119      * @param length The child count.
120      * @param ordered True if content must be ordered.
121      */
122     public MixedContentModel(QName[] children, int[] type, int offset, int length , boolean ordered) {
123         // Make our own copy now, which is exactly the right size
124         fCount = length;
125         fChildren = new QName[fCount];
126         fChildrenType = new int[fCount];
127         for (int i = 0; i < fCount; i++) {
128             fChildren[i] = new QName(children[offset + i]);
129             fChildrenType[i] = type[offset + i];
130         }
131         fOrdered = ordered;
132 
133     }
134 
135     //
136     // ContentModelValidator methods
137     //
138 
139 
140     /**
141      * Check that the specified content is valid according to this
142      * content model. This method can also be called to do 'what if'
143      * testing of content models just to see if they would be valid.
144      * <p>
145      * A value of -1 in the children array indicates a PCDATA node. All other
146      * indexes will be positive and represent child elements. The count can be
147      * zero, since some elements have the EMPTY content model and that must be
148      * confirmed.
149      *
150      * @param children The children of this element.  Each integer is an index within
151      *                 the <code>StringPool</code> of the child element name.  An index
152      *                 of -1 is used to indicate an occurrence of non-whitespace character
153      *                 data.
154      * @param offset Offset into the array where the children starts.
155      * @param length The number of entries in the <code>children</code> array.
156      *
157      * @return The value -1 if fully valid, else the 0 based index of the child
158      *         that first failed. If the value returned is equal to the number
159      *         of children, then the specified children are valid but additional
160      *         content is required to reach a valid ending state.
161      *
162      */
163     public int validate(QName[] children, int offset, int length) {
164 
165         // must match order
166         if (fOrdered) {
167             int inIndex = 0;
168             for (int outIndex = 0; outIndex < length; outIndex++) {
169 
170                 // ignore mixed text
171                 final QName curChild = children[offset + outIndex];
172                 if (curChild.localpart == null) {
173                     continue;
174                 }
175 
176                 // element must match
177                 int type = fChildrenType[inIndex];
178                 if (type == XMLContentSpec.CONTENTSPECNODE_LEAF) {
179                     if (fChildren[inIndex].rawname != children[offset + outIndex].rawname) {
180                         return outIndex;
181                     }
182                 }
183                 else if (type == XMLContentSpec.CONTENTSPECNODE_ANY) {
184                     String uri = fChildren[inIndex].uri;
185                     if (uri != null && uri != children[outIndex].uri) {
186                         return outIndex;
187                     }
188                 }
189                 else if (type == XMLContentSpec.CONTENTSPECNODE_ANY_LOCAL) {
190                     if (children[outIndex].uri != null) {
191                         return outIndex;
192                     }
193                 }
194                 else if (type == XMLContentSpec.CONTENTSPECNODE_ANY_OTHER) {
195                     if (fChildren[inIndex].uri == children[outIndex].uri) {
196                         return outIndex;
197                     }
198                 }
199 
200                 // advance index
201                 inIndex++;
202             }
203         }
204 
205         // can appear in any order
206         else {
207             for (int outIndex = 0; outIndex < length; outIndex++)
208             {
209                 // Get the current child out of the source index
210                 final QName curChild = children[offset + outIndex];
211 
212                 // If its PCDATA, then we just accept that
213                 if (curChild.localpart == null)
214                     continue;
215 
216                 // And try to find it in our list
217                 int inIndex = 0;
218                 for (; inIndex < fCount; inIndex++)
219                 {
220                     int type = fChildrenType[inIndex];
221                     if (type == XMLContentSpec.CONTENTSPECNODE_LEAF) {
222                         if (curChild.rawname == fChildren[inIndex].rawname) {
223                             break;
224                         }
225                     }
226                     else if (type == XMLContentSpec.CONTENTSPECNODE_ANY) {
227                         String uri = fChildren[inIndex].uri;
228                         if (uri == null || uri == children[outIndex].uri) {
229                             break;
230                         }
231                     }
232                     else if (type == XMLContentSpec.CONTENTSPECNODE_ANY_LOCAL) {
233                         if (children[outIndex].uri == null) {
234                             break;
235                         }
236                     }
237                     else if (type == XMLContentSpec.CONTENTSPECNODE_ANY_OTHER) {
238                         if (fChildren[inIndex].uri != children[outIndex].uri) {
239                             break;
240                         }
241                     }
242                     // REVISIT: What about checking for multiple ANY matches?
243                     //          The content model ambiguity *could* be checked
244                     //          by the caller before constructing the mixed
245                     //          content model.
246                 }
247 
248                 // We did not find this one, so the validation failed
249                 if (inIndex == fCount)
250                     return outIndex;
251             }
252         }
253 
254         // Everything seems to be in order, so return success
255         return -1;
256     } // validate
257 
258 } // class MixedContentModel
