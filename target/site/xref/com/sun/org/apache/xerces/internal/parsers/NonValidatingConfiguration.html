51  /**
52   * This is the non validating parser configuration. It extends the basic
53   * configuration with the set of following parser components:
54   * Document scanner, DTD scanner, namespace binder, document handler.
55   * <p>
56   * Xerces parser that uses this configuration is <strong>not</strong> <a href="http://www.w3.org/TR/REC-xml#sec-conformance">conformant</a>
57   * non-validating XML processor, since conformant non-validating processor is required
58   * to process "all the declarations they read in the internal DTD subset ... must use the information in those declarations to normalize attribute values,
59   * include the replacement text of internal entities, and supply default attribute values".
60   *
61   * @author Elena Litani, IBM
62   * @version $Id: NonValidatingConfiguration.java,v 1.7 2010-11-01 04:40:09 joehw Exp $
63   */

461     /**
462      * Parses the document in a pull parsing fashion.
463      *
464      * @param complete True if the pull parser should parse the
465      *                 remaining document completely.
466      *
467      * @return True if there is more document to parse.
468      *
469      * @exception XNIException Any XNI exception, possibly wrapping
470      *                         another exception.
471      * @exception IOException  An IO exception from the parser, possibly
472      *                         from a byte stream or character stream
473      *                         supplied by the parser.
474      *
475      * @see #setInputSource
476      */
477     public boolean parse(boolean complete) throws XNIException, IOException {
478         //
479         // reset and configure pipeline and set InputSource.
480         if (fInputSource !=null) {
481             try {
482                 // resets and sets the pipeline.
483                 reset();
484                 fScanner.setInputSource(fInputSource);
485                 fInputSource = null;
486             }
487             catch (XNIException ex) {
488                 if (PRINT_EXCEPTION_STACK_TRACE)
489                     ex.printStackTrace();
490                 throw ex;
491             }
492             catch (IOException ex) {
493                 if (PRINT_EXCEPTION_STACK_TRACE)
494                     ex.printStackTrace();
495                 throw ex;
496             }
497             catch (RuntimeException ex) {
498                 if (PRINT_EXCEPTION_STACK_TRACE)
499                     ex.printStackTrace();
500                 throw ex;
501             }
502             catch (Exception ex) {
503                 if (PRINT_EXCEPTION_STACK_TRACE)
504                     ex.printStackTrace();
505                 throw new XNIException(ex);
506             }
507         }
508 
509         try {
510             return fScanner.scanDocument(complete);
511         }
512         catch (XNIException ex) {
513             if (PRINT_EXCEPTION_STACK_TRACE)
514                 ex.printStackTrace();
515             throw ex;
516         }
517         catch (IOException ex) {
518             if (PRINT_EXCEPTION_STACK_TRACE)
519                 ex.printStackTrace();
520             throw ex;
521         }
522         catch (RuntimeException ex) {
523             if (PRINT_EXCEPTION_STACK_TRACE)
524                 ex.printStackTrace();
525             throw ex;
526         }
527         catch (Exception ex) {
528             if (PRINT_EXCEPTION_STACK_TRACE)
529                 ex.printStackTrace();
530             throw new XNIException(ex);
531         }
532 
533     } // parse(boolean):boolean

556     public void parse(XMLInputSource source) throws XNIException, IOException {
557 
558         if (fParseInProgress) {
559             // REVISIT - need to add new error message
560             throw new XNIException("FWK005 parse may not be called while parsing.");
561         }
562         fParseInProgress = true;
563 
564         try {
565             setInputSource(source);
566             parse(true);
567         }
568         catch (XNIException ex) {
569             if (PRINT_EXCEPTION_STACK_TRACE)
570                 ex.printStackTrace();
571             throw ex;
572         }
573         catch (IOException ex) {
574             if (PRINT_EXCEPTION_STACK_TRACE)
575                 ex.printStackTrace();
576             throw ex;
577         }
578         catch (RuntimeException ex) {
579             if (PRINT_EXCEPTION_STACK_TRACE)

582         }
583         catch (Exception ex) {
584             if (PRINT_EXCEPTION_STACK_TRACE)
585                 ex.printStackTrace();
586             throw new XNIException(ex);
587         }
588         finally {
589             fParseInProgress = false;
590             // close all streams opened by xerces
591             this.cleanup();
592         }

615     /** Configures the pipeline. */
616     protected void configurePipeline() {
617         // create appropriate scanner
618         // and register it as one of the components.
619         if (fFeatures.get(NAMESPACES) == Boolean.TRUE) {
620             if (fNamespaceScanner == null) {
621                 fNamespaceScanner = new XMLNSDocumentScannerImpl();
622                 addComponent((XMLComponent)fNamespaceScanner);
623             }
624             fProperties.put(DOCUMENT_SCANNER, fNamespaceScanner);
625             fNamespaceScanner.setDTDValidator(null);
626             fScanner = fNamespaceScanner;
627         }
628         else {
629             if (fNonNSScanner == null) {
630                 fNonNSScanner = new XMLDocumentScannerImpl();
631                 addComponent((XMLComponent)fNonNSScanner);
632             }
633             fProperties.put(DOCUMENT_SCANNER, fNonNSScanner);
634             fScanner = fNonNSScanner;
635         }
636 
637         fScanner.setDocumentHandler(fDocumentHandler);
638         fLastComponent = fScanner;
639         // setup dtd pipeline
640         if (fDTDScanner != null) {
641                 fDTDScanner.setDTDHandler(fDTDHandler);
642                 fDTDScanner.setDTDContentModelHandler(fDTDContentModelHandler);
643         }
644 
645 
646     } // configurePipeline()

650     /**
651      * Check a feature. If feature is know and supported, this method simply
652      * returns. Otherwise, the appropriate exception is thrown.
653      *
654      * @param featureId The unique identifier (URI) of the feature.
655      *
656      * @throws XMLConfigurationException Thrown for configuration error.
657      *                                   In general, components should
658      *                                   only throw this exception if
659      *                                   it is <strong>really</strong>
660      *                                   a critical error.
661      */
662     protected FeatureState checkFeature(String featureId)
663         throws XMLConfigurationException {
664 
665         //
666         // Xerces Features
667         //
668 
669         if (featureId.startsWith(Constants.XERCES_FEATURE_PREFIX)) {
670             final int suffixLength = featureId.length() - Constants.XERCES_FEATURE_PREFIX.length();
671 
672             //
673             // http://apache.org/xml/features/validation/dynamic
674             //   Allows the parser to validate a document only when it
675             //   contains a grammar. Validation is turned on/off based
676             //   on each document instance, automatically.
677             //
678             if (suffixLength == Constants.DYNAMIC_VALIDATION_FEATURE.length() &&
679                 featureId.endsWith(Constants.DYNAMIC_VALIDATION_FEATURE)) {
680                 return FeatureState.RECOGNIZED;
681             }
682             //
683             // http://apache.org/xml/features/validation/default-attribute-values
684             //
685             if (suffixLength == Constants.DEFAULT_ATTRIBUTE_VALUES_FEATURE.length() &&

731      * Check a property. If the property is know and supported, this method
732      * simply returns. Otherwise, the appropriate exception is thrown.
733      *
734      * @param propertyId The unique identifier (URI) of the property
735      *                   being set.
736      *
737      * @throws XMLConfigurationException Thrown for configuration error.
738      *                                   In general, components should
739      *                                   only throw this exception if
740      *                                   it is <strong>really</strong>
741      *                                   a critical error.
742      */
743     protected PropertyState checkProperty(String propertyId)
744         throws XMLConfigurationException {
745 
746         //
747         // Xerces Properties
748         //
749 
750         if (propertyId.startsWith(Constants.XERCES_PROPERTY_PREFIX)) {
751             final int suffixLength = propertyId.length() - Constants.XERCES_PROPERTY_PREFIX.length();
752 
753             if (suffixLength == Constants.DTD_SCANNER_PROPERTY.length() &&
754                 propertyId.endsWith(Constants.DTD_SCANNER_PROPERTY)) {
755                 return PropertyState.RECOGNIZED;
756             }
757         }
758 
759         if (propertyId.startsWith(Constants.JAXP_PROPERTY_PREFIX)) {
760             final int suffixLength = propertyId.length() - Constants.JAXP_PROPERTY_PREFIX.length();
761 
762             if (suffixLength == Constants.SCHEMA_SOURCE.length() &&
763                 propertyId.endsWith(Constants.SCHEMA_SOURCE)) {
764                 return PropertyState.RECOGNIZED;
765             }
766         }
767 
768         //
769         // Not recognized
770         //
771 
772         return super.checkProperty(propertyId);
773 
774     } // checkProperty(String)
