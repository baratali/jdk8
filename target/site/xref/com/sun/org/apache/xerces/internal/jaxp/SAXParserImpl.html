126     /**
127      * Create a SAX parser with the associated features
128      * @param features Hashtable of SAX features, may be null
129      */
130     SAXParserImpl(SAXParserFactoryImpl spf, Hashtable features, boolean secureProcessing)
131         throws SAXException
132     {
133         fSecurityManager = new XMLSecurityManager(secureProcessing);
134         fSecurityPropertyMgr = new XMLSecurityPropertyManager();
135         // Instantiate a SAXParser directly and not through SAX so that we use the right ClassLoader
136         xmlReader = new JAXPSAXParser(this, fSecurityPropertyMgr, fSecurityManager);
137 
138         // JAXP "namespaceAware" == SAX Namespaces feature
139         // Note: there is a compatibility problem here with default values:
140         // JAXP default is false while SAX 2 default is true!
141         xmlReader.setFeature0(NAMESPACES_FEATURE, spf.isNamespaceAware());
142 
143         // SAX "namespaces" and "namespace-prefixes" features should not
144         // both be false.  We make them opposite for backward compatibility
145         // since JAXP 1.0 apps may want to receive xmlns* attributes.
146         xmlReader.setFeature0(NAMESPACE_PREFIXES_FEATURE, !spf.isNamespaceAware());
147 
148         // Avoid setting the XInclude processing feature if the value is false.
149         // This will keep the configuration from throwing an exception if it
150         // does not support XInclude.
151         if (spf.isXIncludeAware()) {
152             xmlReader.setFeature0(XINCLUDE_FEATURE, true);
153         }
154 
155         xmlReader.setProperty0(XML_SECURITY_PROPERTY_MANAGER, fSecurityPropertyMgr);
156 
157         xmlReader.setProperty0(SECURITY_MANAGER, fSecurityManager);
158 
159         if (secureProcessing) {
160             /**
161              * By default, secure processing is set, no external access is allowed.
162              * However, we need to check if it is actively set on the factory since we
163              * allow the use of the System Property or jaxp.properties to override
164              * the default value
165              */
166             if (features != null) {
167 
168                 Object temp = features.get(XMLConstants.FEATURE_SECURE_PROCESSING);
169                 if (temp != null) {
170                     boolean value = ((Boolean) temp).booleanValue();
171                     if (value && Constants.IS_JDK8_OR_ABOVE) {
172                         fSecurityPropertyMgr.setValue(XMLSecurityPropertyManager.Property.ACCESS_EXTERNAL_DTD,
173                                 XMLSecurityPropertyManager.State.FSP, Constants.EXTERNAL_ACCESS_DEFAULT_FSP);
174                         fSecurityPropertyMgr.setValue(XMLSecurityPropertyManager.Property.ACCESS_EXTERNAL_SCHEMA,
175                                 XMLSecurityPropertyManager.State.FSP, Constants.EXTERNAL_ACCESS_DEFAULT_FSP);
176 
177                     }
178                 }
179             }
180         }
181 
182         // Set application's features, followed by validation features.
183         setFeatures(features);
184 
185         // If validating, provide a default ErrorHandler that prints
186         // validation errors with a warning telling the user to set an
187         // ErrorHandler.
188         if (spf.isValidating()) {
189             fInitErrorHandler = new DefaultValidationErrorHandler(xmlReader.getLocale());
190             xmlReader.setErrorHandler(fInitErrorHandler);
191         }
192         else {
193             fInitErrorHandler = xmlReader.getErrorHandler();
194         }
195         xmlReader.setFeature0(VALIDATION_FEATURE, spf.isValidating());
196 
197         // Get the Schema object from the factory
198         this.grammar = spf.getSchema();
199         if (grammar != null) {
200             XMLParserConfiguration config = xmlReader.getXMLParserConfiguration();
201             XMLComponent validatorComponent = null;
202             /** For Xerces grammars, use built-in schema validator. **/
203             if (grammar instanceof XSGrammarPoolContainer) {
204                 validatorComponent = new XMLSchemaValidator();
205                 fSchemaValidationManager = new ValidationManager();
206                 fUnparsedEntityHandler = new UnparsedEntityHandler(fSchemaValidationManager);
207                 config.setDTDHandler(fUnparsedEntityHandler);
208                 fUnparsedEntityHandler.setDTDHandler(xmlReader);
209                 xmlReader.setDTDSource(fUnparsedEntityHandler);
210                 fSchemaValidatorComponentManager = new SchemaValidatorConfiguration(config,
211                         (XSGrammarPoolContainer) grammar, fSchemaValidationManager);
212             }
213             /** For third party grammars, use the JAXP validator component. **/
214             else {
215                 validatorComponent = new JAXPValidatorComponent(grammar.newValidatorHandler());
216                 fSchemaValidationManager = null;
217                 fUnparsedEntityHandler = null;
218                 fSchemaValidatorComponentManager = config;
219             }
220             config.addRecognizedFeatures(validatorComponent.getRecognizedFeatures());
221             config.addRecognizedProperties(validatorComponent.getRecognizedProperties());
222             config.setDocumentHandler((XMLDocumentHandler) validatorComponent);
223             ((XMLDocumentSource)validatorComponent).setDocumentHandler(xmlReader);
224             xmlReader.setDocumentSource((XMLDocumentSource) validatorComponent);
225             fSchemaValidator = validatorComponent;
226         }
227         else {
228             fSchemaValidationManager = null;
229             fUnparsedEntityHandler = null;
230             fSchemaValidatorComponentManager = null;
231             fSchemaValidator = null;
232         }
233 
234         // Initial EntityResolver
235         fInitEntityResolver = xmlReader.getEntityResolver();
236     }

238     /**
239      * Set any features of our XMLReader based on any features set on the
240      * SAXParserFactory.
241      *
242      * XXX Does not handle possible conflicts between SAX feature names and
243      * JAXP specific feature names, eg. SAXParserFactory.isValidating()
244      */
245     private void setFeatures(Hashtable features)
246         throws SAXNotSupportedException, SAXNotRecognizedException {
247         if (features != null) {
248             Iterator entries = features.entrySet().iterator();
249             while (entries.hasNext()) {
250                 Map.Entry entry = (Map.Entry) entries.next();
251                 String feature = (String) entry.getKey();
252                 boolean value = ((Boolean) entry.getValue()).booleanValue();
253                 xmlReader.setFeature0(feature, value);
254             }
255         }
256     }

355     public void reset() {
356         try {
357             /** Restore initial values of features and properties. **/
358             xmlReader.restoreInitState();
359         }
360         catch (SAXException exc) {
361             // This should never happen. We only store recognized
362             // features and properties in the hash maps. For now
363             // just ignore it.
364         }
365         /** Restore various handlers. **/
366         xmlReader.setContentHandler(null);
367         xmlReader.setDTDHandler(null);
368         if (xmlReader.getErrorHandler() != fInitErrorHandler) {
369             xmlReader.setErrorHandler(fInitErrorHandler);
370         }
371         if (xmlReader.getEntityResolver() != fInitEntityResolver) {
372             xmlReader.setEntityResolver(fInitEntityResolver);
373         }
374     }

397     public static class JAXPSAXParser extends com.sun.org.apache.xerces.internal.parsers.SAXParser {

399         private final HashMap fInitFeatures = new HashMap();
400         private final HashMap fInitProperties = new HashMap();
401         private final SAXParserImpl fSAXParser;
402         private XMLSecurityManager fSecurityManager;
403         private XMLSecurityPropertyManager fSecurityPropertyMgr;

410         JAXPSAXParser(SAXParserImpl saxParser, XMLSecurityPropertyManager securityPropertyMgr,
411                 XMLSecurityManager securityManager) {
412             super();
413             fSAXParser = saxParser;
414             fSecurityManager = securityManager;
415             fSecurityPropertyMgr = securityPropertyMgr;
416             /**
417              * This class may be used directly. So initialize the security manager if
418              * it is null.
419              */
420             if (fSecurityManager == null) {
421                 fSecurityManager = new XMLSecurityManager(true);
422                 try {
423                     super.setProperty(SECURITY_MANAGER, fSecurityManager);
424                 } catch (SAXException e) {
425                     throw new UnsupportedOperationException(
426                     SAXMessageFormatter.formatMessage(fConfiguration.getLocale(),
427                     "property-not-recognized", new Object [] {SECURITY_MANAGER}), e);
428                 }
429             }
430             if (fSecurityPropertyMgr == null) {
431                 fSecurityPropertyMgr = new XMLSecurityPropertyManager();
432                 try {
433                     super.setProperty(XML_SECURITY_PROPERTY_MANAGER, fSecurityPropertyMgr);
434                 } catch (SAXException e) {
435                     throw new UnsupportedOperationException(
436                     SAXMessageFormatter.formatMessage(fConfiguration.getLocale(),
437                     "property-not-recognized", new Object [] {SECURITY_MANAGER}), e);
438                 }
439             }
440         }

442         /**
443          * Override SAXParser's setFeature method to track the initial state
444          * of features. This keeps us from affecting the performance of the
445          * SAXParser when it is created with XMLReaderFactory.
446          */
447         public synchronized void setFeature(String name, boolean value)
448             throws SAXNotRecognizedException, SAXNotSupportedException {
449             if (name == null) {
450                 // TODO: Add localized error message.
451                 throw new NullPointerException();
452             }
453             if (name.equals(XMLConstants.FEATURE_SECURE_PROCESSING)) {
454                 try {
455                     fSecurityManager.setSecureProcessing(value);
456                     setProperty(SECURITY_MANAGER, fSecurityManager);
457                 }
458                 catch (SAXNotRecognizedException exc) {
459                     // If the property is not supported
460                     // re-throw the exception if the value is true.
461                     if (value) {
462                         throw exc;
463                     }
464                 }
465                 catch (SAXNotSupportedException exc) {
466                     // If the property is not supported
467                     // re-throw the exception if the value is true.
468                     if (value) {
469                         throw exc;
470                     }
471                 }
472                 return;
473             }
474             if (!fInitFeatures.containsKey(name)) {
475                 boolean current = super.getFeature(name);
476                 fInitFeatures.put(name, current ? Boolean.TRUE : Boolean.FALSE);
477             }
478             /** Forward feature to the schema validator if there is one. **/
479             if (fSAXParser != null && fSAXParser.fSchemaValidator != null) {
480                 setSchemaValidatorFeature(name, value);
481             }
482             super.setFeature(name, value);
483         }

485         public synchronized boolean getFeature(String name)
486             throws SAXNotRecognizedException, SAXNotSupportedException {
487             if (name == null) {
488                 // TODO: Add localized error message.
489                 throw new NullPointerException();
490             }
491             if (name.equals(XMLConstants.FEATURE_SECURE_PROCESSING)) {
492                 return fSecurityManager.isSecureProcessing();
493             }
494             return super.getFeature(name);
495         }

497         /**
498          * Override SAXParser's setProperty method to track the initial state
499          * of properties. This keeps us from affecting the performance of the
500          * SAXParser when it is created with XMLReaderFactory.
501          */
502         public synchronized void setProperty(String name, Object value)
503             throws SAXNotRecognizedException, SAXNotSupportedException {
504             if (name == null) {
505                 // TODO: Add localized error message.
506                 throw new NullPointerException();
507             }
508             if (fSAXParser != null) {
509                 // JAXP 1.2 support
510                 if (JAXP_SCHEMA_LANGUAGE.equals(name)) {
511                     // The spec says if a schema is given via SAXParserFactory
512                     // the JAXP 1.2 properties shouldn't be allowed.
513                     if (fSAXParser.grammar != null) {
514                         throw new SAXNotSupportedException(
515                                 SAXMessageFormatter.formatMessage(fConfiguration.getLocale(), "schema-already-specified", new Object[] {name}));
516                     }
517                     if ( W3C_XML_SCHEMA.equals(value) ) {
518                         //None of the properties will take effect till the setValidating(true) has been called
519                         if( fSAXParser.isValidating() ) {
520                             fSAXParser.schemaLanguage = W3C_XML_SCHEMA;
521                             setFeature(XMLSCHEMA_VALIDATION_FEATURE, true);
522                             // this will allow the parser not to emit DTD-related
523                             // errors, as the spec demands
524                             if (!fInitProperties.containsKey(JAXP_SCHEMA_LANGUAGE)) {
525                                 fInitProperties.put(JAXP_SCHEMA_LANGUAGE, super.getProperty(JAXP_SCHEMA_LANGUAGE));
526                             }
527                             super.setProperty(JAXP_SCHEMA_LANGUAGE, W3C_XML_SCHEMA);
528                         }
529 
530                     }
531                     else if (value == null) {
532                         fSAXParser.schemaLanguage = null;
533                         setFeature(XMLSCHEMA_VALIDATION_FEATURE, false);
534                     }
535                     else {
536                         // REVISIT: It would be nice if we could format this message
537                         // using a user specified locale as we do in the underlying
538                         // XMLReader -- mrglavas
539                         throw new SAXNotSupportedException(
540                             SAXMessageFormatter.formatMessage(fConfiguration.getLocale(), "schema-not-supported", null));
541                     }
542                     return;
543                 }
544                 else if (JAXP_SCHEMA_SOURCE.equals(name)) {
545                     // The spec says if a schema is given via SAXParserFactory
546                     // the JAXP 1.2 properties shouldn't be allowed.
547                     if (fSAXParser.grammar != null) {
548                         throw new SAXNotSupportedException(
549                                 SAXMessageFormatter.formatMessage(fConfiguration.getLocale(), "schema-already-specified", new Object[] {name}));
550                     }
551                     String val = (String)getProperty(JAXP_SCHEMA_LANGUAGE);
552                     if ( val != null && W3C_XML_SCHEMA.equals(val) ) {
553                         if (!fInitProperties.containsKey(JAXP_SCHEMA_SOURCE)) {
554                             fInitProperties.put(JAXP_SCHEMA_SOURCE, super.getProperty(JAXP_SCHEMA_SOURCE));
555                         }
556                         super.setProperty(name, value);
557                     }
558                     else {
559                         throw new SAXNotSupportedException(
560                             SAXMessageFormatter.formatMessage(fConfiguration.getLocale(),
561                             "jaxp-order-not-supported",
562                             new Object[] {JAXP_SCHEMA_LANGUAGE, JAXP_SCHEMA_SOURCE}));
563                     }
564                     return;
565                 }
566             }
567             /** Forward property to the schema validator if there is one. **/
568             if (fSAXParser != null && fSAXParser.fSchemaValidator != null) {
569                 setSchemaValidatorProperty(name, value);
570             }
571 
572             //check if the property is managed by security manager
573             if (fSecurityManager == null ||
574                     !fSecurityManager.setLimit(name, XMLSecurityManager.State.APIPROPERTY, value)) {
575                 //check if the property is managed by security property manager
576                 if (fSecurityPropertyMgr == null ||
577                         !fSecurityPropertyMgr.setValue(name, XMLSecurityPropertyManager.State.APIPROPERTY, value)) {
578                     //fall back to the existing property manager
579                     if (!fInitProperties.containsKey(name)) {
580                         fInitProperties.put(name, super.getProperty(name));
581                     }
582                     super.setProperty(name, value);
583                 }
584             }
585 
586         }

588         public synchronized Object getProperty(String name)
589             throws SAXNotRecognizedException, SAXNotSupportedException {
590             if (name == null) {
591                 // TODO: Add localized error message.
592                 throw new NullPointerException();
593             }
594             if (fSAXParser != null && JAXP_SCHEMA_LANGUAGE.equals(name)) {
595                 // JAXP 1.2 support
596                 return fSAXParser.schemaLanguage;
597             }
598 
599             /** Check to see if the property is managed by the security manager **/
600             String propertyValue = (fSecurityManager != null) ?
601                     fSecurityManager.getLimitAsString(name) : null;
602             if (propertyValue != null) {
603                 return propertyValue;
604             } else {
605                 propertyValue = (fSecurityPropertyMgr != null) ?
606                     fSecurityPropertyMgr.getValue(name) : null;
607                 if (propertyValue != null) {
608                     return propertyValue;
609                 }
610             }
611 
612             return super.getProperty(name);
613         }

615         synchronized void restoreInitState()
616             throws SAXNotRecognizedException, SAXNotSupportedException {
617             Iterator iter;
618             if (!fInitFeatures.isEmpty()) {
619                 iter = fInitFeatures.entrySet().iterator();
620                 while (iter.hasNext()) {
621                     Map.Entry entry = (Map.Entry) iter.next();
622                     String name = (String) entry.getKey();
623                     boolean value = ((Boolean) entry.getValue()).booleanValue();
624                     super.setFeature(name, value);
625                 }
626                 fInitFeatures.clear();
627             }
628             if (!fInitProperties.isEmpty()) {
629                 iter = fInitProperties.entrySet().iterator();
630                 while (iter.hasNext()) {
631                     Map.Entry entry = (Map.Entry) iter.next();
632                     String name = (String) entry.getKey();
633                     Object value = entry.getValue();
634                     super.setProperty(name, value);
635                 }
636                 fInitProperties.clear();
637             }
638         }

640         public void parse(InputSource inputSource)
641             throws SAXException, IOException {
642             if (fSAXParser != null && fSAXParser.fSchemaValidator != null) {
643                 if (fSAXParser.fSchemaValidationManager != null) {
644                     fSAXParser.fSchemaValidationManager.reset();
645                     fSAXParser.fUnparsedEntityHandler.reset();
646                 }
