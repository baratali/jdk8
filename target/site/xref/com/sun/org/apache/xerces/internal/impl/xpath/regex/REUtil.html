33      static final int composeFromSurrogates(int high, int low) {
34          return 0x10000 + ((high-0xd800)<<10) + low-0xdc00;
35      }
36  
37      static final boolean isLowSurrogate(int ch) {
38          return (ch & 0xfc00) == 0xdc00;
39      }
40  
41      static final boolean isHighSurrogate(int ch) {
42          return (ch & 0xfc00) == 0xd800;
43      }
44  
45      static final String decomposeToSurrogates(int ch) {
46          char[] chs = new char[2];
47          ch -= 0x10000;
48          chs[0] = (char)((ch>>10)+0xd800);
49          chs[1] = (char)((ch&0x3ff)+0xdc00);
50          return new String(chs);
51      }

53      static final String substring(CharacterIterator iterator, int begin, int end) {
54          char[] src = new char[end-begin];
55          for (int i = 0;  i < src.length;  i ++)
56              src[i] = iterator.setIndex(i+begin);
57          return new String(src);
58      }

62      static final int getOptionValue(int ch) {
63          int ret = 0;
64          switch (ch) {
65            case 'i':
66              ret = RegularExpression.IGNORE_CASE;
67              break;
68            case 'm':
69              ret = RegularExpression.MULTIPLE_LINES;
70              break;
71            case 's':
72              ret = RegularExpression.SINGLE_LINE;
73              break;
74            case 'x':
75              ret = RegularExpression.EXTENDED_COMMENT;
76              break;
77            case 'u':
78              ret = RegularExpression.USE_UNICODE_CATEGORY;
79              break;
80            case 'w':
81              ret = RegularExpression.UNICODE_WORD_BOUNDARY;
82              break;
83            case 'F':
84              ret = RegularExpression.PROHIBIT_FIXED_STRING_OPTIMIZATION;
85              break;
86            case 'H':
87              ret = RegularExpression.PROHIBIT_HEAD_CHARACTER_OPTIMIZATION;
88              break;
89            case 'X':
90              ret = RegularExpression.XMLSCHEMA_MODE;
91              break;
92            case ',':
93              ret = RegularExpression.SPECIAL_COMMA;
94              break;
95            default:
96          }
97          return ret;
98      }

100     static final int parseOptions(String opts) throws ParseException {
101         if (opts == null)  return 0;
102         int options = 0;
103         for (int i = 0;  i < opts.length();  i ++) {
104             int v = getOptionValue(opts.charAt(i));
105             if (v == 0)
106                 throw new ParseException("Unknown Option: "+opts.substring(i), -1);
107             options |= v;
108         }
109         return options;
110     }

112     static final String createOptionString(int options) {
113         StringBuffer sb = new StringBuffer(9);
114         if ((options & RegularExpression.PROHIBIT_FIXED_STRING_OPTIMIZATION) != 0)
115             sb.append((char)'F');
116         if ((options & RegularExpression.PROHIBIT_HEAD_CHARACTER_OPTIMIZATION) != 0)
117             sb.append((char)'H');
118         if ((options & RegularExpression.XMLSCHEMA_MODE) != 0)
119             sb.append((char)'X');
120         if ((options & RegularExpression.IGNORE_CASE) != 0)
121             sb.append((char)'i');
122         if ((options & RegularExpression.MULTIPLE_LINES) != 0)
123             sb.append((char)'m');
124         if ((options & RegularExpression.SINGLE_LINE) != 0)
125             sb.append((char)'s');
126         if ((options & RegularExpression.USE_UNICODE_CATEGORY) != 0)
127             sb.append((char)'u');
128         if ((options & RegularExpression.UNICODE_WORD_BOUNDARY) != 0)
129             sb.append((char)'w');
130         if ((options & RegularExpression.EXTENDED_COMMENT) != 0)
131             sb.append((char)'x');
132         if ((options & RegularExpression.SPECIAL_COMMA) != 0)
133             sb.append((char)',');
134         return sb.toString().intern();
135     }

139     static String stripExtendedComment(String regex) {
140         int len = regex.length();
141         StringBuffer buffer = new StringBuffer(len);
142         int offset = 0;
143         while (offset < len) {
144             int ch = regex.charAt(offset++);
145                                                 // Skips a white space.
146             if (ch == '\t' || ch == '\n' || ch == '\f' || ch == '\r' || ch == ' ')
147                 continue;
148 
149             if (ch == '#') {                    // Skips chracters between '#' and a line end.
150                 while (offset < len) {
151                     ch = regex.charAt(offset++);
152                     if (ch == '\r' || ch == '\n')
153                         break;
154                 }
155                 continue;
156             }
157 
158             int next;                           // Strips an escaped white space.
159             if (ch == '\\' && offset < len) {
160                 if ((next = regex.charAt(offset)) == '#'
161                     || next == '\t' || next == '\n' || next == '\f'
162                     || next == '\r' || next == ' ') {
163                     buffer.append((char)next);
164                     offset ++;
165                 } else {                        // Other escaped character.
166                     buffer.append((char)'\\');
167                     buffer.append((char)next);
168                     offset ++;
169                 }
170             } else                              // As is.
171                 buffer.append((char)ch);
172         }
173         return buffer.toString();
174     }

178     /**
179      * Sample entry.
180      * <div>Usage: <KBD>com.sun.org.apache.xerces.internal.utils.regex.REUtil &lt;regex&gt; &lt;string&gt;</KBD></div>
181      */
182     public static void main(String[] argv) {
183         String pattern = null;
184         try {
185             String options = "";
186             String target = null;
187             if( argv.length == 0 ) {
188                 System.out.println( "Error:Usage: java REUtil -i|-m|-s|-u|-w|-X regularExpression String" );
189                 System.exit( 0 );
190             }
191             for (int i = 0;  i < argv.length;  i ++) {
192                 if (argv[i].length() == 0 || argv[i].charAt(0) != '-') {
193                     if (pattern == null)
194                         pattern = argv[i];
195                     else if (target == null)
196                         target = argv[i];
197                     else
198                         System.err.println("Unnecessary: "+argv[i]);
199                 } else if (argv[i].equals("-i")) {
200                     options += "i";
201                 } else if (argv[i].equals("-m")) {
202                     options += "m";
203                 } else if (argv[i].equals("-s")) {
204                     options += "s";
205                 } else if (argv[i].equals("-u")) {
206                     options += "u";
207                 } else if (argv[i].equals("-w")) {
208                     options += "w";
209                 } else if (argv[i].equals("-X")) {
210                     options += "X";
211                 } else {
212                     System.err.println("Unknown option: "+argv[i]);
213                 }
214             }
215             RegularExpression reg = new RegularExpression(pattern, options);
216             System.out.println("RegularExpression: "+reg);
217             Match match = new Match();
218             reg.matches(target, match);
219             for (int i = 0;  i < match.getNumberOfGroups();  i ++) {
220                 if (i == 0 )  System.out.print("Matched range for the whole pattern: ");
221                 else System.out.print("["+i+"]: ");
222                 if (match.getBeginning(i) < 0)
223                     System.out.println("-1");
224                 else {
225                     System.out.print(match.getBeginning(i)+", "+match.getEnd(i)+", ");
226                     System.out.println("\""+match.getCapturedText(i)+"\"");
227                 }
228             }
229         } catch (ParseException pe) {
230             if (pattern == null) {
231                 pe.printStackTrace();
232             } else {
233                 System.err.println("com.sun.org.apache.xerces.internal.utils.regex.ParseException: "+pe.getMessage());
234                 String indent = "        ";
235                 System.err.println(indent+pattern);
236                 int loc = pe.getLocation();
237                 if (loc >= 0) {
238                     System.err.print(indent);
239                     for (int i = 0;  i < loc;  i ++)  System.err.print("-");
240                     System.err.println("^");
241                 }
242             }
243         } catch (Exception e) {
244             e.printStackTrace();
245         }
246     }

248     static final int CACHESIZE = 20;
249     static final RegularExpression[] regexCache = new RegularExpression[CACHESIZE];
250     /**
251      * Creates a RegularExpression instance.
252      * This method caches created instances.
253      *
254      * @see RegularExpression#RegularExpression(java.lang.String, java.lang.String)
255      */
256     public static RegularExpression createRegex(String pattern, String options)
257         throws ParseException {
258         RegularExpression re = null;
259         int intOptions = REUtil.parseOptions(options);
260         synchronized (REUtil.regexCache) {
261             int i;
262             for (i = 0;  i < REUtil.CACHESIZE;  i ++) {
263                 RegularExpression cached = REUtil.regexCache[i];
264                 if (cached == null) {
265                     i = -1;
266                     break;
267                 }
268                 if (cached.equals(pattern, intOptions)) {
269                     re = cached;
270                     break;
271                 }
272             }
273             if (re != null) {
274                 if (i != 0) {
275                     System.arraycopy(REUtil.regexCache, 0, REUtil.regexCache, 1, i);
276                     REUtil.regexCache[0] = re;
277                 }
278             } else {
279                 re = new RegularExpression(pattern, options);
280                 System.arraycopy(REUtil.regexCache, 0, REUtil.regexCache, 1, REUtil.CACHESIZE-1);
281                 REUtil.regexCache[0] = re;
282             }
283         }
284         return re;
285     }

308     public static String quoteMeta(String literal) {
309         int len = literal.length();
310         StringBuffer buffer = null;
311         for (int i = 0;  i < len;  i ++) {
312             int ch = literal.charAt(i);
313             if (".*+?{[()|\\^$".indexOf(ch) >= 0) {
314                 if (buffer == null) {
315                     buffer = new StringBuffer(i+(len-i)*2);
316                     if (i > 0)  buffer.append(literal.substring(0, i));
317                 }
318                 buffer.append((char)'\\');
319                 buffer.append((char)ch);
320             } else if (buffer != null)
321                 buffer.append((char)ch);
322         }
323         return buffer != null ? buffer.toString() : literal;
324     }
