254     /**
255      * Create a new simple type which is derived by restriction from another
256      * simple type.
257      *
258      * @param name              name of the new type, could be null
259      * @param targetNamespace   target namespace of the new type, could be null
260      * @param finalSet          value of "final"
261      * @param base              base type of the new type
262      * @param annotations       set of annotations
263      * @return                  the newly created simple type
264      */
265     public XSSimpleType createTypeRestriction(String name, String targetNamespace,
266                                               short finalSet, XSSimpleType base, XSObjectList annotations) {
267 
268         if (fDeclPool != null) {
269            XSSimpleTypeDecl st= fDeclPool.getSimpleTypeDecl();
270            return st.setRestrictionValues((XSSimpleTypeDecl)base, name, targetNamespace, finalSet, annotations);
271         }
272         return new XSSimpleTypeDecl((XSSimpleTypeDecl)base, name, targetNamespace, finalSet, false, annotations);
273     }
274 
275     /**
276      * Create a new simple type which is derived by list from another simple
277      * type.
278      *
279      * @param name              name of the new type, could be null
280      * @param targetNamespace   target namespace of the new type, could be null
281      * @param finalSet          value of "final"
282      * @param itemType          item type of the list type
283      * @param annotations       set of annotations
284      * @return                  the newly created simple type
285      */
286     public XSSimpleType createTypeList(String name, String targetNamespace,
287                                        short finalSet, XSSimpleType itemType,
288                                        XSObjectList annotations) {
289         if (fDeclPool != null) {
290            XSSimpleTypeDecl st= fDeclPool.getSimpleTypeDecl();
291            return st.setListValues(name, targetNamespace, finalSet, (XSSimpleTypeDecl)itemType, annotations);
292         }
293         return new XSSimpleTypeDecl(name, targetNamespace, finalSet, (XSSimpleTypeDecl)itemType, false, annotations);
294     }
295 
296     /**
297      * Create a new simple type which is derived by union from a list of other
298      * simple types.
299      *
300      * @param name              name of the new type, could be null
301      * @param targetNamespace   target namespace of the new type, could be null
302      * @param finalSet          value of "final"
303      * @param memberTypes       member types of the union type
304      * @param annotations       set of annotations
305      * @return                  the newly created simple type
306      */
307     public XSSimpleType createTypeUnion(String name, String targetNamespace,
308                                         short finalSet, XSSimpleType[] memberTypes,
309                                         XSObjectList annotations) {
310         int typeNum = memberTypes.length;
311         XSSimpleTypeDecl[] mtypes = new XSSimpleTypeDecl[typeNum];
312         System.arraycopy(memberTypes, 0, mtypes, 0, typeNum);
313 
314         if (fDeclPool != null) {
315            XSSimpleTypeDecl st= fDeclPool.getSimpleTypeDecl();
316            return st.setUnionValues(name, targetNamespace, finalSet, mtypes, annotations);
317         }
318         return new XSSimpleTypeDecl(name, targetNamespace, finalSet, mtypes, annotations);
319     }
