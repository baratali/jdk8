31  /**
32   * This class implements an optimized iterator for
33   * children patterns that have a node test, and possibly a predicate.
34   * @see com.sun.org.apache.xpath.internal.axes.BasicTestIterator
35   * @xsl.usage advanced
36   */
37  public class ChildTestIterator extends BasicTestIterator

47    /**
48     * Create a ChildTestIterator object.
49     *
50     * @param compiler A reference to the Compiler that contains the op map.
51     * @param opPos The position within the op map, which contains the
52     * location path expression for this itterator.
53     *
54     * @throws javax.xml.transform.TransformerException
55     */
56    ChildTestIterator(Compiler compiler, int opPos, int analysis)
57            throws javax.xml.transform.TransformerException
58    {
59      super(compiler, opPos, analysis);
60    }
61  
62    /**
63     * Create a ChildTestIterator object.
64     *
65     * @param traverser Traverser that tells how the KeyIterator is to be handled.
66     *
67     * @throws javax.xml.transform.TransformerException
68     */
69    public ChildTestIterator(DTMAxisTraverser traverser)
70    {
71  
72      super(null);
73  
74      m_traverser = traverser;
75    }
76  
77    /**
78     * Get the next node via getNextXXX.  Bottlenecked for derived class override.
79     * @return The next node on the axis, or DTM.NULL.
80     */
81    protected int getNextNode()
82    {
83      if(true /* 0 == m_extendedTypeID */)
84      {
85        m_lastFetched = (DTM.NULL == m_lastFetched)
86                     ? m_traverser.first(m_context)
87                     : m_traverser.next(m_context, m_lastFetched);
88      }
89  //    else
90  //    {
91  //      m_lastFetched = (DTM.NULL == m_lastFetched)
92  //                   ? m_traverser.first(m_context, m_extendedTypeID)
93  //                   : m_traverser.next(m_context, m_lastFetched,
94  //                                      m_extendedTypeID);
95  //    }
96  
97      return m_lastFetched;
98    }

101   /**
102    *  Get a cloned Iterator that is reset to the beginning
103    *  of the query.
104    *
105    *  @return A cloned NodeIterator set of the start of the query.
106    *
107    *  @throws CloneNotSupportedException
108    */
109   public DTMIterator cloneWithReset() throws CloneNotSupportedException
110   {
111 
112     ChildTestIterator clone = (ChildTestIterator) super.cloneWithReset();
113     clone.m_traverser = m_traverser;
114 
115     return clone;
116   }

151   /**
152    * Returns the axis being iterated, if it is known.
153    *
154    * @return Axis.CHILD, etc., or -1 if the axis is not known or is of multiple
155    * types.
156    */
157   public int getAxis()
158   {
159     return com.sun.org.apache.xml.internal.dtm.Axis.CHILD;
160   }

162   /**
163    *  Detaches the iterator from the set which it iterated over, releasing
164    * any computational resources and placing the iterator in the INVALID
165    * state. After<code>detach</code> has been invoked, calls to
166    * <code>nextNode</code> or<code>previousNode</code> will raise the
167    * exception INVALID_STATE_ERR.
168    */
169   public void detach()
170   {
171     if(m_allowDetach)
172     {
173       m_traverser = null;
174 
175       // Always call the superclass detach last!
176       super.detach();
177     }
178   }
