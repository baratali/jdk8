109   /**
110    * Convert a string to a double -- Allowed input is in fixed
111    * notation ddd.fff.
112    *
113    * @return A double value representation of the string, or return Double.NaN
114    * if the string can not be converted.
115    */
116   public double toDouble()
117   {
118     /* XMLCharacterRecognizer.isWhiteSpace(char c) methods treats the following
119      * characters as white space characters.
120      * ht - horizontal tab, nl - newline , cr - carriage return and sp - space
121      * trim() methods by default also takes care of these white space characters
122      * So trim() method is used to remove leading and trailing white spaces.
123      */
124         XMLString s = trim();
125         double result = Double.NaN;
126         for (int i = 0; i < s.length(); i++)
127         {
128                 char c = s.charAt(i);
129     if (c != '-' && c != '.' && ( c < 0X30 || c > 0x39)) {
130             // The character is not a '-' or a '.' or a digit
131             // then return NaN because something is wrong.
132                         return result;
133         }
134         }
135         try
136         {
137                 result = Double.parseDouble(s.toString());
138         } catch (NumberFormatException e){}
139 
140         return result;
141 }

284   /**
285    * Tell if two objects are functionally equal.
286    *
287    * @param obj2 Object to compare this to
288    *
289    * @return true if the two objects are equal
290    *
291    * @throws javax.xml.transform.TransformerException
292    */
293   public boolean equals(XObject obj2)
294   {
295 
296     // In order to handle the 'all' semantics of
297     // nodeset comparisons, we always call the
298     // nodeset function.
299     int t = obj2.getType();
300     try
301     {
302             if (XObject.CLASS_NODESET == t)
303               return obj2.equals(this);
304             // If at least one object to be compared is a boolean, then each object
305             // to be compared is converted to a boolean as if by applying the
306             // boolean function.
307             else if(XObject.CLASS_BOOLEAN == t)
308                 return obj2.bool() == bool();
309             // Otherwise, if at least one object to be compared is a number, then each object
310             // to be compared is converted to a number as if by applying the number function.
311             else if(XObject.CLASS_NUMBER == t)
312                 return obj2.num() == num();
313     }
314     catch(javax.xml.transform.TransformerException te)
315     {
316         throw new com.sun.org.apache.xml.internal.utils.WrappedRuntimeException(te);
317     }
318 
319     // Otherwise, both objects to be compared are converted to strings as
320     // if by applying the string function.
321     return xstr().equals(obj2.xstr());
322   }

415   /**
416    * Compares two strings lexicographically.
417    *
418    * @param   xstr   the <code>String</code> to be compared.
419    *
420    * @return  the value <code>0</code> if the argument string is equal to
421    *          this string; a value less than <code>0</code> if this string
422    *          is lexicographically less than the string argument; and a
423    *          value greater than <code>0</code> if this string is
424    *          lexicographically greater than the string argument.
425    * @exception java.lang.NullPointerException if <code>anotherString</code>
426    *          is <code>null</code>.
427    */
428   public int compareTo(XMLString xstr)
429   {
430 
431     int len1 = this.length();
432     int len2 = xstr.length();
433     int n = Math.min(len1, len2);
434     int i = 0;
435     int j = 0;
436 
437     while (n-- != 0)
438     {
439       char c1 = this.charAt(i);
440       char c2 = xstr.charAt(j);
441 
442       if (c1 != c2)
443       {
444         return c1 - c2;
445       }
446 
447       i++;
448       j++;
449     }
450 
451     return len1 - len2;
452   }

530   /**
531    * Tests if this string starts with the specified prefix beginning
532    * a specified index.
533    *
534    * @param   prefix    the prefix.
535    * @param   toffset   where to begin looking in the string.
536    * @return  <code>true</code> if the character sequence represented by the
537    *          argument is a prefix of the substring of this object starting
538    *          at index <code>toffset</code>; <code>false</code> otherwise.
539    *          The result is <code>false</code> if <code>toffset</code> is
540    *          negative or greater than the length of this
541    *          <code>String</code> object; otherwise the result is the same
542    *          as the result of the expression
543    *          <pre>
544    *          this.subString(toffset).startsWith(prefix)
545    *          </pre>
546    * @exception java.lang.NullPointerException if <code>prefix</code> is
547    *          <code>null</code>.
548    */
549   public boolean startsWith(XMLString prefix, int toffset)
550   {
551 
552     int to = toffset;
553     int tlim = this.length();
554     int po = 0;
555     int pc = prefix.length();
556 
557     // Note: toffset might be near -1>>>1.
558     if ((toffset < 0) || (toffset > tlim - pc))
559     {
560       return false;
561     }
562 
563     while (--pc >= 0)
564     {
565       if (this.charAt(to) != prefix.charAt(po))
566       {
567         return false;
568       }
569 
570       to++;
571       po++;
572     }
573 
574     return true;
575   }

1013   /**
1014    * Conditionally trim all leading and trailing whitespace in the specified String.
1015    * All strings of white space are
1016    * replaced by a single space character (#x20), except spaces after punctuation which
1017    * receive double spaces if doublePunctuationSpaces is true.
1018    * This function may be useful to a formatter, but to get first class
1019    * results, the formatter should probably do it's own white space handling
1020    * based on the semantics of the formatting object.
1021    *
1022    * @param   trimHead    Trim leading whitespace?
1023    * @param   trimTail    Trim trailing whitespace?
1024    * @param   doublePunctuationSpaces    Use double spaces for punctuation?
1025    * @return              The trimmed string.
1026    */
1027   public XMLString fixWhiteSpace(boolean trimHead, boolean trimTail,
1028                                  boolean doublePunctuationSpaces)
1029   {
1030 
1031     // %OPT% !!!!!!!
1032     int len = this.length();
1033     char[] buf = new char[len];
1034 
1035     this.getChars(0, len, buf, 0);
1036 
1037     boolean edit = false;
1038     int s;
1039 
1040     for (s = 0; s < len; s++)
1041     {
1042       if (isSpace(buf[s]))
1043       {
1044         break;
1045       }
1046     }
1047 
1048     /* replace S to ' '. and ' '+ -> single ' '. */
1049     int d = s;
1050     boolean pres = false;
1051 
1052     for (; s < len; s++)
1053     {
1054       char c = buf[s];
1055 
1056       if (isSpace(c))
1057       {
1058         if (!pres)
1059         {
1060           if (' ' != c)
1061           {
1062             edit = true;
1063           }
1064 
1065           buf[d++] = ' ';
1066 
1067           if (doublePunctuationSpaces && (s != 0))
1068           {
1069             char prevChar = buf[s - 1];
1070 
1071             if (!((prevChar == '.') || (prevChar == '!')
1072                   || (prevChar == '?')))
1073             {
1074               pres = true;
1075             }
1076           }
1077           else
1078           {
1079             pres = true;
1080           }
1081         }
1082         else
1083         {
1084           edit = true;
1085           pres = true;
1086         }
1087       }
1088       else
1089       {
1090         buf[d++] = c;
1091         pres = false;
1092       }
1093     }
1094 
1095     if (trimTail && 1 <= d && ' ' == buf[d - 1])
1096     {
1097       edit = true;
1098 
1099       d--;
1100     }
1101 
1102     int start = 0;
1103 
1104     if (trimHead && 0 < d && ' ' == buf[0])
1105     {
1106       edit = true;
1107 
1108       start++;
1109     }
1110 
1111     XMLStringFactory xsf = XMLStringFactoryImpl.getFactory();
1112 
1113     return edit ? xsf.newstr(new String(buf, start, d - start)) : this;
1114   }
