23  package com.sun.org.apache.xpath.internal.operations;
24  
25  import com.sun.org.apache.xpath.internal.XPathContext;
26  import com.sun.org.apache.xpath.internal.objects.XBoolean;
27  import com.sun.org.apache.xpath.internal.objects.XObject;
28  
29  /**
30   * The 'or' operation expression executer.
31   */
32  public class Or extends Operation
33  {
34      static final long serialVersionUID = -644107191353853079L;
35  
36    /**
37     * OR two expressions and return the boolean result. Override
38     * superclass method for optimization purposes.
39     *
40     * @param xctxt The runtime execution context.
41     *
42     * @return {@link com.sun.org.apache.xpath.internal.objects.XBoolean#S_TRUE} or
43     * {@link com.sun.org.apache.xpath.internal.objects.XBoolean#S_FALSE}.
44     *
45     * @throws javax.xml.transform.TransformerException
46     */
47    public XObject execute(XPathContext xctxt) throws javax.xml.transform.TransformerException
48    {
49  
50      XObject expr1 = m_left.execute(xctxt);
51  
52      if (!expr1.bool())
53      {
54        XObject expr2 = m_right.execute(xctxt);
55  
56        return expr2.bool() ? XBoolean.S_TRUE : XBoolean.S_FALSE;
57      }
58      else
59        return XBoolean.S_TRUE;
60    }
61  
62    /**
63     * Evaluate this operation directly to a boolean.
64     *
65     * @param xctxt The runtime execution context.
66     *
67     * @return The result of the operation as a boolean.
68     *
69     * @throws javax.xml.transform.TransformerException
70     */
71    public boolean bool(XPathContext xctxt)
72            throws javax.xml.transform.TransformerException
73    {
74      return (m_left.bool(xctxt) || m_right.bool(xctxt));
75    }
76  
77  }
