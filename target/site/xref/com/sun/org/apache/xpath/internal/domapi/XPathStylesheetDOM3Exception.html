42  package com.sun.org.apache.xpath.internal.domapi;
43  
44  import javax.xml.transform.SourceLocator;
45  import javax.xml.transform.TransformerException;
46  
47  /**
48   *
49   * A new exception to add support for DOM Level 3 XPath API.
50   * This class is needed to throw a org.w3c.dom.DOMException with proper error code in
51   * createExpression method of XPathEvaluatorImpl (a DOM Level 3 class).
52   *
53   * This class extends TransformerException because the error message includes information
54   * about where the XPath problem is in the stylesheet as well as the XPath expression itself.
55   *
56   * @xsl.usage internal
57   */
58  final public class XPathStylesheetDOM3Exception extends TransformerException {
59          public XPathStylesheetDOM3Exception(String msg, SourceLocator arg1)
60          {
61                  super(msg, arg1);
62          }
63  }
