75       * implementation of the SmartTransformerFactory. This factory
76       * uses com.sun.org.apache.xalan.internal.xsltc.trax.TransformerFactory
77       * to return Templates objects; and uses
78       * com.sun.org.apache.xalan.internal.processor.TransformerFactory

83      private void createXSLTCTransformerFactory() {
84          _xsltcFactory = new TransformerFactoryImpl();
85          _currFactory = _xsltcFactory;
86      }

88      private void createXalanTransformerFactory() {
89          final String xalanMessage =
90              "com.sun.org.apache.xalan.internal.xsltc.trax.SmartTransformerFactoryImpl "+
91              "could not create an "+
92              "com.sun.org.apache.xalan.internal.processor.TransformerFactoryImpl.";
93          // try to create instance of Xalan factory...
94          try {
95              Class xalanFactClass = ObjectFactory.findProviderClass(
96                  "com.sun.org.apache.xalan.internal.processor.TransformerFactoryImpl",
97                  true);
98              _xalanFactory = (SAXTransformerFactory)
99                  xalanFactClass.newInstance();
100         }
101         catch (ClassNotFoundException e) {
102             System.err.println(xalanMessage);
103         }
104         catch (InstantiationException e) {
105             System.err.println(xalanMessage);
106         }
107         catch (IllegalAccessException e) {
108             System.err.println(xalanMessage);
109         }
110         _currFactory = _xalanFactory;
111     }

123     public Object getAttribute(String name)
124         throws IllegalArgumentException
125     {
126         // GTM: NB: 'debug' should change to something more unique...
127         if ((name.equals("translet-name")) || (name.equals("debug"))) {
128             if (_xsltcFactory == null) {
129                 createXSLTCTransformerFactory();
130             }
131             return _xsltcFactory.getAttribute(name);
132         }
133         else {
134             if (_xalanFactory == null) {
135                 createXalanTransformerFactory();
136             }
137             return _xalanFactory.getAttribute(name);
138         }
139     }

158     /**
159      * <p>Set a feature for this <code>SmartTransformerFactory</code> and <code>Transformer</code>s
160      * or <code>Template</code>s created by this factory.</p>
161      *
162      * <p>
163      * Feature names are fully qualified {@link java.net.URI}s.
164      * Implementations may define their own features.
165      * An {@link TransformerConfigurationException} is thrown if this <code>TransformerFactory</code> or the
166      * <code>Transformer</code>s or <code>Template</code>s it creates cannot support the feature.
167      * It is possible for an <code>TransformerFactory</code> to expose a feature value but be unable to change its state.
168      * </p>
169      *
170      * <p>See {@link javax.xml.transform.TransformerFactory} for full documentation of specific features.</p>
171      *
172      * @param name Feature name.
173      * @param value Is feature state <code>true</code> or <code>false</code>.
174      *
175      * @throws TransformerConfigurationException if this <code>TransformerFactory</code>
176      *   or the <code>Transformer</code>s or <code>Template</code>s it creates cannot support this feature.
177      * @throws NullPointerException If the <code>name</code> parameter is null.
178      */
179     public void setFeature(String name, boolean value)
180         throws TransformerConfigurationException {
181 
182         // feature name cannot be null
183         if (name == null) {
184             ErrorMsg err = new ErrorMsg(ErrorMsg.JAXP_SET_FEATURE_NULL_NAME);
185             throw new NullPointerException(err.toString());
186         }
187         // secure processing?
188         else if (name.equals(XMLConstants.FEATURE_SECURE_PROCESSING)) {
189             featureSecureProcessing = value;
190             // all done processing feature
191             return;
192         }
193         else {
194             // unknown feature
195             ErrorMsg err = new ErrorMsg(ErrorMsg.JAXP_UNSUPPORTED_FEATURE, name);
196             throw new TransformerConfigurationException(err.toString());
197         }
198     }

200     /**
201      * javax.xml.transform.sax.TransformerFactory implementation.
202      * Look up the value of a feature (to see if it is supported).
203      * This method must be updated as the various methods and features of this
204      * class are implemented.
205      *
206      * @param name The feature name
207      * @return 'true' if feature is supported, 'false' if not
208      */
209     public boolean getFeature(String name) {
210         // All supported features should be listed here
211         String[] features = {
212             DOMSource.FEATURE,
213             DOMResult.FEATURE,
214             SAXSource.FEATURE,
215             SAXResult.FEATURE,
216             StreamSource.FEATURE,
217             StreamResult.FEATURE
218         };
219 
220         // feature name cannot be null
221         if (name == null) {
222             ErrorMsg err = new ErrorMsg(ErrorMsg.JAXP_GET_FEATURE_NULL_NAME);
223             throw new NullPointerException(err.toString());
224         }
225 
226         // Inefficient, but it really does not matter in a function like this
227         for (int i = 0; i < features.length; i++) {
228             if (name.equals(features[i]))
229                 return true;
230         }
231 
232         // secure processing?
233         if (name.equals(XMLConstants.FEATURE_SECURE_PROCESSING)) {
234             return featureSecureProcessing;
235         }
236 
237         // unknown feature
238         return false;
239     }

262      * result. Uses the com.sun.org.apache.xalan.internal.processor.TransformerFactory.
263      * @return A Transformer object.
264      */
265     public Transformer newTransformer()
266         throws TransformerConfigurationException
267     {
268         if (_xalanFactory == null) {
269             createXalanTransformerFactory();
270         }
271         if (_errorlistener != null) {
272             _xalanFactory.setErrorListener(_errorlistener);
273         }
274         if (_uriresolver != null) {
275             _xalanFactory.setURIResolver(_uriresolver);
276         }
277         _currFactory = _xalanFactory;
278         return _currFactory.newTransformer();
279     }

428     /*
429      * Create an XMLFilter that uses the given source as the
430      * transformation instructions. Uses
431      * com.sun.org.apache.xalan.internal.xsltc.trax.TransformerFactory.
432      */
433     public XMLFilter newXMLFilter(Templates templates)
434         throws TransformerConfigurationException {
435         try {
436             return new com.sun.org.apache.xalan.internal.xsltc.trax.TrAXFilter(templates);
437         }
438         catch(TransformerConfigurationException e1) {
439             if (_xsltcFactory == null) {
440                 createXSLTCTransformerFactory();
441             }
442             ErrorListener errorListener = _xsltcFactory.getErrorListener();
443             if(errorListener != null) {
444                 try {
445                     errorListener.fatalError(e1);
446                     return null;
447                 }
448                 catch( TransformerException e2) {
449                     new TransformerConfigurationException(e2);
450                 }
451             }
452             throw e1;
453         }
