24  package com.sun.org.apache.xalan.internal.xsltc;
25  
26  
27  /**
28   * Admin class that assigns a version number to the XSLTC software.
29   * The version number is made up from three fields as in:
30   * MAJOR.MINOR[.DELTA]. Fields are incremented based on the following:
31   * DELTA field: changes for each bug fix, developer fixing the bug should
32   *              increment this field.
33   * MINOR field: API changes or a milestone culminating from several
34   *              bug fixes. DELTA field goes to zero and MINOR is
35   *              incremented such as: {1.0,1.0.1,1.0.2,1.0.3,...1.0.18,1.1}
36   * MAJOR field: milestone culminating in fundamental API changes or
37   *              architectural changes.  MINOR field goes to zero
38   *              and MAJOR is incremented such as: {...,1.1.14,1.2,2.0}
39   * Stability of a release follows: X.0 > X.X > X.X.X
40   * @author G. Todd Miller
41   */
42  public class ProcessorVersion {
43      private static int MAJOR = 1;
44      private static int MINOR = 0;
45      private static int DELTA = 0;
46  
47      public static void main(String[] args) {
48          System.out.println("XSLTC version " + MAJOR + "." + MINOR +
49              ((DELTA > 0) ? ("."+DELTA) : ("")));
50      }
51  }
