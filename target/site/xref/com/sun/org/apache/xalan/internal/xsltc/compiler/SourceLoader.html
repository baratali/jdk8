24  package com.sun.org.apache.xalan.internal.xsltc.compiler;
25  
26  import org.xml.sax.InputSource;
27  
28  /**
29   * @author Morten Jorgensen
30   */
31  public interface SourceLoader {
32  
33      /**
34       * This interface is used to plug external document loaders into XSLTC
35       * (used with the <xsl:include> and <xsl:import> elements.
36       *
37       * @param href The URI of the document to load
38       * @param context The URI of the currently loaded document
39       * @param xsltc The compiler that resuests the document
40       * @return An InputSource with the loaded document
41       */
42      public InputSource loadSource(String href, String context, XSLTC xsltc);
43  
44  }
