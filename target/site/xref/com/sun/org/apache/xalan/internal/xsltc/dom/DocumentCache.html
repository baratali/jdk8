69      /*
70       * Inner class containing a DOMImpl object and DTD handler
71       */
72      public final class CachedDocument {
73  
74          // Statistics data
75          private long _firstReferenced;
76          private long _lastReferenced;
77          private long _accessCount;
78          private long _lastModified;
79          private long _lastChecked;
80          private long _buildTime;
81  
82          // DOM and DTD handler references
83          private DOMEnhancedForDTM _dom = null;
84  
85          /**
86           * Constructor - load document and initialise statistics
87           */
88          public CachedDocument(String uri) {
89              // Initialise statistics variables
90              final long stamp = System.currentTimeMillis();
91              _firstReferenced = stamp;
92              _lastReferenced  = stamp;
93              _accessCount     = 0;
94              loadDocument(uri);
95  
96              _buildTime = System.currentTimeMillis() - stamp;
97          }
98  
99          /**
100          * Loads the document and updates build-time (latency) statistics
101          */
102         public void loadDocument(String uri) {
103 
104             try {
105                 final long stamp = System.currentTimeMillis();
106                 _dom = (DOMEnhancedForDTM)_dtmManager.getDTM(
107                                  new SAXSource(_reader, new InputSource(uri)),
108                                  false, null, true, false);
109                 _dom.setDocumentURI(uri);
110 
111                 // The build time can be used for statistics for a better
112                 // priority algorithm (currently round robin).
113                 final long thisTime = System.currentTimeMillis() - stamp;
114                 if (_buildTime > 0)
115                     _buildTime = (_buildTime + thisTime) >>> 1;
116                 else
117                     _buildTime = thisTime;
118             }
119             catch (Exception e) {
120                 _dom = null;
121             }
122         }
123 
124         public DOM getDocument()       { return(_dom); }
125 
126         public long getFirstReferenced()   { return(_firstReferenced); }
127 
128         public long getLastReferenced()    { return(_lastReferenced); }
129 
130         public long getAccessCount()       { return(_accessCount); }
131 
132         public void incAccessCount()       { _accessCount++; }
133 
134         public long getLastModified()      { return(_lastModified); }
135 
136         public void setLastModified(long t){ _lastModified = t; }
137 
138         public long getLatency()           { return(_buildTime); }
139 
140         public long getLastChecked()       { return(_lastChecked); }
141 
142         public void setLastChecked(long t) { _lastChecked = t; }
143 
144         public long getEstimatedSize() {
145             if (_dom != null)
146                 return(_dom.getSize() << 5); // ???
147             else
148                 return(0);
149         }
150 
151     }

153     /**
154      * DocumentCache constructor
155      */
156     public DocumentCache(int size) throws SAXException {
157         this(size, null);
158         try {
159             _dtmManager = XSLTCDTMManager.createNewDTMManagerInstance();
160         } catch (Exception e) {
161             throw new SAXException(e);
162         }
163     }
164 
165     /**
166      * DocumentCache constructor
167      */
168     public DocumentCache(int size, XSLTCDTMManager dtmManager) throws SAXException {
169         _dtmManager = dtmManager;
170         _count = 0;
171         _current = 0;
172         _size  = size;
173         _references = new Hashtable(_size+2);
174         _URIs = new String[_size];
175 
176         try {
177             // Create a SAX parser and get the XMLReader object it uses
178             final SAXParserFactory factory = SAXParserFactory.newInstance();
179             try {
180                 factory.setFeature(Constants.NAMESPACE_FEATURE,true);
181             }
182             catch (Exception e) {
183                 factory.setNamespaceAware(true);
184             }
185             _parser = factory.newSAXParser();
186             _reader = _parser.getXMLReader();
187         }
188         catch (ParserConfigurationException e) {
189             BasisLibrary.runTimeError(BasisLibrary.NAMESPACES_SUPPORT_ERR);
190         }
191     }

193     /**
194      * Returns the time-stamp for a document's last update
195      */
196     private final long getLastModified(String uri) {
197         try {
198             URL url = new URL(uri);
199             URLConnection connection = url.openConnection();
200             long timestamp = connection.getLastModified();
201             // Check for a "file:" URI (courtesy of Brian Ewins)
202             if (timestamp == 0){ // get 0 for local URI
203                 if ("file".equals(url.getProtocol())){
204                     File localfile = new File(URLDecoder.decode(url.getFile()));
205                     timestamp = localfile.lastModified();
206                 }
207             }
208             return(timestamp);
209         }
210         // Brutal handling of all exceptions
211         catch (Exception e) {
212             return(System.currentTimeMillis());
213         }
214     }

216     /**
217      *
218      */
219     private CachedDocument lookupDocument(String uri) {
220         return((CachedDocument)_references.get(uri));
221     }
222 
223     /**
224      *
225      */
226     private synchronized void insertDocument(String uri, CachedDocument doc) {
227         if (_count < _size) {
228             // Insert out URI in circular buffer
229             _URIs[_count++] = uri;
230             _current = 0;
231         }
232         else {
233             // Remove oldest URI from reference Hashtable
234             _references.remove(_URIs[_current]);
235             // Insert our URI in circular buffer
236             _URIs[_current] = uri;
237             if (++_current >= _size) _current = 0;
238         }
239         _references.put(uri, doc);
240     }
241 
242     /**
243      *
244      */
245     private synchronized void replaceDocument(String uri, CachedDocument doc) {
246         CachedDocument old = (CachedDocument)_references.get(uri);
247         if (doc == null)
248             insertDocument(uri, doc);
249         else
250             _references.put(uri, doc);
251     }

253     /**
254      * Returns a document either by finding it in the cache or
255      * downloading it and putting it in the cache.
256      */
257     @Override
258     public DOM retrieveDocument(String baseURI, String href, Translet trs) {
259         CachedDocument doc;
260 
261     String uri = href;
262     if (baseURI != null && !baseURI.equals("")) {
263         try {
264             uri = SystemIDResolver.getAbsoluteURI(uri, baseURI);
265         } catch (TransformerException te) {
266             // ignore
267         }
268     }
269 
270         // Try to get the document from the cache first
271         if ((doc = lookupDocument(uri)) == null) {
272             doc = new CachedDocument(uri);
273             if (doc == null) return null; // better error handling needed!!!
274             doc.setLastModified(getLastModified(uri));
275             insertDocument(uri, doc);
276         }
277         // If the document is in the cache we must check if it is still valid
278         else {
279             long now = System.currentTimeMillis();
280             long chk = doc.getLastChecked();
281             doc.setLastChecked(now);
282             // Has the modification time for this file been checked lately?
283             if (now > (chk + REFRESH_INTERVAL)) {
284                 doc.setLastChecked(now);
285                 long last = getLastModified(uri);
286                 // Reload document if it has been modified since last download
287                 if (last > doc.getLastModified()) {
288                     doc = new CachedDocument(uri);
289                     if (doc == null) return null;
290                     doc.setLastModified(getLastModified(uri));
291                     replaceDocument(uri, doc);
292                 }
293             }
294 
295         }
296 
297         // Get the references to the actual DOM and DTD handler
298         final DOM dom = doc.getDocument();
299 
300         // The dom reference may be null if the URL pointed to a
301         // non-existing document
302         if (dom == null) return null;
303 
304         doc.incAccessCount(); // For statistics
305 
306         final AbstractTranslet translet = (AbstractTranslet)trs;
307 
308         // Give the translet an early opportunity to extract any
309         // information from the DOM object that it would like.
310         translet.prepassDocument(dom);
311 
312         return(doc.getDocument());
313     }

315     /**
316      * Outputs the cache statistics
317      */
318     public void getStatistics(PrintWriter out) {
319         out.println("<h2>DOM cache statistics</h2><center><table border=\"2\">"+
320                     "<tr><td><b>Document URI</b></td>"+
321                     "<td><center><b>Build time</b></center></td>"+
322                     "<td><center><b>Access count</b></center></td>"+
323                     "<td><center><b>Last accessed</b></center></td>"+
324                     "<td><center><b>Last modified</b></center></td></tr>");
325 
326         for (int i=0; i<_count; i++) {
327             CachedDocument doc = (CachedDocument)_references.get(_URIs[i]);
328             out.print("<tr><td><a href=\""+_URIs[i]+"\">"+
329                       "<font size=-1>"+_URIs[i]+"</font></a></td>");
330             out.print("<td><center>"+doc.getLatency()+"ms</center></td>");
331             out.print("<td><center>"+doc.getAccessCount()+"</center></td>");
332             out.print("<td><center>"+(new Date(doc.getLastReferenced()))+
333                       "</center></td>");
334             out.print("<td><center>"+(new Date(doc.getLastModified()))+
335                       "</center></td>");
336             out.println("</tr>");
337         }
338 
339         out.println("</table></center>");
340     }
