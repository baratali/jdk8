25  /**
26   * Primary constants used by the XSLT Processor
27   * @xsl.usage advanced
28   */
29  public class Constants
30  {
31  
32    /**
33     * Mnemonics for standard XML Namespace URIs, as Java Strings:
34     * <ul>
35     * <li>S_XMLNAMESPACEURI (http://www.w3.org/XML/1998/namespace) is the
36     * URI permanantly assigned to the "xml:" prefix. This is used for some
37     * features built into the XML specification itself, such as xml:space
38     * and xml:lang. It was defined by the W3C's XML Namespaces spec.</li>
39     * <li>S_XSLNAMESPACEURL (http://www.w3.org/1999/XSL/Transform) is the
40     * URI which indicates that a name may be an XSLT directive. In most
41     * XSLT stylesheets, this is bound to the "xsl:" prefix. It's defined
42     * by the W3C's XSLT Recommendation.</li>
43     * <li>S_OLDXSLNAMESPACEURL (http://www.w3.org/XSL/Transform/1.0) was
44     * used in early prototypes of XSLT processors for much the same purpose
45     * as S_XSLNAMESPACEURL. It is now considered obsolete, and the version
46     * of XSLT which it signified is not fully compatable with the final
47     * XSLT Recommendation, so what it really signifies is a badly obsolete
48     * stylesheet.</li>
49     * </ul> */
50    public static final String
51          S_XMLNAMESPACEURI = "http://www.w3.org/XML/1998/namespace",
52          S_XSLNAMESPACEURL = "http://www.w3.org/1999/XSL/Transform",
53          S_OLDXSLNAMESPACEURL = "http://www.w3.org/XSL/Transform/1.0";
54  
55    /** Authorship mnemonics, as Java Strings. Not standardized,
56     * as far as I know.
57     * <ul>
58     * <li>S_VENDOR -- the name of the organization/individual who published
59     * this XSLT processor. </li>
60     * <li>S_VENDORURL -- URL where one can attempt to retrieve more
61     * information about this publisher and product.</li>
62     * </ul>
63     */
64    public static final String
65          S_VENDOR = "Apache Software Foundation",
66          S_VENDORURL = "http://xml.apache.org";
67  
68    /** S_BUILTIN_EXTENSIONS_URL is a mnemonic for the XML Namespace
69     *(http://xml.apache.org/xalan) predefined to signify Xalan's
70     * built-in XSLT Extensions. When used in stylesheets, this is often
71     * bound to the "xalan:" prefix.
72     */
73    public static final String
74      S_BUILTIN_EXTENSIONS_URL = "http://xml.apache.org/xalan";
