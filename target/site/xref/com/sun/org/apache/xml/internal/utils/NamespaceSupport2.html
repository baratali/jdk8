30  /**
31   * Encapsulate Namespace tracking logic for use by SAX drivers.
32   *
33   * <p>This class is an attempt to rewrite the SAX NamespaceSupport
34   * "helper" class for improved efficiency. It can be used to track the
35   * namespace declarations currently in scope, providing lookup
36   * routines to map prefixes to URIs and vice versa.</p>
37   *
38   * <p>ISSUE: For testing purposes, I've extended NamespaceSupport even
39   * though I'm completely reasserting all behaviors and fields.
40   * Wasteful.... But SAX did not put an interface under that object and
41   * we seem to have written that SAX class into our APIs... and I don't
42   * want to argue with it right now. </p>
43   *
44   * @see org.xml.sax.helpers.NamespaceSupport
45   * */
46  public class NamespaceSupport2
47      extends org.xml.sax.helpers.NamespaceSupport
48  {

90      /**
91       * Reset this Namespace support object for reuse.
92       *
93       * <p>It is necessary to invoke this method before reusing the
94       * Namespace support object for a new session.</p>
95       */
96      public void reset ()
97      {
98          // Discarding the whole stack doesn't save us a lot versus
99          // creating a new NamespaceSupport. Do we care, or should we
100         // change this to just reset the root context?
101         currentContext = new Context2(null);
102         currentContext.declarePrefix("xml", XMLNS);
103     }

106     /**
107      * Start a new Namespace context.
108      *
109      * <p>Normally, you should push a new context at the beginning
110      * of each XML element: the new context will automatically inherit
111      * the declarations of its parent context, but it will also keep
112      * track of which declarations were made within this context.</p>
113      *
114      * <p>The Namespace support object always starts with a base context
115      * already in force: in this context, only the "xml" prefix is
116      * declared.</p>
117      *
118      * @see #popContext
119      */
120     public void pushContext ()
121     {
122         // JJK: Context has a parent pointer.
123         // That means we don't need a stack to pop.
124         // We may want to retain for reuse, but that can be done via
125         // a child pointer.
126 
127         Context2 parentContext=currentContext;
128         currentContext = parentContext.getChild();
129         if (currentContext == null){
130                 currentContext = new Context2(parentContext);
131             }
132         else{
133             // JJK: This will wipe out any leftover data
134             // if we're reusing a previously allocated Context.
135             currentContext.setParent(parentContext);
136         }
137     }
138 
139 
140     /**
141      * Revert to the previous Namespace context.
142      *
143      * <p>Normally, you should pop the context at the end of each
144      * XML element.  After popping the context, all Namespace prefix
145      * mappings that were previously in force are restored.</p>
146      *
147      * <p>You must not attempt to declare additional Namespace
148      * prefixes after popping a context, unless you push another
149      * context first.</p>
150      *
151      * @see #pushContext
152      */
153     public void popContext ()
154     {
155         Context2 parentContext=currentContext.getParent();
156         if(parentContext==null)
157             throw new EmptyStackException();
158         else
159             currentContext = parentContext;
160     }

169     /**
170      * Declare a Namespace prefix.
171      *
172      * <p>This method declares a prefix in the current Namespace
173      * context; the prefix will remain in force until this context
174      * is popped, unless it is shadowed in a descendant context.</p>
175      *
176      * <p>To declare a default Namespace, use the empty string.  The
177      * prefix must not be "xml" or "xmlns".</p>
178      *
179      * <p>Note that you must <em>not</em> declare a prefix after
180      * you've pushed and popped another Namespace.</p>
181      *
182      * <p>Note that there is an asymmetry in this library: while {@link
183      * #getPrefix getPrefix} will not return the default "" prefix,
184      * even if you have declared one; to check for a default prefix,
185      * you have to look it up explicitly using {@link #getURI getURI}.
186      * This asymmetry exists to make it easier to look up prefixes
187      * for attribute names, where the default prefix is not allowed.</p>
188      *
189      * @param prefix The prefix to declare, or null for the empty
190      *        string.
191      * @param uri The Namespace URI to associate with the prefix.
192      * @return true if the prefix was legal, false otherwise
193      * @see #processName
194      * @see #getURI
195      * @see #getPrefix
196      */
197     public boolean declarePrefix (String prefix, String uri)
198     {
199         if (prefix.equals("xml") || prefix.equals("xmlns")) {
200             return false;
201         } else {
202             currentContext.declarePrefix(prefix, uri);
203             return true;
204         }
205     }

208     /**
209      * Process a raw XML 1.0 name.
210      *
211      * <p>This method processes a raw XML 1.0 name in the current
212      * context by removing the prefix and looking it up among the
213      * prefixes currently declared.  The return value will be the
214      * array supplied by the caller, filled in as follows:</p>
215      *
216      * <dl>
217      * <dt>parts[0]</dt>
218      * <dd>The Namespace URI, or an empty string if none is
219      *  in use.</dd>
220      * <dt>parts[1]</dt>
221      * <dd>The local name (without prefix).</dd>
222      * <dt>parts[2]</dt>
223      * <dd>The original raw name.</dd>
224      * </dl>
225      *
226      * <p>All of the strings in the array will be internalized.  If
227      * the raw name has a prefix that has not been declared, then
228      * the return value will be null.</p>
229      *
230      * <p>Note that attribute names are processed differently than
231      * element names: an unprefixed element name will received the
232      * default Namespace (if any), while an unprefixed element name
233      * will not.</p>
234      *
235      * @param qName The raw XML 1.0 name to be processed.
236      * @param parts A string array supplied by the caller, capable of
237      *        holding at least three members.
238      * @param isAttribute A flag indicating whether this is an
239      *        attribute name (true) or an element name (false).
240      * @return The supplied array holding three internalized strings
241      *        representing the Namespace URI (or empty string), the
242      *        local name, and the raw XML 1.0 name; or null if there
243      *        is an undeclared prefix.
244      * @see #declarePrefix
245      * @see java.lang.String#intern */
246     public String [] processName (String qName, String[] parts,
247                                   boolean isAttribute)
248     {
249         String[] name=currentContext.processName(qName, isAttribute);
250         if(name==null)
251             return null;
252 
253         // JJK: This recopying is required because processName may return
254         // a cached result. I Don't Like It. *****
255         System.arraycopy(name,0,parts,0,3);
256         return parts;
257     }

321     /**
322      * Return an enumeration of all prefixes currently declared for a URI.
323      *
324      * <p>This method returns prefixes mapped to a specific Namespace
325      * URI.  The xml: prefix will be included.  If you want only one
326      * prefix that's mapped to the Namespace URI, and you don't care
327      * which one you get, use the {@link #getPrefix getPrefix}
328      *  method instead.</p>
329      *
330      * <p><strong>Note:</strong> the empty (default) prefix is
331      * <em>never</em> included in this enumeration; to check for the
332      * presence of a default Namespace, use the {@link #getURI getURI}
333      * method with an argument of "".</p>
334      *
335      * @param uri The Namespace URI.
336      * @return An enumeration of all prefixes declared in the
337      *         current context.
338      * @see #getPrefix
339      * @see #getDeclaredPrefixes
340      * @see #getURI */
341     public Enumeration getPrefixes (String uri)
342     {
343         // JJK: The old code involved creating a vector, filling it
344         // with all the matching prefixes, and then getting its
345         // elements enumerator. Wastes storage, wastes cycles if we
346         // don't actually need them all. Better to either implement
347         // a specific enumerator for these prefixes... or a filter
348         // around the all-prefixes enumerator, which comes out to
349         // roughly the same thing.
350         //
351         // **** Currently a filter. That may not be most efficient
352         // when I'm done restructuring storage!
353         return new PrefixForUriEnumerator(this,uri,getPrefixes());
354     }

366      * @see #getPrefixes

384 /**
385  * Implementation of Enumeration filter, wrapped
386  * aroung the get-all-prefixes version of the operation. This is NOT
387  * necessarily the most efficient approach; finding the URI and then asking
388  * what prefixes apply to it might make much more sense.
389  */
390 class PrefixForUriEnumerator implements Enumeration
391 {
392     private Enumeration allPrefixes;
393     private String uri;
394     private String lookahead=null;
395     private NamespaceSupport2 nsup;
396 
397     // Kluge: Since one can't do a constructor on an
398     // anonymous class (as far as I know)...
399     PrefixForUriEnumerator(NamespaceSupport2 nsup,String uri, Enumeration allPrefixes)
400     {
401         this.nsup=nsup;
402         this.uri=uri;
403         this.allPrefixes=allPrefixes;
404     }
405 
406     public boolean hasMoreElements()
407     {
408         if(lookahead!=null)
409             return true;
410 
411         while(allPrefixes.hasMoreElements())
412             {
413                 String prefix=(String)allPrefixes.nextElement();
414                 if(uri.equals(nsup.getURI(prefix)))
415                     {
416                         lookahead=prefix;
417                         return true;
418                     }
419             }
420         return false;
421     }
422 
423     public Object nextElement()
424     {
425         if(hasMoreElements())
426             {
427                 String tmp=lookahead;
428                 lookahead=null;
429                 return tmp;
430             }
431         else
432             throw new java.util.NoSuchElementException();
433     }
434 }

436 /**
437  * Internal class for a single Namespace context.
438  *
439  * <p>This module caches and reuses Namespace contexts, so the number allocated
440  * will be equal to the element depth of the document, not to the total
441  * number of elements (i.e. 5-10 rather than tens of thousands).</p>
442  */
443 final class Context2 {
444 
445     ////////////////////////////////////////////////////////////////
446     // Manefest Constants
447     ////////////////////////////////////////////////////////////////
448 
449     /**
450      * An empty enumeration.
451      */
452     private final static Enumeration EMPTY_ENUMERATION =
453         new Vector().elements();
454 
455     ////////////////////////////////////////////////////////////////
456     // Protected state.
457     ////////////////////////////////////////////////////////////////
458 
459     Hashtable prefixTable;
460     Hashtable uriTable;
461     Hashtable elementNameTable;
462     Hashtable attributeNameTable;
463     String defaultNS = null;
464 
465     ////////////////////////////////////////////////////////////////
466     // Internal state.
467     ////////////////////////////////////////////////////////////////
468 
469     private Vector declarations = null;
470     private boolean tablesDirty = false;
471     private Context2 parent = null;
472     private Context2 child = null;
473 
474     /**
475      * Create a new Namespace context.
476      */
477     Context2 (Context2 parent)
478     {
479         if(parent==null)
480             {
481                 prefixTable = new Hashtable();
482                 uriTable = new Hashtable();
483                 elementNameTable=null;
484                 attributeNameTable=null;
485             }
486         else
487             setParent(parent);
488     }
489 
490 
491     /**
492      * @returns The child Namespace context object, or null if this
493      * is the last currently on the chain.
494      */
495     Context2 getChild()
496     {
497         return child;
498     }
499 
500     /**
501      * @returns The parent Namespace context object, or null if this
502      * is the root.
503      */
504     Context2 getParent()
505     {
506         return parent;
507     }
508 
509     /**
510      * (Re)set the parent of this Namespace context.
511      * This is separate from the c'tor because it's re-applied
512      * when a Context2 is reused by push-after-pop.
513      *
514      * @param parent The parent Namespace context object.
515      */
516     void setParent (Context2 parent)
517     {
518         this.parent = parent;
519         parent.child = this;        // JJK: Doubly-linked
520         declarations = null;
521         prefixTable = parent.prefixTable;
522         uriTable = parent.uriTable;
523         elementNameTable = parent.elementNameTable;
524         attributeNameTable = parent.attributeNameTable;
525         defaultNS = parent.defaultNS;
526         tablesDirty = false;
527     }
528 
529 
530     /**
531      * Declare a Namespace prefix for this context.
532      *
533      * @param prefix The prefix to declare.
534      * @param uri The associated Namespace URI.
535      * @see org.xml.sax.helpers.NamespaceSupport2#declarePrefix
536      */
537     void declarePrefix (String prefix, String uri)
538     {
539                                 // Lazy processing...
540         if (!tablesDirty) {
541             copyTables();
542         }
543         if (declarations == null) {
544             declarations = new Vector();
545         }
546 
547         prefix = prefix.intern();
548         uri = uri.intern();
549         if ("".equals(prefix)) {
550             if ("".equals(uri)) {
551                 defaultNS = null;
552             } else {
553                 defaultNS = uri;
554             }
555         } else {
556             prefixTable.put(prefix, uri);
557             uriTable.put(uri, prefix); // may wipe out another prefix
558         }
559         declarations.addElement(prefix);
560     }
561 
562 
563     /**
564      * Process a raw XML 1.0 name in this context.
565      *
566      * @param qName The raw XML 1.0 name.
567      * @param isAttribute true if this is an attribute name.
568      * @return An array of three strings containing the
569      *         URI part (or empty string), the local part,
570      *         and the raw name, all internalized, or null
571      *         if there is an undeclared prefix.
572      * @see org.xml.sax.helpers.NamespaceSupport2#processName
573      */
574     String [] processName (String qName, boolean isAttribute)
575     {
576         String name[];
577         Hashtable table;
578 
579                                 // Select the appropriate table.
580         if (isAttribute) {
581             if(elementNameTable==null)
582                 elementNameTable=new Hashtable();
583             table = elementNameTable;
584         } else {
585             if(attributeNameTable==null)
586                 attributeNameTable=new Hashtable();
587             table = attributeNameTable;
588         }
589 
590                                 // Start by looking in the cache, and
591                                 // return immediately if the name
592                                 // is already known in this content
593         name = (String[])table.get(qName);
594         if (name != null) {
595             return name;
596         }
597 
598                                 // We haven't seen this name in this
599                                 // context before.
600         name = new String[3];
601         int index = qName.indexOf(':');
602 
603 
604                                 // No prefix.
605         if (index == -1) {
606             if (isAttribute || defaultNS == null) {
607                 name[0] = "";
608             } else {
609                 name[0] = defaultNS;
610             }
611             name[1] = qName.intern();
612             name[2] = name[1];
613         }
614 
615                                 // Prefix
616         else {
617             String prefix = qName.substring(0, index);
618             String local = qName.substring(index+1);
619             String uri;
620             if ("".equals(prefix)) {
621                 uri = defaultNS;
622             } else {
623                 uri = (String)prefixTable.get(prefix);
624             }
625             if (uri == null) {
626                 return null;
627             }
628             name[0] = uri;
629             name[1] = local.intern();
630             name[2] = qName.intern();
631         }
632 
633                                 // Save in the cache for future use.
634         table.put(name[2], name);
635         tablesDirty = true;
636         return name;
637     }
638 
639 
640     /**
641      * Look up the URI associated with a prefix in this context.
642      *
643      * @param prefix The prefix to look up.
644      * @return The associated Namespace URI, or null if none is
645      *         declared.
646      * @see org.xml.sax.helpers.NamespaceSupport2#getURI
647      */
648     String getURI (String prefix)
649     {
650         if ("".equals(prefix)) {
651             return defaultNS;
652         } else if (prefixTable == null) {
653             return null;
654         } else {
655             return (String)prefixTable.get(prefix);
656         }
657     }
658 
659 
660     /**
661      * Look up one of the prefixes associated with a URI in this context.
662      *
663      * <p>Since many prefixes may be mapped to the same URI,
664      * the return value may be unreliable.</p>
665      *
666      * @param uri The URI to look up.
667      * @return The associated prefix, or null if none is declared.
668      * @see org.xml.sax.helpers.NamespaceSupport2#getPrefix
669      */
670     String getPrefix (String uri)
671     {
672         if (uriTable == null) {
673             return null;
674         } else {
675             return (String)uriTable.get(uri);
676         }
677     }
678 
679 
680     /**
681      * Return an enumeration of prefixes declared in this context.
682      *
683      * @return An enumeration of prefixes (possibly empty).
684      * @see org.xml.sax.helpers.NamespaceSupport2#getDeclaredPrefixes
685      */
686     Enumeration getDeclaredPrefixes ()
687     {
688         if (declarations == null) {
689             return EMPTY_ENUMERATION;
690         } else {
691             return declarations.elements();
692         }
693     }
694 
695 
696     /**
697      * Return an enumeration of all prefixes currently in force.
698      *
699      * <p>The default prefix, if in force, is <em>not</em>
700      * returned, and will have to be checked for separately.</p>
701      *
702      * @return An enumeration of prefixes (never empty).
703      * @see org.xml.sax.helpers.NamespaceSupport2#getPrefixes
704      */
705     Enumeration getPrefixes ()
706     {
707         if (prefixTable == null) {
708             return EMPTY_ENUMERATION;
709         } else {
710             return prefixTable.keys();
711         }
712     }
713 
714     ////////////////////////////////////////////////////////////////
715     // Internal methods.
716     ////////////////////////////////////////////////////////////////
717 
718     /**
719      * Copy on write for the internal tables in this context.
720      *
721      * <p>This class is optimized for the normal case where most
722      * elements do not contain Namespace declarations. In that case,
723      * the Context2 will share data structures with its parent.
724      * New tables are obtained only when new declarations are issued,
725      * so they can be popped off the stack.</p>
726      *
727      * <p> JJK: **** Alternative: each Context2 might declare
728      *  _only_ its local bindings, and delegate upward if not found.</p>
729      */
730     private void copyTables ()
731     {
732         // Start by copying our parent's bindings
733         prefixTable = (Hashtable)prefixTable.clone();
734         uriTable = (Hashtable)uriTable.clone();
735 
736         // Replace the caches with empty ones, rather than
737         // trying to determine which bindings should be flushed.
738         // As far as I can tell, these caches are never actually
739         // used in Xalan... More efficient to remove the whole
740         // cache system? ****
741         if(elementNameTable!=null)
742             elementNameTable=new Hashtable();
743         if(attributeNameTable!=null)
744             attributeNameTable=new Hashtable();
745         tablesDirty = true;
746     }

748 }
