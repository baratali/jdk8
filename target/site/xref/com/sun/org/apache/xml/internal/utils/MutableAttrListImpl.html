23  package com.sun.org.apache.xml.internal.utils;
24  
25  import java.io.Serializable;
26  
27  import org.xml.sax.Attributes;
28  import org.xml.sax.helpers.AttributesImpl;
29  
30  /**
31   * Mutable version of AttributesImpl.
32   * @xsl.usage advanced
33   */
34  public class MutableAttrListImpl extends AttributesImpl
35          implements Serializable
36  {
37      static final long serialVersionUID = 6289452013442934470L;
38  
39  /**
40   * Construct a new, empty AttributesImpl object.
41   */
42  
43  public MutableAttrListImpl()
44    {
45      super();
46    }
47  
48    /**
49     * Copy an existing Attributes object.
50     *
51     * <p>This constructor is especially useful inside a start
52     * element event.</p>
53     *
54     * @param atts The existing Attributes object.
55     */
56    public MutableAttrListImpl(Attributes atts)
57    {
58      super(atts);
59    }
60  
61    /**
62     * Add an attribute to the end of the list.
63     *
64     * <p>For the sake of speed, this method does no checking
65     * to see if the attribute is already in the list: that is
66     * the responsibility of the application.</p>
67     *
68     * @param uri The Namespace URI, or the empty string if
69     *        none is available or Namespace processing is not
70     *        being performed.
71     * @param localName The local name, or the empty string if
72     *        Namespace processing is not being performed.
73     * @param qName The qualified (prefixed) name, or the empty string
74     *        if qualified names are not available.
75     * @param type The attribute type as a string.
76     * @param value The attribute value.
77     */
78    public void addAttribute(String uri, String localName, String qName,
79                             String type, String value)
80    {
81  
82      if (null == uri)
83        uri = "";
84  
85      // getIndex(qName) seems to be more reliable than getIndex(uri, localName),
86      // in the case of the xmlns attribute anyway.
87      int index = this.getIndex(qName);
88      // int index = this.getIndex(uri, localName);
89  
90      // System.out.println("MutableAttrListImpl#addAttribute: "+uri+":"+localName+", "+index+", "+qName+", "+this);
91  
92      if (index >= 0)
93        this.setAttribute(index, uri, localName, qName, type, value);
94      else
95        super.addAttribute(uri, localName, qName, type, value);
96    }
97  
98    /**
99     * Add the contents of the attribute list to this list.
100    *
101    * @param atts List of attributes to add to this list
102    */
103   public void addAttributes(Attributes atts)
104   {
105 
106     int nAtts = atts.getLength();
107 
108     for (int i = 0; i < nAtts; i++)
109     {
110       String uri = atts.getURI(i);
111 
112       if (null == uri)
113         uri = "";
114 
115       String localName = atts.getLocalName(i);
116       String qname = atts.getQName(i);
117       int index = this.getIndex(uri, localName);
118       // System.out.println("MutableAttrListImpl#addAttributes: "+uri+":"+localName+", "+index+", "+atts.getQName(i)+", "+this);
119       if (index >= 0)
120         this.setAttribute(index, uri, localName, qname, atts.getType(i),
121                           atts.getValue(i));
122       else
123         addAttribute(uri, localName, qname, atts.getType(i),
124                      atts.getValue(i));
125     }
126   }
127 
128   /**
129    * Return true if list contains the given (raw) attribute name.
130    *
131    * @param name Raw name of attribute to look for
132    *
133    * @return true if an attribute is found with this name
134    */
135   public boolean contains(String name)
136   {
137     return getValue(name) != null;
138   }
139 }
