215     /**
216      * Method translateURItoJCEID
217      *
218      * @param algorithmURI
219      * @return the JCE standard name corresponding to the given URI
220      */
221     public static String translateURItoJCEID(String algorithmURI) {
222         if (log.isLoggable(java.util.logging.Level.FINE)) {
223             log.log(java.util.logging.Level.FINE, "Request for URI " + algorithmURI);
224         }
225 
226         Algorithm algorithm = algorithmsMap.get(algorithmURI);
227         if (algorithm != null) {
228             return algorithm.jceName;
229         }
230         return null;
231     }

300     /**
301      * Represents the Algorithm xml element
302      */
303     public static class Algorithm {
304 
305         final String requiredKey;
306         final String jceName;
307         final String algorithmClass;
308         final int keyLength;
309 
310         /**
311          * Gets data from element
312          * @param el
313          */
314         public Algorithm(Element el) {
315             requiredKey = el.getAttribute("RequiredKey");
316             jceName = el.getAttribute("JCEName");
317             algorithmClass = el.getAttribute("AlgorithmClass");
318             if (el.hasAttribute("KeyLength")) {
319                 keyLength = Integer.parseInt(el.getAttribute("KeyLength"));
320             } else {
321                 keyLength = 0;
322             }
323         }
324 
325         public Algorithm(String requiredKey, String jceName) {
326             this(requiredKey, jceName, null, 0);
327         }
328 
329         public Algorithm(String requiredKey, String jceName, String algorithmClass) {
330             this(requiredKey, jceName, algorithmClass, 0);
331         }
332 
333         public Algorithm(String requiredKey, String jceName, int keyLength) {
334             this(requiredKey, jceName, null, keyLength);
335         }
336 
337         public Algorithm(String requiredKey, String jceName, String algorithmClass, int keyLength) {
338             this.requiredKey = requiredKey;
339             this.jceName = jceName;
340             this.algorithmClass = algorithmClass;
341             this.keyLength = keyLength;
342         }
343     }
