60  import java.util.LinkedList;
61  import com.sun.org.apache.bcel.internal.classfile.JavaClass;
62  
63  /**
64   * Utility class implementing a (typesafe) queue of JavaClass
65   * objects.
66   *
67   * @author <A HREF="mailto:markus.dahm@berlin.de">M. Dahm</A>
68   * @see ClassVector
69  */
70  public class ClassQueue implements java.io.Serializable {
71    protected LinkedList vec  = new LinkedList();
72  
73    public void enqueue(JavaClass clazz) { vec.addLast(clazz); }
74  
75    public JavaClass dequeue()                {
76      return (JavaClass)vec.removeFirst();
77    }
78  
79    public boolean empty() { return vec.isEmpty(); }
80  
81    public String toString() {
82      return vec.toString();
83    }
84  }
