67          super.installDefaults();
68  
69          titlePaneHeight = UIManager.getInt("InternalFrame.titlePaneHeight");
70          buttonWidth     = UIManager.getInt("InternalFrame.titleButtonWidth")  - 4;
71          buttonHeight    = UIManager.getInt("InternalFrame.titleButtonHeight") - 4;
72  
73          Object obj      = UIManager.get("InternalFrame.titleButtonToolTipsOn");
74          hotTrackingOn = (obj instanceof Boolean) ? (Boolean)obj : true;
75  
76  
77          if (XPStyle.getXP() != null) {
78              // Fix for XP bug where sometimes these sizes aren't updated properly
79              // Assume for now that height is correct and derive width using the
80              // ratio from the uxtheme part
81              buttonWidth = buttonHeight;
82              Dimension d = XPStyle.getPartSize(Part.WP_CLOSEBUTTON, State.NORMAL);
83              if (d != null && d.width != 0 && d.height != 0) {
84                  buttonWidth = (int) ((float) buttonWidth * d.width / d.height);
85              }
86          } else {
87              buttonWidth += 2;
88              selectedTitleGradientColor =
89                      UIManager.getColor("InternalFrame.activeTitleGradient");
90              notSelectedTitleGradientColor =
91                      UIManager.getColor("InternalFrame.inactiveTitleGradient");
92              Color activeBorderColor =
93                      UIManager.getColor("InternalFrame.activeBorderColor");
94              setBorder(BorderFactory.createLineBorder(activeBorderColor, 1));
95          }
96      }

123     public void paintComponent(Graphics g)  {
124         XPStyle xp = XPStyle.getXP();
125 
126         paintTitleBackground(g);
127 
128         String title = frame.getTitle();
129         if (title != null) {
130             boolean isSelected = frame.isSelected();
131             Font oldFont = g.getFont();
132             Font newFont = (titleFont != null) ? titleFont : getFont();
133             g.setFont(newFont);
134 
135             // Center text vertically.
136             FontMetrics fm = SwingUtilities2.getFontMetrics(frame, g, newFont);
137             int baseline = (getHeight() + fm.getAscent() - fm.getLeading() -
138                     fm.getDescent()) / 2;
139 
140             Rectangle lastIconBounds = new Rectangle(0, 0, 0, 0);
141             if (frame.isIconifiable()) {
142                 lastIconBounds = iconButton.getBounds();
143             } else if (frame.isMaximizable()) {
144                 lastIconBounds = maxButton.getBounds();
145             } else if (frame.isClosable()) {
146                 lastIconBounds = closeButton.getBounds();
147             }
148 
149             int titleX;
150             int titleW;
151             int gap = 2;
152             if (WindowsGraphicsUtils.isLeftToRight(frame)) {
153                 if (lastIconBounds.x == 0) { // There are no icons
154                     lastIconBounds.x = frame.getWidth() - frame.getInsets().right;
155                 }
156                 titleX = systemLabel.getX() + systemLabel.getWidth() + gap;
157                 if (xp != null) {
158                     titleX += 2;
159                 }
160                 titleW = lastIconBounds.x - titleX - gap;
161             } else {
162                 if (lastIconBounds.x == 0) { // There are no icons
163                     lastIconBounds.x = frame.getInsets().left;
164                 }
165                 titleW = SwingUtilities2.stringWidth(frame, fm, title);
166                 int minTitleX = lastIconBounds.x + lastIconBounds.width + gap;
167                 if (xp != null) {
168                     minTitleX += 2;
169                 }
170                 int availableWidth = systemLabel.getX() - gap - minTitleX;
171                 if (availableWidth > titleW) {
172                     titleX = systemLabel.getX() - gap - titleW;
173                 } else {
174                     titleX = minTitleX;
175                     titleW = availableWidth;
176                 }
177             }
178             title = getTitle(frame.getTitle(), fm, titleW);
179 
180             if (xp != null) {
181                 String shadowType = null;
182                 if (isSelected) {
183                     shadowType = xp.getString(this, Part.WP_CAPTION,
184                                               State.ACTIVE, Prop.TEXTSHADOWTYPE);
185                 }
186                 if ("single".equalsIgnoreCase(shadowType)) {
187                     Point shadowOffset = xp.getPoint(this, Part.WP_WINDOW, State.ACTIVE,
188                                                      Prop.TEXTSHADOWOFFSET);
189                     Color shadowColor  = xp.getColor(this, Part.WP_WINDOW, State.ACTIVE,
190                                                      Prop.TEXTSHADOWCOLOR, null);
191                     if (shadowOffset != null && shadowColor != null) {
192                         g.setColor(shadowColor);
193                         SwingUtilities2.drawString(frame, g, title,
194                                      titleX + shadowOffset.x,
195                                      baseline + shadowOffset.y);
196                     }
197                 }
198             }
199             g.setColor(isSelected ? selectedTextColor : notSelectedTextColor);
200             SwingUtilities2.drawString(frame, g, title, titleX, baseline);
201             g.setFont(oldFont);
202         }
203     }

209     public Dimension getMinimumSize() {
210         Dimension d = new Dimension(super.getMinimumSize());
211         d.height = titlePaneHeight + 2;
212 
213         XPStyle xp = XPStyle.getXP();
214         if (xp != null) {
215             // Note: Don't know how to calculate height on XP,
216             // the captionbarheight is 25 but native caption is 30 (maximized 26)
217             if (frame.isMaximum()) {
218                 d.height -= 1;
219             } else {
220                 d.height += 3;
221             }
222         }
223         return d;
224     }

226     protected void paintTitleBackground(Graphics g) {
227         XPStyle xp = XPStyle.getXP();
228         if (xp != null) {
229             Part part = frame.isIcon() ? Part.WP_MINCAPTION
230                                        : (frame.isMaximum() ? Part.WP_MAXCAPTION
231                                                             : Part.WP_CAPTION);
232             State state = frame.isSelected() ? State.ACTIVE : State.INACTIVE;
233             Skin skin = xp.getSkin(this, part);
234             skin.paintSkin(g, 0,  0, getWidth(), getHeight(), state);
235         } else {
236             Boolean gradientsOn = (Boolean)LookAndFeel.getDesktopPropertyValue(
237                 "win.frame.captionGradientsOn", Boolean.valueOf(false));
238             if (gradientsOn.booleanValue() && g instanceof Graphics2D) {
239                 Graphics2D g2 = (Graphics2D)g;
240                 Paint savePaint = g2.getPaint();
241 
242                 boolean isSelected = frame.isSelected();
243                 int w = getWidth();
244 
245                 if (isSelected) {
246                     GradientPaint titleGradient = new GradientPaint(0,0,
247                             selectedTitleColor,
248                             (int)(w*.75),0,
249                             selectedTitleGradientColor);
250                     g2.setPaint(titleGradient);
251                 } else {
252                     GradientPaint titleGradient = new GradientPaint(0,0,
253                             notSelectedTitleColor,
254                             (int)(w*.75),0,
255                             notSelectedTitleGradientColor);
256                     g2.setPaint(titleGradient);
257                 }
258                 g2.fillRect(0, 0, getWidth(), getHeight());
259                 g2.setPaint(savePaint);
260             } else {
261                 super.paintTitleBackground(g);
262             }
263         }
264     }

266     protected void assembleSystemMenu() {
267         systemPopupMenu = new JPopupMenu();
268         addSystemMenuItems(systemPopupMenu);
269         enableActions();
270         systemLabel = new JLabel(frame.getFrameIcon()) {
271             protected void paintComponent(Graphics g) {
272                 int x = 0;
273                 int y = 0;
274                 int w = getWidth();
275                 int h = getHeight();
276                 g = g.create();  // Create scratch graphics
277                 if (isOpaque()) {
278                     g.setColor(getBackground());
279                     g.fillRect(0, 0, w, h);
280                 }
281                 Icon icon = getIcon();
282                 int iconWidth;
283                 int iconHeight;
284                 if (icon != null &&
285                     (iconWidth = icon.getIconWidth()) > 0 &&
286                     (iconHeight = icon.getIconHeight()) > 0) {
287 
288                     // Set drawing scale to make icon scale to our desired size
289                     double drawScale;
290                     if (iconWidth > iconHeight) {
291                         // Center icon vertically
292                         y = (h - w*iconHeight/iconWidth) / 2;
293                         drawScale = w / (double)iconWidth;
294                     } else {
295                         // Center icon horizontally
296                         x = (w - h*iconWidth/iconHeight) / 2;
297                         drawScale = h / (double)iconHeight;
298                     }
299                     ((Graphics2D)g).translate(x, y);
300                     ((Graphics2D)g).scale(drawScale, drawScale);
301                     icon.paintIcon(this, g, 0, 0);
302                 }
303                 g.dispose();
304             }
305         };
306         systemLabel.addMouseListener(new MouseAdapter() {
307             public void mouseClicked(MouseEvent e) {
308                 if (e.getClickCount() == 2 && frame.isClosable() &&
309                     !frame.isIcon()) {
310                     systemPopupMenu.setVisible(false);
311                     frame.doDefaultCloseAction();
312                 }
313                 else {
314                     super.mouseClicked(e);
315                 }
316             }
317             public void mousePressed(MouseEvent e) {
318                 try {
319                     frame.setSelected(true);
320                 } catch(PropertyVetoException pve) {
321                 }
322                 showSystemPopupMenu(e.getComponent());
323             }
324         });
325     }

356     private void showSystemPopupMenu(Component invoker){
357         Dimension dim = new Dimension();
358         Border border = frame.getBorder();
359         if (border != null) {
360             dim.width += border.getBorderInsets(frame).left +
361                 border.getBorderInsets(frame).right;
362             dim.height += border.getBorderInsets(frame).bottom +
363                 border.getBorderInsets(frame).top;
364         }
365         if (!frame.isIcon()) {
366             systemPopupMenu.show(invoker,
367                 getX() - dim.width,
368                 getY() + getHeight() - dim.height);
369         } else {
370             systemPopupMenu.show(invoker,
371                 getX() - dim.width,
372                 getY() - systemPopupMenu.getPreferredSize().height -
373                      dim.height);
374         }
375     }

385     public class WindowsTitlePaneLayout extends BasicInternalFrameTitlePane.TitlePaneLayout {
386         private Insets captionMargin = null;
387         private Insets contentMargin = null;
388         private XPStyle xp = XPStyle.getXP();
389 
390         WindowsTitlePaneLayout() {
391             if (xp != null) {
392                 Component c = WindowsInternalFrameTitlePane.this;
393                 captionMargin = xp.getMargin(c, Part.WP_CAPTION, null, Prop.CAPTIONMARGINS);
394                 contentMargin = xp.getMargin(c, Part.WP_CAPTION, null, Prop.CONTENTMARGINS);
395             }
396             if (captionMargin == null) {
397                 captionMargin = new Insets(0, 2, 0, 2);
398             }
399             if (contentMargin == null) {
400                 contentMargin = new Insets(0, 0, 0, 0);
401             }
402         }

404         private int layoutButton(JComponent button, Part part,
405                                  int x, int y, int w, int h, int gap,
406                                  boolean leftToRight) {
407             if (!leftToRight) {
408                 x -= w;
409             }
410             button.setBounds(x, y, w, h);
411             if (leftToRight) {
412                 x += w + 2;
413             } else {
414                 x -= 2;
415             }
416             return x;
417         }
418 
419         public void layoutContainer(Container c) {
420             boolean leftToRight = WindowsGraphicsUtils.isLeftToRight(frame);
421             int x, y;
422             int w = getWidth();
423             int h = getHeight();
424 
425             // System button
426             // Note: this icon is square, but the buttons aren't always.
427             int iconSize = (xp != null) ? (h-2)*6/10 : h-4;
428             if (xp != null) {
429                 x = (leftToRight) ? captionMargin.left + 2 : w - captionMargin.right - 2;
430             } else {
431                 x = (leftToRight) ? captionMargin.left : w - captionMargin.right;
432             }
433             y = (h - iconSize) / 2;
434             layoutButton(systemLabel, Part.WP_SYSBUTTON,
435                          x, y, iconSize, iconSize, 0,
436                          leftToRight);
437 
438             // Right hand buttons
439             if (xp != null) {
440                 x = (leftToRight) ? w - captionMargin.right - 2 : captionMargin.left + 2;
441                 y = 1;  // XP seems to ignore margins and offset here
442                 if (frame.isMaximum()) {
443                     y += 1;
444                 } else {
445                     y += 5;
446                 }
447             } else {
448                 x = (leftToRight) ? w - captionMargin.right : captionMargin.left;
449                 y = (h - buttonHeight) / 2;
450             }
451 
452             if(frame.isClosable()) {
453                 x = layoutButton(closeButton, Part.WP_CLOSEBUTTON,
454                                  x, y, buttonWidth, buttonHeight, 2,
455                                  !leftToRight);
456             }
457 
458             if(frame.isMaximizable()) {
459                 x = layoutButton(maxButton, Part.WP_MAXBUTTON,
460                                  x, y, buttonWidth, buttonHeight, (xp != null) ? 2 : 0,
461                                  !leftToRight);
462             }
463 
464             if(frame.isIconifiable()) {
465                 layoutButton(iconButton, Part.WP_MINBUTTON,
466                              x, y, buttonWidth, buttonHeight, 0,
467                              !leftToRight);
468             }
469         }
470     } // end WindowsTitlePaneLayout

486     /**
487      * A versatile Icon implementation which can take an array of Icon
488      * instances (typically <code>ImageIcon</code>s) and choose one that gives the best
489      * quality for a given Graphics2D scale factor when painting.
490      * <p>
491      * The class is public so it can be instantiated by UIDefaults.ProxyLazyValue.
492      * <p>
493      * Note: We assume here that icons are square.
494      */
495     public static class ScalableIconUIResource implements Icon, UIResource {
496         // We can use an arbitrary size here because we scale to it in paintIcon()
497         private static final int SIZE = 16;
498 
499         private Icon[] icons;
500 
501         /**
502          * @params objects an array of Icon or UIDefaults.LazyValue
503          * <p>
504          * The constructor is public so it can be called by UIDefaults.ProxyLazyValue.
505          */
506         public ScalableIconUIResource(Object[] objects) {
507             this.icons = new Icon[objects.length];
508 
509             for (int i = 0; i < objects.length; i++) {
510                 if (objects[i] instanceof UIDefaults.LazyValue) {
511                     icons[i] = (Icon)((UIDefaults.LazyValue)objects[i]).createValue(null);
512                 } else {
513                     icons[i] = (Icon)objects[i];
514                 }
515             }
516         }
517 
518         /**
519          * @return the <code>Icon</code> closest to the requested size
520          */
521         protected Icon getBestIcon(int size) {
522             if (icons != null && icons.length > 0) {
523                 int bestIndex = 0;
524                 int minDiff = Integer.MAX_VALUE;
525                 for (int i=0; i < icons.length; i++) {
526                     Icon icon = icons[i];
527                     int iconSize;
528                     if (icon != null && (iconSize = icon.getIconWidth()) > 0) {
529                         int diff = Math.abs(iconSize - size);
530                         if (diff < minDiff) {
531                             minDiff = diff;
532                             bestIndex = i;
533                         }
534                     }
535                 }
536                 return icons[bestIndex];
537             } else {
538                 return null;
539             }
540         }

542         public void paintIcon(Component c, Graphics g, int x, int y) {
543             Graphics2D g2d = (Graphics2D)g.create();
544             // Calculate how big our drawing area is in pixels
545             // Assume we are square
546             int size = getIconWidth();
547             double scale = g2d.getTransform().getScaleX();
548             Icon icon = getBestIcon((int)(size * scale));
549             int iconSize;
550             if (icon != null && (iconSize = icon.getIconWidth()) > 0) {
551                 // Set drawing scale to make icon act true to our reported size
552                 double drawScale = size / (double)iconSize;
553                 g2d.translate(x, y);
554                 g2d.scale(drawScale, drawScale);
555                 icon.paintIcon(c, g2d, 0, 0);
556             }
557             g2d.dispose();
558         }
