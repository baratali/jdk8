26  package com.sun.java.swing.plaf.windows;
27  
28  import java.awt.*;
29  import javax.swing.JSplitPane;
30  import javax.swing.UIManager;
31  import javax.swing.plaf.basic.BasicSplitPaneUI;
32  import javax.swing.plaf.basic.BasicSplitPaneDivider;
33  
34  
35  /**
36   * Divider used for Windows split pane.
37   * <p>
38   * <strong>Warning:</strong>
39   * Serialized objects of this class will not be compatible with
40   * future Swing releases.  The current serialization support is appropriate
41   * for short term storage or RMI between applications running the same
42   * version of Swing.  A future release of Swing will provide support for
43   * long term persistence.
44   *
45   * @author Jeff Dinkins
46   */
47  public class WindowsSplitPaneDivider extends BasicSplitPaneDivider
48  {
49  
50      /**
51       * Creates a new Windows SplitPaneDivider
52       */
53      public WindowsSplitPaneDivider(BasicSplitPaneUI ui) {
54          super(ui);
55      }
56  
57      /**
58        * Paints the divider.
59        */
60      public void paint(Graphics g) {
61          Color bgColor = (splitPane.hasFocus()) ?
62                              UIManager.getColor("SplitPane.shadow") :
63                              getBackground();
64          Dimension size = getSize();
65  
66          if(bgColor != null) {
67              g.setColor(bgColor);
68              g.fillRect(0, 0, size.width, size.height);
69          }
70          super.paint(g);
71      }
72  }
