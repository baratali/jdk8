25  package com.sun.jmx.snmp.mpm;
26  
27  import com.sun.jmx.snmp.SnmpSecurityParameters;
28  import com.sun.jmx.snmp.SnmpMsg;
29  /**
30   * The translator interface is implemented by classes dealing with a specific SNMP protocol version. SnmpMsgTranslator are used in conjonction with SnmpMsgProcessingModel implementations.
31   * <p><b>This API is a Sun Microsystems internal API  and is subject
32   * to change without notice.</b></p>
33   * @since 1.5
34   */
35  public interface SnmpMsgTranslator {
36      /**
37       * Returns the request or message Id contained in the passed message. The message is a generic one that is narrowed in the object implementing this interface.
38       */
39      public int getMsgId(SnmpMsg msg);
40      /**
41       * Returns the response max message size. The message is a generic one that is narrowed in the object implementing this interface.
42       */
43      public int getMsgMaxSize(SnmpMsg msg);
44      /**
45       * Returns the message flags. The message is a generic one that is narrowed in the object implementing this interface.
46       */
47      public byte getMsgFlags(SnmpMsg msg);
48      /**
49       * Returns the message security model. The message is a generic one that is narrowed in the object implementing this interface.
50       */
51      public int getMsgSecurityModel(SnmpMsg msg);
52      /**
53       * Returns the message security level. The message is a generic one that is narrowed in the object implementing this interface.
54       */
55      public int getSecurityLevel(SnmpMsg msg);
56       /**
57       * Returns an encoded representation of security parameters contained in the passed msg. The message is a generic one that is narrowed in the object implementing this interface.
58       */
59      public byte[] getFlatSecurityParameters(SnmpMsg msg);
60      /**
61       * Returns the message security parameters. The message is a generic one that is narrowed in the object implementing this interface.
62       */
63      public SnmpSecurityParameters getSecurityParameters(SnmpMsg msg);
64      /**
65       * Returns the message context Engine Id. The message is a generic one that is narrowed in the object implementing this interface.
66       */
67      public byte[] getContextEngineId(SnmpMsg msg);
68      /**
69       * Returns the message context name. The message is a generic one that is narrowed in the object implementing this interface.
70       */
71      public byte[] getContextName(SnmpMsg msg);
72      /**
73       * Returns the raw message context name. Raw mean as it is received from the network, without translation. It can be useful when some data are piggy backed in the context name.The message is a generic one that is narrowed in the object implementing this interface.
74       */
75      public byte[] getRawContextName(SnmpMsg msg);
76      /**
77       * Returns the message accesscontext name. This access context name is used when dealing with access rights (eg: community for V1/V2 or context name for V3).The message is a generic one that is narrowed in the object implementing this interface.
78       */
79      public byte[] getAccessContext(SnmpMsg msg);
80      /**
81       * Returns the message encrypted pdu or null if no encryption. The message is a generic one that is narrowed in the object implementing this interface.
82       */
83      public byte[] getEncryptedPdu(SnmpMsg msg);
84      /**
85       * Set the context name of the passed message.
86       */
87      public void setContextName(SnmpMsg req, byte[] contextName);
88       /**
89       * Set the context engine Id of the passed message.
90       */
91      public void setContextEngineId(SnmpMsg req, byte[] contextEngineId);
92  }
