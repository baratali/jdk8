8   package com.sun.jmx.snmp;
9   
10  import java.io.Serializable;
11  import java.io.UnsupportedEncodingException;
12  import java.util.Objects;
13  
14  import com.sun.jmx.snmp.SnmpDefinitions;
15  import com.sun.jmx.snmp.SnmpStatusException;
16  
17  
18  /**
19   * Contains a set of resources that are used by while sending or receiving
20   * packets to and from an <CODE>SnmpPeer</CODE>. An <CODE>SnmpPeer</CODE> can
21   * be configured explicitly to use a specific <CODE>SnmpParameter</CODE>.
22   * A number of <CODE>SnmpPeer</CODE> objects can share a single parameter
23   * object.
24   * <P>
25   * <B>Note</B>: Changing values for an <CODE>SnmpParameter</CODE> object
26   * affects all <CODE>SnmpPeer</CODE> objects that share the parameter object.
27   *
28   * @see com.sun.jmx.snmp.SnmpPeer
29   *
30   * <p><b>This API is a Sun Microsystems internal API  and is subject
31   * to change without notice.</b></p>
32   */
33  
34  
35  public class SnmpParameters extends SnmpParams implements Cloneable, Serializable {
36      private static final long serialVersionUID = -1822462497931733790L;
37  
38      /**
39       * Creates an <CODE>SnmpParameters</CODE> object with defaults set up.
40       * By default, <CODE>set</CODE> operations are not allowed, the read community and
41       * the inform community strings to use is "public" and the SNMP version is V1.
42       */
43      public SnmpParameters() {
44          _readCommunity = defaultRdCommunity ;
45          _informCommunity = defaultRdCommunity ;
46      }
47  
48      /**
49       * Creates an <CODE>SnmpParameters</CODE> object.
50       * This constructor allows the specification of the read/write community strings.
51       * The inform community string to use is "public".
52       *
53       * @param rdc community string to use for <CODE>get</CODE> operations.
54       * @param wrc community string to use for <CODE>set</CODE> operations.
55       */
56      public SnmpParameters(String rdc, String wrc) {
57          _readCommunity = rdc ;
58          _writeCommunity = wrc ;
59          _informCommunity = defaultRdCommunity ;
60      }
61  
62      /**
63       * Creates an <CODE>SnmpParameters</CODE> object.
64       * This constructor allows the specification of the read/write/inform community strings.
65       *
66       * @param rdc community string to use for <CODE>get</CODE> operations.
67       * @param wrc community string to use for <CODE>set</CODE> operations.
68       * @param inform community string to use for <CODE>inform</CODE> requests.
69       */
70      public SnmpParameters(String rdc, String wrc, String inform) {
71          _readCommunity = rdc ;
72          _writeCommunity = wrc ;
73          _informCommunity = inform ;
74      }
75  
76      /**
77       * Gets the community to be used when issuing <CODE>get</CODE> operations.
78       * @return The community string.
79       */
80      public String getRdCommunity() {
81          return _readCommunity ;
82      }
83  
84      /**
85       * Sets the community string to use when performing <CODE>get</CODE> operations.
86       * @param read The community string.
87       */
88      public synchronized void setRdCommunity(String read) {
89          if (read == null)
90              _readCommunity = defaultRdCommunity ;
91          else
92              _readCommunity = read ;
93      }
94  
95      /**
96       * Gets the community to be used when issuing <CODE>set</CODE> operations.
97       * @return The community string.
98       */
99      public String getWrCommunity() {
100         return _writeCommunity ;
101     }
102 
103     /**
104      * Sets the community to be used when issuing <CODE>set</CODE> operations.
105      * @param write The community string.
106      */
107     public void setWrCommunity(String write) {
108         _writeCommunity = write;
109     }
110 
111     /**
112      * Gets the community to be used when issuing <CODE>inform</CODE> requests.
113      * @return The community string.
114      */
115     public String getInformCommunity() {
116         return _informCommunity ;
117     }
118 
119     /**
120      * Sets the community string to use when performing <CODE>inform</CODE> requests.
121      * @param inform The community string.
122      */
123     public void setInformCommunity(String inform) {
124         if (inform == null)
125             _informCommunity = defaultRdCommunity ;
126         else
127             _informCommunity = inform ;
128     }
129 
130     /**
131      * Checks whether parameters are in place for an SNMP <CODE>set</CODE> operation.
132      * @return <CODE>true</CODE> if parameters are in place, <CODE>false</CODE> otherwise.
133      */
134     public boolean allowSnmpSets() {
135         return _writeCommunity != null ;
136     }
137 
138     /**
139      * Compares two objects.
140      * Two <CODE>SnmpParameters</CODE> are equal if they correspond to the same protocol version,
141      * read community and write community.
142      * @param obj The object to compare <CODE>this</CODE> with.
143      * @return <CODE>true</CODE> if <CODE>this</CODE> and the specified object are equal, <CODE>false</CODE> otherwise.
144      */
145     @Override
146     public synchronized boolean equals(Object obj) {
147         if (!( obj instanceof SnmpParameters))
148             return false;
149 
150         if (this == obj)
151             return true ;
152         SnmpParameters param = (SnmpParameters) obj ;
153         if (_protocolVersion == param._protocolVersion)
154             if (_readCommunity.equals(param._readCommunity))
155                 return true ;
156         return false ;
157     }
158 
159     @Override
160     public synchronized int hashCode() {
161         return (_protocolVersion * 31) ^ Objects.hashCode(_readCommunity);
162     }
163 
164     /**
165      * Clones the object and its content.
166      * @return The object clone.
167      */
168     public synchronized Object clone() {
169         SnmpParameters par = null ;
170         try {
171             par = (SnmpParameters) super.clone() ;
172             //par._retryPolicy = _retryPolicy ;
173             par._readCommunity = _readCommunity ;
174             par._writeCommunity = _writeCommunity ;
175             par._informCommunity = _informCommunity ;
176         } catch (CloneNotSupportedException e) {
177             throw new InternalError() ; // VM bug.
178         }
179         return par ;
180     }
181 
182     /**
183      * For SNMP Runtime internal use only.
184      */
185     public byte[] encodeAuthentication(int snmpCmd)
186         throws SnmpStatusException {
187         //
188         // Returns the community string associated to the specified command.
189         //
190         try {
191             if (snmpCmd == pduSetRequestPdu)
192                 return _writeCommunity.getBytes("8859_1");
193             else if (snmpCmd == pduInformRequestPdu)
194                 return _informCommunity.getBytes("8859_1") ;
195             else
196                 return _readCommunity.getBytes("8859_1") ;
197         }catch(UnsupportedEncodingException e) {
198             throw new SnmpStatusException(e.getMessage());
199         }
200     }
201 
202     /**
203      * Specify the default community string to use for <CODE>get</CODE> operations.
204      * By default, the value is "public".
205      */
206     final static String defaultRdCommunity = "public" ;
207 
208     /**
209      * The protocol version.
210      * By default, the value is SNMP version 1.
211      * @serial
212      */
213     private int         _protocolVersion = snmpVersionOne ;
214     /**
215      * The community to be used when issuing <CODE>get</CODE> operations.
216      * @serial
217      */
218     private String      _readCommunity ;
219     /**
220      * The community to be used when issuing <CODE>set</CODE> operations.
221      * @serial
222      */
223     private String      _writeCommunity ;
224     /**
225      * The community to be used when issuing <CODE>inform</CODE> requests.
226      * @serial
227      */
228     private String      _informCommunity ;
229     /**
230      */
231     //private int               _retryPolicy ;  // not implemented as yet.
232 }
