13  /**
14   * Contains a list of <CODE>SnmpVarBind</CODE> objects.
15   * This class helps to create an <CODE>SnmpVarBindList</CODE> from a list of MIB variable names.
16   * In addition, it contains different forms of methods which can copy or clone the list.
17   * This list is required by any SNMP entity which specifies a list of variables to query.
18   * <p><b>This API is a Sun Microsystems internal API  and is subject
19   * to change without notice.</b></p>
20   */
21  
22  public class SnmpVarBindList extends Vector<SnmpVarBind> {
23      private static final long serialVersionUID = -7203997794636430321L;

25      /**
26       * A name given to the <CODE>SnmpVarBindList</CODE>. Useful for debugging.
27       * The default name is "VarBindList".

33       * Valid only for <CODE>SnmpGet</CODE> and <CODE>SnmpGetNext</CODE> operations.
34       * <CODE>SnmpTimestamp</CODE> is null by default.
35       * Also, when the list is cloned without value the timestamp is not copied.

79      /**
80       * Creates a new <CODE>SnmpVarBindList</CODE> object from a plain vector of <CODE>SnmpVarBind</CODE> objects.
81       * Objects in the specified vector can be <CODE>SnmpVarBind</CODE> objects or derivatives.
82       * @param list The vector of <CODE>SnmpVarBind</CODE> objects to copy.
83       */
84      public SnmpVarBindList(Vector<SnmpVarBind> list) {
85          super(list.size(), 5);
86          for (Enumeration<SnmpVarBind> e = list.elements(); e.hasMoreElements();) {
87              final SnmpVarBind varBind = e.nextElement();
88              addElement(varBind.clone());
89          }
90      }

99          this(list);
100         identity = name;

152     /**
153      * Replaces the current variable binding list of <CODE>SnmpVarBind</CODE> with the new specified variable binding
154      * list of <CODE>SnmpVarBind</CODE> objects.
155      * This method only clones the vector. It does not clone the <CODE>SnmpVarBind</CODE> objects
156      * contained in the list.
157      * @param list A vector of <CODE>SnmpVarBind</CODE> objects.
158      */
159     public final synchronized void setVarBindList(Vector<SnmpVarBind> list) {
160         setVarBindList(list, false) ;
161     }
162 
163     /**
164      * Replaces the current variable binding list of <CODE>SnmpVarBind</CODE> objects with the new variable binding
165      * list of <CODE>SnmpVarBind</CODE> objects.
166      * If <CODE>copy</CODE> is <CODE>true</CODE>, it will clone each <CODE>SnmpVarBind</CODE> object
167      * contained in the list.
168      * @param list A vector of <CODE>SnmpVarBind</CODE> objects.
169      * @param copy The flag indicating whether each object in the list should be cloned.
170      */
171     public final synchronized void setVarBindList(Vector<SnmpVarBind> list, boolean copy) {
172         synchronized (list) {
173             final int max = list.size();
174             setSize(max) ;
175             list.copyInto(this.elementData) ;
176             if (copy) {         // do deepcopy of all vars.
177                 for (int i = 0; i < max ; i++) {
178                     SnmpVarBind avar = (SnmpVarBind)elementData[i] ;
179                     elementData[i] = avar.clone() ;
180                 }
181             }
182         }
183     }

225     /**
226      * Prepares a vector of <CODE>SnmpVarBindList</CODE> from an array of SNMP MIB variables and instances.
227      * @param list An array of <CODE>String</CODE> containing MIB variable names.
228      * @param inst A common instance for each of the MIB variables in <CODE>vlist</CODE>.
229      * @exception SnmpStatusException An error occurred while accessing a MIB node.
230      */
231     public final synchronized void addVarBind(String list[], String inst) throws SnmpStatusException {
232         for (int i = 0; i < list.length; i++) {
233             SnmpVarBind avar = new SnmpVarBind(list[i]) ;
234             avar.addInstance(inst) ;
235             addElement(avar) ;
236         }

239     /**
240      * Removes the array of SNMP MIB variables and instances from the existing <CODE>SnmpVarBindList</CODE>.
241      * @param list An array of <CODE>String</CODE> containing MIB variable names.
242      * @param inst A common instance for each of the MIB variables in <CODE>vlist</CODE>.
243      * @return <CODE>true</CODE> if all the SNMP MIB variables were components of this <CODE>SnmpVarBindList</CODE>,
244      * <CODE>false</CODE> otherwise.
245      * @exception SnmpStatusException An error occurred while accessing a MIB node.
246      */
247     public synchronized boolean removeVarBind(String list[], String inst) throws SnmpStatusException {
248         boolean result = true;
249         for (int i = 0; i < list.length; i++) {
250             SnmpVarBind avar = new SnmpVarBind(list[i]) ;
251             avar.addInstance(inst) ;
252             int indexOid = indexOfOid(avar) ;
253             try {
254                 removeElementAt(indexOid) ;
255             } catch (ArrayIndexOutOfBoundsException e) {
256                 result = false ;
257             }
258         }
259         return result ;
260     }

367     /**
368      * Returns <CODE>false</CODE> if any of the variables does not contain a valid value.
369      * Typically used for <CODE>SnmpSet</CODE> operations.
370      * @return <CODE>false</CODE> if any of the variables does not contain a valid value, <CODE>true</CODE> otherwise.
371      */
372     public synchronized boolean checkForValidValues() {
373         int max= this.size();
374         for (int i = 0; i < max ; i++) {
375             SnmpVarBind avar = (SnmpVarBind)elementData[i] ;
376             if (avar.isValidValue() == false)

378         }

396     /**
397      * Splits the <CODE>SnmpVarBindList</CODE>.
398      * @param pos The position at which to split the <CODE>SnmpVarBindList</CODE>
399      * @return The <CODE>SnmpVarBindList</CODE> list from the beginning up to the split position.
400      */
401     public synchronized SnmpVarBindList splitAt(int pos) {
402         SnmpVarBindList splitVb = null ;
403         if (pos > elementCount)
404             return splitVb ;
405         splitVb = new SnmpVarBindList() ; // size() - atPosition) ;
406         int max= size();
407         for (int i = pos; i < max ; i++)
408             splitVb.addElement((SnmpVarBind) elementData[i]) ;
409         elementCount = pos ;
410         trimToSize() ;
411         return splitVb ;
412     }

414     /**
415      * Gives the index of an OID in the <CODE>SnmpVarBindList</CODE>.
416      * The index returned must be greater than or equal to the <CODE>start</CODE> parameter
417      * and smaller than the <CODE>end</CODE> parameter. Otherwise the method returns -1.
418      * @param var The <CODE>SnmpVarBind</CODE> object with the requested OID.
419      * @param min The min index in <CODE>SnmpVarBindList</CODE>.
420      * @param max The max index in <CODE>SnmpVarBindList</CODE>.
421      * @return The index of the OID in <CODE>SnmpVarBindList</CODE>.
422      */
423     public synchronized int indexOfOid(SnmpVarBind var, int min, int max) {
424         SnmpOid oidarg = var.getOid() ;
425         for (int i = min; i < max ; i++) {
426             SnmpVarBind avar = (SnmpVarBind)elementData[i] ;
427             if (oidarg.equals(avar.getOid()))
428                 return i ;
429         }
430         return -1 ;
431     }

499     /**
500      * Copies the <CODE>SnmpVarBindList</CODE> into a plain vector of <CODE>SnmpVarBind</CODE> objects.
501      * If the <code>copy</code> flag is false, does a shallow copy of the list. Otherwise,
502      * individual elements will be cloned.
503      * @param copy The flag indicating whether each object in the list should be cloned.
504      * @return A new vector of <CODE>SnmpVarBind</CODE> objects.
505      */
506     public synchronized Vector<SnmpVarBind> toVector(boolean copy) {
507         final int count = elementCount;
508         if (copy == false) return new Vector<>(this);
509         Vector<SnmpVarBind> result = new Vector<>(count,5);
510         for (int i = 0; i < count ; i++) {
511             SnmpVarBind avar = (SnmpVarBind)elementData[i] ;
512             result.addElement(avar.clone()) ;
513         }
514         return result;
515     }
