307     // API error status enumeration.
308     //------------------------------
309     /**
310      * Error code specific to the API.
311      * <P>Used if the request PDU exceeds the <CODE>maxTries</CODE> limit
312      * defined in the {@link com.sun.jmx.snmp.SnmpPeer SnmpPeer} object.
313      */
314     final public static int snmpReqTimeout          = (0xE0) ;
315 
316     /**
317      * Error code specific to the API.
318      * <P>Used if the pending request PDU is cancelled.
319      */
320     final public static int snmpReqAborted          = (0xE1) ;
321 
322     /**
323      * Error code specific to the API.
324      * <P>Used if the response to the request PDU fails at decoding time.
325      */
326     final public static int snmpRspDecodingError    = (0xE2) ;
327 
328     /**
329      * Error code specific to the API.
330      * <P>Used if the request PDU does not retreive the community string at
331      * authentication encoding time.
332      */
333     final public static int snmpReqEncodingError    = (0xE3) ;
334 
335     /**
336      * Error code specific to the API.
337      * <P>Used if the request PDU does not fit in the <CODE>maxSnmpPktSize</CODE> limit
338      * defined in the {@link com.sun.jmx.snmp.SnmpPeer SnmpPeer} object.
339      */
340     final public static int snmpReqPacketOverflow   = (0xE4) ;
341 
342     /**
343      * Error code specific to the API.
344      * <P>Used for the last variable returned by a <CODE>Walk</CODE> request PDU.
345      */
346     final public static int snmpRspEndOfTable       = (0xE5) ;
347 
348     /**
349      * Error code specific to the API.
350      * <P>Used if the request PDU is retried after a <CODE>noSuchName</CODE> error fixed.
351      * This is an internal temporary status.
352      */
353     final public static int snmpReqRefireAfterVbFix = (0xE6) ;
354 
355     /**
356      * Error code specific to the API.
357      * <P>Used if the size of the request PDU <CODE>varBindList</CODE> exceeds
358      * the <CODE>maxVarBindLimit</CODE> limit defined in the
359      * {@link com.sun.jmx.snmp.SnmpPeer SnmpPeer} object.
360      */
361     final public static int snmpReqHandleTooBig     = (0xE7) ;
362 
363     /**
364      * Error code specific to the API.
365      * <P>Used if the request PDU fails with a <CODE>tooBig</CODE> error status
366      * but its <CODE>varBindList</CODE> contains just one element.
367      */
368     final public static int snmpReqTooBigImpossible = (0xE8) ;
