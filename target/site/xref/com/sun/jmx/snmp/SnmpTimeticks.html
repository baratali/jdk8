10  /**
11   * Contains an <CODE>SnmpTimeTick</CODE> value which
12   * has units of 1/100th of a second.
13   *
14   * <p><b>This API is a Sun Microsystems internal API  and is subject
15   * to change without notice.</b></p>
16   *
17   */
18  
19  public class SnmpTimeticks extends SnmpUnsignedInt {
20  
21      // CONSTRUCTORS
22      //-------------
23      /**
24       * Constructs a new <CODE>SnmpTimeticks</CODE> from the specified
25       * integer value.
26       * @param v The initialization value.
27       * @exception IllegalArgumentException The specified value is negative.
28       */
29      public SnmpTimeticks(int v) throws IllegalArgumentException {
30          super(v) ;
31      }
32  
33      /**
34       * Constructs a new <CODE>SnmpTimeticks</CODE> from the specified
35       * <CODE>Integer</CODE> value.
36       * @param v The initialization value.
37       * @exception IllegalArgumentException The specified value is negative.
38       */
39      public SnmpTimeticks(Integer v) throws IllegalArgumentException {
40          super(v) ;
41      }
42  
43      /**
44       * Constructs a new <CODE>SnmpTimeticks</CODE> from the specified long
45       * value.
46       * <p>If the specified value is greater than {@link
47       * SnmpUnsignedInt#MAX_VALUE SnmpUnsignedInt.MAX_VALUE}, the SnmpTimeTicks
48       * will be initialized with <code>v%(SnmpUnsignedInt.MAX_VALUE+1)</code>.
49       * @param v The initialization value.
50       * @exception IllegalArgumentException if the specified value is negative.
51       */
52      public SnmpTimeticks(long v) throws IllegalArgumentException {
53          super(((v>0)?v&SnmpUnsignedInt.MAX_VALUE:v)) ;
54      }
55  
56      /**
57       * Constructs a new <CODE>SnmpTimeticks</CODE> from the specified
58       * <CODE>Long</CODE> value.
59       * <p>If the specified value is greater than {@link
60       * SnmpUnsignedInt#MAX_VALUE SnmpUnsignedInt.MAX_VALUE}, the SnmpTimeTicks
61       * will be initialized with <code>v%(SnmpUnsignedInt.MAX_VALUE+1)</code>.
62       * @param v The initialization value.
63       * @exception IllegalArgumentException if the specified value is negative.
64       */
65      public SnmpTimeticks(Long v) throws IllegalArgumentException {
66          this(v.longValue()) ;
67      }
68  
69      // PUBLIC METHODS
70      //---------------
71      /**
72       * Parses the specified long value with time units and
73       * returns a <CODE>String</CODE> of the form <CODE>d days hh:mm:ss</CODE>.
74       * @param timeticks The value to be parsed.
75       * @return The <CODE>String</CODE> representation of the value.
76       */
77      final static public String printTimeTicks(long timeticks) {
78          int seconds, minutes, hours, days;
79          StringBuffer buf = new StringBuffer() ;
80  
81          timeticks /= 100;
82          days = (int)(timeticks / (60 * 60 * 24));
83          timeticks %= (60 * 60 * 24);
84  
85          hours = (int)(timeticks / (60 * 60)) ;
86          timeticks %= (60 * 60);
87  
88          minutes = (int)(timeticks / 60) ;
89          seconds = (int)(timeticks % 60) ;
90  
91          if (days == 0) {
92              buf.append(hours + ":" + minutes + ":" + seconds) ;
93              return buf.toString() ;
94          }
95          if (days == 1) {
96              buf.append("1 day ") ;
97          } else {
98              buf.append(days + " days ") ;
99          }
100         buf.append(hours + ":" + minutes + ":" + seconds) ;
101         return buf.toString() ;
102     }
103 
104     /**
105      * Converts the timeticks value to its <CODE>String</CODE> form.
106      * The format of the returned <CODE>String</CODE> is <CODE>d days hh:mm:ss</CODE>.
107      * <BR>Note: this method simply calls the {@link #printTimeTicks printTimeTicks} method.
108      * @return The <CODE>String</CODE> representation of the value.
109      */
110     final public String toString() {
111         return printTimeTicks(value) ;
112     }
113 
114     /**
115      * Returns a textual description of the type object.
116      * @return ASN.1 textual description.
117      */
118     final public String getTypeName() {
119         return name;
120     }
121 
122     // VARIABLES
123     //----------
124     /**
125      * Name of the type.
126      */
127     final static String name = "TimeTicks" ;
128     static final private long serialVersionUID = -5486435222360030630L;
129 }
