26  package com.sun.corba.se.spi.ior;
27  
28  import java.util.List;
29  import java.util.ArrayList;
30  import java.util.Iterator;
31  
32  import com.sun.corba.se.impl.ior.FreezableList ;
33  
34  import com.sun.corba.se.spi.ior.TaggedComponent ;
35  import com.sun.corba.se.spi.ior.Identifiable ;
36  
37  /** Convenience class for defining objects that contain lists of Identifiables.
38   * Mainly implements iteratorById.  Also note that the constructor creates the
39   * list, which here is always an ArrayList, as this is much more efficient overall
40   * for short lists.
41   * @author  Ken Cavanaugh
42   */
43  public class IdentifiableContainerBase extends FreezableList
44  {
45      /** Create this class with an empty list of identifiables.
46       * The current implementation uses an ArrayList.
47       */
48      public IdentifiableContainerBase()
49      {
50          super( new ArrayList() ) ;
51      }
52  
53      /** Return an iterator which iterates over all contained Identifiables
54       * with type given by id.
55       */
56      public Iterator iteratorById( final int id)
57      {
58          return new Iterator() {
59              Iterator iter = IdentifiableContainerBase.this.iterator() ;
60              Object current = advance() ;
61  
62              private Object advance()
63              {
64                  while (iter.hasNext()) {
65                      Identifiable ide = (Identifiable)(iter.next()) ;
66                      if (ide.getId() == id)
67                          return ide ;
68                  }
69  
70                  return null ;
71              }
72  
73              public boolean hasNext()
74              {
75                  return current != null ;
76              }
77  
78              public Object next()
79              {
80                  Object result = current ;
81                  current = advance() ;
82                  return result ;
83              }
84  
85              public void remove()
86              {
87                  iter.remove() ;
88              }
89          } ;
90      }
91  }
