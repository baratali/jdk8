26  package com.sun.corba.se.spi.orbutil.threadpool;
27  
28  
29  public interface ThreadPoolChooser
30  {
31      /**
32       * This method is used to return an instance of ThreadPool based on the
33       * strategy/policy implemented in the ThreadPoolChooser from the set of
34       * ThreadPools allowed to be used by the ORB. Typically, the set of
35       * ThreadPools would be specified by passing the threadpool-ids
36       * configured in the ORB element of the domain.xml of the appserver.
37       */
38      public ThreadPool getThreadPool();
39  
40      /**
41       * This method is used to return an instance of ThreadPool that is obtained
42       * by using the id argument passed to it. This method will be used in
43       * situations where the threadpool id is known to the caller e.g. by the
44       * connection object or looking at the high order bits of the request id
45       */
46      public ThreadPool getThreadPool(int id);
47  
48      /**
49       * This method is a convenience method to see what threadpool-ids are being
50       * used by the ThreadPoolChooser
51       */
52      public String[] getThreadPoolIds();
53  }
