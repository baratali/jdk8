26  package org.omg.CORBA;
27  
28  /**
29   * A class that contains user exceptions returned by the server.
30   * When the client uses the DII to make an invocation, any user exception
31   * returned from the server is enclosed in an <code>Any</code> object contained in the
32   * <code>UnknownUserException</code> object. This is available from the
33   * <code>Environment</code> object returned by the method <code>Request.env</code>.
34   *
35   * @see <A href="../../../../technotes/guides/idl/jidlExceptions.html">documentation on
36   * Java&nbsp;IDL exceptions</A>
37   * @see Request
38   */
39  
40  public final class UnknownUserException extends UserException {
41  
42      /** The <code>Any</code> instance that contains the actual user exception thrown
43       *  by the server.
44       * @serial
45       */
46      public Any except;
47  
48      /**
49       * Constructs an <code>UnknownUserException</code> object.
50       */
51      public UnknownUserException() {
52          super();
53      }
54  
55      /**
56       * Constructs an <code>UnknownUserException</code> object that contains the given
57       * <code>Any</code> object.
58       *
59       * @param a an <code>Any</code> object that contains a user exception returned
60       *         by the server
61       */
62      public UnknownUserException(Any a) {
63          super();
64          except = a;
65      }
66  }
