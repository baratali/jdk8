26  package org.omg.CORBA;
27  
28  /**
29   * This exception is raised if a client attempts to retrieve the result
30   * of a deferred synchronous call, but the response for the request is
31   * not yet available.<P>
32   * It contains a minor code, which gives more detailed information about
33   * what caused the exception, and a completion status. It may also contain
34   * a string describing the exception.
35   *
36   * @see <A href="../../../../technotes/guides/idl/jidlExceptions.html">documentation on
37   * Java&nbsp;IDL exceptions</A>
38   * @since       JDK1.2
39   */
40  
41  public final class NO_RESPONSE extends SystemException {
42      /**
43       * Constructs a <code>NO_RESPONSE</code> exception with a default minor code
44       * of 0, a completion state of CompletionStatus.COMPLETED_NO,
45       * and a null description.
46       */
47      public NO_RESPONSE() {
48          this("");
49      }
50  
51      /**
52       * Constructs a <code>NO_RESPONSE</code> exception with the specified description message,
53       * a minor code of 0, and a completion state of COMPLETED_NO.
54       * @param s the String containing a description message
55       */
56      public NO_RESPONSE(String s) {
57          this(s, 0, CompletionStatus.COMPLETED_NO);
58      }
59  
60      /**
61       * Constructs a <code>NO_RESPONSE</code> exception with the specified
62       * minor code and completion status.
63       * @param minor the minor code
64       * @param completed the completion status
65       */
66      public NO_RESPONSE(int minor, CompletionStatus completed) {
67          this("", minor, completed);
68      }
69  
70      /**
71       * Constructs a <code>NO_RESPONSE</code> exception with the specified description
72       * message, minor code, and completion status.
73       * @param s the String containing a description message
74       * @param minor the minor code
75       * @param completed the completion status
76       */
77      public NO_RESPONSE(String s, int minor, CompletionStatus completed) {
78          super(s, minor, completed);
79      }
80  }
