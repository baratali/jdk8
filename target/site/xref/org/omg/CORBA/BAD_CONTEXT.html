26  package org.omg.CORBA;
27  
28  /**
29   * Exception thrown when an operation is invoked by a client but the passed
30   * context does not contain the context values required by the operation.<P>
31   * It contains a minor code, which gives more detailed information about
32   * what caused the exception, and a completion status. It may also contain
33   * a string describing the exception.
34   *
35   * @see <A href="../../../../technotes/guides/idl/jidlExceptions.html">documentation on
36   * Java&nbsp;IDL exceptions</A>
37   * @since    JDK1.2
38   */
39  
40  public final class BAD_CONTEXT extends SystemException {
41  
42      /**
43       * Constructs a <code>BAD_CONTEXT</code> exception
44       * with a default minor code
45       * of 0 and a completion state of COMPLETED_NO.
46       */
47      public BAD_CONTEXT() {
48          this("");
49      }
50  
51      /**
52       * Constructs a <code>BAD_CONTEXT</code> exception
53       * with the specified detail message, a minor code
54       * of 0 and a completion state of COMPLETED_NO.
55       * @param s a <code>String</code> object containing a detail message
56       */
57      public BAD_CONTEXT(String s) {
58          this(s, 0, CompletionStatus.COMPLETED_NO);
59      }
60  
61      /**
62       * Constructs a <code>BAD_CONTEXT</code> exception
63       * with the specified
64       * minor code and completion status.
65       * @param minor the minor code
66       * @param completed an instance of <code>CompletionStatus</code> indicating
67       *                  the completion status
68       */
69      public BAD_CONTEXT(int minor, CompletionStatus completed) {
70          this("", minor, completed);
71      }
72  
73      /**
74       * Constructs a <code>BAD_CONTEXT</code> exception
75       * with the specified detail
76       * message, minor code, and completion status.
77       * A detail message is a String that describes this particular exception.
78       * @param s the String containing a detail message
79       * @param minor the minor code
80       * @param completed an instance of <code>CompletionStatus</code> indicating
81       *                  the completion status
82       */
83      public BAD_CONTEXT(String s, int minor, CompletionStatus completed) {
84          super(s, minor, completed);
85      }
86  }
