32  package org.omg.CORBA.portable;
33  /**
34   * The org.omg.CORBA.portable.UnknownException is used for reporting
35   * unknown exceptions between ties and ORBs and between ORBs and stubs.
36   * It provides a Java representation of an UNKNOWN system exception
37   * that has an UnknownExceptionInfo service context.
38   * If the CORBA system exception org.omg.CORBA.portable.UnknownException
39   * is thrown, then the stub does one of the following:
40   * (1) Translates it to org.omg.CORBA.UNKNOWN.
41   * (2) Translates it to the nested exception that the UnknownException contains.
42   * (3) Passes it on directly to the user.
43   */
44  public class UnknownException extends org.omg.CORBA.SystemException {
45      /**
46       * A throwable--the original exception that was wrapped in a CORBA
47       * UnknownException.
48       */
49      public Throwable originalEx;
50      /**
51       * Constructs an UnknownException object.
52       * @param ex a Throwable object--to be wrapped in this exception.
53       */
54      public UnknownException(Throwable ex) {
55          super("", 0, org.omg.CORBA.CompletionStatus.COMPLETED_MAYBE);
56          originalEx = ex;
57      }
58  }
