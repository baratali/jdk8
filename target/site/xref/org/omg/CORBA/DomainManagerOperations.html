26  package org.omg.CORBA;
27  
28  /**
29   *  Provides the <tt>DomainManager</tt> with the means to access policies.
30   *  <P>
31   *  The <tt>DomainManager</tt> has associated with it the policy objects for a
32   *  particular domain. The domain manager also records the membership of
33   *  the domain and provides the means to add and remove members. The domain
34   *  manager is itself a member of a domain, possibly the domain it manages.
35   *  The domain manager provides mechanisms for establishing and navigating
36   *  relationships to superior and subordinate domains and
37   *  creating and accessing policies.
38   */
39  
40  public interface DomainManagerOperations
41  {
42      /** This returns the policy of the specified type for objects in
43       *  this domain.  The types of policies available are domain specific.
44       *  See the CORBA specification for a list of standard ORB policies.
45       *
46       *@param policy_type Type of policy to request
47       */
48      public org.omg.CORBA.Policy get_domain_policy(int policy_type);
49  }
