42  package org.w3c.dom;
43  
44  /**
45   *  The <code>DOMStringList</code> interface provides the abstraction of an
46   * ordered collection of <code>DOMString</code> values, without defining or
47   * constraining how this collection is implemented. The items in the
48   * <code>DOMStringList</code> are accessible via an integral index, starting
49   * from 0.
50   * <p>See also the <a href='http://www.w3.org/TR/2004/REC-DOM-Level-3-Core-20040407'>Document Object Model (DOM) Level 3 Core Specification</a>.
51   * @since DOM Level 3
52   */
53  public interface DOMStringList {
54      /**
55       *  Returns the <code>index</code>th item in the collection. If
56       * <code>index</code> is greater than or equal to the number of
57       * <code>DOMString</code>s in the list, this returns <code>null</code>.
58       * @param index Index into the collection.
59       * @return  The <code>DOMString</code> at the <code>index</code>th
60       *   position in the <code>DOMStringList</code>, or <code>null</code> if
61       *   that is not a valid index.
62       */
63      public String item(int index);
64  
65      /**
66       * The number of <code>DOMString</code>s in the list. The range of valid
67       * child node indices is 0 to <code>length-1</code> inclusive.
68       */
69      public int getLength();
70  
71      /**
72       *  Test if a string is part of this <code>DOMStringList</code>.
73       * @param str  The string to look for.
74       * @return  <code>true</code> if the string has been found,
75       *   <code>false</code> otherwise.
76       */
77      public boolean contains(String str);
78  
79  }
